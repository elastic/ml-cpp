57 T pop() {
58 std::unique_lock<std::mutex> lock(m_Mutex);
59 this->waitWhileEmpty(lock);
60
61 size_t oldSize{m_Queue.size()};
62 T result{std::move(m_Queue.front())};
63 m_Queue.pop_front();
64
65 this->notifyIfNoLongerFull(lock, oldSize);

67 }

142private:
143 void waitWhileEmpty(std::unique_lock<std::mutex>& lock) {
144 m_ConsumerCondition.wait(lock, [this] { return m_Queue.size() > 0; });
145 }
146 void waitWhileFull(std::unique_lock<std::mutex>& lock) {
147 m_ProducerCondition.wait(
148 lock, [this] { return m_Queue.size() < m_QueueCapacity; });
149 }
150
151 void notifyIfNoLongerFull(std::unique_lock<std::mutex>& lock, std::size_t oldSize) {
152 lock.unlock();
153 if (oldSize >= m_NotifyCapacity) {
154 m_ProducerCondition.notify_all();
155 }
156 }
157 void notifyIfNoLongerEmpty(std::unique_lock<std::mutex>& lock, std::size_t oldSize) {
158 lock.unlock();
159 if (oldSize == 0) {
160 m_ConsumerCondition.notify_all();
161 }
162 }
