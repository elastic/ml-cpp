52template<class STATISTIC>
53class CMetricPartialStatistic {
54public:
55 using TDouble1Vec = core::CSmallVector<double, 1>;
56 using TMeanAccumulator =
57 maths::common::CBasicStatistics::SSampleMean<maths::common::CDoublePrecisionStorage>::TAccumulator;
58
59public:
60 static const std::string VALUE_TAG;
61 static const std::string TIME_TAG;
62
63public:
64 CMetricPartialStatistic(std::size_t dimension)
65 : m_Value(CMetricStatisticWrappers::template make<STATISTIC>(dimension)) {}
66
68 void persist(core::CStatePersistInserter& inserter) const {
69 CMetricStatisticWrappers::persist(m_Value, VALUE_TAG, inserter);
70 inserter.insertValue(TIME_TAG, m_Time.toDelimited());
71 }
72
74 bool restore(core::CStateRestoreTraverser& traverser) {
75 do {
76 const std::string& name = traverser.name();
77 if (name == VALUE_TAG) {
78 if (CMetricStatisticWrappers::restore(traverser, m_Value) == false) {
79 LOG_ERROR(<< "Invalid statistic in " << traverser.value());
80 return false;
81 }
82 } else if (name == TIME_TAG) {
83 if (m_Time.fromDelimited(traverser.value()) == false) {
84 LOG_ERROR(<< "Invalid time in " << traverser.value());
85 return false;
86 }
87 }
88 } while (traverser.next());
89 return true;
90 }
91
97 inline void add(const TDouble1Vec& value, core_t::TTime time, unsigned int count) {
98 CMetricStatisticWrappers::add(value, count, m_Value);
99 m_Time.add(static_cast<double>(time), count);
100 }
101
103 inline TDouble1Vec value() const {
104 return CMetricStatisticWrappers::value(m_Value);
105 }
106
108 inline double count() const {
109 return maths::common::CBasicStatistics::count(m_Time);
110 }
111
113 inline core_t::TTime time() const {
114 return static_cast<core_t::TTime>(maths::common::CBasicStatistics::mean(m_Time) + 0.5);
115 }
116
118 inline const CMetricPartialStatistic& operator+=(const CMetricPartialStatistic& rhs) {
119 m_Value += rhs.m_Value;
120 m_Time += rhs.m_Time;
121 return *this;
122 }
123
125 inline uint64_t checksum(uint64_t seed) const {
126 seed = maths::common::CChecksum::calculate(seed, m_Value);
127 return maths::common::CChecksum::calculate(seed, m_Time);
128 }
129
131 inline void debugMemoryUsage(const core::CMemoryUsage::TMemoryUsagePtr& mem) const {
132 mem->setName("CMetricPartialStatistic", sizeof(*this));
133 core::memory_debug::dynamicSize("m_Value", m_Value, mem);
134 core::memory_debug::dynamicSize("m_Time", m_Time, mem);
135 }
136
138 inline std::size_t memoryUsage() const {
139 return sizeof(*this) + core::memory::dynamicSize(m_Value) +
140 core::memory::dynamicSize(m_Time);
141 }
142
144 inline std::string print() const {
145 std::ostringstream result;
146 result << m_Value << ' ' << maths::common::CBasicStatistics::mean(m_Time);
147 return result.str();
148 }
149
150private:
151 STATISTIC m_Value;
152 TMeanAccumulator m_Time;
153};

159}
