98template<typename POINT, EInfoCriterionType TYPE>
99class CSphericalGaussianInfoCriterion {
100public:
101 using TPointVec = std::vector<POINT>;
102 using TPointVecVec = std::vector<TPointVec>;
103 using TBarePoint = typename SUnannotated<POINT>::Type;
104 using TBarePointPrecise = typename SFloatingPoint<TBarePoint, double>::Type;
105 using TCoordinate = typename SCoordinate<TBarePointPrecise>::Type;
106 using TMeanVarAccumulator =
107 typename CBasicStatistics::SSampleMeanVar<TBarePointPrecise>::TAccumulator;
108
109public:
110 CSphericalGaussianInfoCriterion() = default;
111 explicit CSphericalGaussianInfoCriterion(const TPointVecVec& x) {
112 this->add(x);
113 }
114 explicit CSphericalGaussianInfoCriterion(const TPointVec& x) {
115 this->add(x);
116 }
117
119 void add(const TPointVecVec& x) {
120 for (const auto& xi : x) {
121 this->add(xi);
122 }
123 }
124
126 void add(const TPointVec& x) {
127 if (x.size() > 0) {
128 TMeanVarAccumulator moments(las::zero(x[0]));
129 moments.add(x);
130 this->add(moments);
131 }
132 }
133
135 void add(const TMeanVarAccumulator& moments) {
136 double ni = CBasicStatistics::count(moments);
137 const TBarePointPrecise& mean{CBasicStatistics::mean(moments)};
138 const TBarePointPrecise& covarianceDiag{
139 CBasicStatistics::maximumLikelihoodVariance(moments)};
140 std::size_t d{las::dimension(covarianceDiag)};
141 double vi{0.0};
142 for (std::size_t i = 0; i < d; ++i) {
143 vi += covarianceDiag(i);
144 }
145 vi = std::max(vi, EPS * las::norm(mean));
146
147 m_D = static_cast<double>(d);
148 m_K += 1.0;
149 m_N += ni;
150 if (ni > 1.0) {
151 double upper = information_criteria_detail::confidence(ni - 1.0);
152 m_Likelihood += ni * log(ni) - 0.5 * m_D * ni *
153 (1.0 + core::constants::LOG_TWO_PI +
154 std::log(upper * vi / m_D));
155 } else {
156 m_Likelihood += ni * log(ni) - 0.5 * m_D * ni *
157 (1.0 + core::constants::LOG_TWO_PI +
158 core::constants::LOG_MAX_DOUBLE);
159 }
160 }
161
163 double calculate(double p = 0.0) const {
164 if (m_N != 0.0) {
165 double logN{std::log(m_N)};
166 p += m_D * m_K + 2.0 * m_K - 1.0;
167 switch (TYPE) {
168 case E_BIC:
169 return -2.0 * (m_Likelihood - m_N * logN) + p * logN;
170 case E_AICc:
171 return -2.0 * (m_Likelihood - m_N * logN) + 2.0 * p +
172 p * (p + 1.0) / (m_N - p - 1.0);
173 }
174 }
175 return 0.0;
176 }
177
178private:
179 static constexpr double EPS{10.0 * std::numeric_limits<TCoordinate>::epsilon()};
180
181private:
183 double m_D = 0.0;
185 double m_K = 0.0;
187 double m_N = 0.0;
189 double m_Likelihood = 0.0;
190};

200template<typename POINT, EInfoCriterionType TYPE>
201class CGaussianInfoCriterion {
202public:
203 using TPointVec = std::vector<POINT>;
204 using TPointVecVec = std::vector<TPointVec>;
205 using TBarePoint = typename SUnannotated<POINT>::Type;
206 using TBarePointPrecise = typename SFloatingPoint<TBarePoint, double>::Type;
207 using TCoordinate = typename SCoordinate<TBarePointPrecise>::Type;
208 using TCovariances = CBasicStatistics::SSampleCovariances<TBarePointPrecise>;
209
210public:
211 CGaussianInfoCriterion() = default;
212 explicit CGaussianInfoCriterion(const TPointVecVec& x) { this->add(x); }
213 explicit CGaussianInfoCriterion(const TPointVec& x) { this->add(x); }
214
216 void add(const TPointVecVec& x) {
217 for (const auto& xi : x) {
218 this->add(xi);
219 }
220 }
221
223 void add(const TPointVec& x) {
224 if (x.size() > 0) {
225 TCovariances covariances(las::dimension(x[0]));
226 covariances.add(x);
227 this->add(covariances);
228 }
229 }
230
232 void add(const TCovariances& covariance) {
233 double ni{CBasicStatistics::count(covariance)};
234 m_D = static_cast<double>(las::dimension(CBasicStatistics::mean(covariance)));
235 m_K += 1.0;
236 m_N += ni;
237 m_Likelihood += ni * log(ni) -
238 0.5 * ni *
239 (m_D + m_D * core::constants::LOG_TWO_PI +
240 (ni <= m_D + 1.0 ? core::constants::LOG_MAX_DOUBLE
241 : this->logDeterminant(covariance)));
242 }
243
245 double calculate(double p = 0.0) const {
246 if (m_N != 0.0) {
247 double logN{std::log(m_N)};
248 p += m_D * (1.0 + 0.5 * (m_D + 1.0)) * m_K + m_K - 1.0;
249 switch (TYPE) {
250 case E_BIC:
251 return -2.0 * (m_Likelihood - m_N * logN) + p * logN;
252 case E_AICc:
253 return -2.0 * (m_Likelihood - m_N * logN) + 2.0 * p +
254 p * (p + 1.0) / (m_N - p - 1.0);
255 }
256 }
257 return 0.0;
258 }
259
260private:
262 double logDeterminant(const TCovariances& covariance) const {
263 double n{CBasicStatistics::count(covariance)};
264 const auto& c = CBasicStatistics::maximumLikelihoodCovariances(covariance);
265 double upper{information_criteria_detail::confidence(n - m_D - 1.0)};
266 return information_criteria_detail::logDeterminant(c, upper);
267 }
268
269private:
271 double m_D = 0.0;
273 double m_K = 0.0;
275 double m_N = 0.0;
277 double m_Likelihood = 0.0;
278};
279}
