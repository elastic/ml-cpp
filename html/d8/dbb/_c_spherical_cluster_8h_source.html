137template<typename POINT>
138struct SCovariancesLedoitWolf<CAnnotatedVector<POINT, SCountAndVariance>> {
139 template<typename OTHER_POINT>
140 static void
141 estimate(const std::vector<CAnnotatedVector<POINT, SCountAndVariance>>& points,
142 CBasicStatistics::SSampleCovariances<OTHER_POINT>& covariances) {
143 if (points.empty()) {
144 return;
145 }
146
147 using TCoordinate = typename SCoordinate<OTHER_POINT>::Type;
148 using TMatrix = typename SConformableMatrix<OTHER_POINT>::Type;
149
150 std::size_t dimension{las::dimension(points[0])};
151 TCoordinate d{static_cast<TCoordinate>(dimension)};
152
153 TCoordinate n{CBasicStatistics::count(covariances)};
154 if (n == TCoordinate{0}) {
155 return;
156 }
157
158 const OTHER_POINT& m{CBasicStatistics::mean(covariances)};
159 const TMatrix& s{CBasicStatistics::maximumLikelihoodCovariances(covariances)};
160
161 TCoordinate mn{s.trace() / d};
162 TCoordinate norm(0);
163 for (std::size_t i = 0; i < dimension; ++i) {
164 norm += pow2(s(i, i) - mn);
165 for (std::size_t j = 0; j < i; ++j) {
166 norm += TCoordinate(2) * pow2(s(i, j));
167 }
168 }
169 TCoordinate dn{norm / d};
170 TCoordinate bn(0);
171 TCoordinate z{n * n};
172 for (const auto& point : points) {
173 TCoordinate ni{static_cast<TCoordinate>(point.annotation().s_Count)};
174 TCoordinate vi{static_cast<TCoordinate>(point.annotation().s_Variance)};
175 norm = TCoordinate(0);
176 for (std::size_t i = 0; i < dimension; ++i) {
177 norm += pow2(pow2(TCoordinate(point(i)) - m(i)) + vi - s(i, i));
178 for (std::size_t j = 0; j < i; ++j) {
179 norm += TCoordinate(2) * pow2((TCoordinate(point(i)) - m(i)) *
180 (TCoordinate(point(j)) - m(j)) -
181 s(i, j));
182 }
183 }
184 bn += ni * norm / d / z;
185 }
186 bn = std::min(bn, dn);
187 LOG_TRACE(<< "m = " << mn << ", d = " << dn << ", b = " << bn);
188
189 covariances.s_Covariances *=
190 std::max((TCoordinate(1) - (bn == dn ? 1.0 : bn / dn)), 0.0);
191 for (std::size_t i = 0; i < dimension; ++i) {
192 covariances.s_Covariances(i, i) += (bn == dn ? 1.0 : bn / dn) * mn;
193 }
194 }
195
196 template<typename U>
197 static U pow2(U x) {
198 return x * x;
199 }
200};
201}
