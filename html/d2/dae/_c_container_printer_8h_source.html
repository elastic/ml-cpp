79class CElementPrinter<false> {
80public:
81 template<typename T>
82 static std::string print(const T& value) {
83 // Use CStringUtils if possible: it's much faster but only supports
84 // fundamental types.
85 return print_(value, std::is_arithmetic<T>{});
86 }
87
88private:
90 template<typename T>
91 inline static std::string print_(T value, true_ /*is arithmetic*/) {
92 // For signed types only.
93 if (value != T(0) && value == std::numeric_limits<T>::lowest()) {
94 return "\"min\"";
95 }
96 if (value == std::numeric_limits<T>::max()) {
97 return "\"max\"";
98 }
99 return CStringUtils::typeToStringPretty(value);
100 }
101
103 inline static std::string print_(bool value, true_ /*is arithmetic*/) {
104 return CStringUtils::typeToStringPretty(value);
105 }
106
108 inline static std::string print_(float value, true_ /*is arithmetic*/) {
109 return std::to_string(value);
110 }
111
113 template<typename T>
114 inline static std::string print_(const T& value, false_ /*is arithmetic*/) {
115 std::ostringstream result;
116 result << value;
117 return result.str();
118 }

206class CORE_EXPORT CContainerPrinter : private CNonInstantiatable {
207private:
208 static const std::string NULL_STR;
209 static constexpr std::size_t SIZE_OF_ELEMENT{20};

212 template<typename T>
213 static std::string printElement(const T& value) {
214 using namespace printer_detail;
215 using Forwarder = CPrinterForwarder<CContainerPrinter, has_const_iterator<T>::value>;
216 return Forwarder::print(value);
217 }

250 // it isn't printing please feel free to add an overload here.

253 template<typename T>
254 static std::string printElement(const std::optional<T>& value) {
255 return value == std::nullopt ? NULL_STR : printElement(*value);
256 }

259 template<typename U, typename V>
260 static std::string printElement(const std::pair<U, V>& value) {
261 std::string result;
262 result.reserve(2 * (SIZE_OF_ELEMENT + 2) + 2);
263 result += "(";
264 result += printElement(value.first);
265 result += ", ";
266 result += printElement(value.second);
267 result += ")";
268 return result;
269 }
270
272 template<typename... ARGS>
273 static std::string printElement(const std::tuple<ARGS...>& value) {
274 std::string result;
275 result.reserve((SIZE_OF_ELEMENT + 2) * sizeof...(ARGS) + 2);
276 std::apply(
277 [&result](ARGS const&... args) {
278 result += '(';
279 std::size_t n{0};
280 ((result += printElement(args) + (++n != sizeof...(ARGS) ? ", " : "")), ...);
281 result += ')';
282 },
283 value);
284 return result;
285 }
286
288 static const std::string& printElement(const std::string& value) {
289 return value;
290 }
291
292public:
294 template<typename T>
295 static auto print(const T& t) -> decltype(printElement(t)) {
296 return printElement(t);
297 }
298
302 template<typename ITR>
303 static std::string print(ITR begin, ITR end) {
304 std::string result;
305 result.reserve((SIZE_OF_ELEMENT + 2) * std::distance(begin, end) + 2);
306 result += "[";
307 if (begin != end) {
308 for (;;) {
309 result += printElement(*begin);
310 if (++begin == end) {
311 break;
312 }
313 result += ", ";
314 }
315 }
316 result += "]";
317 return result;
318 }

321 template<typename T, std::size_t SIZE>
322 static std::string print(const T (&array)[SIZE]) {
323 return print(std::begin(array), std::end(array));
324 }
