164 std::uint32_t operator()(std::uint32_t x1, std::uint32_t x2) const {
165 // Note by construction:
166 // (a(1) * x(1)) mod p + a(2) * x(2) + b
167 // < p^2 + 2*p
168 // < 2^64
169 std::uint64_t h = (static_cast<std::uint64_t>(m_A[0]) * x1) % BIG_PRIME +
170 static_cast<std::uint64_t>(m_A[1]) * x2;
171 return static_cast<std::uint32_t>(
172 ((h + static_cast<std::uint64_t>(m_B)) % BIG_PRIME) %
173 static_cast<std::uint64_t>(m_M));
174 }
175
179 std::uint32_t operator()(const TUInt32Vec& x) const {
180 // Note we variously use that:
181 // a(1) * x(1)
182 // < h mod p + a(i) * x(i)
183 // < h mod p + a(n) * x(n) + b
184 // < p^2 + 2*p
185 // < 2^64
186 std::uint64_t h = static_cast<std::uint64_t>(m_A[0]) * x[0];
187 for (std::size_t i = 1; i < x.size(); ++i) {
188 h = (h % BIG_PRIME + static_cast<std::uint64_t>(m_A[i]) * x[i]);
189 }
190 return static_cast<std::uint32_t>(
191 ((h + static_cast<std::uint64_t>(m_B)) % BIG_PRIME) %
192 static_cast<std::uint64_t>(m_M));
193 }
