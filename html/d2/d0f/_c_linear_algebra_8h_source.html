168 double frobenius(std::size_t d) const {
169 double result = 0.0;
170 for (std::size_t i = 0, i_ = 0; i < d; ++i, ++i_) {
171 for (std::size_t j = 0; j < i; ++j, ++i_) {
172 result += 2.0 * m_LowerTriangle[i_] * m_LowerTriangle[i_];
173 }
174 result += m_LowerTriangle[i_] * m_LowerTriangle[i_];

192 }

195 }
