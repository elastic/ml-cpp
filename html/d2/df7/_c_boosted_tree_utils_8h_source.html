152public:

157 bool empty() const { return m_Size == 0; }
158
160 std::size_t size() const { return m_Size; }

163 std::size_t upperBound(common::CFloatStorage x) const {
164 // These branches should be predictably false most of the time for our
165 // usage and so almost free.
166 if (m_Size == 0 || x < m_Min) {
167 return 0;
168 }
169 if (x >= INF) {
170 return m_Size;
171 }
172
173 std::size_t node{0};
174 // The tree structure doesn't arrange all values in order so we need to
175 // keep track of the number of out-of-order values preceding a position
176 // in the tree to map it back to its position in the sorted collection.
177 std::size_t numberOutOfOrderValues{0};
178 auto vecx = ml_broadcast_load_128(&x.cstorage());
179
180 for (auto branchSize : m_BranchSizes) {
181 std::size_t branch{selectBranch(&m_Values[node], vecx)};
182 LOG_TRACE(<< "node = " << node << "/" << this->printNode(node)
183 << ", branch = " << branch);
184
185 // Note that node is a multiple of 4. This follows from the fact that
186 // the step size 4 + 5^n - 1 is a multiple of 4 which can be shown by
187 // induction:
188 //
189 // 5^n - 1 = 5 * (5^(n - 1) - 1) + 4 and 5 - 1 = 1 * 4.
190 //
191 // This means that since m_Values are 16 byte aligned the values at node
192 // are 16 byte aligned and we can safely read them using an aligned load.
193 node += 4 + (branchSize - 1) * branch;
194
195 // Each split value which is greater than x is out of order w.r.t. this
196 // value and must be subtracted from node position to get the correct
197 // upper bound.
198 numberOutOfOrderValues += 4 - branch;
199 }
200
201 std::size_t branch{selectBranch(&m_Values[node], vecx)};
202 LOG_TRACE(<< "x = " << x << ", node = " << node << "/"
203 << this->printNode(node) << ", branch = " << branch
204 << ", number out of order values = " << numberOutOfOrderValues);
205
206 return node + branch + 1 - numberOutOfOrderValues;
207 }
208
209private:
210 using TAlignedFloatVec = std::vector<float, core::CAlignedAllocator<float>>;
211 using TSizeAry = std::array<std::size_t, 16>;
212
213private:
214 static constexpr float INF{std::numeric_limits<float>::infinity()};
215
216private:
217 void build(const TFloatVec& values, std::size_t a, std::size_t b);
218 std::string printNode(std::size_t node) const;
219 static constexpr TSizeAry MASK_TO_BRANCH_MAP{4, 0, 0, 0, 0, 0, 0, 0,
220 3, 0, 0, 0, 2, 0, 1, 0};
221 static std::size_t selectBranch(const float* values, ml_vec_128 vecx) {
222 auto vecv = ml_aligned_load_128(values);
223 auto less = ml_less_than_128(vecx, vecv);
224 auto mask = ml_bits_mov_mask_128(less);
225 return MASK_TO_BRANCH_MAP[mask];
226 }
227
228private:
229 std::size_t m_Size{0};
230 TSizeVec m_BranchSizes;
231 float m_Min{-INF};
232 TAlignedFloatVec m_Values;
