114 template<typename INT_TYPE>
115 static INT_TYPE gcd(std::vector<INT_TYPE> c) {
116 if (c.empty()) {
117 return 0;
118 }
119 if (c.size() == 1) {
120 return c[0];
121 }
122
123 // Repeatedly apply Euclid's algorithm and use the fact that
124 // gcd(a, b, c) = gcd(gcd(a, b), c).
125 INT_TYPE result{gcd(c[0], c[1])};
126 for (std::size_t i = 2; i < c.size(); ++i) {
127 result = gcd(result, c[i]);
128 }
129 return result;
130 }
