40class CIsCloseEnough {

42 using result_type = boost::test_tools::assertion_result;

44public:
45 template<typename T>
46 result_type operator()(const T& lhs, const T& rhs, const SAbsoluteTolerance<T>& tol) const {
47
48 T diff{(lhs < rhs) ? (rhs - lhs) : (lhs - rhs)};
49 result_type result{diff <= tol.s_Tolerance};
50 if (!result) {
