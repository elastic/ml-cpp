127 void add(std::size_t count, const TMemoryMappedFloatVector& derivatives) {
128 this->flatView() += derivatives;
129 *m_Count += static_cast<double>(count);
130 }
131
133 void add(const CDerivatives& rhs) {
134 this->flatView() += const_cast<CDerivatives*>(&rhs)->flatView();
135 *m_Count += *rhs.m_Count;
136 }
137
139 void subtract(const CDerivatives& rhs) {
140 *m_Count -= *rhs.m_Count;
141 if (*m_Count > 0) {
142 m_Gradient -= rhs.m_Gradient;
143 m_Curvature -= rhs.m_Curvature;
144 // None of our loss functions have negative curvature therefore we
145 // shouldn't allow the cumulative curvature to be negative either.
146 // In this case we force it to be a very small multiple of the
147 // magnitude of the gradient since this is the closest feasible
148 // estimate.
149 for (int i = 0; i < m_Gradient.size(); ++i) {
150 m_Curvature(i, i) = std::max(m_Curvature(i, i),
151 SMALLEST_RELATIVE_CURVATURE *
152 std::fabs(m_Gradient(i)));
153 }
154 } else {
155 // Numeric errors mean that it's possible the sum curvature for a
156 // candidate split is identically zero while the gradient is epsilon.
157 // This can cause the node gain to appear infinite (when there is no
158 // weight regularisation) which in turns causes problems initialising
159 // the region we search for optimal hyperparameter values. We can
160 // safely force the gradient and curvature to be zero if we detect
161 // that the count is zero.
162 m_Gradient.setZero();
163 m_Curvature.setZero();
164 }
165 }
166
168 void remapCurvature() {
169 // For performance, we accumulate curvatures into the first n + n (n + 1) / 2
170 // elements of the array backing flatView. However, the memory mapped matrix
171 // class expects them to be stored column major in the lower triangle of an
172 // n x n matrix. This copies them backwards to their correct positions.
173 TMemoryMappedDoubleVector derivatives{this->flatView()};
174 for (std::ptrdiff_t j = m_Curvature.cols() - 1, k = derivatives.rows() - 1;
175 j >= 0; --j) {
176 for (std::ptrdiff_t i = m_Curvature.rows() - 1; i >= j; --i, --k) {
177 m_Curvature(i, j) = derivatives(k);
178 }
179 }
180 }
181
183 std::uint64_t checksum(std::uint64_t seed = 0) const;
184
185 private:
186 TMemoryMappedDoubleVector flatView() {
187 // Gradient + upper triangle of the Hessian.
188 auto n = m_Gradient.rows();
189 return {m_Gradient.data(), n * (n + 3) / 2};
190 }
191
192 private:
193 double* m_Count;
194 TMemoryMappedDoubleVector m_Gradient;
195 TMemoryMappedDoubleMatrix m_Curvature;
196 };

484 template<typename SPLITS>
485 void mapStorage(const SPLITS& splits) {
486 // This function maps the memory in a single presized buffer containing
487 // enough space to store all gradient vectors and curvatures. For each
488 // feature the layout in this buffer is as follows:
489 //
490 // "split grad" "split hessian" "count" "missing grad" "missing hessian" "count"
491 // | | | | | |
492 // V V V V V V
493 // | n | n^2 | 1 | ... | n | n^2 | 1 |
494 //
495 // Note we ensure 16 byte alignment because we're using aligned memory
496 // mapped vectors which have better performance.
497
498 auto dimensionGradient = static_cast<int>(m_DimensionGradient);
499 std::size_t numberFeatures{splits.size()};
500 std::size_t numberGradients{this->numberGradients()};
501 std::size_t countOffset{this->countOffset()};
502 std::size_t splitSize{this->splitSize()};
503 double* storage{m_Storage.data()};
504 for (std::size_t i = 0; i < numberFeatures; ++i, storage += splitSize) {
505 std::size_t size{numberSplits(splits[i])};
506 m_Derivatives[i].reserve(size);
507 for (std::size_t j = 0; j < size; ++j, storage += splitSize) {
508 m_Derivatives[i].emplace_back(dimensionGradient, storage,
509 storage + numberGradients,
510 storage + countOffset);
511 }
512 }
513 }
