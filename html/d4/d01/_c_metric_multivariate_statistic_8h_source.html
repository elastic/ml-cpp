47template<class STATISTIC>
48class CMetricMultivariateStatistic {
49public:
50 using TDouble1Vec = core::CSmallVector<double, 1>;
51
52public:
53 static const std::string VALUE_TAG;
54
55public:
56 CMetricMultivariateStatistic(std::size_t n) : m_Values(n) {}
57
59 void persist(core::CStatePersistInserter& inserter) const {
60 for (std::size_t i = 0; i < m_Values.size(); ++i) {
61 CMetricStatisticWrappers::persist(m_Values[i], VALUE_TAG, inserter);
62 }
63 }
64
66 bool restore(core::CStateRestoreTraverser& traverser) {
67 std::size_t i = 0;
68 do {
69 const std::string& name = traverser.name();
70 if (name == VALUE_TAG) {
71 if (CMetricStatisticWrappers::restore(traverser, m_Values[i++]) == false) {
72 LOG_ERROR(<< "Invalid statistic in " << traverser.value());
73 return false;
74 }
75 }
76 } while (traverser.next());
77 return true;
78 }
79
84 void add(const TDouble1Vec& value, unsigned int count) {
85 if (value.size() != m_Values.size()) {
86 LOG_ERROR(<< "Inconsistent input data:"
87 << " # values = " << value.size() << ", expected "
88 << m_Values.size());
89 return;
90 }
91 for (std::size_t i = 0; i < value.size(); ++i) {
92 m_Values[i].add(value[i], count);
93 }
94 }
95
97 TDouble1Vec value() const {
98 std::size_t dimension = m_Values.size();
99 TDouble1Vec result(dimension);
100 for (std::size_t i = 0; i < dimension; ++i) {
101 TDouble1Vec vi = CMetricStatisticWrappers::value(m_Values[i]);
102 if (vi.size() > 1) {
103 result.resize(vi.size() * dimension);
104 }
105 for (std::size_t j = 0; j < vi.size(); ++j) {
106 result[i + j * dimension] = vi[j];
107 }
108 }
109 return result;
110 }
111
114 TDouble1Vec influencerValue() const {
115 std::size_t dimension = m_Values.size();
116 TDouble1Vec result(dimension);
117 for (std::size_t i = 0; i < dimension; ++i) {
118 TDouble1Vec vi = CMetricStatisticWrappers::influencerValue(m_Values[i]);
119 if (vi.size() > 1) {
120 result.resize(vi.size() * dimension);
121 }
122 for (std::size_t j = 0; j < vi.size(); ++j) {
123 result[i + j * dimension] = vi[j];
124 }
125 }
126 return result;
127 }
128
130 double count() const {
131 return CMetricStatisticWrappers::count(m_Values[0]);
132 }
133
135 const CMetricMultivariateStatistic& operator+=(const CMetricMultivariateStatistic& rhs) {
136 for (std::size_t i = 0; i < m_Values.size(); ++i) {
137 m_Values[i] += rhs.m_Values[i];
138 }
139 return *this;
140 }
141
143 uint64_t checksum(uint64_t seed) const {
144 return maths::common::CChecksum::calculate(seed, m_Values);
145 }
146
148 void debugMemoryUsage(const core::CMemoryUsage::TMemoryUsagePtr& mem) const {
149 mem->setName("CMetricPartialStatistic", sizeof(*this));
150 core::memory_debug::dynamicSize("m_Value", m_Values, mem);
151 }
152
154 std::size_t memoryUsage() const {
155 return sizeof(*this) + core::memory::dynamicSize(m_Values);
156 }
157
158private:
159 using TStatistic2Vec = core::CSmallVector<STATISTIC, 2>;
160
161private:
162 TStatistic2Vec m_Values;
163
164 friend std::ostream&
165 operator<<(std::ostream& o, const CMetricMultivariateStatistic<STATISTIC>& statistic) {
166 return o << core::CContainerPrinter::print(statistic.m_Values);
167 }
168};
