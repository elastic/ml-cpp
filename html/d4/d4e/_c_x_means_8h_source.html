167 void run(std::size_t improveParamsKmeansIterations,
168 std::size_t improveStructureClusterSeeds,
169 std::size_t improveStructureKmeansIterations) {
170 while (this->improveStructure(improveStructureClusterSeeds,
171 improveStructureKmeansIterations)) {
172 this->improveParams(improveParamsKmeansIterations);
173 }
174 this->polish(10 * improveParamsKmeansIterations);

183 void improveParams(std::size_t kmeansIterations) {
184 using TClusterCPtrVec = std::vector<const CCluster*>;
185
186 std::size_t n = m_Clusters.size();
187
188 // Setup k-means to run on the current centres and create
189 // sorted lookup of the current clusters.
190 TPointVec oldCentres;
191 oldCentres.reserve(n);
192 TClusterCPtrVec oldClusters;
193 oldClusters.reserve(n);
194 for (std::size_t i = 0; i < n; ++i) {
195 oldCentres.push_back(m_Clusters[i].centre());
196 oldClusters.push_back(&m_Clusters[i]);
197 }
198 std::sort(oldClusters.begin(), oldClusters.end(), COrderings::SDerefLess{});
199 m_Kmeans.setCentres(oldCentres);
200
201 // k-means to improve parameters.
202 m_Kmeans.run(kmeansIterations);
203 const TPointVec& newCentres = m_Kmeans.centres();
204 TPointVecVec newClusterPoints;
205 m_Kmeans.clusters(newClusterPoints);
206
207 // Note that oldClusters holds pointers to the current
208 // clusters so we can't overwrite them until after the
209 // following loop.
210
211 TClusterVec newClusters;
212 newClusters.reserve(newCentres.size());
213 TUInt64USet preserved;
214 COST cost_;
215
216 for (std::size_t i = 0; i < n; ++i) {
217 newClusters.push_back(CCluster());
218 CCluster& cluster = newClusters.back();
219 cluster.centre(newCentres[i]);
220 cluster.points(newClusterPoints[i]);
221 auto j = std::lower_bound(oldClusters.begin(), oldClusters.end(),
222 &cluster, COrderings::SDerefLess{});
223 if (j != oldClusters.end() && **j == cluster) {
224 cluster.cost((*j)->cost());
225 preserved.insert(cluster.checksum());
226 } else {
227 cluster.cost(COST(cluster.points()).calculate());
228 }
229 cost_.add(cluster.points());
230 }
231
232 // Refresh the clusters and inactive list.
233 m_Clusters.swap(newClusters);
234 for (auto i = m_Inactive.begin(); i != m_Inactive.end(); ) {
235 if (preserved.count(*i) > 0) {
236 ++i;
237 } else {
238 i = m_Inactive.erase(i);
239 }
240 }
241
242 // Refresh the best clustering found so far.
243 double cost = cost_.calculate();
244 if (cost < m_MinCost) {
245 m_BestCentres.clear();
246 for (const auto& cluster : m_Clusters) {
247 m_BestCentres.push_back(cluster.centre());
248 }
249 m_MinCost = cost;
250 }
251 }

260 bool improveStructure(std::size_t clusterSeeds, std::size_t kmeansIterations) {
261 if (m_Clusters.empty()) {
262 return false;
263 }
264
265 // Declared outside the loop to minimize allocations.
266 CKMeans<POINT> kmeans;
267 TPointVec points;
268 TPointVec seedCentres;
269 TPointVecVec clusterPoints;
270 TPointVec bestClusterCentres;
271 TPointVecVec bestClusterPoints;
272
273 std::size_t largest = 0;
274 for (const auto& cluster : m_Clusters) {
275 largest = std::max(largest, cluster.size());
276 }
277
278 kmeans.reserve(largest);
279 points.reserve(largest);
280 seedCentres.reserve(2);
281 clusterPoints.reserve(2);
282 bestClusterCentres.reserve(2);
283 bestClusterPoints.reserve(2);
284
285 bool split = false;
286
287 for (std::size_t i = 0, n = m_Clusters.size();
288 i < n && m_Clusters.size() < m_Kmax; ++i) {
289 if (m_Inactive.count(m_Clusters[i].checksum()) > 0) {
290 continue;
291 }
292
293 LOG_TRACE(<< "Working on cluster at " << m_Clusters[i].centre());
294 LOG_TRACE(<< "Cluster cost = " << m_Clusters[i].cost());
295
296 points.reserve(m_Clusters[i].size());
297 points.assign(m_Clusters[i].points().begin(), m_Clusters[i].points().end());
298 kmeans.setPoints(points);
299 double minCost = std::numeric_limits<double>::max();
300
301 for (std::size_t j = 0; j < clusterSeeds; ++j) {
302 this->generateSeedCentres(points, 2, seedCentres);
303 LOG_TRACE(<< "seed centres = " << seedCentres);
304
305 kmeans.setCentres(seedCentres);
306 kmeans.run(kmeansIterations);
307
308 const TPointVec& centres = kmeans.centres();
309 LOG_TRACE(<< "centres = " << centres);
310 clusterPoints.clear();
311 kmeans.clusters(clusterPoints);
312
313 double cost = COST(clusterPoints).calculate();
314 LOG_TRACE(<< "cost = " << cost);
315
316 if (cost < minCost) {
317 minCost = cost;
318 bestClusterCentres.assign(centres.begin(), centres.end());
319 bestClusterPoints.swap(clusterPoints);
320 }
321 }
322
323 // Check if we should split.
324 if (minCost < m_Clusters[i].cost()) {
325 m_Inactive.erase(m_Clusters[i].checksum());
326 m_Clusters[i].cost(COST(bestClusterPoints[0]).calculate());
327 m_Clusters[i].centre(bestClusterCentres[0]);
328 m_Clusters[i].points(bestClusterPoints[0]);
329 for (std::size_t j = 1; j < bestClusterCentres.size(); ++j) {
330 m_Clusters.push_back(CCluster());
331 m_Clusters.back().cost(COST(bestClusterPoints[j]).calculate());
332 m_Clusters.back().centre(bestClusterCentres[j]);
333 m_Clusters.back().points(bestClusterPoints[j]);
334 }
335 split = true;
336 } else {
337 LOG_TRACE(<< "Setting inactive = " << m_Clusters[i].checksum());
338 m_Inactive.insert(m_Clusters[i].checksum());
339 }
340 }
341
342 return split;
343 }

363 void polish(std::size_t kmeansIterations) {
364 if (m_BestCentres.size() > 1) {
365 m_Kmeans.setCentres(m_BestCentres);
366 m_Kmeans.run(kmeansIterations);
367 m_BestCentres = m_Kmeans.centres();
368 TPointVecVec polishedClusterPoints;
369 m_Kmeans.clusters(polishedClusterPoints);
370 m_Clusters.clear();
371 m_Clusters.reserve(m_BestCentres.size());
372 for (std::size_t i = 0; i < m_BestCentres.size(); ++i) {
373 m_Clusters.push_back(CCluster());
374 CCluster& cluster = m_Clusters.back();
375 cluster.cost(COST(polishedClusterPoints[i]).calculate());
376 cluster.centre(m_BestCentres[i]);
377 cluster.points(polishedClusterPoints[i]);
378 }
379 }
