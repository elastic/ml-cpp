78 CBucketQueue(std::size_t latencyBuckets,
79 core_t::TTime bucketLength,
80 core_t::TTime latestBucketStart,
81 T initial = T())
82 : m_Queue(latencyBuckets + 1),

185 void acceptPersistInserter(core::CStatePersistInserter& inserter) const {
186 for (std::size_t i = 0; i < m_Queue.size(); i++) {
187 inserter.insertValue(INDEX_TAG, i);
188 core::CPersistUtils::persist(BUCKET_TAG, m_Queue[i], inserter);
189 }
190 }
191
194 bool acceptRestoreTraverser(core::CStateRestoreTraverser& traverser) {
195 std::size_t i = 0;
196 do {
197 if (traverser.name() == INDEX_TAG) {
198 if (core::CStringUtils::stringToType(traverser.value(), i) == false) {
199 LOG_ERROR(<< "Bad index in " << traverser.value());
200 return false;
201 }
202 } else if (traverser.name() == BUCKET_TAG) {
203 if (i >= m_Queue.size()) {
204 LOG_ERROR(<< "Bucket queue is smaller on restore than on persist: " << i
205 << " >= " << m_Queue.size() << ". Restoration failed.");
206 return false;
207 }
208 if (!(core::CPersistUtils::restore(BUCKET_TAG, m_Queue[i], traverser))) {
209 LOG_ERROR(<< "Invalid bucket");
210 return false;
211 }
212 }
213 } while (traverser.next());
214 return true;
215 }
216
217private:
219 template<typename F>
220 void persist(F bucketPersist, core::CStatePersistInserter& inserter) const {
221 for (std::size_t i = 0; i < m_Queue.size(); i++) {
222 inserter.insertValue(INDEX_TAG, i);
223 inserter.insertLevel(
224 BUCKET_TAG, std::bind<void>(bucketPersist, std::cref(m_Queue[i]),
225 std::placeholders::_1));
226 }
227 }
228
230 template<typename F>
231 bool restore(F bucketRestore, const T& initial, core::CStateRestoreTraverser& traverser) {
232 std::size_t i = 0;
233 do {
234 if (traverser.name() == INDEX_TAG) {
235 if (core::CStringUtils::stringToType(traverser.value(), i) == false) {
236 LOG_DEBUG(<< "Bad index in " << traverser.value());
237 return false;
238 }
239 } else if (traverser.name() == BUCKET_TAG) {
240 if (i >= m_Queue.size()) {
241 LOG_ERROR(<< "Bucket queue is smaller on restore than on persist: " << i
242 << " >= " << m_Queue.size() << ". Restoration failed.");
243 return false;
244 }
245 m_Queue[i] = initial;
246 if (traverser.hasSubLevel()) {
247 if (traverser.traverseSubLevel(
248 std::bind<bool>(bucketRestore, std::ref(m_Queue[i]),
249 std::placeholders::_1)) == false) {
250 LOG_ERROR(<< "Invalid bucket");
251 return false;
252 }
253 }
254 }
255 } while (traverser.next());
256 return true;
257 }
258
260 void fill(const T& initial) {
261 for (std::size_t i = 0; i < m_Queue.capacity(); ++i) {
262 this->push(initial);
263 }
264 }
265
267 std::size_t index(core_t::TTime time) const {
268 if (m_BucketLength == 0) {
269 LOG_ERROR(<< "Invalid bucketLength for queue!");
270 return 0;
271 }
272 std::size_t index = static_cast<std::size_t>((m_LatestBucketEnd - time) / m_BucketLength);
273 std::size_t size = this->size();
274 if (index >= size) {
275 LOG_ERROR(<< "Time " << time << " is out of range. Returning earliest bucket index.");
276 return size - 1;
277 }
278 return index;
279 }
280
281private:
282 TQueue m_Queue;
283 core_t::TTime m_LatestBucketEnd;
284 core_t::TTime m_BucketLength;
285};
