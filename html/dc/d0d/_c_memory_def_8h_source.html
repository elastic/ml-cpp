96template<typename T, std::size_t N>
97bool inplace(const CSmallVector<T, N>& t) {
98 const char* address = reinterpret_cast<const char*>(&t);
99 const char* storage = reinterpret_cast<const char*>(t.data());
100 return storage >= address && storage < address + sizeof t;
101}

149template<typename T>
150std::size_t dynamicSize(const T& t, std::enable_if_t<!std::is_pointer_v<T>>*) {
151 // If you hit this assert there are one of three reasons:
152 // 1. You need to add a memoryUsage function to your type,
153 // 2. You need to add a dynamicSizeAlwaysZero function to your type,
154 // 3. You need to include CMemoryDefStd.h.
155 static_assert(!memory_detail::expectMemoryOverload<T>(),
156 "Maybe miss accounting for memory usage");
157 if constexpr (!memory_detail::SDynamicSizeAlwaysZero<T>::value()) {
158 return memory_detail::SMemoryDynamicSize<T>::dispatch(t);
159 }
160 return 0;
161}

174template<typename T>
175std::size_t dynamicSize(const std::shared_ptr<T>& t) {
176 // The check for nullptr here may seem unnecessary but there are situations
177 // where an unset shared_ptr can have a use_count greater than 0, see
178 // https://stackoverflow.com/questions/48885252/c-sharedptr-use-count-for-nullptr/48885643
179 long uc{t == nullptr ? 0 : t.use_count()};
180 if (uc == 0) {
181 return 0;
182 }
183 // Note we add on sizeof(long) here to account for the memory
184 // used by the shared reference count. Also, round up.
185 return (sizeof(long) + memory::staticSize(*t) + memory::dynamicSize(*t) +
186 std::size_t(uc - 1)) /
187 uc;
188}

291class CORE_EXPORT CAnyVisitor {
292public:
293 using TDynamicSizeFunc = std::size_t (*)(const std::any& any);
294 using TTypeInfoDynamicSizeFuncPr = std::pair<TTypeInfoCRef, TDynamicSizeFunc>;
295 using TTypeInfoDynamicSizeFuncPrVec = std::vector<TTypeInfoDynamicSizeFuncPr>;
296
297public:
298 static CAnyVisitor& instance() {
299 static CAnyVisitor instance;
300 return instance;
301 }
302
303public:
304 CAnyVisitor(const CAnyVisitor&) = delete;
305 CAnyVisitor& operator=(const CAnyVisitor&) = delete;
306
309 template<typename T>
310 bool registerCallback() {
311 auto i = std::lower_bound(m_Callbacks.begin(), m_Callbacks.end(),
312 std::cref(typeid(T)), memory_detail::STypeInfoLess());
313 if (i == m_Callbacks.end()) {
314 m_Callbacks.emplace_back(std::cref(typeid(T)),
315 &CAnyVisitor::dynamicSizeCallback<T>);
316 return true;
317 }
318 if (i->first.get() != typeid(T)) {
319 m_Callbacks.insert(
320 i, {std::cref(typeid(T)), &CAnyVisitor::dynamicSizeCallback<T>});
321 return true;
322 }
323 return false;
324 }
325
328 std::size_t dynamicSize(const std::any& x) const {
329 if (x.has_value()) {
330 auto i = std::lower_bound(m_Callbacks.begin(), m_Callbacks.end(),
331 std::cref(x.type()),
332 memory_detail::STypeInfoLess());
333 if (i != m_Callbacks.end() && i->first.get() == x.type()) {
334 return (*i->second)(x);
335 }
336 LOG_ERROR(<< "No callback registered for " << x.type().name());
337 }
338 return 0;
339 }
340
341private:
342 CAnyVisitor() = default;
343
344private:
346 template<typename T>
347 static std::size_t dynamicSizeCallback(const std::any& any) {
348 try {
349 return sizeof(T) + memory::dynamicSize(std::any_cast<const T&>(any));
350 } catch (const std::exception& e) {
351 LOG_ERROR(<< "Failed to calculate size " << e.what());
352 }
353 return 0;
354 }
355
356private:
357 TTypeInfoDynamicSizeFuncPrVec m_Callbacks;
358};

408template<typename T>
409void dynamicSize(const char* name,
410 const std::shared_ptr<T>& t,
411 const CMemoryUsage::TMemoryUsagePtr& mem) {
412 // The check for nullptr here may seem unnecessary but there are situations
413 // where an unset shared_ptr can have a use_count greater than 0, see
414 // https://stackoverflow.com/questions/48885252/c-sharedptr-use-count-for-nullptr/48885643
415 long uc{t == nullptr ? 0 : t.use_count()};
416 if (uc == 0) {
417 return;
418 }
419 // If the pointer is shared by multiple users, each one
420 // might count it, so divide by the number of users.
421 // However, if only 1 user has it, do a full debug.
422 std::string ptrName(name);
423 if (uc == 1) {
424 // Note we add on sizeof(long) here to account for
425 // the memory used by the shared reference count.
426 ptrName += "_shared_ptr";
427 // Note we add on sizeof(long) here to account for
428 // the memory used by the shared reference count.
429 mem->addItem(ptrName, sizeof(long) + memory::staticSize(*t));
430 memory_debug::dynamicSize(name, *t, mem);
431 } else {
432 ptrName += "_shared_ptr (x" + std::to_string(uc) + ")";
433 // Note we add on sizeof(long) here to account for
434 // the memory used by the shared reference count.
435 // Also, round up.
436 mem->addItem(ptrName, (sizeof(long) + memory::staticSize(*t) +
437 memory::dynamicSize(*t) + std::size_t(uc - 1)) /
438 uc);
439 }
440}
