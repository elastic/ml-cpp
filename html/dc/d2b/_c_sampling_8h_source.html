52 template<typename T>
53 class CStreamSampler {
54 public:
55 virtual ~CStreamSampler() = default;
56
58 CPRNG::CXorOShiro128Plus& rng() { return m_Rng; }
59
61 std::size_t sampleSize() const { return m_SampleSize; }
62
64 std::size_t targetSampleSize() const { return m_TargetSampleSize; }
65
67 virtual void reset() {
68 m_SampleSize = 0;
69 m_StreamWeight = m_SampleWeight = 0.0;
70 }
71
73 void sample(const T& x) {
74 if (m_SampleSize < m_TargetSampleSize) {
75 this->doSample(m_SampleSize++, x, 1.0);
76 if (m_SampleSize == m_TargetSampleSize) {
77 m_StreamWeight = m_SampleWeight = this->seedWeights();
78 }
79 return;
80 }
81
82 double weight{this->weight(x)};
83 double p{uniformSample(m_Rng, 0.0, 1.0)};
84 double exchangeWeight{weight - this->minWeight()};
85 if (p * (m_StreamWeight + exchangeWeight) < m_SampleWeight + exchangeWeight) {
86 std::size_t slot{this->sampleToReplace(weight)};
87 if (slot < m_TargetSampleSize) {
88 m_SampleWeight += weight - this->doSample(slot, x, weight);
89 }
90 }
91 m_StreamWeight += weight;
92 }
93
94 protected:
95 explicit CStreamSampler(std::size_t targetSampleSize,
96 const CPRNG::CXorOShiro128Plus& rng = CPRNG::CXorOShiro128Plus{})
97 : m_Rng{rng}, m_TargetSampleSize{targetSampleSize} {}
98
99 double sampleWeight() const { return m_SampleWeight; }
100
101 private:
102 virtual double seedWeights() = 0;
103 virtual double weight(const T& x) = 0;
104 virtual double minWeight() const = 0;
105 virtual std::size_t sampleToReplace(double weight) = 0;
106 virtual double doSample(std::size_t slot, const T& x, double weight) = 0;
107
108 private:
109 CPRNG::CXorOShiro128Plus m_Rng;
110 std::size_t m_TargetSampleSize;
111 std::size_t m_SampleSize{0};
112 double m_StreamWeight{0.0};
113 double m_SampleWeight{0.0};
114 };

196 template<typename T>
197 class CVectorDissimilaritySampler final : public CStreamSampler<T> {
198 public:
199 using TVec = std::vector<T>;
200
201 public:
202 explicit CVectorDissimilaritySampler(std::size_t targetSampleSize,
203 const CPRNG::CXorOShiro128Plus& rng = CPRNG::CXorOShiro128Plus{})
204 : CStreamSampler<T>(targetSampleSize, rng),
205 m_SampleWeights(targetSampleSize, 0.0) {
206 m_Samples.reserve(targetSampleSize);
207 m_Probabilities.reserve(targetSampleSize + 1);
208 }
209
210 TVec& samples() {
211 this->deduplicate();
212 return m_Samples;
213 }
214 TDoubleVec& sampleWeights() {
215 this->deduplicate();
216 return m_SampleWeights;
217 }
218
219 void reset() override {
220 this->CStreamSampler<T>::reset();
221 m_SampleWeights.assign(this->targetSampleSize(), 0.0);
222 m_Samples.clear();
223 }
224
225 void merge(const CVectorDissimilaritySampler& other) {
226 m_Samples.insert(m_Samples.end(), other.m_Samples.begin(),
227 other.m_Samples.end());
228 this->deduplicate();
229 if (m_SampleWeights.size() > this->targetSampleSize()) {
230 TSizeVec selected;
231 m_Probabilities.assign(m_SampleWeights.begin(), m_SampleWeights.end());
232 categoricalSampleWithoutReplacement(this->rng(), m_Probabilities,
233 this->targetSampleSize(), selected);
234 std::sort(selected.begin(), selected.end());
235 for (std::size_t i = 0; i < selected.size(); ++i) {
236 m_Samples[i] = m_Samples[selected[i]];
237 m_SampleWeights[i] = m_SampleWeights[selected[i]];
238 }
239 m_Samples.resize(selected.size());
240 m_SampleWeights.resize(selected.size());
241 }
242 }
243
244 private:
245 double seedWeights() override {
246 double result{0.0};
247 for (std::size_t i = 0; i < m_Samples.size(); ++i) {
248 m_SampleWeights[i] = this->weight(m_Samples[i]);
249 m_MinWeight = std::min(m_MinWeight, m_SampleWeights[i]);
250 result += m_SampleWeights[i];
251 }
252 return result;
253 }
254
255 double weight(const T& x) override {
256 double result{0.0};
257 if (m_Samples.size() < 10) {
258 for (const auto& sample : m_Samples) {
259 result += las::distance(x, sample);
260 }
261 } else {
262 for (std::size_t i = 0; i < 10; ++i) {
263 const auto& sample =
264 m_Samples[uniformSample(this->rng(), 0, m_Samples.size())];
265 result += las::distance(x, sample);
266 }
267 }
268 return result / std::min(static_cast<double>(m_Samples.size()), 10.0);
269 }
270
271 double minWeight() const override { return m_MinWeight; }
272
273 std::size_t sampleToReplace(double weight) override {
274 if (this->sampleWeight() == 0.0) {
275 return uniformSample(this->rng(), 0, this->targetSampleSize() + 1);
276 }
277 // We're choosing the sample to _evict_ so use probabilities proportional
278 // to inverse weight.
279 std::size_t n{m_SampleWeights.size()};
280 m_Probabilities.resize(n + 1);
281 double wmin{1e-6 * this->sampleWeight() / static_cast<double>(n + 1)};
282 for (std::size_t i = 0; i < n; ++i) {
283 m_Probabilities[i] = 1.0 / std::max(m_SampleWeights[i], wmin);
284 }
285 m_Probabilities[n] = 1.0 / std::max(weight, wmin);
286 return categoricalSample(this->rng(), m_Probabilities);
287 }
288
289 double doSample(std::size_t slot, const T& x, double weight) override {
290 if (m_Samples.size() <= slot) {
291 m_Samples.resize(slot + 1, x);
292 // Weights are not available until the sample set is its target size.
293 } else {
294 m_Samples[slot] = x;
295 std::swap(m_SampleWeights[slot], weight);
296 m_MinWeight = weight == m_MinWeight
297 ? *std::min_element(m_SampleWeights.begin(),
298 m_SampleWeights.end())
299 : std::min(m_MinWeight, weight);
300 }
301 m_Deduplicated = false;
302 return weight;
303 }
304
305 void deduplicate() {
306 if (m_Deduplicated) {
307 return;
308 }
309 std::sort(m_Samples.begin(), m_Samples.end());
310 m_Samples.erase(std::unique(m_Samples.begin(), m_Samples.end()),
311 m_Samples.end());
312 m_SampleWeights.resize(m_Samples.size());
313 m_MinWeight = std::numeric_limits<double>::max();
314 for (std::size_t i = 0; i < m_Samples.size(); ++i) {
315 m_SampleWeights[i] = this->weight(m_Samples[i]);
316 m_MinWeight = std::min(m_MinWeight, m_SampleWeights[i]);
317 }
318 m_Deduplicated = true;
319 }
320
321 private:
322 bool m_Deduplicated{true};
323 double m_MinWeight{std::numeric_limits<double>::max()};
324 TVec m_Samples;
325 TDoubleVec m_SampleWeights;
326 TDoubleVec m_Probabilities;
327 };

595 static void random_shuffle(RNG& rng, ITR first, ITR last) {
596 auto d = last - first;
597 if (d > 1) {
598 CUniform0nGenerator<RNG> rand(rng);
599 for (--last; first < last; ++first, --d) {
600 auto i = rand(d);
601 if (i > 0) {
602 std::iter_swap(first, first + i);
603 }
604 }
605 }
