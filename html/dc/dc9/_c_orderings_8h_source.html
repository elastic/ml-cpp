313 template<std::size_t I, typename PRED, typename... ARGS>
314 static bool lexicographicalCompareAt(const PRED& pred, const ARGS&... args) {
315 static_assert(sizeof...(ARGS) % 2 == 0, "The number of values to compare must be equal");
316 const auto& lhs = std::get<I>(std::forward_as_tuple(args...));
317 const auto& rhs = std::get<I + sizeof...(ARGS) / 2>(std::forward_as_tuple(args...));
318 if (pred(lhs, rhs)) {
319 return true;
320 }
321 if constexpr (sizeof...(args) > 2 * (I + 1)) {
322 return pred(rhs, lhs) == false &&
323 lexicographicalCompareAt<I + 1>(pred, args...);
324 }
325 return false;
326 }

332 template<typename PRED, typename... ARGS>
333 static bool lexicographicalCompareWith(const PRED& pred, const ARGS&... args) {
334 return lexicographicalCompareAt<0>(pred, args...);
335 }
336
338 template<typename... ARGS>
339 static bool lexicographicalCompare(const ARGS&... args) {
340 return lexicographicalCompareWith(std::less<>{}, args...);
341 }
