83 template<typename F, typename G>
84 std::pair<VECTOR, double>
85 minimize(const F& f, const G& g, VECTOR x0, double eps = 1e-8, std::size_t iterations = 50) {
86
87 this->reinitialize(f, x0);
88
89 VECTOR x{std::move(x0)};
90
91 for (std::size_t i = 0; i < iterations; ++i) {
92 this->updateDescentDirection(g, x);
93 x = this->lineSearch(f);
94 if (this->converged(eps)) {
95 break;
96 }
97 }
98
99 return {std::move(x), m_Fx};
100 }

118 template<typename F, typename G>
119 std::pair<VECTOR, double> constrainedMinimize(const F& f,
120 const G& g,
121 const VECTOR& a,
122 const VECTOR& b,
123 VECTOR x0,
124 double rho,
125 double eps = 1e-8,
126 std::size_t iterations = 50) {
127
128 // We use an ADMM update scheme to compute the minimum with respect to the
129 // bounding box constraints. This is less efficient but significantly simpler
130 // than L-BFGS-B.
131 //
132 // In particular, we want to solve
133 // argmin_x f(x)
134 // s.t. a <= x <= b (1)
135 //
136 // We convert the inequality to an affine constraint (suitable for ADMM) by
137 // introducing slack variables, i.e. we transform (1) as follows
138 // argmin_x,z f(x) + g(z_1) g(z_2)
139 // s.t. x - z_1 = a
140 // x + z_2 = b
141 //
142 // where the function g(z) is infinite for z < 0 and 0 otherwise. Note that
143 // the minimization over z_1 and z_2 can be done in closed form and are just
144 // the closest non-negative points to x - a + w_1 and b - x - w_2, respectively.
145
146 // Get the closest feasible point to x0.
147 VECTOR x{std::move(x0)};
148 las::max(a, x);
149 las::min(b, x);
150
151 VECTOR zero{las::zero(x)};
152
153 VECTOR z1{zero};
154 VECTOR z2{zero};
155 VECTOR w1{zero};
156 VECTOR w2{zero};
157 VECTOR r1;
158 VECTOR r2;
159
160 // Functions to compute the augmented Lagrangian and its gradient w.r.t. x.
161 auto al = [&](const VECTOR& x_) -> double {
162 r1 = x_ - z1 + w1 - a;
163 r2 = x_ + z2 + w2 - b;
164 double n1{las::norm(r1)};
165 double n2{las::norm(r2)};
166 // Explicitly construct the return type before returning from the lambda,
167 // otherwise the auto return type may be an Eigen closure type which
168 // references temporaries
169 return f(x_) + 0.5 * rho * (n1 * n1 + n2 * n2);
170 };
171 auto gal = [&](const VECTOR& x_) -> VECTOR {
172 // Explicitly construct the return type before returning from the lambda,
173 // otherwise the auto return type may be an Eigen closure type which
174 // references temporaries
175 return g(x_) + rho * (2 * x_ - z1 + z2 - a - b + w1 + w2);
176 };
177
178 VECTOR xmin{x};
179 double fmin{f(x)};
180
181 double f0{fmin};
182 double fl{fmin};
183 double fi;
184 VECTOR xi;
185 for (std::size_t i = 0; i < iterations; ++i) {
186 // x-minimization.
187 std::tie(x, std::ignore) = this->minimize(al, gal, x, eps, iterations);
188
189 // z-minimization.
190 z1 = x - a + w1;
191 z2 = b - x - w2;
192 las::max(zero, z1);
193 las::max(zero, z2);
194
195 // Dual update.
196 w1 += x - z1 - a;
197 w2 += x + z2 - b;
198
199 // Snap to nearest feasible point to test for convergence.
200 xi = x;
201 las::max(a, xi);
202 las::min(b, xi);
203 fi = f(xi);
204
205 // Snapping can increase the objective so make sure we remember the best
206 // we've seen.
207 if (fi < fmin) {
208 xmin = xi;
209 fmin = fi;
210 }
211
212 if (converged(fi, fl, f0, eps)) {
213 return {xmin, fmin};
214 }
215
216 fl = fmin;
217 }
218
219 return {xmin, fmin};
220 }

227 template<typename F>
228 void reinitialize(const F& f, const VECTOR& x0) {
229 m_Initial = true;
230 m_F0 = m_Fl = m_Fx = f(x0);
231 m_X = x0;
232 m_Dx.clear();
233 m_Dg.clear();
234 m_Dx.set_capacity(std::min(m_Rank, las::dimension(x0)));
235 m_Dg.set_capacity(std::min(m_Rank, las::dimension(x0)));
236 m_Rho.clear();
237 m_Alpha.clear();
238 }
239
240 bool converged(double eps) const {
241 return converged(m_Fx, m_Fl, m_F0, eps);
242 }
243
244 static bool converged(double fx, double fl, double f0, double eps) {
245 // We use less than or equal so that this is true if fx == fl == f0.
246 return std::fabs(fx - fl) <= eps * std::fabs(fx - f0);
247 }

249 template<typename F>
250 VECTOR lineSearch(const F& f) {
251
252 // This uses the Armijo condition that the decrease is bounded below by
253 // some multiple of the decrease promised by the gradient for the step
254 // size.
255
256 double s{1.0};
257 VECTOR xs{m_X - m_P};
258 double fs{f(xs)};
259
260 for (std::size_t i = 0; i < MAXIMUM_BACK_TRACKING_ITERATIONS &&
261 fs - m_Fx > this->minimumDecrease(s);
262 ++i) {
263 s *= m_StepScale;
264 xs = m_X - s * m_P;
265 fs = f(xs);
266 }
267
268 // This can happen if s |m_P| < epsilon |m_X|.
269 if (las::distance(xs, m_X) == 0.0) {
270 this->fullLineSearch(f, xs, fs);
271 }
272
273 m_Fl = m_Fx;
274 m_Fx = fs;
275
276 return xs;
277 }

279 template<typename F>
280 void fullLineSearch(const F& f, VECTOR& xs, double& fs) {
281 VECTOR step{las::norm(m_P) == 0.0 ? this->randomStep(f)
282 : this->minimumStepSize() * las::norm(m_X) /
283 las::norm(m_P) * m_P};
284
285 if (las::norm(step) == 0.0) {
286 xs = m_X;
287 fs = m_Fx;
288 return;
289 }
290
291 // We probe a selection of points along the step direction looking for
292 // a minimum and polish up with a few iterations of Brent's method.
293
294 TDoubleVec probes(MAXIMUM_BACK_TRACKING_ITERATIONS);
295 probes[0] = 1.0;
296 for (std::size_t i = 1; i < MAXIMUM_BACK_TRACKING_ITERATIONS; ++i) {
297 probes[i] = probes[i - 1] / m_StepScale;
298 }
299
300 auto f_ = [&](double s_) { return f(m_X - s_ * step); };
301
302 double smin;
303 double fmin;
304 double fsd;
305 CSolvers::globalMinimize(probes, f_, smin, fmin, fsd);
306 xs = m_X - smin * step;
307 fs = f(xs);
308 }

310 template<typename F>
311 VECTOR randomStep(const F& f) {
312 double eps{this->minimumStepSize() * las::norm(m_X)};
313 for (std::size_t i = 0; eps > 0.0 && i < MAXIMUM_RANDOM_PROBES; ++i) {
314 // Randomly probe looking for a decrease in the function.
315 VECTOR step{las::zero(m_X)};
316 for (std::size_t j = 0; j < las::dimension(m_X); ++j) {
317 step(j) = CSampling::uniformSample(m_Rng, -eps, eps);
318 }
319 if (f(m_X + step) < f(m_X)) {
320 return -step;
321 }
322 }
323 // We couldn't find a direction in which the fuction decreases. Returning
324 // zero means we'll exit on the convergence condition.
325 return las::zero(m_X);
326 }

328 template<typename G>
329 void updateDescentDirection(const G& g, const VECTOR& x) {
330
331 // This uses the L-BFGS Hessian approximation scheme.
332
333 m_P = g(x);
334
335 if (m_Initial == false) {
336 m_Dx.push_back(x - m_X);
337 m_Dg.push_back(m_P - m_Gx);
338 }
339
340 m_X = x;
341 m_Gx = m_P;
342
343 if (m_Initial == false && las::norm(m_Gx) > 0.0) {
344 double eps{std::numeric_limits<double>::epsilon() * las::norm(m_Gx)};
345
346 std::size_t k{m_Dx.size()};
347 m_Rho.resize(k);
348 m_Alpha.resize(k);
349 for (std::size_t i = k; i > 0; --i) {
350 m_Rho[i - 1] = 1.0 / (las::inner(m_Dg[i - 1], m_Dx[i - 1]) + eps);
351 m_Alpha[i - 1] = m_Rho[i - 1] * las::inner(m_Dx[i - 1], m_P);
352 m_P -= m_Alpha[i - 1] * m_Dg[i - 1];
353 }
354
355 // The initialisation choice is free, this is an estimate for the
356 // size of the true Hessian along the most recent search direction
357 // and tends to mean that the initial step size is chosen in the
358 // line search.
359
360 double hmax{las::norm(m_Gx) / this->minimumStepSize()};
361
362 double lambda{(las::inner(m_Dg[k - 1], m_Dx[k - 1]) + eps) /
363 (las::inner(m_Dg[k - 1], m_Dg[k - 1]) + eps * eps)};
364 double h0{std::copysign(std::min(std::fabs(lambda), hmax), lambda)};
365
366 m_P *= h0;
367
368 for (std::size_t i = 0; i < k; ++i) {
369 double beta{m_Rho[i] * (las::inner(m_Dg[i], m_P) + eps)};
370 double gk{m_Alpha[i] - beta};
371 double gmax{hmax / las::norm(m_Dx[i])};
372 m_P += std::copysign(std::min(std::fabs(gk), gmax), gk) * m_Dx[i];
373 }
374
375 if (las::inner(m_Gx, m_P) <= 0.0) {
376 m_P = (las::norm(m_P) / (las::norm(m_Gx) + eps)) * m_Gx;
377 }
378 } else {
379 m_Initial = false;
380 }
381 }
382
383 double minimumDecrease(double s) const {
384 double normP{las::norm(m_P)};
385 return m_BacktrackingMinDecrease * s *
386 (normP == 0.0 ? las::norm(m_Gx) : las::inner(m_Gx, m_P) / normP);
387 }
388
389 double minimumStepSize() const {
390 return CTools::stable(std::pow(
391 m_StepScale, static_cast<double>(MAXIMUM_BACK_TRACKING_ITERATIONS)));
392 }
