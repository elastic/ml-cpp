47template<typename VECTOR>
48struct SDoAssign {
49 static const CVectorRange<VECTOR>& dispatch(CVectorRange<VECTOR>& lhs,
50 const CVectorRange<VECTOR>& rhs) {
51 if (rhs.base() != lhs.base()) {
52 lhs.assign(rhs.begin(), rhs.end());
53 } else {
54 VECTOR tmp{rhs.begin(), rhs.end()};
55 lhs.assign(tmp.begin(), tmp.end());
56 }
57 return lhs;
58 }
59};
60template<typename VECTOR>
61struct SDoAssign<const VECTOR> {
62 static const CVectorRange<const VECTOR>&
63 dispatch(CVectorRange<const VECTOR>& lhs, const CVectorRange<const VECTOR>& rhs) {
64 CVectorRange<const VECTOR> tmp(*rhs.base(), rhs.a(), rhs.b());
65 lhs.swap(tmp);
66 return lhs;
67 }

98 template<typename T>
99 void assign(size_type n, const T& value) {
100 std::fill_n(this->begin(), std::min(this->size(), n), value);
101 if (n > this->size()) {
102 m_Vector->insert(this->end(), n - this->size(), value);
103 } else if (n < this->size()) {
104 m_Vector->erase(this->begin() + n, this->end());
105 }
106 m_B = m_A + n;
107 }
109 template<typename ITR>
110 void assign(ITR begin, ITR end) {
111 size_type size = std::distance(begin, end);
112 std::copy(begin, begin + std::min(this->size(), size), this->begin());
113 if (size > this->size()) {
114 m_Vector->insert(this->end(), begin + this->size(), end);
115 } else if (size < this->size()) {
116 m_Vector->erase(this->begin() + size, this->end());
117 }
118 m_B = m_A + size;
119 }

257 void range_check(size_type pos) const {
258 if (m_A + pos >= m_B) {
259 throw std::out_of_range("out of range: " + std::to_string(pos) +
260 " >= " + std::to_string(m_B - m_A));
261 }
262 }
