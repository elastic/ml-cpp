242 void add(const T& x, const TCoordinate& n, int) {
243 if (n == TCoordinate{0}) {
244 return;
245 }
246
247 s_Count += n;
248
249 // Note we don't trap the case alpha is less than epsilon,
250 // because then we'd have to compute epsilon and it is very
251 // unlikely the count will get big enough.
252 TCoordinate alpha{n / s_Count};
253 TCoordinate beta{TCoordinate{1} - alpha};
254
255 T mean{s_Moments[0]};
256 s_Moments[0] = beta * mean + alpha * x;
257
258 if constexpr (ORDER > 1) {
259 T r{x - s_Moments[0]};
260 T r2{las::componentwise(r) * las::componentwise(r)};
261 T dMean{mean - s_Moments[0]};
262 T dMean2{las::componentwise(dMean) * las::componentwise(dMean)};
263 T variance{s_Moments[1]};
264
265 s_Moments[1] = beta * (variance + dMean2) + alpha * r2;
266
267 if constexpr (ORDER > 2) {
268 T skew{s_Moments[2]};
269 T dSkew{TCoordinate(3) * variance + dMean2};
270 dSkew = las::componentwise(dSkew) * las::componentwise(dMean);
271
272 s_Moments[2] = beta * (skew + dSkew) + alpha * r2 * r;
273 }
274 }
275 }

279 template<typename U>
280 const SSampleCentralMoments& operator+=(const SSampleCentralMoments<U, ORDER>& rhs) {
281 if (rhs.s_Count == TCoordinate{0}) {
282 return *this;
283 }
284
285 s_Count = s_Count + rhs.s_Count;
286
287 // Note we don't trap the case alpha is less than epsilon,
288 // because then we'd have to compute epsilon and it is very
289 // unlikely the count will get big enough.
290 TCoordinate alpha{rhs.s_Count / s_Count};
291 TCoordinate beta{TCoordinate{1} - alpha};
292
293 T meanLhs{s_Moments[0]};
294 T meanRhs{rhs.s_Moments[0]};
295
296 s_Moments[0] = beta * meanLhs + alpha * meanRhs;
297
298 if constexpr (ORDER > 1) {
299 T dMeanLhs{meanLhs - s_Moments[0]};
300 T dMean2Lhs{las::componentwise(dMeanLhs) * las::componentwise(dMeanLhs)};
301 T varianceLhs{s_Moments[1]};
302 T dMeanRhs{meanRhs - s_Moments[0]};
303 T dMean2Rhs{las::componentwise(dMeanRhs) * las::componentwise(dMeanRhs)};
304 T varianceRhs{rhs.s_Moments[1]};
305
306 s_Moments[1] = beta * (varianceLhs + dMean2Lhs) +
307 alpha * (varianceRhs + dMean2Rhs);
308
309 if constexpr (ORDER > 2) {
310 T skewLhs{s_Moments[2]};
311 T dSkewLhs{TCoordinate{3} * varianceLhs + dMean2Lhs};
312 dSkewLhs = las::componentwise(dSkewLhs) * las::componentwise(dMeanLhs);
313
314 T skewRhs{rhs.s_Moments[2]};
315 T dSkewRhs{TCoordinate{3} * varianceRhs + dMean2Rhs};
316 dSkewRhs = las::componentwise(dSkewRhs) * las::componentwise(dMeanRhs);
317
318 s_Moments[2] = beta * (skewLhs + dSkewLhs) + alpha * (skewRhs + dSkewRhs);
319 }
320 }
321
322 return *this;
323 }

339 template<typename U>
340 const SSampleCentralMoments& operator-=(const SSampleCentralMoments<U, ORDER>& rhs) {
341 if (rhs.s_Count == TCoordinate{0}) {
342 return *this;
343 }
344
345 using std::max;
346
347 s_Count = max(s_Count - rhs.s_Count, TCoordinate{0});
348
349 if (s_Count == TCoordinate{0}) {
350 std::fill_n(s_Moments, ORDER, T{0});
351 return *this;
352 }
353
354 // Note we don't trap the case alpha is less than epsilon,
355 // because then we'd have to compute epsilon and it is very
356 // unlikely the count will get big enough.
357 TCoordinate alpha{rhs.s_Count / s_Count};
358 TCoordinate beta{TCoordinate{1} + alpha};
359
360 T meanLhs{s_Moments[0]};
361 T meanRhs{rhs.s_Moments[0]};
362
363 s_Moments[0] = beta * meanLhs - alpha * meanRhs;
364
365 if constexpr (ORDER > 1) {
366 T dMeanLhs{s_Moments[0] - meanLhs};
367 T dMean2Lhs{las::componentwise(dMeanLhs) * las::componentwise(dMeanLhs)};
368 T dMeanRhs{meanRhs - meanLhs};
369 T dMean2Rhs{las::componentwise(dMeanRhs) * las::componentwise(dMeanRhs)};
370 T varianceRhs{rhs.s_Moments[1]};
371
372 s_Moments[1] = max(beta * (s_Moments[1] - dMean2Lhs) -
373 alpha * (varianceRhs + dMean2Rhs - dMean2Lhs),
374 T{0});
375
376 if constexpr (ORDER > 2) {
377 T skewLhs{s_Moments[2]};
378 T dSkewLhs{TCoordinate{3} * s_Moments[1] + dMean2Lhs};
379 dSkewLhs = las::componentwise(dSkewLhs) * las::componentwise(dMeanLhs);
380 T skewRhs{rhs.s_Moments[2]};
381 T dSkewRhs{TCoordinate{3} * varianceRhs + dMean2Rhs};
382 dSkewRhs = las::componentwise(dSkewRhs) * las::componentwise(dMeanRhs);
383
384 s_Moments[2] = beta * (skewLhs - dSkewLhs) -
385 alpha * (skewRhs + dSkewRhs - dSkewLhs);
386 }
387 }
388
389 return *this;
390 }

633 template<typename T, unsigned int N>
634 static inline T skewness(const SSampleCentralMoments<T, N>& accumulator) {
635 using TCoordinate = typename SSampleCentralMoments<T, N>::TCoordinate;
636
637 static_assert(N >= 3, "N must be at least 3");
638
639 if (accumulator.s_Count <= TCoordinate{2}) {
640 return T{0};
641 }
642
643 T normalization{variance(accumulator)};
644 using std::sqrt;
645 normalization = normalization * sqrt(normalization);
646
647 return accumulator.s_Moments[2] == T{0} ? T{0} : accumulator.s_Moments[2] / normalization;
648 }
