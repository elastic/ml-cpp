33template<typename RNG, typename Distribution, typename Container>
34void CRandomNumbers::generateSamples(RNG& randomNumberGenerator,
35 const Distribution& distribution,
36 std::size_t numberSamples,
37 Container& samples) {
38 samples.clear();
39 samples.reserve(numberSamples);
40 std::generate_n(std::back_inserter(samples), numberSamples,
41 [&distribution, &randomNumberGenerator]() {
42 return const_cast<Distribution&>(distribution)(randomNumberGenerator);
43 });
44}

46template<typename T, std::size_t N>
47void CRandomNumbers::generateRandomMultivariateNormals(
48 const TSizeVec& sizes,
49 std::vector<maths::common::CVectorNx1<T, N>>& means,
50 std::vector<maths::common::CSymmetricMatrixNxN<T, N>>& covariances,
51 std::vector<std::vector<maths::common::CVectorNx1<T, N>>>& points) {
52 means.clear();
53 covariances.clear();
54 points.clear();
55
56 std::size_t k = sizes.size();
57
58 TDoubleVec means_;
59 this->generateUniformSamples(-100.0, 100.0, N * k, means_);
60 for (std::size_t i = 0; i < N * k; i += N) {
61 maths::common::CVectorNx1<T, N> mean(&means_[i], &means_[i + N]);
62 means.push_back(mean);
63 }
64
65 TDoubleVec variances;
66 this->generateUniformSamples(10.0, 100.0, N * k, variances);
67 for (std::size_t i = 0; i < k; ++i) {
68 constexpr int N_{static_cast<int>(N)};
69 Eigen::Matrix<T, N_, N_> covariance = Eigen::Matrix<T, N_, N_>::Zero();
70
71 for (std::size_t j = 0; j < N; ++j) {
72 covariance(j, j) = variances[i * N + j];
73 }
74
75 // Generate random rotations in two planes.
76 TSizeVec coordinates;
77 this->generateUniformSamples(0, N, 4, coordinates);
78 std::sort(coordinates.begin(), coordinates.end());
79 coordinates.erase(std::unique(coordinates.begin(), coordinates.end()),
80 coordinates.end());
81
82 TDoubleVec thetas;
83 this->generateUniformSamples(0.0, boost::math::constants::two_pi<double>(), 2, thetas);
84
85 Eigen::Matrix<T, N_, N_> rotation = Eigen::Matrix<T, N_, N_>::Identity();
86 for (std::size_t j = 1; j < coordinates.size(); j += 2) {
87 double ct = std::cos(thetas[j / 2]);
88 double st = std::sin(thetas[j / 2]);
89
90 Eigen::Matrix<T, N_, N_> r = Eigen::Matrix<T, N_, N_>::Identity();
91 r(coordinates[j / 2], coordinates[j / 2]) = ct;
92 r(coordinates[j / 2], coordinates[j / 2 + 1]) = -st;
93 r(coordinates[j / 2 + 1], coordinates[j / 2]) = st;
94 r(coordinates[j / 2 + 1], coordinates[j / 2 + 1]) = ct;
95 rotation *= r;
96 }
97 covariance = rotation.transpose() * covariance * rotation;
98
99 covariances.emplace_back(maths::common::fromDenseMatrix(covariance));
100 }
101
102 points.resize(k);
103 TDoubleVecVec pointsi;
104 for (std::size_t i = 0; i < k; ++i) {
105 LOG_TRACE(<< "mean = " << means[i]);
106 LOG_TRACE(<< "covariance = " << covariances[i]);
107 this->generateMultivariateNormalSamples(
108 means[i].template toVector<TDoubleVec>(),
109 covariances[i].template toVectors<TDoubleVecVec>(), sizes[i], pointsi);
110 for (std::size_t j = 0; j < pointsi.size(); ++j) {
111 points[i].emplace_back(pointsi[j]);
112 }
113 }
114}
115}
116}
