175 double value(double x) const {
176 if (this->knots().empty()) {
177 return 0.0;
178 }
179
180 std::size_t k = CTools::truncate(
181 std::size_t(std::lower_bound(this->knots().begin(), this->knots().end(), x) -
182 this->knots().begin()),
183 std::size_t(1), this->knots().size() - 1);
184
185 if (x == this->knots()[k]) {
186 return this->values()[k];
187 }
188
189 switch (m_Type) {
190 case E_Linear: {
191 double h = this->knots()[k] - this->knots()[k - 1];
192 double c = (this->values()[k] - this->values()[k - 1]) / h;
193 double d = this->values()[k - 1];
194 double r = x - this->knots()[k - 1];
195 return c * r + d;
196 }
197 case E_Cubic: {
198 double h = this->knots()[k] - this->knots()[k - 1];
199 double a = (this->curvatures()[k] - this->curvatures()[k - 1]) / 6.0 / h;
200 double b = this->curvatures()[k - 1] / 2.0;
201 double c = (this->values()[k] - this->values()[k - 1]) / h -
202 (this->curvatures()[k] / 6.0 + this->curvatures()[k - 1] / 3.0) * h;
203 double d = this->values()[k - 1];
204 double r = x - this->knots()[k - 1];
205 return ((a * r + b) * r + c) * r + d;
206 }
207 }
208
209 LOG_ABORT(<< "Unexpected type " << m_Type);
210 }

213 double mean() const {
214 if (this->knots().empty()) {
215 return 0.0;
216 }
217
218 std::size_t n = this->knots().size();
219 double interval = (this->knots()[n - 1] - this->knots()[0]);
220
221 TMeanAccumulator result;
222 switch (m_Type) {
223 case E_Linear:
224 for (std::size_t i = 1; i < this->knots().size(); ++i) {
225 double h = this->knots()[i] - this->knots()[i - 1];
226 double c = (this->values()[i] - this->values()[i - 1]) / h;
227 double d = this->values()[i - 1];
228 result.add(c / 2.0 * h + d, h / interval);
229 }
230 break;
231
232 case E_Cubic:
233 for (std::size_t i = 1; i < this->knots().size(); ++i) {
234 double h = this->knots()[i] - this->knots()[i - 1];
235 double a = (this->curvatures()[i] - this->curvatures()[i - 1]) / 6.0 / h;
236 double b = this->curvatures()[i - 1] / 2.0;
237 double c = (this->values()[i] - this->values()[i - 1]) / h -
238 (this->curvatures()[i] / 6.0 + this->curvatures()[i - 1] / 3.0) * h;
239 double d = this->values()[i - 1];
240 result.add(((a * h / 4.0 + b / 3.0) * h + c / 2.0) * h + d, h / interval);
241 }
242 break;
243 }
244
245 return CBasicStatistics::mean(result);
246 }

253 double slope(double x) const {
254 if (this->knots().empty()) {
255 return 0.0;
256 }
257
258 std::size_t k = CTools::truncate(
259 std::size_t(std::lower_bound(this->knots().begin(), this->knots().end(), x) -
260 this->knots().begin()),
261 std::size_t(1), this->knots().size() - 1);
262
263 switch (m_Type) {
264 case E_Linear: {
265 double h = this->knots()[k] - this->knots()[k - 1];
266 return (this->values()[k] - this->values()[k - 1]) / h;
267 }
268 case E_Cubic: {
269 double h = this->knots()[k] - this->knots()[k - 1];
270 double a = (this->curvatures()[k] - this->curvatures()[k - 1]) / 6.0 / h;
271 double b = this->curvatures()[k - 1] / 2.0;
272 double c = (this->values()[k] - this->values()[k - 1]) / h -
273 (this->curvatures()[k] / 6.0 + this->curvatures()[k - 1] / 3.0) * h;
274 double r = x - this->knots()[k - 1];
275 return ((3.0 * a * r + 2.0 * b) * r + c);
276 }
277 }
278
279 LOG_ABORT(<< "Unexpected type " << m_Type);
280 }

288 double absSlope() const {
289 double result = 0.0;
290
291 std::size_t n = this->knots().size();
292
293 switch (m_Type) {
294 case E_Linear:
295 for (std::size_t i = 1; i < n; ++i) {
296 result += std::fabs((this->values()[i] - this->values()[i - 1]));
297 }
298 break;
299
300 case E_Cubic:
301 for (std::size_t i = 1; i < n; ++i) {
302 double a = this->knots()[i - 1];
303 double b = this->knots()[i];
304 double h = b - a;
305 double ai = (this->curvatures()[i] - this->curvatures()[i - 1]) / 6.0 / h;
306 double bi = this->curvatures()[i - 1] / 2.0;
307 double ci = (this->values()[i] - this->values()[i - 1]) / h -
308 (this->curvatures()[i] / 6.0 + this->curvatures()[i - 1] / 3.0) * h;
309
310 double descriminant = bi * bi - 3.0 * ai * ci;
311 if (descriminant < 0.0) {
312 result += std::fabs(((ai * h + bi) * h + ci) * h);
313 continue;
314 }
315 double rl = CTools::truncate(a - (bi + descriminant) / 3.0 / ai, a, b);
316 double rr = CTools::truncate(a + (-bi + descriminant) / 3.0 / ai, a, b);
317 if (rl > rr) {
318 std::swap(rl, rr);
319 }
320 result +=
321 std::fabs(((ai * (rl - a) + bi) * (rl - a) + ci) * (rl - a)) +
322 std::fabs(((ai * (rr - rl) + bi) * (rr - rl) + ci) * (rr - rl)) +
323 std::fabs(((ai * (b - rr) + bi) * (b - rr) + ci) * (b - rr));
324 }
325 break;
326 }
327
328 return result / (this->knots()[n - 1] - this->knots()[0]);
329 }

338 void coefficients(TDoubleVec* a = nullptr,
339 TDoubleVec* b = nullptr,
340 TDoubleVec* c = nullptr,
341 TDoubleVec* d = nullptr) const {
342 if (a)
343 a->reserve(this->values().size());
344 if (b)
345 b->reserve(this->values().size());
346 if (c)
347 c->reserve(this->values().size());
348 if (d)
349 d->reserve(this->values().size());
350
351 switch (m_Type) {
352 case E_Linear:
353 for (std::size_t i = 1; i < this->knots().size(); ++i) {
354 double h = this->knots()[i] - this->knots()[i - 1];
355 if (a)
356 a->push_back(0.0);
357 if (b)
358 b->push_back(0.0);
359 if (c)
360 c->push_back((this->values()[i] - this->values()[i - 1]) / h);
361 if (d)
362 d->push_back(this->values()[i - 1]);
363 }
364 break;
365
366 case E_Cubic:
367 for (std::size_t i = 1; i < this->knots().size(); ++i) {
368 double h = this->knots()[i] - this->knots()[i - 1];
369 if (a)
370 a->push_back((this->curvatures()[i] - this->curvatures()[i - 1]) / 6.0 / h);
371 if (b)
372 b->push_back(this->curvatures()[i - 1] / 2.0);
373 if (c)
374 c->push_back(
375 (this->values()[i] - this->values()[i - 1]) / h -
376 (this->curvatures()[i] / 6.0 + this->curvatures()[i - 1] / 3.0) * h);
377 if (d)
378 d->push_back(this->values()[i - 1]);
379 }
380 break;
381 }

400 bool interpolate(const TDoubleVec& knots, const TDoubleVec& values, EBoundaryCondition boundary) {
401 if (knots.size() < 2) {
402 LOG_ERROR(<< "Insufficient knot points supplied");
403 return false;
404 }
405 if (knots.size() != values.size()) {
406 LOG_ERROR(<< "Number knots not equal to number of values: "
407 << " knots = " << knots << " values = " << values);
408 return false;
409 }
410
411 TNonConstKnots oldKnots;
412 TNonConstValues oldValues;
413 oldKnots.swap(this->knotsRef());
414 oldValues.swap(this->valuesRef());
415 this->knotsRef().assign(knots.begin(), knots.end());
416 this->valuesRef().assign(values.begin(), values.end());
417
418 // If two knots are equal to working precision then we
419 // de-duplicate and use the average of the values at the
420 // duplicate knots. The exit condition from this loop is
421 // rather subtle and ensures that "last" always points
422 // to the last element in the reduced knot set.
423 std::size_t last = std::numeric_limits<std::size_t>::max();
424 std::size_t n = this->knots().size();
425 for (std::size_t i = 1; i <= n; ++i) {
426 std::size_t i_ = i - 1;
427 double knot = this->knots()[i_];
428 for (; i < n && this->knots()[i] == knot; ++i) {
429 }
430 if (i - i_ > 1) {
431 TMeanAccumulator value;
432 for (std::size_t j = i_; j < i; ++j) {
433 value.add(this->values()[j]);
434 }
435 this->valuesRef()[i_] = CBasicStatistics::mean(value);
436 }
437 if (++last != i_) {
438 this->knotsRef()[last] = this->knots()[i_];
439 this->valuesRef()[last] = this->values()[i_];
440 }
441 }
442 this->knotsRef().erase(this->knotsRef().begin() + last + 1,
443 this->knotsRef().end());
444 this->valuesRef().erase(this->valuesRef().begin() + last + 1,
445 this->valuesRef().end());
446 n = this->knots().size();
447 LOG_TRACE(<< "knots = " << this->knots());
448 LOG_TRACE(<< "values = " << this->values());
449
450 if (this->knots().size() < 2) {
451 LOG_ERROR(<< "Insufficient distinct knot points supplied");
452 this->knotsRef().swap(oldKnots);
453 this->valuesRef().swap(oldValues);
454 return false;
455 }
456
457 switch (m_Type) {
458 case E_Linear:
459 // Curvatures are all zero and we don't bother to store them.
460 break;
461
462 case E_Cubic: {
463 this->curvaturesRef().clear();
464 this->curvaturesRef().reserve(n);
465
466 // Construct the diagonals: a is the subdiagonal, b is the
467 // main diagonal and c is the superdiagonal.
468
469 TDoubleVec a;
470 TDoubleVec b;
471 TDoubleVec c;
472 a.reserve(n - 1);
473 b.reserve(n);
474 c.reserve(n - 1);
475
476 double h = this->knots()[1] - this->knots()[0];
477 double h_ = this->knots()[n - 1] - this->knots()[n - 2];
478
479 switch (boundary) {
480 case E_Natural:
481 b.push_back(1.0);
482 c.push_back(0.0);
483 this->curvaturesRef().push_back(0.0);
484 break;
485
486 case E_ParabolicRunout:
487 b.push_back(1.0);
488 c.push_back(-1.0);
489 this->curvaturesRef().push_back(0.0);
490 break;
491
492 case E_Periodic:
493 b.push_back(2.0 * (h + h_));
494 c.push_back(h - 1.0);
495 this->curvaturesRef().push_back(
496 6.0 * ((this->values()[1] - this->values()[0]) / h -
497 (this->values()[0] - this->values()[n - 2]) / h_));
498 break;
499 }
500
501 for (std::size_t i = 1; i + 1 < n; ++i) {
502 h_ = h;
503 h = this->knots()[i + 1] - this->knots()[i];
504 a.push_back(h_);
505 b.push_back(2.0 * (h + h_));
506 c.push_back(h);
507 this->curvaturesRef().push_back(
508 6.0 * ((this->values()[i + 1] - this->values()[i]) / h -
509 (this->values()[i] - this->values()[i - 1]) / h_));
510 }
511
512 h_ = h;
513 h = this->knots()[1] - this->knots()[0];
514
515 switch (boundary) {
516 case E_Natural:
517 a.push_back(0.0);
518 b.push_back(1.0);
519 this->curvaturesRef().push_back(0.0);
520 if (!spline_detail::solveTridiagonal(a, b, c, this->curvaturesRef())) {
521 LOG_ERROR(<< "Failed to calculate curvatures");
522 return false;
523 }
524 break;
525
526 case E_ParabolicRunout:
527 a.push_back(-1.0);
528 b.push_back(1.0);
529 this->curvaturesRef().push_back(0.0);
530 if (!spline_detail::solveTridiagonal(a, b, c, this->curvaturesRef())) {
531 LOG_ERROR(<< "Failed to calculate curvatures");
532 return false;
533 }
534 break;
535
536 case E_Periodic: {
537 a.push_back(h_ * (1.0 - h));
538 b.push_back(2.0 * (h + h_));
539 TDoubleVec u(n, 0.0);
540 u[0] = 1.0;
541 u[n - 1] = h;
542 TDoubleVec v(n, 0.0);
543 v[1] = 1.0;
544 v[n - 2] = h_;
545 this->curvaturesRef().push_back(
546 6.0 * ((this->values()[1] - this->values()[n - 1]) / h -
547 (this->values()[n - 1] - this->values()[n - 2]) / h_));
548 if (!spline_detail::solvePeturbedTridiagonal(
549 a, b, c, u, v, this->curvaturesRef())) {
550 LOG_ERROR(<< "Failed to calculate curvatures");
551 return false;
552 }
553 } break;
554 }
555 break;
556 }
557 }
558
559 return true;
560 }

566 void describe(const std::string& indent, std::string& result) const {
567 result += "\n" + indent + "cubic spline";
568 if (!this->initialized()) {
569 result += " zero everywhere";
570 return;
571 }
572
573 result += ":";
574 switch (m_Type) {
575 case E_Linear:
576 for (std::size_t i = 1; i < this->knots().size(); ++i) {
577 double h = this->knots()[i] - this->knots()[i - 1];
578 double c = (this->values()[i] - this->values()[i - 1]) / h;
579 double d = this->values()[i - 1];
580 std::string kl =
581 core::CStringUtils::typeToStringPretty(this->knots()[i - 1]);
582 result += "\n" + indent +
583 core::CStringUtils::typeToStringPretty(c) + " (x - " +
584 kl + ") + " + core::CStringUtils::typeToStringPretty(d) +
585 " x in [" + kl + "," +
586 core::CStringUtils::typeToStringPretty(this->knots()[i]) + ")";
587 }
588 break;
589
590 case E_Cubic:
591 for (std::size_t i = 1; i < this->knots().size(); ++i) {
592 double h = this->knots()[i] - this->knots()[i - 1];
593 double a = (this->curvatures()[i] - this->curvatures()[i - 1]) / 6.0 / h;
594 double b = this->curvatures()[i - 1] / 2.0;
595 double c = (this->values()[i] - this->values()[i - 1]) / h -
596 (this->curvatures()[i] / 6.0 + this->curvatures()[i - 1] / 3.0) * h;
597 double d = this->values()[i - 1];
598 std::string kl =
599 core::CStringUtils::typeToStringPretty(this->knots()[i - 1]);
600 result += "\n" + indent + core::CStringUtils::typeToStringPretty(a) +
601 " (x - " + kl + ")^3 + " +
602 core::CStringUtils::typeToStringPretty(b) + " (x - " + kl +
603 ")^2 + " + core::CStringUtils::typeToStringPretty(c) + " (x - " +
604 kl + ") + " + core::CStringUtils::typeToStringPretty(d) +
605 " x in [" + kl + "," +
606 core::CStringUtils::typeToStringPretty(this->knots()[i]) + ")";
607 }
608 break;
609 }
