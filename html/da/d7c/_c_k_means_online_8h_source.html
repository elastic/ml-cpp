239 template<typename RNG>
240 static bool kmeans(RNG& rng,
241 TSphericalClusterVec clusters,
242 std::size_t k,
243 TSphericalClusterVecVec& result,
244 std::size_t numberSeeds = NUMBER_SEEDS,
245 std::size_t maxIterations = MAX_ITERATIONS) {
246 result.clear();
247
248 if (k == 0) {
249 LOG_ERROR(<< "Bad request for zero categories");
250 return false;
251 }
252 if (clusters.empty()) {
253 return true;
254 }
255
256 result.reserve(std::min(k, clusters.size()));
257
258 if (k >= clusters.size()) {
259 TSphericalClusterVec cluster(1);
260 for (std::size_t i = 0; i < clusters.size(); ++i) {
261 cluster[0] = clusters[i];
262 result.push_back(cluster);
263 }
264 return true;
265 } else if (k == 1) {
266 result.push_back(clusters);
267 return true;
268 }
269
270 CKMeans<TSphericalCluster> kmeans;
271 kmeans.setPoints(std::move(clusters));
272
273 CBasicStatistics::SMin<double>::TAccumulator minCost;
274 TSphericalClusterVec centres;
275 TSphericalClusterVecVec candidates;
276 CKMeansPlusPlusInitialization<TSphericalCluster, RNG> seed{rng};
277 for (std::size_t i = 0; i < numberSeeds; ++i) {
278 seed.run(kmeans.beginPoints(), kmeans.endPoints(), k, centres);
279 kmeans.setCentres(centres);
280 kmeans.run(maxIterations);
281 kmeans.clusters(candidates);
282 candidates.erase(std::remove_if(candidates.begin(), candidates.end(),
283 [](TSphericalClusterVec& cluster) {
284 return cluster.empty();
285 }),
286 candidates.end());
287 CSphericalGaussianInfoCriterion<TSphericalCluster, E_BIC> criterion;
288 criterion.add(candidates);
289 double cost{criterion.calculate()};
290 if (minCost.add(cost)) {
291 result.swap(candidates);
292 }
293 }
294
295 LOG_TRACE(<< "result = " << result);
296
297 return true;
298 }

306 bool split(const TSizeVecVec& split, TKMeansOnlineVec& result) {
307 result.clear();
308 if (!this->checkSplit(split)) {
309 return false;
310 }
311
312 result.reserve(split.size());
313 TStoragePointMeanAccumulatorDoublePrVec clusters;
314 for (std::size_t i = 0; i < split.size(); ++i) {
315 clusters.clear();
316 clusters.reserve(split[i].size());
317 for (std::size_t j = 0; j < split[i].size(); ++j) {
318 clusters.push_back(m_Clusters[split[i][j]]);
319 }
320 result.emplace_back(clusters, m_K, m_DecayRate, m_MinClusterSize,
321 m_BufferSize, m_NumberSeeds, m_MaxIterations);
322 }
323
324 return true;
325 }

398 void sample(std::size_t numberSamples, TDoublePointVec& result) const {
399 result.clear();
400 if (numberSamples == 0) {
401 return;
402 }
403
404 using TDoubleVec = std::vector<double>;
405 using TDoubleSizePr = std::pair<double, std::size_t>;
406
407 // See, for example, Effective C++ item 3.
408 const_cast<CKMeansOnline*>(this)->reduce();
409 LOG_TRACE(<< "categories = " << m_Clusters);
410
411 TDoubleVec counts;
412 counts.reserve(m_Clusters.size());
413 double Z{0.0};
414 for (std::size_t i = 0; i < m_Clusters.size(); ++i) {
415 double ni{CBasicStatistics::count(m_Clusters[i].first)};
416 counts.push_back(ni);
417 Z += ni;
418 }
419 Z /= static_cast<double>(numberSamples);
420 for (std::size_t i = 0; i < counts.size(); ++i) {
421 counts[i] /= Z;
422 }
423 LOG_TRACE(<< "weights = " << counts << ", Z = " << Z << ", n = " << numberSamples);
424
425 result.reserve(2 * numberSamples);
426
427 TDoubleVec weights;
428 TDoublePointVec clusterSamples;
429 for (std::size_t i = 0; i < m_Clusters.size(); ++i) {
430 double ni{counts[i]};
431
432 clusterSamples.clear();
433 TDoublePoint m{CBasicStatistics::mean(m_Clusters[i].first)};
434 if (m_Clusters[i].second == 0.0) {
435 clusterSamples.push_back(m);
436 } else {
437 std::size_t ni_{static_cast<std::size_t>(std::ceil(ni))};
438 TDoublePoint v(m_Clusters[i].second);
439 sampleGaussian(ni_, m, v.asDiagonal(), clusterSamples);
440 }
441
442 ni /= static_cast<double>(clusterSamples.size());
443
444 result.insert(result.end(), clusterSamples.begin(), clusterSamples.end());
445 weights.insert(weights.end(), clusterSamples.size(), ni);
446 }
447 LOG_TRACE(<< "samples = " << result);
448 LOG_TRACE(<< "weights = " << weights);
449
450 TDoublePointVec finalSamples;
451 finalSamples.reserve(static_cast<std::size_t>(
452 std::ceil(std::accumulate(weights.begin(), weights.end(), 0.0))));
453 TDoublePointMeanAccumulator sample{las::zero(result[0])};
454 for (;;) {
455 CBasicStatistics::SMin<TDoubleSizePr>::TAccumulator nearest;
456 const TDoublePoint& sample_{CBasicStatistics::mean(sample)};
457 for (std::size_t j = 0; j < result.size(); ++j) {
458 if (weights[j] > 0.0) {
459 nearest.add({las::distance(result[j], sample_), j});
460 }
461 }
462 if (nearest.count() == 0) {
463 break;
464 }
465
466 std::size_t j{nearest[0].second};
467 const TDoublePoint& xj{result[j]};
468 do {
469 double nj{std::min(1.0 - CBasicStatistics::count(sample), weights[j])};
470 sample.add(xj, nj);
471 weights[j] -= nj;
472 if (CBasicStatistics::count(sample) > ALMOST_ONE) {
473 finalSamples.push_back(CBasicStatistics::mean(sample));
474 sample = TDoublePointMeanAccumulator{las::zero(result[0])};
475 }
476 } while (weights[j] > 0.0);
477 }
478
479 result = std::move(finalSamples);
480 LOG_TRACE(<< "# samples = " << result.size());
481 LOG_TRACE(<< "samples = " << result);
482 }

530 void reduce() {
531 deduplicate(m_Clusters);
532
533 if (m_Clusters.size() < m_K) {
534 return;
535 }
536
537 LOG_TRACE(<< "clusters = " << m_Clusters);
538 LOG_TRACE(<< "# clusters = " << m_Clusters.size());
539
540 TSphericalClusterVec oldClusters;
541 this->clusters(oldClusters, std::true_type{});
542
543 TDoublePointMeanVarAccumulator empty{las::zero(oldClusters[0])};
544
545 TSphericalClusterVecVec newClusters;
546 kmeans(m_Rng, std::move(oldClusters), m_K, newClusters, m_NumberSeeds, m_MaxIterations);
547
548 m_Clusters.resize(newClusters.size());
549
550 TDoublePointMeanVarAccumulator centroid;
551 for (std::size_t i = 0; i < newClusters.size(); ++i) {
552 centroid = empty;
553 for (const auto& point : newClusters[i]) {
554 centroid.add(point);
555 }
556 double n{CBasicStatistics::count(centroid)};
557 TDoublePoint& m{CBasicStatistics::moment<0>(centroid)};
558 CBasicStatistics::count(m_Clusters[i].first) = n;
559 CBasicStatistics::moment<0>(m_Clusters[i].first) = std::move(m);
560 m_Clusters[i].second = variance(centroid);
561 }
562
563 LOG_TRACE(<< "reduced clusters = " << m_Clusters);
564 LOG_TRACE(<< "# reduced clusters = " << m_Clusters.size());
565 }

570 static void deduplicate(TStoragePointMeanAccumulatorDoublePrVec& clusters) {
571 if (clusters.size() > 1) {
572 std::sort(clusters.begin(), clusters.end(), [](const auto& lhs, const auto& rhs) {
573 return CBasicStatistics::mean(lhs.first) <
574 CBasicStatistics::mean(rhs.first);
575 });
576 auto back = clusters.begin();
577 for (auto i = back + 1; i != clusters.end(); ++i) {
578 if (CBasicStatistics::mean(back->first) == CBasicStatistics::mean(i->first)) {
579 back->first += i->first;
580 double n[]{CBasicStatistics::count(back->first),
581 CBasicStatistics::count(i->first)};
582 back->second = (n[0] * back->second + n[1] * i->second) /
583 (n[0] + n[1]);
584 } else if (++back != i) {
585 *back = std::move(*i);
586 }
587 }
588 clusters.erase(back + 1, clusters.end());
589 }

595 result.clear();
