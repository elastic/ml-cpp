309 template<int BITS>
310 class CLookupTableForFastLog {
311 public:
312 static const std::size_t BINS = 1 << BITS;
313
314 public:
315 using TArray = std::array<double, BINS>;
316
317 public:
319 CLookupTableForFastLog() {
320 // Notes:
321 // 1) The shift is the maximum mantissa / BINS.
322 // 2) The sign bit is set to 0 which is positive.
323 // 3) The exponent is set to 1022, which is 0 in two's
324 // complement.
325 // 4) This implementation is endian neutral because it
326 // is constructing a look up from the mantissa value
327 // (interpreted as an integer) to the corresponding
328 // double value and fastLog uses the same approach
329 // to extract the mantissa.
330 std::uint64_t dx = 0x10000000000000ull / BINS;
331 core::CIEEE754::SDoubleRep x;
332 x.s_Sign = 0;
333 x.s_Mantissa = (dx / 2) & core::CIEEE754::IEEE754_MANTISSA_MASK;
334 x.s_Exponent = 1022;
335 for (std::size_t i = 0; i < BINS;
336 ++i, x.s_Mantissa = (x.s_Mantissa + dx) & core::CIEEE754::IEEE754_MANTISSA_MASK) {
337 double value;
338 static_assert(sizeof(double) == sizeof(core::CIEEE754::SDoubleRep),
339 "SDoubleRep definition unsuitable for memcpy to double");
340 // Use memcpy() rather than union to adhere to strict
341 // aliasing rules
342 std::memcpy(&value, &x, sizeof(double));
343 m_Table[i] = stable(std::log2(value));
344 }
345 }
346
348 const double& operator[](std::uint64_t mantissa) const {
349 return m_Table[mantissa >> FAST_LOG_SHIFT];
350 }
351
352 private:
354 TArray m_Table;
355 };

367 static double fastLog(double x) {
368 std::uint64_t mantissa;
369 int log2;
370 core::CIEEE754::decompose(x, mantissa, log2);
371 return 0.693147180559945 * (FAST_LOG_TABLE[mantissa] + log2);
372 }
