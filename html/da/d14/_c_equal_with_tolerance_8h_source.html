78template<typename T>
79class CEqualWithTolerance : public CToleranceTypes {
80public:
81 CEqualWithTolerance(unsigned int toleranceType, const T& eps)
82 : m_ToleranceType(toleranceType), m_AbsoluteEps(abs(norm(eps))),
83 m_RelativeEps(abs(norm(eps))) {}
84
85 CEqualWithTolerance(unsigned int toleranceType, const T& absoluteEps, const T& relativeEps)
86 : m_ToleranceType(toleranceType), m_AbsoluteEps(abs(norm(absoluteEps))),
87 m_RelativeEps(abs(norm(relativeEps))) {}
88
89 bool operator()(const T& lhs, const T& rhs) const {
90 const T& max = norm(rhs) > norm(lhs) ? rhs : lhs;
91 const T& min = norm(rhs) > norm(lhs) ? lhs : rhs;
92 const T& maxAbs = abs(norm(rhs)) > abs(norm(lhs)) ? rhs : lhs;
93
94 T difference = max - min;
95
96 switch (m_ToleranceType) {
97 case 2: // absolute & relative
98 return (norm(difference) <= m_AbsoluteEps) &&
99 (norm(difference) <= m_RelativeEps * abs(norm(maxAbs)));
100 case 3: // absolute
101 return norm(difference) <= m_AbsoluteEps;
102 case 6: // relative
103 return norm(difference) <= m_RelativeEps * abs(norm(maxAbs));
104 case 7: // absolute | relative
105 return (norm(difference) <= m_AbsoluteEps) ||
106 (norm(difference) <= m_RelativeEps * abs(norm(maxAbs)));
107 }
108 LOG_ERROR(<< "Unexpected tolerance type " << m_ToleranceType);
109 return false;
110 }
111
112private:
113 using TNorm =
114 decltype(equal_with_tolerance_detail::SNorm<T>::dispatch(std::declval<T>()));
115
116private:
118 template<typename U>
119 static inline U abs(const U& x) {
120 return x < U(0) ? -x : x;
121 }
122
124 static auto norm(const T& t) {
125 return equal_with_tolerance_detail::SNorm<T>::dispatch(t);
126 }
127
128private:
129 unsigned int m_ToleranceType;
130 TNorm m_AbsoluteEps;
131 TNorm m_RelativeEps;
132};
133}
