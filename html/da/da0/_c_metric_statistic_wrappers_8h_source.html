112 static TDouble1Vec value(const TVarianceAccumulator& stat) {
113 TDouble1Vec result;
114 if (maths::common::CBasicStatistics::count(stat) >= 2.0) {
115 result.assign({maths::common::CBasicStatistics::maximumLikelihoodVariance(stat),
116 maths::common::CBasicStatistics::mean(stat)});
117 }
118 return result;
119 }
121 static TDouble1Vec value(const TMedianAccumulator& stat) {
122 double result;
123 if (!stat.quantile(50.0, result)) {
124 return TDouble1Vec{0.0};
125 }
126 return TDouble1Vec{result};
127 }

140 static TDouble1Vec influencerValue(const TVarianceAccumulator& stat) {
141 // We always return an influence value (independent of the count)
142 // because this is not used to directly compute a variance only
143 // to adjust the bucket variance.
144 TDouble1Vec result(2);
145 result[0] = maths::common::CBasicStatistics::maximumLikelihoodVariance(stat);
146 result[1] = maths::common::CBasicStatistics::mean(stat);
147 return result;
148 }
