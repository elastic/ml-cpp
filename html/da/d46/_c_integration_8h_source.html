112 template<EOrder ORDER, typename F, typename T>
113 static bool gaussLegendre(const F& function, double a, double b, T& result) {
114 result = T();
115
116 const double* weights = CGaussLegendreQuadrature::weights(ORDER);
117 const double* abscissas = CGaussLegendreQuadrature::abscissas(ORDER);
118
119 // Evaluate f(x) at the abscissas and compute the weighted sum
120 // of the quadrature.
121 double centre = (a + b) / 2.0;
122 double range = (b - a) / 2.0;
123 for (unsigned int i = 0; i < ORDER; ++i) {
124 T fx;
125 if (!function(centre + range * abscissas[i], fx)) {
126 return false;
127 }
128 fx *= weights[i];
129 result += fx;
130 }
131 result *= range;
132
133 return true;
134 }

164 template<EOrder ORDER, typename F, typename G, typename U, typename V>
165 static bool productGaussLegendre(const F& f,
166 const G& g,
167 double a,
168 double b,
169 U& productIntegral,
170 U& fIntegral,
171 V& gIntegral) {
172 productIntegral = U();
173 fIntegral = U();
174 gIntegral = V();
175
176 const double* weights = CGaussLegendreQuadrature::weights(ORDER);
177 const double* abscissas = CGaussLegendreQuadrature::abscissas(ORDER);
178
179 // Evaluate f(x) at the abscissas and compute the weighted sum
180 // of the quadrature.
181 double centre = (a + b) / 2.0;
182 double range = (b - a) / 2.0;
183 for (unsigned int i = 0; i < ORDER; ++i) {
184 U fx{};
185 V gx{};
186 if (!f(centre + range * abscissas[i], fx) ||
187 !g(centre + range * abscissas[i], gx)) {
188 return false;
189 }
190 double weight = weights[i];
191 productIntegral += fx * gx * weight;
192 fIntegral += fx * weight;
193 gIntegral += gx * weight;
194 }
195
196 productIntegral *= range;
197 fIntegral *= range;
198 gIntegral *= range;
199
200 return true;
201 }

224 template<EOrder ORDER, typename F>
225 static bool logGaussLegendre(const F& function, double a, double b, double& result) {
226 result = 0.0;
227
228 if (b <= a) {
229 std::swap(a, b);
230 }
231
232 const double* weights = CGaussLegendreQuadrature::weights(ORDER);
233 const double* abscissas = CGaussLegendreQuadrature::abscissas(ORDER);
234
235 double fx[ORDER] = {0.0};
236
237 // Evaluate f(x) at the abscissas.
238 double centre = (a + b) / 2.0;
239 double range = (b - a) / 2.0;
240 for (unsigned int i = 0; i < ORDER; ++i) {
241 if (!function(centre + range * abscissas[i], fx[i])) {
242 return false;
243 }
244 }
245
246 // Re-normalize and then take exponentials to avoid underflow.
247 double fmax = *std::max_element(fx, fx + ORDER);
248 for (unsigned int i = 0; i < ORDER; ++i) {
249 fx[i] = std::exp(fx[i] - fmax);
250 }
251
252 // Quadrature.
253 for (unsigned int i = 0; i < ORDER; ++i) {
254 result += weights[i] * fx[i];
255 }
256 result *= range;
257 result = result <= 0.0 ? core::constants::LOG_MIN_DOUBLE : fmax + std::log(result);
258
259 return true;
260 }

287 template<EOrder ORDER, typename F>
288 static bool adaptiveGaussLegendre(const F& f,
289 TDoubleDoublePrVec& intervals,
290 TDoubleVec& fIntervals,
291 std::size_t refinements,
292 std::size_t splitsPerRefinement,
293 double tolerance,
294 double& result) {
295 if (intervals.size() != fIntervals.size()) {
296 LOG_ERROR(<< "Inconsistent intervals and function integrals: " << intervals
297 << " " << fIntervals);
298 return false;
299 }
300
301 result += std::accumulate(fIntervals.begin(), fIntervals.end(), 0.0);
302 LOG_TRACE(<< "initial = " << result);
303
304 TDoubleVec corrections;
305 corrections.reserve(fIntervals.size());
306 for (std::size_t i = 0; i < fIntervals.size(); ++i) {
307 corrections.push_back(std::fabs(fIntervals[i]));
308 }
309
310 for (std::size_t i = 0; !intervals.empty() && i < refinements; ++i) {
311 std::size_t n = intervals.size();
312 double cutoff = tolerance * std::fabs(result) / static_cast<double>(n);
313
314 std::size_t end = 0;
315 for (std::size_t j = 0; j < corrections.size(); ++j) {
316 if (corrections[j] > cutoff) {
317 std::swap(intervals[end], intervals[j]);
318 std::swap(fIntervals[end], fIntervals[j]);
319 std::swap(corrections[end], corrections[j]);
320 ++end;
321 }
322 }
323 if (end != corrections.size()) {
324 intervals.erase(intervals.begin() + end, intervals.end());
325 fIntervals.erase(fIntervals.begin() + end, fIntervals.end());
326 corrections.erase(corrections.begin() + end, corrections.end());
327 }
328 n = intervals.size();
329
330 if (i + 1 < refinements) {
331 intervals.reserve(splitsPerRefinement * n);
332 fIntervals.reserve(splitsPerRefinement * n);
333 corrections.reserve(splitsPerRefinement * n);
334 }
335
336 for (std::size_t j = 0; j < n; ++j) {
337 if (corrections[j] <= cutoff) {
338 corrections[j] = 0.0;
339 continue;
340 }
341
342 double fjOld = fIntervals[j];
343 double fjNew = 0.0;
344
345 double aj = intervals[j].first;
346 double dj = (intervals[j].second - intervals[j].first) /
347 static_cast<double>(splitsPerRefinement);
348 for (std::size_t k = 0; k < splitsPerRefinement; ++k, aj += dj) {
349 double df;
350 if (CIntegration::gaussLegendre<ORDER>(f, aj, aj + dj, df)) {
351 fjNew += df;
352 if (i + 1 < refinements) {
353 if (k == 0) {
354 intervals[j] = TDoubleDoublePr(aj, aj + dj);
355 fIntervals[j] = df;
356 } else {
357 intervals.push_back(TDoubleDoublePr(aj, aj + dj));
358 fIntervals.push_back(df);
359 }
360 }
361 } else {
362 LOG_ERROR(<< "Couldn't integrate f over [" << aj << ","
363 << aj + dj << "]");
364 return false;
365 }
366 }
367
368 LOG_TRACE(<< "fjNew = " << fjNew << ", fjOld = " << fjOld);
369 double correction = fjNew - fjOld;
370 if (i + 1 < refinements) {
371 corrections[j] = std::fabs(correction);
372 corrections.resize(corrections.size() + splitsPerRefinement - 1,
373 std::fabs(correction));
374 }
375
376 result += correction;
377 cutoff = tolerance * std::fabs(result) / static_cast<double>(n);
378 }
379 }
380
381 return true;
382 }

444 static bool next(std::size_t d, TUIntVec& indices, TUIntVec& stop) {
445 for (;;) {
446 ++indices[d];
447 if (indices[d] > stop[d]) {
448 if (d == DIMENSION - 1) {
449 break;
450 }
451 indices[d] = 1;
452 ++d;
453 } else {
454 for (std::size_t j = 0; j < d; ++j) {
455 stop[j] = stop[d] - indices[d] + 1;
456 }
457 indices[0] = stop[0];
458 d = 0;
459 return true;
460 }
461 }
462 return false;

465 CSparseGaussLegendreQuadrature() {
466 // Generate the weights. We don't exploit the weight and
467 // abscissa symmetries to reduce the static storage since
468 // this reduces the speed of integration and since we limit
469 // the dimension the maximum number of points will be 8761.
470 //
471 // Note this uses the construction:
472 // Q^d_l = \sum{l <= ||k||_1 <= l+d-1}{ (-1)^(l+d-||k||_1-1) (d-1 ||k||_1-l) (Q^1_k_1 x ... x Q^1_k_d) }
473
474 using TVectorDoubleMap = std::map<TVector, double>;
475
476 TVectorDoubleMap ordered;
477
478 for (unsigned int l = ORDER > DIMENSION ? ORDER - DIMENSION : 0;
479 l < ORDER; ++l) {
480 LOG_TRACE(<< "order = " << l);
481 std::size_t d = 0;
482 TUIntVec indices(DIMENSION, 1);
483 indices[0] = l + 1;
484 TUIntVec stop(DIMENSION, l + 1);
485
486 double sign = (ORDER - l - 1) % 2 == 1 ? -1.0 : 1.0;
487 double scale = sign * CIntegerTools::binomial(DIMENSION - 1,
488 DIMENSION + l - ORDER);
489 LOG_TRACE(<< "scale = " << scale);
490
491 do {
492 LOG_TRACE(<< "indices = " << indices);
493
494 unsigned int n = 1;
495 for (std::size_t i = 0; i < indices.size(); ++i) {
496 n *= indices[i];
497 }
498 LOG_TRACE(<< "Number of points = " << n);
499
500 TDoubleVec weights(n, 1.0);
501 TVectorVec points(n, TVector(0.0));
502 for (unsigned int i = 0; i < n; ++i) {
503 for (unsigned int i_ = i, j = 0; j < indices.size();
504 i_ /= indices[j], ++j) {
505 EOrder order = static_cast<EOrder>(indices[j]);
506 const double* w = CGaussLegendreQuadrature::weights(order);
507 const double* a = CGaussLegendreQuadrature::abscissas(order);
508 std::size_t k = i_ % indices[j];
509 weights[i] *= w[k];
510 points[i](j) = a[k];
511 }
512 }
513 LOG_TRACE(<< "weights = " << weights);
514 LOG_TRACE(<< "points = " << points);
515 for (std::size_t i = 0; i < n; ++i) {
516 ordered[points[i]] += scale * weights[i];
517 }
518 } while (next(d, indices, stop));
519 }
520
521 m_Weights.reserve(ordered.size());
522 m_Points.reserve(ordered.size());
523 for (const auto& i : ordered) {
524 m_Weights.push_back(i.second);
525 m_Points.push_back(i.first);
526 }
527 }

531 TDoubleVec m_Weights;
532 TVectorVec m_Points;
533 };

557 template<EOrder ORDER, EDimension DIMENSION, typename F, typename T>
558 static bool
559 sparseGaussLegendre(const F& function, const TDoubleVec& a, const TDoubleVec& b, T& result) {
560 using TSparseQuadrature = CSparseGaussLegendreQuadrature<ORDER, DIMENSION>;
561 using TVector = typename TSparseQuadrature::TVector;
562 using TVectorVec = typename TSparseQuadrature::TVectorVec;
563
564 result = T();
565
566 if (a.size() != static_cast<std::size_t>(DIMENSION)) {
567 LOG_ERROR(<< "Bad lower limits: " << a);
568 return false;
569 }
570 if (b.size() != static_cast<std::size_t>(DIMENSION)) {
571 LOG_ERROR(<< "Bad upper limits: " << b);
572 return false;
573 }
574
575 const TDoubleVec& weights = TSparseQuadrature::instance().weights();
576 const TVectorVec& points = TSparseQuadrature::instance().points();
577
578 // Evaluate f(x) at the abscissas and compute the weighted sum
579 // of the quadrature.
580
581 TVector a_(a.begin(), a.end());
582 TVector b_(b.begin(), b.end());
583 TVector centre = (a_ + b_) / 2.0;
584 TVector range = (b_ - a_) / 2.0;
585
586 for (std::size_t i = 0; i < weights.size(); ++i) {
587 T fx;
588 if (!function(centre + range * points[i], fx)) {
589 return false;
590 }
591 fx *= weights[i];
592 result += fx;
593 }
594
595 for (std::size_t i = 0; i < DIMENSION; ++i) {
596 result *= range(i);
597 }
598
599 return true;
600 }
