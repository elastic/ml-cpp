44template<typename T, typename A>
45std::size_t dynamicSize(const std::list<T, A>& t) {
46 return memory::elementDynamicSize(t) +
47 (memory_detail::EXTRA_NODES + t.size()) *
48 (sizeof(T) + memory::storageNodeOverhead(t));
49}

51template<typename T, typename A>
52std::size_t dynamicSize(const std::deque<T, A>& t) {
53 // std::deque is a pointer to an array of pointers to pages
54 std::size_t pageSize = std::max(sizeof(T), memory_detail::MIN_DEQUE_PAGE_SIZE);
55 std::size_t itemsPerPage = pageSize / sizeof(T);
56 // This could be an underestimate if items have been removed
57 std::size_t numPages = (t.size() + itemsPerPage - 1) / itemsPerPage;
58 // This could also be an underestimate if items have been removed
59 std::size_t pageVecEntries = std::max(numPages, memory_detail::MIN_DEQUE_PAGE_VEC_ENTRIES);
60
61 return memory::elementDynamicSize(t) +
62 pageVecEntries * sizeof(std::size_t) + numPages * pageSize;
63}

100template<typename T, typename A>
101void dynamicSize(const char* name,
102 const std::list<T, A>& t,
103 const CMemoryUsage::TMemoryUsagePtr& mem) {
104 // std::list appears to use 2 pointers per list node
105 // (prev and next pointers).
106 std::string componentName(name);
107 componentName += "_list";
108
109 std::size_t listSize = (memory_detail::EXTRA_NODES + t.size()) *
110 (sizeof(T) + memory::storageNodeOverhead(t));
111
112 CMemoryUsage::SMemoryUsage usage(componentName, listSize);
113 CMemoryUsage::TMemoryUsagePtr ptr = mem->addChild();
114 ptr->setName(usage);
115
116 memory_debug::elementDynamicSize(std::move(componentName), t, mem);
117}

119template<typename T, typename C, typename A>
120void dynamicSize(const char* name,
121 const std::deque<T, A>& t,
122 const CMemoryUsage::TMemoryUsagePtr& mem) {
123 // std::deque is a pointer to an array of pointers to pages
124 std::string componentName(name);
125 componentName += "_deque";
126
127 std::size_t pageSize = std::max(sizeof(T), memory_detail::MIN_DEQUE_PAGE_SIZE);
128 std::size_t itemsPerPage = pageSize / sizeof(T);
129 // This could be an underestimate if items have been removed
130 std::size_t numPages = (t.size() + itemsPerPage - 1) / itemsPerPage;
131 // This could also be an underestimate if items have been removed
132 std::size_t pageVecEntries = std::max(numPages, memory_detail::MIN_DEQUE_PAGE_VEC_ENTRIES);
133
134 std::size_t dequeTotal = pageVecEntries * sizeof(std::size_t) + numPages * pageSize;
135 std::size_t dequeUsed = numPages * sizeof(std::size_t) + t.size() * sizeof(T);
136
137 CMemoryUsage::SMemoryUsage usage(componentName, dequeTotal, dequeTotal - dequeUsed);
138 CMemoryUsage::TMemoryUsagePtr ptr = mem->addChild();
139 ptr->setName(usage);
140
141 memory_debug::elementDynamicSize(std::move(componentName), t, mem);
142}
