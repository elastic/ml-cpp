58template<typename POINT, typename NEAREST_NEIGHBOURS>
59class CNearestNeighbourMethod {
60public:
61 using TPointVec = std::vector<POINT>;
62 using TMatrix = typename common::SConformableMatrix<POINT>::Type;
63
64public:
65 CNearestNeighbourMethod(bool computeFeatureInfluence,
66 std::size_t k,
67 NEAREST_NEIGHBOURS lookup,
68 TProgressCallback recordProgress)
69 : m_ComputeFeatureInfluence{computeFeatureInfluence}, m_K{k},
70 m_Lookup{std::move(lookup)}, m_RecordProgress{std::move(recordProgress)} {}
71 virtual ~CNearestNeighbourMethod() = default;
72
74 bool computeFeatureInfluence() const { return m_ComputeFeatureInfluence; }
75
77 std::size_t n() const { return m_Lookup.size(); }
78
80 std::size_t k() const { return m_K; }
81
83 const NEAREST_NEIGHBOURS& lookup() const { return m_Lookup; }
84
86 TDouble1VecVec2Vec
87 run(const TPointVec& points, const TMatrix& projection, double eps, std::size_t numberScores) {
88
89 this->setup(points, projection);
90
91 // We call add exactly once for each point. Scores is presized
92 // so any writes to it are safe.
93 TDouble1VecVec2Vec scores(this->numberMethods(), TDouble1VecVec(numberScores));
94 core::parallel_for_each(
95 points.begin(), points.end(),
96 [&, neighbours = TPointVec{} ](const POINT& point) mutable {
97 m_Lookup.nearestNeighbours(m_K + 1, point, neighbours);
98 this->add(point, projection, eps, neighbours, scores);
99 },
100 [this](double fractionalProgress) {
101 this->recordProgress(fractionalProgress);
102 });
103
104 this->compute(points, projection, eps, scores);
105
106 return scores;
107 }
108
110 virtual void recoverMemory() {}
111
113 virtual std::size_t staticSize() const { return sizeof(*this); }
114
116 virtual std::size_t memoryUsage() const { return 0; }
117
119
120
121 TProgressCallback& progressRecorder() { return m_RecordProgress; }
123 void recordProgress(double fractionalProgress) {
124 m_RecordProgress(fractionalProgress);
125 }
127
129 virtual std::string print() const {
130 return this->name() + "(n = " + std::to_string(this->n()) +
131 ", k = " + std::to_string(m_K) + ")";
132 }
133
134 virtual void setup(const TPointVec&, const TMatrix&) {}
135 virtual void add(const POINT&, const TMatrix&, double, const TPointVec&, TDouble1VecVec&) {
136 }
137 virtual void compute(const TPointVec&, const TMatrix&, double, TDouble1VecVec&) {}
138
139protected:
140 static double distanceAtEps(const TMatrix& projection,
141 std::size_t j,
142 double eps,
143 const POINT& x,
144 const POINT& y) {
145 double result{0.0};
146 for (std::size_t i = 0; i < common::las::dimension(x); ++i) {
147 result += common::CTools::pow2(x(i) + eps * projection(i, j) - y(i));
148 }
149 return std::sqrt(result);
150 }
151
152private:
153 virtual void add(const POINT& point,
154 const TMatrix& projection,
155 double eps,
156 const TPointVec& neighbours,
157 TDouble1VecVec2Vec& scores) {
158 this->add(point, projection, eps, neighbours, scores[0]);
159 }
160 virtual void compute(const TPointVec& points,
161 const TMatrix& projection,
162 double eps,
163 TDouble1VecVec2Vec& scores) {
164 this->compute(points, projection, eps, scores[0]);
165 }
166 virtual std::size_t numberMethods() const { return 1; }
167 virtual std::string name() const = 0;
168
169private:
170 bool m_ComputeFeatureInfluence;
171 std::size_t m_K;
172 NEAREST_NEIGHBOURS m_Lookup;
173 TProgressCallback m_RecordProgress;

177template<typename POINT, typename NEAREST_NEIGHBOURS>
178class CLof final : public CNearestNeighbourMethod<POINT, NEAREST_NEIGHBOURS> {
179public:
180 using TPointVec = std::vector<POINT>;
181 using TMatrix = typename common::SConformableMatrix<POINT>::Type;
182 using TCoordinate = typename common::SCoordinate<POINT>::Type;
183 using TCoordinateVec = std::vector<TCoordinate>;
184 using TUInt32CoordinatePr = std::pair<std::uint32_t, TCoordinate>;
185 using TUInt32CoordinatePrVec = std::vector<TUInt32CoordinatePr>;
186 using CNearestNeighbourMethod<POINT, NEAREST_NEIGHBOURS>::distanceAtEps;
187 static const TCoordinate UNSET_DISTANCE;
188
189public:
190 CLof(bool computeFeatureInfluence,
191 std::size_t k,
192 TProgressCallback recordProgress,
193 NEAREST_NEIGHBOURS lookup = NEAREST_NEIGHBOURS())
194 : CNearestNeighbourMethod<POINT, NEAREST_NEIGHBOURS>{
195 computeFeatureInfluence, k, std::move(lookup), std::move(recordProgress)} {}
196
197 void recoverMemory() override {
198 m_KDistances.resize(this->k() * m_StartAddresses);
199 m_KDistances.shrink_to_fit();
200 m_Lrd.resize(m_StartAddresses);
201 m_Lrd.shrink_to_fit();
202 if (this->computeFeatureInfluence()) {
203 m_LrdAtEps.resize(m_NumberInfluences * m_StartAddresses);
204 m_LrdAtEps.shrink_to_fit();
205 }
206 }
207
208 std::size_t staticSize() const override { return sizeof(*this); }
209
210 std::size_t memoryUsage() const override {
211 return core::memory::dynamicSize(m_KDistances) +
212 core::memory::dynamicSize(m_Lrd) + core::memory::dynamicSize(m_LrdAtEps);
213 }
214
215 static std::size_t estimateOwnMemoryOverhead(bool computeFeatureInfluence,
216 std::size_t k,
217 std::size_t numberPoints,
218 std::size_t dimension) {
219 return numberPoints *
220 (k * sizeof(TUInt32CoordinatePr) +
221 (computeFeatureInfluence ? dimension + 1 : 1) * sizeof(TCoordinate));
222 }
223
224private:
225 void setup(const TPointVec& points, const TMatrix& projection) override {
226
227 m_NumberInfluences = common::las::columns(projection);
228
229 auto minmax = std::minmax_element(
230 points.begin(), points.end(), [](const POINT& lhs, const POINT& rhs) {
231 return lhs.annotation() < rhs.annotation();
232 });
233 m_StartAddresses = minmax.first->annotation();
234 m_EndAddresses = minmax.second->annotation() + 1;
235
236 // In the following, we first shrink then grow to ensure we overwrite
237 // values in the range [start addresses, end addresses).
238
239 std::size_t k{this->k()};
240 m_KDistances.resize(k * m_StartAddresses, {std::uint32_t{0}, UNSET_DISTANCE});
241 m_KDistances.resize(k * m_EndAddresses, {std::uint32_t{0}, UNSET_DISTANCE});
242 m_Lrd.resize(m_StartAddresses, UNSET_DISTANCE);
243 m_Lrd.resize(m_EndAddresses, UNSET_DISTANCE);
244 if (this->computeFeatureInfluence()) {
245 m_LrdAtEps.resize(m_NumberInfluences * m_StartAddresses, UNSET_DISTANCE);
246 m_LrdAtEps.resize(m_NumberInfluences * m_EndAddresses, UNSET_DISTANCE);
247 }
248 }

250 void add(const POINT& point, const TMatrix&, double, const TPointVec& neighbours, TDouble1VecVec&) override {
251 // This is called exactly once for each point therefore an element
252 // of m_KDistances is only ever written by one thread.
253 if (neighbours.size() < 2) {
254 return;
255 }
256 std::size_t i{point.annotation()};
257 std::size_t a(point == neighbours[0] ? 1 : 0);
258 std::size_t b{std::min(this->k() + a - 1, neighbours.size() + a - 2)};
259 for (std::size_t j = a; j <= b; ++j) {
260 std::size_t index{this->kDistanceIndex(i, j - a)};
261 m_KDistances[index].first =
262 static_cast<std::uint32_t>(neighbours[j].annotation());
263 m_KDistances[index].second = common::las::distance(point, neighbours[j]);
264 }
265 }
266
267 void compute(const TPointVec& points,
268 const TMatrix& projection,
269 double eps,
270 TDouble1VecVec& scores) override {
271 this->computeLocalReachabilityDistances(points, projection, eps);
272 this->computeLocalOutlierFactors(points, eps, scores);
273 }

275 std::string name() const override { return "lof"; }
276
277 void computeLocalReachabilityDistances(const TPointVec& points,
278 const TMatrix& projection,
279 double eps) {
280
281 // We bind minimum accumulators (by value) to each lambda (since
282 // one copy is then accessed by each thread) and take the minimum
283 // of these at the end.
284
285 using TMinAccumulator = common::CBasicStatistics::SMin<double>::TAccumulator;
286
287 auto results = core::parallel_for_each(
288 points.begin(), points.end(),
289 core::bindRetrievableState(
290 [this](TMinAccumulator& min, const POINT& point) mutable {
291 std::size_t i{point.annotation()};
292 TMeanAccumulator reachability_;
293 for (std::size_t j = 0; j < this->k(); ++j) {
294 const auto& neighbour = m_KDistances[this->kDistanceIndex(i, j)];
295 if (distance(neighbour) != UNSET_DISTANCE) {
296 reachability_.add(this->reachabilityDistance(neighbour));
297 }
298 }
299 double reachability{common::CBasicStatistics::mean(reachability_)};
300 if (reachability > 0.0) {
301 m_Lrd[i] = 1.0 / reachability;
302 min.add(reachability);
303 }
304 },
305 TMinAccumulator{}));
306
307 TMinAccumulator min;
308 for (const auto& result : results) {
309 min += result.s_FunctionState;
310 }
311 if (min.count() > 0) {
312 // Use twice the maximum "density" at any other point if there are
313 // k-fold duplicates. Note it is possible that all lookup points are
314 // duplicates, in which case we need to set their local reachability
315 // density in this loop. The overwritten densities are reset in setup.
316 for (std::size_t i = 0; i < m_EndAddresses; ++i) {
317 if (m_Lrd[i] == UNSET_DISTANCE) {
318 m_Lrd[i] = 2.0 / min[0];
319 }
320 }
321 }
322
323 if (eps <= 0.0 || this->computeFeatureInfluence() == false) {
324 return;
325 }
326
327 // Unfortunately, we need to look up nearest neighbours again or cache
328 // the distance for each coordinate perturbation for each neighbour
329 // which is prohibitive.
330
331 core::parallel_for_each(
332 points.begin(), points.end(),
333 [&, neighbours = TPointVec{} ](const POINT& point) mutable {
334 this->lookup().nearestNeighbours(this->k() + 1, point, neighbours);
335 if (neighbours.size() < 2) {
336 return;
337 }
338 std::size_t i{point.annotation()};
339 std::size_t a(point == neighbours[0] ? 1 : 0);
340 std::size_t b{std::min(this->k() + a - 1, neighbours.size() + a - 2)};
341 for (std::size_t k = 0; k < m_NumberInfluences; ++k) {
342 TMeanAccumulator reachability_;
343 for (std::size_t j = a; j <= b; ++j) {
344 reachability_.add(this->reachabilityDistance(
345 {neighbours[j].annotation(),
346 distanceAtEps(projection, k, eps, point, neighbours[j])}));
347 }
348 double reachability{common::CBasicStatistics::mean(reachability_)};
349 m_LrdAtEps[this->epsLrdIndex(i, k)] =
350 1.0 / std::max(reachability, min[0] / 2.0);
351 }
352 });
353 }
354
355 void computeLocalOutlierFactors(const TPointVec& points, double eps, TDouble1VecVec& scores) {
356
357 core::parallel_for_each(points.begin(), points.end(), [&](const POINT& point) mutable {
358
359 std::size_t i{point.annotation()};
360
361 TMeanAccumulator neighbourhoodLrd;
362 for (std::size_t j = 0; j < this->k(); ++j) {
363 const auto& neighbour = m_KDistances[this->kDistanceIndex(i, j)];
364 if (distance(neighbour) != UNSET_DISTANCE) {
365 neighbourhoodLrd.add(m_Lrd[index(neighbour)]);
366 }
367 }
368
369 scores[i].resize(eps > 0.0 && this->computeFeatureInfluence()
370 ? m_NumberInfluences + 1
371 : 1);
372 scores[i][0] = common::CBasicStatistics::mean(neighbourhoodLrd) / m_Lrd[i];
373
374 // We choose to ignore the impact of moving the point on its
375 // neighbours' local reachability distances when computing
376 // scores for each coordinate.
377 for (std::size_t j = 1; j < scores[i].size(); ++j) {
378 scores[i][j] = common::CBasicStatistics::mean(neighbourhoodLrd) /
379 m_LrdAtEps[this->epsLrdIndex(i, j - 1)];
380 }
381 });
382 }
383
384 std::size_t kDistanceIndex(std::size_t index, std::size_t neighbourIndex) const {
385 return index * this->k() + neighbourIndex;
386 }
387 std::size_t epsLrdIndex(std::size_t index, std::size_t coordinate) const {
388 return index * m_NumberInfluences + coordinate;
389 }
390
391 static std::size_t index(const TUInt32CoordinatePr& neighbour) {
392 return neighbour.first;
393 }
394 static double distance(const TUInt32CoordinatePr& neighbour) {
395 return neighbour.second;
396 }
397
398 double reachabilityDistance(const TUInt32CoordinatePr& neighbour) const {
399 return std::max(this->kdistance(index(neighbour)), distance(neighbour));
400 }
401 double kdistance(std::size_t i) const {
402 for (std::size_t j = this->k(); j > 0; --j) {
403 double dist{distance(m_KDistances[this->kDistanceIndex(i, j - 1)])};
404 if (dist != UNSET_DISTANCE) {
405 return dist;
406 }
407 }
408 return UNSET_DISTANCE;
409 }
410
411private:
412 std::size_t m_NumberInfluences;
413 std::size_t m_StartAddresses;
414 std::size_t m_EndAddresses;
415 // The k distances to the neighbouring points of each point are stored
416 // flattened: [neighbours of 0, neighbours of 1,...].
417 TUInt32CoordinatePrVec m_KDistances;
418 TCoordinateVec m_Lrd;
419 // The epsilon local reachability distances are stored flattened:
420 // [coordinates of 0, coordinates of 2, ...].
421 TCoordinateVec m_LrdAtEps;
422};

429template<typename POINT, typename NEAREST_NEIGHBOURS>
430class CLdof final : public CNearestNeighbourMethod<POINT, NEAREST_NEIGHBOURS> {
431public:
432 CLdof(bool computeFeatureInfluence,
433 std::size_t k,
434 TProgressCallback recordProgress,
435 NEAREST_NEIGHBOURS lookup = NEAREST_NEIGHBOURS())
436 : CNearestNeighbourMethod<POINT, NEAREST_NEIGHBOURS>{
437 computeFeatureInfluence, k, std::move(lookup), std::move(recordProgress)} {}
438
439private:
440 using TMatrix = typename common::SConformableMatrix<POINT>::Type;
441 using CNearestNeighbourMethod<POINT, NEAREST_NEIGHBOURS>::distanceAtEps;
442
443private:
444 void add(const POINT& point,
445 const TMatrix& projection,
446 double eps,
447 const std::vector<POINT>& neighbours,
448 TDouble1VecVec& scores) override {
449
450 std::size_t numberInfluences{common::las::columns(projection)};
451 auto& score = scores[point.annotation()];
452 score.assign(
453 eps > 0.0 && this->computeFeatureInfluence() ? numberInfluences + 1 : 1, 0.0);
454
455 if (neighbours.size() < 2) {
456 return;
457 }
458
459 auto ldof = [](const TMeanAccumulator& d, const TMeanAccumulator& D) {
460 return common::CBasicStatistics::mean(D) > 0.0
461 ? common::CBasicStatistics::mean(d) /
462 common::CBasicStatistics::mean(D)
463 : 0.0;
464 };
465
466 std::size_t a(point == neighbours[0] ? 1 : 0);
467 std::size_t b{std::min(this->k() + a - 1, neighbours.size() + a - 2)};
468
469 TMeanAccumulator d;
470 TMeanAccumulator D;
471 for (std::size_t i = a; i <= b; ++i) {
472 d.add(common::las::distance(point, neighbours[i]));
473 for (std::size_t j = 1; j < i; ++j) {
474 D.add(common::las::distance(neighbours[i], neighbours[j]));
475 }
476 }
477 score[0] = ldof(d, D);
478
479 for (std::size_t j = 1; j < score.size(); ++j) {
480 d = TMeanAccumulator{};
481 for (std::size_t i = a; i <= b; ++i) {
482 d.add(distanceAtEps(projection, j - 1, eps, point, neighbours[i]));
483 }
484 score[j] = ldof(d, D);
485 }
486 }
487
488 std::string name() const override { return "ldof"; }
489};
