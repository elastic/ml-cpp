64template<std::size_t N>
65class CRandomProjectionClusterer {
66public:
67 using TDoubleVec = std::vector<double>;

70public:
71 virtual ~CRandomProjectionClusterer() = default;

74 virtual bool initialise(std::size_t numberProjections, std::size_t dimension) {
75 m_Dimension = dimension;
76 if (this->generateProjections(numberProjections) == false) {
77 LOG_ERROR(<< "Failed to generate projections");
78 return false;
79 }
80 return true;

83protected:

89protected:
91 CPRNG::CXorShift1024Mult& rng() const { return m_Rng; }
92
94 const TVectorArrayVec& projections() const { return m_Projections; }
95
97 bool generateProjections(std::size_t b) {
98 m_Projections.clear();
99
100 if (b == 0) {
101 return true;
102 }
103
104 if (m_Dimension <= N) {
105 m_Projections.resize(1);
106 TVectorArray& projection = m_Projections[0];
107 for (std::size_t i = 0; i < N; ++i) {
108 projection[i].extend(m_Dimension, 0.0);
109 if (i < m_Dimension) {
110 projection[i](i) = 1.0;
111 }
112 }
113 return true;
114 }
115
116 m_Projections.resize(b);
117
118 TDoubleVec components;
119 CSampling::normalSample(m_Rng, 0.0, 1.0, 2 * b * N * m_Dimension, components);
120 auto projection = m_Projections.begin();
121 for (std::size_t i = 0; i < 2 * b && projection != m_Projections.end(); ++i) {
122 for (std::size_t j = 0; j < N; ++j) {
123 (*projection)[j].assign(&components[(i * N + j) * m_Dimension],
124 &components[(i * N + j + 1) * m_Dimension]);
125 }
126 if (COrthogonaliser::orthonormalBasis(*projection)) {
127 ++projection;
128 }
129 }
130
131 return projection == m_Projections.end();
132 }
133
136 bool extendProjections(std::size_t dimension) {
137 using TDoubleVecArray = std::array<TDoubleVec, N>;
138
139 if (dimension <= m_Dimension) {
140 return true;
141 } else if (dimension <= N) {
142 TVectorArray& projection = m_Projections[0];
143 for (std::size_t i = m_Dimension; i < dimension; ++i) {
144 projection[i](i) = 1.0;
145 }
146 return true;
147 }
148
149 std::size_t b = m_Projections.size();
150 std::size_t d = dimension - m_Dimension;
151 double alpha = static_cast<double>(m_Dimension) / static_cast<double>(dimension);
152 double beta = 1.0 - alpha;
153
154 TDoubleVecArray extension;
155 TDoubleVec components;
156 CSampling::normalSample(m_Rng, 0.0, 1.0, b * N * d, components);
157 for (std::size_t i = 0; i < b; ++i) {
158 for (std::size_t j = 0; j < N; ++j) {
159 extension[j].assign(&components[(i * N + j) * d],
160 &components[(i * N + j + 1) * d]);
161 }
162
163 if (COrthogonaliser::orthonormalBasis(extension) == false) {
164 LOG_ERROR(<< "Failed to construct basis");
165 return false;
166 }
167
168 for (std::size_t j = 0; j < N; ++j) {
169 scale(extension[j], beta);
170 TVector& projection = m_Projections[i][j];
171 projection *= alpha;
172 projection.reserve(dimension);
173 projection.extend(extension[j].begin(), extension[j].end());
174 }
175 }
176
177 return true;
178 }
179
180private:
182 void scale(TDoubleVec& x, double scale) {
183 for (std::size_t i = 0; i < x.size(); ++i) {
184 x[i] *= scale;
185 }
186 }
187
188private:
190 mutable CPRNG::CXorShift1024Mult m_Rng;
191
193 std::size_t m_Dimension;
194
196 TVectorArrayVec m_Projections;
197};

264 template<typename CLUSTERER>
265 void run(CLUSTERER clusterer, TSizeVecVec& result) const {
266 if (m_ProjectedData.empty()) {
267 return;
268 }
269
270 std::size_t b = m_ProjectedData.size();
271
272 // Filled in with the weights of the clusterings.
273 TDoubleVecVec W(b);
274 // Filled in with the sample means of the clusterings.
275 TVectorNx1VecVec M(b);
276 // Filled in with the SVDs of the sample covariances
277 // of the clusterings.
278 TSvdNxNVecVec C(b);
279 // Filled in with the sample points indices.
280 TSizeUSet I;
281
282 // Compute the projected clusterings and sampling.
283 this->clusterProjections(clusterer, W, M, C, I);
284
285 // Compute the sample neighbourhoods.
286 std::size_t h = I.size();
287 TSizeVecVec H(h);
288 this->neighbourhoods(I, H);
289
290 // Compute the cluster similarities.
291 TDoubleVecVec S(h);
292 this->similarities(W, M, C, H, S);
293
294 // Run agglomerative clustering and choose number of clusters.
295 this->clusterNeighbourhoods(S, H, result);
296 }

326 template<typename CLUSTERER>
327 void clusterProjections(CLUSTERER clusterer,
328 TDoubleVecVec& W,
329 TVectorNx1VecVec& M,
330 TSvdNxNVecVec& C,
331 TSizeUSet& I) const {
332 using TVectorNx1CRef = std::reference_wrapper<const TVectorNx1>;
333 using TVectorNx1CRefSizeUMap =
334 boost::unordered_map<TVectorNx1CRef, std::size_t, SHashVector, SVectorsEqual>;
335 using TClusterVec = typename CLUSTERER::TClusterVec;
336 using TSampleCovariancesNxN = CBasicStatistics::SSampleCovariances<TVectorNx1>;
337
338 std::size_t b = m_ProjectedData.size();
339 std::size_t n = m_ProjectedData[0].size();
340
341 // An index lookup for some projected points.
342 TVectorNx1CRefSizeUMap lookup(n);
343 // A placeholder for copy of i'th projected data.
344 TVectorNx1Vec P;
345 // Filled in with the probabilities of sampling the points (i,j)'th
346 // cluster.
347 TDoubleVec pij;
348 // Filled in with a mapping from the candidates for sampling to the
349 // actual points in the (i,j)'th cluster.
350 TSizeVec fij;
351 // Filled in with the samples of the (i,j)'th cluster.
352 TSizeVec sij;
353
354 for (std::size_t i = 0; i < b; ++i) {
355 LOG_TRACE(<< "projection " << i);
356 P = m_ProjectedData[i];
357
358 // Create a lookup of points to their indices.
359 lookup.clear();
360 lookup.rehash(P.size());
361 for (std::size_t j = 0; j < m_ProjectedData[i].size(); ++j) {
362 lookup[std::cref(m_ProjectedData[i][j])] = j;
363 }
364
365 // Cluster the i'th projection.
366 clusterer.setPoints(P);
367 clusterer.run();
368 const TClusterVec& clusters = clusterer.clusters();
369 double ni = static_cast<double>(clusters.size());
370 LOG_TRACE(<< "# clusters = " << ni);
371
372 for (std::size_t j = 0; j < clusters.size(); ++j) {
373 const TVectorNx1Vec& points = clusters[j].points();
374 LOG_TRACE(<< "# points = " << points.size());
375
376 // Compute the number of points to sample from this cluster.
377 std::size_t nij = points.size();
378 double wij = static_cast<double>(nij) / static_cast<double>(n);
379 std::size_t nsij =
380 static_cast<std::size_t>(std::max(m_Compression * wij * ni, 1.0));
381 LOG_TRACE(<< "wij = " << wij << ", nsij = " << nsij);
382
383 // Compute the cluster sample mean and covariance matrix.
384 TSampleCovariancesNxN covariances(N);
385 covariances.add(points);
386 TVectorNx1 mij = CBasicStatistics::mean(covariances);
387 TSvdNxN Cij(toDenseMatrix(CBasicStatistics::covariances(covariances)),
388 Eigen::ComputeFullU | Eigen::ComputeFullV);
389
390 // Compute the probability that a sample from the cluster
391 // is a given point in the cluster.
392 pij.clear();
393 fij.clear();
394 pij.reserve(nij);
395 fij.reserve(nij);
396 double pmax = std::numeric_limits<double>::lowest();
397 for (std::size_t k = 0; k < nij; ++k) {
398 std::size_t index = lookup[std::cref(points[k])];
399 if (I.count(index) == 0) {
400 TEigenVectorNx1 x = toDenseVector(points[k] - mij);
401 pij.push_back(-0.5 * x.transpose() * Cij.solve(x));
402 fij.push_back(index);
403 pmax = std::max(pmax, pij.back());
404 }
405 }
406
407 if (pij.size() > 0) {
408 double Zij = 0.0;
409 for (std::size_t k = 0; k < pij.size(); ++k) {
410 pij[k] = std::exp(pij[k] - pmax);
411 Zij += pij[k];
412 }
413 for (std::size_t k = 0; k < pij.size(); ++k) {
414 pij[k] /= Zij;
415 }
416 LOG_TRACE(<< "pij = " << pij);
417
418 // Sample the cluster.
419 CSampling::categoricalSampleWithoutReplacement(this->rng(),
420 pij, nsij, sij);
421 LOG_TRACE(<< "sij = " << sij);
422
423 // Save the relevant data for the i'th clustering.
424 for (std::size_t k = 0; k < nsij; ++k) {
425 I.insert(fij[sij[k]]);
426 }
427 }
428 W[i].push_back(wij);
429 M[i].push_back(mij);
430 C[i].push_back(Cij);
431 }
432 }
433 }

440 void neighbourhoods(const TSizeUSet& I, TSizeVecVec& H) const {
441 using TVectorSizeUMap = boost::unordered_map<TVector, std::size_t, SHashVector>;
442
443 LOG_TRACE(<< "I = " << I);
444 std::size_t b = m_ProjectedData.size();
445 std::size_t n = m_ProjectedData[0].size();
446
447 // Create a k-d tree of the sampled data points.
448 TVectorVec S;
449 S.reserve(I.size());
450 TVector concat(b * N);
451 for (auto i : I) {
452 for (std::size_t j = 0; j < b; ++j) {
453 for (std::size_t k = 0; k < N; ++k) {
454 concat(N * j + k) = m_ProjectedData[j][i](k);
455 }
456 }
457 LOG_TRACE(<< "concat = " << concat);
458 S.push_back(concat);
459 }
460 TVectorSizeUMap lookup(S.size());
461 for (std::size_t i = 0; i < S.size(); ++i) {
462 lookup[S[i]] = i;
463 }
464 CKdTree<TVector> samples;
465 samples.build(S);
466
467 // Compute the neighbourhoods.
468 for (std::size_t i = 0; i < n; ++i) {
469 for (std::size_t j = 0; j < b; ++j) {
470 for (std::size_t k = 0; k < N; ++k) {
471 concat(N * j + k) = m_ProjectedData[j][i](k);
472 }
473 }
474 const TVector* nn = samples.nearestNeighbour(concat);
475 if (!nn) {
476 LOG_ERROR(<< "No nearest neighbour of " << concat);
477 continue;
478 }
479 LOG_TRACE(<< "nn = " << *nn);
480 H[lookup[*nn]].push_back(i);
481 }
482 LOG_TRACE(<< "H = " << H);
483 }

494 void similarities(const TDoubleVecVec& W,
495 const TVectorNx1VecVec& M,
496 const TSvdNxNVecVec& C,
497 const TSizeVecVec& H,
498 TDoubleVecVec& S) const {
499 std::size_t b = m_ProjectedData.size();
500 std::size_t h = H.size();
501
502 TMeanAccumulatorVecVec S_(h);
503
504 TVectorVec Pi(h);
505 for (std::size_t i = 0; i < b; ++i) {
506 const TVectorNx1Vec& X = m_ProjectedData[i];
507 const TDoubleVec& Wi = W[i];
508 const TVectorNx1Vec& Mi = M[i];
509 const TSvdNxNVec& Ci = C[i];
510 LOG_TRACE(<< "W(i) = " << Wi);
511 LOG_TRACE(<< "M(i) = " << Mi);
512
513 std::size_t nci = Mi.size();
514 std::fill_n(Pi.begin(), h, TVector(nci));
515
516 // Compute the probability each neighbourhood is from
517 // a given cluster.
518 for (std::size_t c = 0; c < nci; ++c) {
519 double wic = std::log(Wi[c]) - 0.5 * this->logDeterminant(Ci[c]);
520 LOG_TRACE(<< " w(" << i << "," << c << ") = " << wic);
521 for (std::size_t j = 0; j < h; ++j) {
522 std::size_t hj = H[j].size();
523 Pi[j](c) = static_cast<double>(hj) * wic;
524 for (std::size_t k = 0; k < hj; ++k) {
525 TEigenVectorNx1 x = toDenseVector(X[H[j][k]] - Mi[c]);
526 Pi[j](c) -= 0.5 * x.transpose() * Ci[c].solve(x);
527 }
528 LOG_TRACE(<< " P(" << j << "," << c << ") = " << Pi[j](c));
529 }
530 }
531 for (std::size_t j = 0; j < h; ++j) {
532 double Pmax = *std::max_element(Pi[j].begin(), Pi[j].end());
533 double Z = 0.0;
534 for (std::size_t c = 0; c < nci; ++c) {
535 Pi[j](c) = std::exp(Pi[j](c) - Pmax);
536 Z += Pi[j](c);
537 }
538 for (std::size_t c = 0; c < nci; ++c) {
539 Pi[j](c) /= Z;
540 }
541 LOG_TRACE(<< " P(" << j << ") = " << Pi[j]);
542 }
543
544 // Compute the similarities.
545 for (std::size_t j = 0; j < h; ++j) {
546 S_[j].resize(j + 1);
547 for (std::size_t k = 0; k <= j; ++k) {
548 S_[j][k].add(-std::log(std::max(
549 Pi[j].inner(Pi[k]), boost::numeric::bounds<double>::smallest())));
550 }
551 }
552 }
553 for (std::size_t i = 0; i < S_.size(); ++i) {
554 S[i].reserve(S_[i].size());
555 for (std::size_t j = 0; j < S_[i].size(); ++j) {
556 S[i].push_back(CBasicStatistics::mean(S_[i][j]));
557 }
558 }
559 }

568 void clusterNeighbourhoods(TDoubleVecVec& S, const TSizeVecVec& H, TSizeVecVec& result) const {
569 using TNode = CAgglomerativeClusterer::CNode;
570 using TDoubleTuple = CNaturalBreaksClassifier::TDoubleTuple;
571 using TDoubleTupleVec = CNaturalBreaksClassifier::TDoubleTupleVec;
572
573 result.clear();
574
575 CAgglomerativeClusterer agglomerative;
576 agglomerative.initialize(S);
577 CAgglomerativeClusterer::TNodeVec tree;
578 agglomerative.run(CAgglomerativeClusterer::E_Average, tree);
579
580 TDoubleTupleVec heights;
581 heights.reserve(tree.size());
582 for (std::size_t i = 0; i < tree.size(); ++i) {
583 heights.push_back(TDoubleTuple());
584 heights.back().add(tree[i].height());
585 }
586 LOG_TRACE(<< "heights = " << heights);
587
588 TSizeVec splits;
589 if (CNaturalBreaksClassifier::naturalBreaks(
590 heights,
591 2, // Number splits
592 0, // Minimum cluster size
593 CNaturalBreaksClassifier::E_TargetDeviation, splits)) {
594 double height = CBasicStatistics::mean(heights[splits[0] - 1]);
595 LOG_TRACE(<< "split = " << splits << ", height = " << height);
596 const TNode& root = tree.back();
597 root.clusteringAt(height, result);
598 for (std::size_t i = 0; i < result.size(); ++i) {
599 TSizeVec& ri = result[i];
600 std::size_t n = ri.size();
601 for (std::size_t j = 0; j < n; ++j) {
602 ri.insert(ri.end(), H[ri[j]].begin(), H[ri[j]].end());
603 }
604 ri.erase(ri.begin(), ri.begin() + n);
605 }
606 } else {
607 LOG_ERROR(<< "Failed to cluster " << heights);
608 }
609 }

615 double logDeterminant(const TSvdNxN& svd) const {
616 double result = 0.0;
617 for (std::size_t i = 0u, rank = static_cast<std::size_t>(svd.rank());
618 i < rank; ++i) {
619 result += std::log(svd.singularValues()[i]);
620 }
621 return result;
622 }
