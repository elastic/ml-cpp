76 setupBinaryClassificationData(fieldNames, fieldValues, analyzer,
77 weights, regressors, targets);

85 template<typename T>
86 static void addPredictionTestData(TLossFunctionType type,
87 const TStrVec& fieldNames,
88 TStrVec fieldValues,
89 api::CDataFrameAnalyzer& analyzer,
90 std::vector<T>& expectedPredictions,
91 std::size_t numberExamples = 100,
92 double alpha = -1.0,
93 double lambda = -1.0,
94 double gamma = -1.0,
95 double softTreeDepthLimit = -1.0,
96 double softTreeDepthTolerance = -1.0,
97 double eta = 0.0,
98 std::size_t maximumNumberTrees = 0,
99 double downsampleFactor = 0.0,
100 double featureBagFraction = 0.0,
101 double lossFunctionParameter = 1.0,
102 TSizeOptional seed = {}) {
103
104 test::CRandomNumbers rng;
105 if (seed) {
106 rng.seed(*seed);
107 }
108
109 TDoubleVec weights;
110 rng.generateUniformSamples(-1.0, 1.0, fieldNames.size() - 3, weights);
111 TDoubleVec regressors;
112 rng.generateUniformSamples(-10.0, 10.0, weights.size() * numberExamples, regressors);
113
114 TStrVec targets;
115 auto frame = [&] {
116 switch (type) {
117 case TLossFunctionType::E_MseRegression:
118 case TLossFunctionType::E_HuberRegression:
119 return setupLinearRegressionData(fieldNames, fieldValues, analyzer,
120 weights, regressors, targets);
121 case TLossFunctionType::E_MsleRegression:
122 return setupLinearRegressionData(fieldNames, fieldValues, analyzer,
123 weights, regressors, targets,
124 [](double x) { return x * x; });
125 case TLossFunctionType::E_BinaryClassification:
126 return setupBinaryClassificationData(fieldNames, fieldValues, analyzer,
127 weights, regressors, targets);
128 case TLossFunctionType::E_MulticlassClassification:
129 // TODO
130 return TDataFrameUPtr{};
131 }
132 return TDataFrameUPtr{};
133 }();
134
135 TLossUPtr loss;
136 switch (type) {
137 case TLossFunctionType::E_MseRegression:
138 loss = std::make_unique<maths::analytics::boosted_tree::CMse>();
139 break;
140 case TLossFunctionType::E_MsleRegression:
141 loss = std::make_unique<maths::analytics::boosted_tree::CMsle>(lossFunctionParameter);
142 break;
143 case TLossFunctionType::E_HuberRegression:
144 loss = std::make_unique<maths::analytics::boosted_tree::CPseudoHuber>(lossFunctionParameter);
145 break;
146 case TLossFunctionType::E_BinaryClassification:
147 loss = std::make_unique<maths::analytics::boosted_tree::CBinomialLogisticLoss>();
148 break;
149 case TLossFunctionType::E_MulticlassClassification:
150 // TODO
151 loss = TLossUPtr{};
152 break;
153 }
154
155 maths::analytics::CBoostedTreeFactory treeFactory{
156 maths::analytics::CBoostedTreeFactory::constructFromParameters(1, std::move(loss))};
157 if (alpha >= 0.0) {
158 treeFactory.depthPenaltyMultiplier({alpha});
159 }
160 if (lambda >= 0.0) {
161 treeFactory.leafWeightPenaltyMultiplier({lambda});
162 }
163 if (gamma >= 0.0) {
164 treeFactory.treeSizePenaltyMultiplier({gamma});
165 }
166 if (softTreeDepthLimit >= 0.0) {
167 treeFactory.softTreeDepthLimit({softTreeDepthLimit});
168 }
169 if (softTreeDepthTolerance >= 0.0) {
170 treeFactory.softTreeDepthTolerance({softTreeDepthTolerance});
171 }
172 if (eta > 0.0) {
173 treeFactory.eta({eta});
174 }
175 if (maximumNumberTrees > 0) {
176 treeFactory.maximumNumberTrees(maximumNumberTrees);
177 }
178 if (downsampleFactor > 0.0) {
179 treeFactory.downsampleFactor({downsampleFactor});
180 }
181 if (featureBagFraction > 0.0) {
182 treeFactory.featureBagFraction({featureBagFraction});
183 }
184
185 ml::api::CDataFrameTrainBoostedTreeInstrumentation instrumentation(
186 "testJob", core::constants::BYTES_IN_GIGABYTES);
187 treeFactory.analysisInstrumentation(instrumentation);
188
189 auto tree = treeFactory.buildForTrain(*frame, weights.size());
190
191 tree->train();
192 tree->predict();
193
194 frame->readRows(1, [&](const TRowItr& beginRows, const TRowItr& endRows) {
195 for (auto row = beginRows; row != endRows; ++row) {
196 auto prediction = tree->adjustedPrediction(*row);
197 appendPrediction(*frame, weights.size(), prediction, expectedPredictions);
198 }
199 });
200 }
