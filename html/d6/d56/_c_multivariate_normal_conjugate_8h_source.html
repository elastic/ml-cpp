194 void addSamples(const TDouble10Vec1Vec& samples,
195 const TDouble10VecWeightsAry1Vec& weights) override {
196 if (samples.empty()) {
197 return;
198 }
199 if (!this->check(samples, weights)) {
200 return;
201 }
202
203 this->CMultivariatePrior::addSamples(samples, weights);
204
205 // Note that if either count weight or outlier weights are supplied
206 // the weight of the sample x(i) is interpreted as its count, so for example
207 // updating with {(x, 2)} is equivalent to updating with {x, x}.
208 //
209 // If the data are discrete then we approximate the discrete distribution
210 // by saying it is uniform on the intervals [n,n+1] for each integral n.
211 // This is like saying that the data are samples from:
212 // X' = X + Z
213 //
214 // where,
215 // {[Z]_i} are IID uniform in the interval [0,1].
216 //
217 // We care about the limiting behaviour of the filter, i.e. as the number
218 // of samples n -> inf. In this case, the law of large numbers give that:
219 // mean(x(i) + z(i))
220 // -> 1/n * Sum_i( x(i) ) + E[Z]
221 //
222 // and
223 // cov(x(i) + z(i))
224 // = Sum_i( (x(i) + z(i) - 1/n * Sum_i( x(i) + z(i) ))'
225 // (x(i) + z(i) - 1/n * Sum_i( x(i) + z(i) )) )
226 // -> Sum_i( (x(i) - 1/n * Sum_j( x(j) ) + z(i) - E[Z])'
227 // (x(i) - 1/n * Sum_j( x(j) ) + z(i) - E[Z]) )
228 // -> cov(x(i)) + n * E[(Z - E[Z])'(Z - E[Z])]
229 //
230 // Since Z is uniform on the interval [0,1]
231 // E[Z] = 1/2 1
232 // E[(Z - E[Z])^2] = 1/12 I
233
234 TPoint numberSamples(0.0);
235 TCovariance covariancePost(N);
236 for (std::size_t i = 0; i < samples.size(); ++i) {
237 TPoint x(samples[i]);
238 TPoint n(maths_t::countForUpdate(weights[i]));
239 TPoint varianceScale = TPoint(maths_t::seasonalVarianceScale(weights[i])) *
240 TPoint(maths_t::countVarianceScale(weights[i]));
241 numberSamples += n;
242 covariancePost.add(x, n / varianceScale);
243 }
244 TPoint scaledNumberSamples = covariancePost.s_Count;
245
246 if (this->isInteger()) {
247 covariancePost.s_Mean += TPoint(0.5);
248 covariancePost.s_Covariances += TPoint(1.0 / 12.0).asDiagonal();
249 }
250
251 if (m_WishartDegreesFreedom > 0.0) {
252 TPoint scale = TPoint(1.0) / m_GaussianPrecision;
253 TMatrix covariances = m_WishartScaleMatrix;
254 scaleCovariances(scale, covariances);
255 TCovariance covariancePrior = CBasicStatistics::covariancesAccumulator(
256 m_GaussianPrecision, m_GaussianMean, covariances);
257 covariancePost += covariancePrior;
258 }
259 m_GaussianMean = CBasicStatistics::mean(covariancePost);
260 m_GaussianPrecision += scaledNumberSamples;
261 m_WishartDegreesFreedom +=
262 this->smallest(numberSamples.template toVector<TDouble10Vec>());
263 m_WishartScaleMatrix = covariancePost.s_Covariances;
264 scaleCovariances(covariancePost.s_Count, m_WishartScaleMatrix);
265
266 // If the coefficient of variation of the data is too small we run
267 // in to numerical problems. We truncate the variation by modeling
268 // the impact of an actual variation (standard deviation divided by
269 // mean) in the data of size MINIMUM_COEFFICIENT_OF_VARATION on the
270 // prior parameters.
271
272 if (!this->isNonInformative()) {
273 double truncatedMean = max(m_GaussianMean, TPoint(1e-8)).euclidean();
274 double minimumDeviation = truncatedMean * MINIMUM_COEFFICIENT_OF_VARIATION;
275 double minimumDiagonal = m_WishartDegreesFreedom * minimumDeviation * minimumDeviation;
276 for (std::size_t i = 0; i < N; ++i) {
277 m_WishartScaleMatrix(i, i) = std::max(m_WishartScaleMatrix(i, i), minimumDiagonal);
278 }
279 }
280
281 LOG_TRACE(<< "numberSamples = " << numberSamples
282 << ", scaledNumberSamples = " << scaledNumberSamples
283 << ", m_WishartDegreesFreedom = " << m_WishartDegreesFreedom
284 << ", m_WishartScaleMatrix = " << m_WishartScaleMatrix
285 << ", m_GaussianMean = " << m_GaussianMean
286 << ", m_GaussianPrecision = " << m_GaussianPrecision);
287
288 if (this->isBad()) {
289 LOG_ERROR(<< "Update failed (" << this->debug() << ")"
290 << ", samples = " << samples << ", weights = " << weights);
291 this->setToNonInformative(this->offsetMargin(), this->decayRate());
292 }

296 void propagateForwardsByTime(double time) override {
297 if (!CMathsFuncs::isFinite(time) || time < 0.0) {
298 LOG_ERROR(<< "Bad propagation time " << time);
299 return;
300 }
301 if (this->isNonInformative()) {
302 // Nothing to be done.
303 return;
304 }
305
306 double alpha = std::exp(-this->scaledDecayRate() * time);
307
308 m_GaussianPrecision = alpha * m_GaussianPrecision +
309 (1.0 - alpha) * TPoint(NON_INFORMATIVE_PRECISION);
310
311 // The mean of the Wishart distribution is n V and the variance
312 // is [V]_ij = n ( V_ij^2 + V_ii * V_jj), note V is the inverse
313 // of teh scale matrix. We want to increase the variance while
314 // holding its mean constant s.t. in the limit t -> inf var -> inf.
315 // Choosing a factor f in the range [0, 1] and the update is as
316 // follows:
317 // n -> f * n
318 // V -> f * V
319 //
320 // Thus the mean is unchanged and variance is increased by 1 / f.
321
322 double factor = std::min((alpha * m_WishartDegreesFreedom +
323 (1.0 - alpha) * NON_INFORMATIVE_DEGREES_FREEDOM) /
324 m_WishartDegreesFreedom,
325 1.0);
326
327 m_WishartDegreesFreedom *= factor;
328 m_WishartScaleMatrix *= factor;
329
330 this->numberSamples(this->numberSamples() * alpha);
331
332 LOG_TRACE(<< "time = " << time << ", alpha = " << alpha
333 << ", m_WishartDegreesFreedom = " << m_WishartDegreesFreedom
334 << ", m_WishartScaleMatrix = " << m_WishartScaleMatrix
335 << ", m_GaussianMean = " << m_GaussianMean
336 << ", m_GaussianPrecision = " << m_GaussianPrecision
337 << ", numberSamples = " << this->numberSamples());
338 }

351 TUnivariatePriorPtrDoublePr univariate(const TSize10Vec& marginalize,
352 const TSizeDoublePr10Vec& condition) const override {
353 if (!this->check(marginalize, condition)) {
354 return {};
355 }
356
357 TSize10Vec i1;
358 this->remainingVariables(marginalize, condition, i1);
359 if (i1.size() != 1) {
360 LOG_ERROR(<< "Invalid variables for computing univariate distribution: "
361 << "marginalize '" << marginalize << "'"
362 << ", condition '" << condition << "'");
363 return {};
364 }
365
366 maths_t::EDataType dataType = this->dataType();
367 double decayRate = this->decayRate();
368 if (this->isNonInformative()) {
369 return {TUnivariatePriorPtr(CNormalMeanPrecConjugate::nonInformativePrior(dataType, decayRate)
370 .clone()),
371 0.0};
372 }
373
374 double p = m_GaussianPrecision(i1[0]);
375 double s = m_WishartDegreesFreedom / 2.0;
376 double v = m_WishartDegreesFreedom - static_cast<double>(N) - 1.0;
377 const TPoint& m = this->mean();
378 TMatrix c = m_WishartScaleMatrix / v;
379
380 double m1 = m(i1[0]);
381 double c11 = c(i1[0], i1[0]);
382 if (condition.empty()) {
383 return {std::make_unique<CNormalMeanPrecConjugate>(
384 dataType, m1, p, s, c11 * v / 2.0, decayRate),
385 0.0};
386 }
387
388 TSize10Vec condition_;
389 condition_.reserve(condition.size() + 1);
390 CDenseVector<double> xc(condition.size());
391 this->unpack(condition, condition_, xc);
392
393 try {
394 std::size_t n = condition_.size();
395 CDenseVector<double> m2 = projectedVector(condition_, m);
396 condition_.push_back(i1[0]);
397 CDenseMatrix<double> cp = projectedMatrix(condition_, c);
398 CDenseVector<double> c12 = cp.topRightCorner(n, 1);
399 auto c22 = cp.topLeftCorner(n, n).jacobiSvd(Eigen::ComputeThinU |
400 Eigen::ComputeThinV);
401 LOG_TRACE(<< "c22 = " << cp.topLeftCorner(n, n) << ", c12 = " << c12
402 << ", a = " << xc << ", m2 = " << m2);
403
404 CDenseVector<double> c22SolvexcMinusm2 = c22.solve(xc - m2);
405
406 double mean = m1 + c12.transpose() * c22SolvexcMinusm2;
407 double variance = std::max(c11 - c12.transpose() * c22.solve(c12),
408 MINIMUM_COEFFICIENT_OF_VARIATION * std::fabs(mean));
409 double weight = 0.5 * (std::log(variance) - (xc - m2).transpose() * c22SolvexcMinusm2);
410 LOG_TRACE(<< "mean = " << mean << ", variance = " << variance
411 << ", weight = " << weight);
412
413 return {std::make_unique<CNormalMeanPrecConjugate>(
414 dataType, mean, p, s, variance * v / 2.0, decayRate),
415 weight};
416 } catch (const std::exception& e) {
417 LOG_ERROR(<< "Failed to get univariate prior: " << e.what());
418 }
419
420 return {};
421 }

435 TPriorPtrDoublePr bivariate(const TSize10Vec& marginalize,
436 const TSizeDoublePr10Vec& condition) const override {
437 if (N == 2) {
438 return {TPriorPtr(this->clone()), 0.0};
439 }
440 if (!this->check(marginalize, condition)) {
441 return {};
442 }
443
444 TSize10Vec i1;
445 this->remainingVariables(marginalize, condition, i1);
446 if (i1.size() != 2) {
447 return {};
448 }
449
450 maths_t::EDataType dataType = this->dataType();
451 double decayRate = this->decayRate();
452 if (this->isNonInformative()) {
453 return {TPriorPtr(CMultivariateNormalConjugate<2>::nonInformativePrior(dataType, decayRate)
454 .clone()),
455 0.0};
456 }
457
458 using TPoint2 = CVectorNx1<double, 2>;
459 using TMatrix2 = CSymmetricMatrixNxN<double, 2>;
460
461 TPoint2 p;
462 p(0) = m_GaussianPrecision(i1[0]);
463 p(1) = m_GaussianPrecision(i1[1]);
464 double f = m_WishartDegreesFreedom;
465 const TPoint& m = this->mean();
466 const TMatrix& c = m_WishartScaleMatrix;
467
468 TPoint2 m1;
469 TMatrix2 c11;
470 for (std::size_t i = 0; i < 2; ++i) {
471 m1(i) = m(i1[i]);
472 for (std::size_t j = 0; j < 2; ++j) {
473 c11(i, j) = c(i1[i], i1[j]);
474 }
475 }
476 if (condition.empty()) {
477 return {std::make_unique<CMultivariateNormalConjugate<2>>(
478 dataType, m1, p, f, c11, decayRate),
479 0.0};
480 }
481
482 TSize10Vec condition_;
483 condition_.reserve(condition.size() + 1);
484 CDenseVector<double> xc(condition.size());
485 this->unpack(condition, condition_, xc);
486
487 try {
488 std::size_t n = condition_.size();
489 CDenseVector<double> m2 = projectedVector(condition_, m);
490 condition_.push_back(i1[0]);
491 condition_.push_back(i1[1]);
492 CDenseMatrix<double> cp = projectedMatrix(condition_, c);
493 CDenseVector<double> c12 = cp.topRightCorner(n, 1);
494 auto c22 = cp.topLeftCorner(n, n).jacobiSvd(Eigen::ComputeThinU |
495 Eigen::ComputeThinV);
496 LOG_TRACE(<< "c22 = " << cp.topLeftCorner(n, n) << ", c12 = " << c12
497 << ", a = " << xc << ", m2 = " << m2);
498
499 CDenseVector<double> c22SolvexcMinusm2 = c22.solve(xc - m2);
500
501 TPoint2 mean(fromDenseVector(toDynamicDenseVector(m1) +
502 c12.transpose() * c22SolvexcMinusm2));
503 TMatrix2 covariance(fromDenseMatrix(toDynamicDenseMatrix(c11) -
504 c12.transpose() * c22.solve(c12)));
505 double weight;
506 logDeterminant(covariance, weight, false);
507 weight -= 0.5 * (xc - m2).transpose() * c22SolvexcMinusm2;
508 LOG_TRACE(<< "mean = " << mean << ", covariance = " << covariance);
509
510 return {std::make_unique<CMultivariateNormalConjugate<2>>(
511 dataType, mean, p, f, covariance, decayRate),
512 weight};
513 } catch (const std::exception& e) {
514 LOG_ERROR(<< "Failed to get univariate prior: " << e.what());
515 }
516
517 return {};
518 }

552 maths_t::EFloatingPointErrorStatus
553 jointLogMarginalLikelihood(const TDouble10Vec1Vec& samples,
554 const TDouble10VecWeightsAry1Vec& weights,
555 double& result) const override {
556 result = 0.0;
557
558 if (samples.empty()) {
559 LOG_ERROR(<< "Can't compute likelihood for empty sample set");
560 return maths_t::E_FpFailed;
561 }
562 if (!this->check(samples, weights)) {
563 return maths_t::E_FpFailed;
564 }
565
566 result = std::numeric_limits<double>::lowest();
567
568 if (this->isNonInformative()) {
569 // The non-informative likelihood is improper and effectively
570 // zero everywhere. We use minus max double because
571 // log(0) = HUGE_VALUE, which causes problems for Windows.
572 // Calling code is notified when the calculation overflows
573 // and should avoid taking the exponential since this will
574 // underflow and pollute the floating point environment. This
575 // may cause issues for some library function implementations
576 // (see fe*exceptflag for more details).
577 return maths_t::E_FpOverflowed;
578 }
579
580 // We evaluate the integral over the latent variable by Monte-Carlo
581 // approximation.
582
583 maths_t::EFloatingPointErrorStatus status;
584
585 if (this->isInteger()) {
586 double logLikelihood;
587 status = this->jointLogMarginalLikelihood(samples, TPoint(0.5),
588 weights, logLikelihood);
589 if (status != maths_t::E_FpNoErrors) {
590 return status;
591 }
592
593 double sum = 0.0;
594 double n = 0.0;
595 double maxLogLikelihood = logLikelihood;
596
597 TDoubleVec z;
598 CSampling::uniformSample(0.0, 1.0, 3 * N, z);
599 for (std::size_t i = 0; i < z.size(); i += N) {
600 status = this->jointLogMarginalLikelihood(
601 samples, TPoint(&z[i], &z[i + N]), weights, logLikelihood);
602 if (status & maths_t::E_FpFailed) {
603 return maths_t::E_FpFailed;
604 }
605 if (status & maths_t::E_FpOverflowed) {
606 continue;
607 }
608
609 if (logLikelihood > maxLogLikelihood) {
610 sum *= std::exp(maxLogLikelihood - logLikelihood);
611 maxLogLikelihood = logLikelihood;
612 }
613 sum += std::exp(logLikelihood - maxLogLikelihood);
614 n += 1.0;
615 }
616
617 result = maxLogLikelihood + std::log(sum / n);
618 } else {
619 status = this->jointLogMarginalLikelihood(samples, TPoint(0.0), weights, result);
620 }
621
622 if (status & maths_t::E_FpFailed) {
623 LOG_ERROR(<< "Failed to compute log likelihood (" << this->debug() << ")");
624 LOG_ERROR(<< "samples = " << samples);
625 LOG_ERROR(<< "weights = " << weights);
626 } else if (status & maths_t::E_FpOverflowed) {
627 LOG_TRACE(<< "Log likelihood overflowed for (" << this->debug() << ")");
628 LOG_TRACE(<< "samples = " << samples);
629 LOG_TRACE(<< "weights = " << weights);
630 }
631 return status;
632 }

654 void sampleMarginalLikelihood(std::size_t numberSamples,
655 TDouble10Vec1Vec& samples) const override {
656 samples.clear();
657
658 if (numberSamples == 0 || this->numberSamples() == 0.0) {
659 return;
660 }
661
662 if (this->isNonInformative()) {
663 // We can't sample the marginal likelihood directly. This should
664 // only happen if we've had one sample so just return that sample.
665 samples.push_back(m_GaussianMean.template toVector<TDouble10Vec>());
666 return;
667 }
668
669 // We sample the moment matched Gaussian (to the marginal likelihood).
670 // Clearly, E_{m*,p*}[ X ] = m. To calculate the covariance matrix we
671 // note that:
672 // E_{m*,p*}[ (X - m)'(X - m) ]
673 // = E_{m*,p*}[ (X - m*)'(X - m*) - (m* - m)'(m* - m) ]
674 // = E_{p*}[ (1 + 1/t) (p* ^ -1) ]
675 // = (1 + 1/t) V ^ (-1) / (v - d - 1)
676 //
677 // In the last line we have used the fact that if X ~ W_d(V, n) and
678 // Y = X^(-1), then Y ~ W_d^(-1)(V^(-1), n), i.e. the inverse Wishart
679 // with the same degrees of freedom, but inverse scale matrix.
680 //
681 // See sampleGaussian for details on the sampling strategy.
682
683 double d = static_cast<double>(N);
684 double v = m_WishartDegreesFreedom - d - 1.0;
685 TPoint mean(m_GaussianMean);
686 TMatrix covariance(m_WishartScaleMatrix);
687 for (std::size_t i = 0; i < N; ++i) {
688 if (m_GaussianPrecision(i) > 0.0 && v > 0.0) {
689 scaleCovariances(i, (1.0 - 1.0 / m_GaussianPrecision(i)) / v, covariance);
690 }
691 }
692 TPointVec samples_;
693 sampleGaussian(numberSamples, mean, covariance, samples_);
694 samples.reserve(samples_.size());
695 for (const auto& sample : samples_) {
696 samples.push_back(sample.template toVector<TDouble10Vec>());
697 }
698 }

724 std::uint64_t checksum(std::uint64_t seed = 0) const override {

787 void randomSamplePrecisionMatrixPrior(std::size_t n, TMatrixVec& result) {
788 // The prior on the precision matrix is Wishart with matrix V equal
789 // to the inverse of the scale matrix and degrees freedom equal to
790 // degrees freedom minus the data dimension. To sample from the Wishart
791 // we use the Bartlett transformation. In particular, if X ~ W(V, n),
792 // V = LL^t (i.e. the Cholesky factorization), then X = L A A^t L^t
793 // with
794 // | c1 0 . . 0 |
795 // | n21 c2 . |
796 // A = | . . . |
797 // | . . 0 |
798 // | nd1 . . . cd |
799 //
800 // Here, ci^2 ~ chi2(n - i + 1) and nij ~ N(0, 1).
801
802 result.clear();
803
804 if (this->isNonInformative()) {
805 return;
806 }
807
808 double d = static_cast<double>(N);
809 double f = m_WishartDegreesFreedom - d - 1.0;
810 LOG_TRACE(<< "f = " << f);
811
812 auto precision =
813 toDenseMatrix(m_WishartScaleMatrix).jacobiSvd(Eigen::ComputeFullU | Eigen::ComputeFullV);
814
815 // Note we can extract the (non-zero vectors of the Cholesky
816 // factorization by noting that U = V^t and multiplying each
817 // column of U by the square root of the corresponding singular
818 // value).
819
820 std::size_t rank = static_cast<std::size_t>(precision.rank());
821
822 TDenseVector diag = precision.singularValues();
823 for (std::size_t i = 0; i < rank; ++i) {
824 diag(i) = 1.0 / std::sqrt(diag(i));
825 }
826 for (std::size_t i = rank; i < N; ++i) {
827 diag(i) = 0.0;
828 }
829 TDenseMatrix L = TDenseMatrix::Zero(N, N);
830 L.leftCols(rank) = precision.matrixU().leftCols(rank);
831 L = L * diag.asDiagonal();
832 LOG_TRACE(<< "L = " << L);
833
834 TDoubleVec chi;
835 TDoubleVec chii;
836 for (std::size_t i = 0; i < rank; ++i) {
837 chii.clear();
838 CSampling::chiSquaredSample(f - static_cast<double>(i), n, chii);
839 chi.insert(chi.end(), chii.begin(), chii.end());
840 }
841 TDoubleVec normal;
842 CSampling::normalSample(0.0, 1.0, n * rank * (rank - 1) / 2, normal);
843
844 TDenseMatrix A(N, N);
845 for (std::size_t s = 0; s < n; ++s) {
846 A.setZero();
847 for (std::size_t i = 0, k = 0; i < rank; ++i) {
848 A(i, i) = std::sqrt(chi[i * n + s]);
849 for (std::size_t j = 0; j < i; ++j, ++k) {
850 A(i, j) = normal[(s * rank * (rank - 1)) / 2 + k];
851 }
852 }
853
854 TDenseMatrix denseMatrix = L * A * A.transpose() * L.transpose();
855 TMatrix resultMatrix{fromDenseMatrix(denseMatrix)};
856 result.emplace_back(resultMatrix);
857 }
858 }

861 void randomSampleMeanPrior(std::size_t n, TPointVec& result) {
862 result.clear();
863
864 if (this->isNonInformative()) {
865 return;
866 }
867
868 // The marginal distribution of the prior is multivariate t with
869 // mean equal to the Gaussian prior mean and covariance matrix
870 // equal to 1 / (f - d + 1) * S where S and f are the scale matrix
871 // and degrees freedom of the Wishart prior and d is the data
872 // dimension. We use the fact that a multivariate t variable
873 // T(m, V, v) = m + (W)^(1/2) N(0, V), where m is its mean and
874 // W ~ v / chi^2(v).
875
876 double d = static_cast<double>(N);
877 double f = m_WishartDegreesFreedom - d + 1.0;
878 TPoint mean(m_GaussianMean);
879 TMatrix covariance = m_WishartScaleMatrix;
880 for (std::size_t i = 0; i < N; ++i) {
881 if (m_GaussianPrecision(i) > 0.0 && f > 0.0) {
882 scaleCovariances(i, 1.0 / (m_GaussianPrecision(i) * f), covariance);
883 }
884 }
885 LOG_TRACE(<< "mean = " << mean);
886 LOG_TRACE(<< "covariance = " << covariance);
887 LOG_TRACE(<< "fmin = " << f);
888
889 TDoubleVec chi;
890 CSampling::chiSquaredSample(f, n, chi);
891 TPoint zero(0.0);
892 CSampling::multivariateNormalSample(zero, covariance, n, result);
893
894 for (std::size_t i = 0; i < n; ++i) {
895 result[i] = mean + (f / chi[i]) * result[i];
896 }
897 }

900 void randomSamplePredictive(std::size_t n, TPointVec& result) {
901 result.clear();
902
903 if (this->isNonInformative()) {
904 return;
905 }
906
907 // The predictive distribution is multivariate t with mean, m,
908 // equal to the Gaussian prior mean, scale matrix, V, equal to
909 // the Wishart prior scale matrix, and degrees freedom, f, equal
910 // to its degrees of freedom. See randomSampleMeanPrior for how
911 // to sample from multivariate t.
912
913 double d = static_cast<double>(N);
914 double f = m_WishartDegreesFreedom - d;
915 TPoint mean(m_GaussianMean);
916
917 TDoubleVec chi;
918 CSampling::chiSquaredSample(f, n, chi);
919 TPoint zero(0.0);
920 CSampling::multivariateNormalSample(zero, m_WishartScaleMatrix, n, result);
921
922 for (std::size_t i = 0; i < n; ++i) {
923 result[i] = mean + (f / chi[i]) * result[i];
924 }
925 }

934 TMatrix covarianceMatrix() const {
935 // This can be found by change of variables from the prior on the
936 // precision matrix. In particular, if X ~ W_d(V, n) and Y = X^(-1),
937 // then Y ~ W_d^(-1)(V^(-1), n), i.e. the inverse Wishart with the
938 // same degrees of freedom, but inverse scale matrix.
939
940 double d = static_cast<double>(N);
941 double v = m_WishartDegreesFreedom - d - 1.0;
942 TMatrix covariance(m_WishartScaleMatrix);
943 for (std::size_t i = 0; i < N; ++i) {
944 if (m_GaussianPrecision(i) > 0.0 && v > 0.0) {
945 scaleCovariances(i, (1.0 - 1.0 / m_GaussianPrecision(i)) / v, covariance);
946 }
947 }
948 return covariance;
949 }

1030 maths_t::EFloatingPointErrorStatus
1031 jointLogMarginalLikelihood(const TDouble10Vec1Vec& samples,
1032 const TPoint& offset,
1033 const TDouble10VecWeightsAry1Vec& weights,
1034 double& result) const {
1035 // As usual, one can find the marginal likelihood by noting that
1036 // it is proportional to the ratio of the normalization factors
1037 // of the conjugate distribution before and after update with the
1038 // samples.
1039
1040 double numberSamples = 0.0;
1041 TCovariance covariancePost(N);
1042 double logCountVarianceScales = 0.0;
1043 TPoint m(this->marginalLikelihoodMean());
1044 for (std::size_t i = 0; i < samples.size(); ++i) {
1045 TPoint x(samples[i]);
1046 TPoint n(maths_t::countForUpdate(weights[i]));
1047 TPoint seasonalScale =
1048 sqrt(TPoint(maths_t::seasonalVarianceScale(weights[i])));
1049 TPoint countVarianceScale(maths_t::countVarianceScale(weights[i]));
1050 x = m + (x + offset - m) / seasonalScale;
1051 numberSamples += this->smallest(n.template toVector<TDouble10Vec>());
1052 covariancePost.add(x, n / countVarianceScale);
1053 for (std::size_t j = 0; j < N; ++j) {
1054 logCountVarianceScales -= 0.5 * std::log(countVarianceScale(j));
1055 }
1056 }
1057 TPoint scaledNumberSamples = covariancePost.s_Count;
1058 TCovariance covariancePrior = CBasicStatistics::covariancesAccumulator(
1059 TPoint(m_WishartDegreesFreedom), m_GaussianMean,
1060 m_WishartScaleMatrix / m_WishartDegreesFreedom);
1061 covariancePost += covariancePrior;
1062
1063 double logGaussianPrecisionPrior = 0.0;
1064 double logGaussianPrecisionPost = 0.0;
1065 for (std::size_t i = 0; i < N; ++i) {
1066 logGaussianPrecisionPrior += std::log(m_GaussianPrecision(i));
1067 logGaussianPrecisionPost +=
1068 std::log(m_GaussianPrecision(i) + scaledNumberSamples(i));
1069 }
1070 double wishartDegreesFreedomPrior = m_WishartDegreesFreedom;
1071 double wishartDegreesFreedomPost = m_WishartDegreesFreedom + numberSamples;
1072 TMatrix wishartScaleMatrixPost = covariancePost.s_Covariances;
1073 scaleCovariances(covariancePost.s_Count, wishartScaleMatrixPost);
1074 double logDeterminantPrior;
1075 if (logDeterminant(m_WishartScaleMatrix, logDeterminantPrior, false) &
1076 maths_t::E_FpFailed) {
1077 LOG_ERROR(<< "Failed to calculate log det " << m_WishartScaleMatrix);
1078 return maths_t::E_FpFailed;
1079 }
1080 double logDeterminantPost;
1081 if (logDeterminant(wishartScaleMatrixPost, logDeterminantPost) & maths_t::E_FpFailed) {
1082 LOG_ERROR(<< "Failed to calculate log det " << wishartScaleMatrixPost);
1083 return maths_t::E_FpFailed;
1084 }
1085 double logGammaPost = 0.0;
1086 double logGammaPrior = 0.0;
1087 double logGammaPostMinusPrior = 0.0;
1088
1089 for (std::size_t i = 0; i < N; ++i) {
1090 if (CTools::lgamma(0.5 * (wishartDegreesFreedomPost - static_cast<double>(i)),
1091 logGammaPost, true) &&
1092 CTools::lgamma(0.5 * (wishartDegreesFreedomPrior - static_cast<double>(i)),
1093 logGammaPrior, true)) {
1094
1095 logGammaPostMinusPrior += logGammaPost - logGammaPrior;
1096 } else {
1097 return maths_t::E_FpOverflowed;
1098 }
1099 }
1100 LOG_TRACE(<< "numberSamples = " << numberSamples);
1101 LOG_TRACE(<< "logGaussianPrecisionPrior = " << logGaussianPrecisionPrior
1102 << ", logGaussianPrecisionPost = " << logGaussianPrecisionPost);
1103 LOG_TRACE(<< "wishartDegreesFreedomPrior = " << wishartDegreesFreedomPrior
1104 << ", wishartDegreesFreedomPost = " << wishartDegreesFreedomPost);
1105 LOG_TRACE(<< "wishartScaleMatrixPrior = " << m_WishartScaleMatrix);
1106 LOG_TRACE(<< "wishartScaleMatrixPost = " << wishartScaleMatrixPost);
1107 LOG_TRACE(<< "logDeterminantPrior = " << logDeterminantPrior
1108 << ", logDeterminantPost = " << logDeterminantPost);
1109 LOG_TRACE(<< "logGammaPostMinusPrior = " << logGammaPostMinusPrior);
1110 LOG_TRACE(<< "logCountVarianceScales = " << logCountVarianceScales);
1111
1112 double d = static_cast<double>(N);
1113 result = 0.5 * (wishartDegreesFreedomPrior * logDeterminantPrior -
1114 wishartDegreesFreedomPost * logDeterminantPost -
1115 d * (logGaussianPrecisionPost - logGaussianPrecisionPrior) +
1116 (wishartDegreesFreedomPost - wishartDegreesFreedomPrior) *
1117 d * core::constants::LOG_TWO +
1118 2.0 * logGammaPostMinusPrior -
1119 numberSamples * d * core::constants::LOG_TWO_PI - logCountVarianceScales);
1120
1121 return static_cast<maths_t::EFloatingPointErrorStatus>(CMathsFuncs::fpStatus(result));
1122 }
