38#define LESS_OR_GREATER(l, r) \
39 if (l < r) { \
40 return true; \
41 } else if (r > l) { \
42 return false; \
43 }

46template<typename SCALAR, int FLAGS, typename STORAGE_INDEX>
47bool operator<(const SparseMatrix<SCALAR, FLAGS, STORAGE_INDEX>& lhs,
48 const SparseMatrix<SCALAR, FLAGS, STORAGE_INDEX>& rhs) {
49 LESS_OR_GREATER(lhs.rows(), rhs.rows())
50 LESS_OR_GREATER(lhs.cols(), rhs.cols())
51 for (STORAGE_INDEX i = 0; i < lhs.rows(); ++i) {
52 for (STORAGE_INDEX j = 0; j < lhs.cols(); ++j) {
53 LESS_OR_GREATER(lhs(i, j), rhs(i, j))
54 }
55 }
56 return false;
57}

60template<typename SCALAR, int FLAGS, typename STORAGE_INDEX>
61bool operator<(const SparseVector<SCALAR, FLAGS, STORAGE_INDEX>& lhs,
62 const SparseVector<SCALAR, FLAGS, STORAGE_INDEX>& rhs) {
63 LESS_OR_GREATER(lhs.size(), rhs.size())
64 for (STORAGE_INDEX i = 0; i < lhs.size(); ++i) {
65 LESS_OR_GREATER(lhs(i), rhs(i))
66 }
67 return false;
68}

71template<typename SCALAR, int ROWS, int COLS, int OPTIONS, int MAX_ROWS, int MAX_COLS>
72bool operator<(const Matrix<SCALAR, ROWS, COLS, OPTIONS, MAX_ROWS, MAX_COLS>& lhs,
73 const Matrix<SCALAR, ROWS, COLS, OPTIONS, MAX_ROWS, MAX_COLS>& rhs) {
74 LESS_OR_GREATER(lhs.rows(), rhs.rows())
75 LESS_OR_GREATER(lhs.cols(), rhs.cols())
76 return std::lexicographical_compare(lhs.data(), lhs.data() + lhs.size(),
77 rhs.data(), rhs.data() + rhs.size());
78}

145public:
146 using iterator_category = std::input_iterator_tag;
147 using value_type = std::ptrdiff_t;
148 using difference_type = std::ptrdiff_t;
149 using pointer = void;
150 using reference = void;
151
152public:
153 CSparseVectorIndexIterator(const CSparseVector<SCALAR, FLAGS>& vector, std::size_t index)
154 : m_Vector(&vector), m_Base(vector, index) {}
155
156 bool operator==(const CSparseVectorIndexIterator& rhs) const {
157 return m_Vector == rhs.m_Vector && m_Base.row() == rhs.m_Base.row() &&
158 m_Base.col() == rhs.m_Base.col();
159 }
160 bool operator!=(const CSparseVectorIndexIterator& rhs) const {
161 return (*this == rhs) == false;
162 }
163
164 std::ptrdiff_t operator*() const {
165 return std::max(m_Base.row(), m_Base.col());
166 }
167
168 CSparseVectorIndexIterator& operator++() {
169 ++m_Base;
170 return *this;
171 }
172 CSparseVectorIndexIterator operator++(int) {
173 CSparseVectorIndexIterator result(*this);
174 ++m_Base;
175 return result;
176 }
177
178private:
179 using TIterator = typename CSparseVector<SCALAR, FLAGS>::InnerIterator;
180
181private:
182 CSparseVector<SCALAR, FLAGS>* m_Vector;
183 TIterator m_Base;
184};
