114 bool checkInvariants(std::size_t dimension) const {
115 if (s_Parent) {
116 if (s_Parent->s_LeftChild != this && s_Parent->s_RightChild != this) {
117 LOG_ERROR(<< "Not parent's child");
118 return false;
119 }
120 }
121
122 std::size_t coordinate{this->depth() % dimension};
123 CCoordinateLess less(coordinate);
124 if (s_LeftChild && less(s_Point, s_LeftChild->s_Point)) {
125 LOG_ERROR(<< "parent = " << s_Point << ", left child = "
126 << s_LeftChild->s_Point << ", coordinate = " << coordinate);
127 return false;
128 }
129 if (s_RightChild && less(s_RightChild->s_Point, s_Point)) {
130 LOG_ERROR(<< "parent = " << s_Point << ", right child = "
131 << s_RightChild->s_Point << ", coordinate = " << coordinate);
132 return false;
133 }
134 return true;
135 }

139 std::size_t depth{0};
140 for (const SNode* ancestor = s_Parent; ancestor; ancestor = ancestor->s_Parent) {
141 ++depth;
142 }

287 void nearestNeighbours(std::size_t n,
288 const TPoint& point,
289 TPoint& distancesToHyperplanes,
290 TPointVec& result) const {
291
292 result.clear();
293
294 if (n > 0 && n < m_Nodes.size()) {
295 auto inf = std::numeric_limits<TCoordinatePrecise>::max();
296
297 // These neighbour points will be completely replaced by the call
298 // to nearestNeighbours, but we need the collection to be initialized
299 // with infinite distances so we get the correct value for the furthest
300 // nearest neighbour at the start of the branch and bound search.
301 COrderings::SLess less;
302 TCoordinatePrecisePointCRefPrVec neighbours(
303 n, {inf, std::cref(m_Nodes[0].s_Point)});
304 this->nearestNeighbours(point, less, m_Nodes[0], distancesToHyperplanes,
305 0 /*split coordinate*/, neighbours);
306
307 result.reserve(n);
308 std::sort_heap(neighbours.begin(), neighbours.end(), less);
309 for (const auto& neighbour : neighbours) {
310 result.push_back(neighbour.second);
311 }
312 } else if (n >= m_Nodes.size()) {
313 TDoubleVec distances;
314 distances.reserve(m_Nodes.size());
315 result.reserve(m_Nodes.size());
316 for (const auto& node : m_Nodes) {
317 distances.push_back(las::distance(point, node.s_Point));
318 result.push_back(node.s_Point);
319 }
320 COrderings::simultaneousSort(distances, result);
321 }
322 }

394 template<typename ITR, typename MOVE>
395 SNode* buildRecursively(SNode* parent, std::size_t coordinate, ITR begin, ITR end, MOVE move) {
396 std::size_t n{static_cast<std::size_t>(end - begin) / 2};
397 ITR median{begin + n};
398 std::nth_element(begin, median, end, CCoordinateLess(coordinate));
399 this->append(move, parent, *median);
400 SNode* node{&m_Nodes.back()};
401 if (median - begin > 0) {
402 std::size_t next{this->nextCoordinate(coordinate)};
403 SNode* leftChild{this->buildRecursively(node, next, begin, median, move)};
404 node->s_LeftChild = leftChild;
405 }
406 if (end - median > 1) {
407 std::size_t next{this->nextCoordinate(coordinate)};
408 SNode* rightChild{this->buildRecursively(node, next, median + 1, end, move)};
409 node->s_RightChild = rightChild;
410 }
411 return node;
412 }

415 const POINT* nearestNeighbour(const TPoint& point,
416 const SNode& node,
417 TPoint& distancesToHyperplanes,
418 std::size_t coordinate,
419 const POINT* nearest,
420 TCoordinatePrecise& distanceToNearest) const {
421
422 TCoordinatePrecise distance{las::distance(point, core::unwrap_ref(node.s_Point))};
423
424 if (distance < distanceToNearest ||
425 (distance == distanceToNearest && core::unwrap_ref(node.s_Point) < point)) {
426 distanceToNearest = distance;
427 nearest = &node.s_Point;
428 }
429
430 const SNode* primary{node.s_LeftChild};
431 const SNode* secondary{node.s_RightChild};
432
433 if (primary != nullptr && secondary != nullptr) {
434 TCoordinate distanceToHyperplane{
435 point(coordinate) - core::unwrap_ref(node.s_Point)(coordinate)};
436
437 if (distanceToHyperplane > TCoordinate{0}) {
438 std::swap(primary, secondary);
439 } else {
440 distanceToHyperplane = std::fabs(distanceToHyperplane);
441 }
442
443 std::size_t nextCoordinate{this->nextCoordinate(coordinate)};
444 nearest = this->nearestNeighbour(point, *primary, distancesToHyperplanes,
445 nextCoordinate, nearest, distanceToNearest);
446 std::swap(distancesToHyperplanes(coordinate), distanceToHyperplane);
447 if (las::norm(distancesToHyperplanes) < distanceToNearest) {
448 nearest = this->nearestNeighbour(point, *secondary,
449 distancesToHyperplanes, nextCoordinate,
450 nearest, distanceToNearest);
451 }
452 std::swap(distancesToHyperplanes(coordinate), distanceToHyperplane);
453 } else if (primary != nullptr) {
454 std::size_t nextCoordinate{this->nextCoordinate(coordinate)};
455 nearest = this->nearestNeighbour(point, *primary, distancesToHyperplanes,
456 nextCoordinate, nearest, distanceToNearest);
457 } else if (secondary != nullptr) {
458 std::size_t nextCoordinate{this->nextCoordinate(coordinate)};
459 nearest = this->nearestNeighbour(point, *secondary, distancesToHyperplanes,
460 nextCoordinate, nearest, distanceToNearest);
461 }
462
463 return nearest;
464 }

467 void nearestNeighbours(const TPoint& point,
468 const COrderings::SLess& less,
469 const SNode& node,
470 TPoint& distancesToHyperplanes,
471 std::size_t coordinate,
472 TCoordinatePrecisePointCRefPrVec& nearest) const {
473
474 TCoordinatePrecise distance{las::distance(point, core::unwrap_ref(node.s_Point))};
475
476 if (distance < nearest.front().first ||
477 (distance == nearest.front().first && core::unwrap_ref(node.s_Point) < point)) {
478 std::pop_heap(nearest.begin(), nearest.end(), less);
479 nearest.back().first = distance;
480 nearest.back().second = std::cref(core::unwrap_ref(node.s_Point));
481 std::push_heap(nearest.begin(), nearest.end(), less);
482 }
483
484 const SNode* primary{node.s_LeftChild};
485 const SNode* secondary{node.s_RightChild};
486
487 if (primary != nullptr && secondary != nullptr) {
488 TCoordinate distanceToHyperplane{
489 point(coordinate) - core::unwrap_ref(node.s_Point)(coordinate)};
490
491 if (distanceToHyperplane > TCoordinate{0}) {
492 std::swap(primary, secondary);
493 } else {
494 distanceToHyperplane = std::fabs(distanceToHyperplane);
495 }
496
497 std::size_t nextCoordinate{this->nextCoordinate(coordinate)};
498 this->nearestNeighbours(point, less, *primary, distancesToHyperplanes,
499 nextCoordinate, nearest);
500 std::swap(distancesToHyperplanes(coordinate), distanceToHyperplane);
501 if (las::norm(distancesToHyperplanes) < nearest.front().first) {
502 this->nearestNeighbours(point, less, *secondary, distancesToHyperplanes,
503 nextCoordinate, nearest);
504 }
505 std::swap(distancesToHyperplanes(coordinate), distanceToHyperplane);
506 } else if (primary != nullptr) {
507 std::size_t nextCoordinate{this->nextCoordinate(coordinate)};
508 this->nearestNeighbours(point, less, *primary, distancesToHyperplanes,
509 nextCoordinate, nearest);
510 } else if (secondary != nullptr) {
511 std::size_t nextCoordinate{this->nextCoordinate(coordinate)};
512 this->nearestNeighbours(point, less, *secondary, distancesToHyperplanes,
513 nextCoordinate, nearest);
514 }
