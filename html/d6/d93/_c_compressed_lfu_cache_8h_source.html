82 void resize(std::size_t maximumMemory) {
83
84 this->guardWrite(DONT_TIME_OUT, [&] {
85 m_MaximumMemory = maximumMemory;
86 this->computeMemoryParameters();
87
88 // Ensure the cache is consuming no more than the current memory limit.
89 while (this->memoryUsage() > m_MaximumMemory) {
90 auto itemToEvict = m_ItemStats.begin();
91 m_RemovedCount += itemToEvict->count();
92 this->removeFromCache(itemToEvict);
93 this->maybeReallocateCache();
94 }
95 });
96 }

104 //| Null values produced by \p computeValue are not written to the cache.
122 bool lookup(KEY key,
123 const TComputeValueCallback& computeValue,
124 const TReadValueCallback& readValue) {
125
126 auto compressedKey = m_CompressKey(m_Dictionary, key);
127
128 if (this->guardRead(TIME_OUT, [&] {
129 ++m_NumberLookups;
130 auto hit = this->hit(compressedKey);
131 if (hit != nullptr) {
132 ++m_NumberHits;
133 readValue(*hit, true);
134 return true;
135 }
136 return false;
137 })) {
138 if (this->guardWrite(TIME_OUT, [&] {
139 this->incrementCount(compressedKey);
140 }) == false) {
141 ++m_LostCount;
142 }
143 return true;
144 }
145
146 auto value = computeValue(std::move(key));
147 if (!value) {
148 return false;
149 }
150
151 std::size_t itemMemoryUsage{memory::dynamicSize(*value)};
152
153 if (this->guardWrite(TIME_OUT, [&] {
154 // It is possible that two values with the same key check the cache
155 // before either takes the write lock. So check if this is already
156 // in the cache before going any further.
157 if (m_ItemCache.find(compressedKey) != m_ItemCache.end()) {
158 readValue(*value, true);
159 this->incrementCount(compressedKey);
160 return;
161 }
162
163 // Update the item counts using the Space-Saving algorithm.
164 std::size_t count{1};
165 std::size_t lastEvictedCount{0};
166 while (this->full(itemMemoryUsage)) {
167 auto itemToEvict = m_ItemStats.begin();
168 // It's possible that the cache is empty yet isn't big
169 // enough to hold this new item.
170 if (itemToEvict == m_ItemStats.end()) {
171 readValue(*value, false);
172 return;
173 }
174 m_RemovedCount += lastEvictedCount;
175 lastEvictedCount = itemToEvict->count();
176 this->removeFromCache(itemToEvict);
177 }
178 readValue(this->insert(compressedKey, *value, itemMemoryUsage,
179 count + lastEvictedCount),
180 false);
181 }) == false) {
182 ++m_LostCount;
183 // guardWrite failed to acquire the lock but the potentially
184 // expensive work of the compute callback is complete.
185 // Return the computed value even if it has not been written
186 // to the cache.
187 readValue(*value, false);
188 }
189
190 return false;
191 }

253 bool checkInvariants() const {
254
255 bool result{true};
256
257 this->guardRead(DONT_TIME_OUT, [&] {
258 if (m_ItemCache.size() != m_ItemStats.size()) {
259 LOG_ERROR(<< "Size mismatch " << m_ItemCache.size() << " vs "
260 << m_ItemStats.size() << ".");
261 result = false;
262 }
263
264 std::size_t itemsMemoryUsage{0};
265 std::uint64_t totalCount{m_RemovedCount};
266
267 for (auto stats = m_ItemStats.begin(); stats != m_ItemStats.end(); ++stats) {
268 auto item = m_ItemCache.find(stats->key());
269 if (item == m_ItemCache.end()) {
270 LOG_ERROR(<< "Missing item '" << stats->key().print() << "'.");
271 result = false;
272 } else if (stats != item->second.stats()) {
273 LOG_ERROR(<< "Link mismatch for '" << stats->key().print() << "'.");
274 result = false;
275 }
276 itemsMemoryUsage += stats->memoryUsage();
277 totalCount += stats->count();
278 }
279
280 if (itemsMemoryUsage != m_ItemsMemoryUsage) {
281 LOG_ERROR(<< "Memory mismatch " << itemsMemoryUsage << " vs "
282 << m_ItemsMemoryUsage << ".");
283 result = false;
284 }
285 // We may be missing counts if a single addition caused multiple
286 // evictions, or if updates can time out
287 if (m_NumberLookups - m_LostCount != totalCount) {
288 LOG_ERROR(<< "Count mismatch " << m_NumberLookups.load() << " (less "
289 << m_LostCount.load() << " lost) vs " << totalCount << ".");
290 result = false;
291 }
292 return true;
293 });
294
295 return result;
296 }

327 m_ItemCache[stats->key()].stats(stats);
328 }
329
330 this->checkRestoredInvariants();
331
332 return true;

446 void maybeReallocateCache() {
447 auto shrunk = std::lower_bound(m_BucketCountSequence.begin(),
448 m_BucketCountSequence.end(),
449 m_ItemCache.bucket_count());
450 if (shrunk != m_BucketCountSequence.begin()) {
451 --shrunk;
452 double shrunkLoadFactor{static_cast<double>(m_ItemCache.bucket_count()) /
453 static_cast<double>(*shrunk) *
454 static_cast<double>(m_ItemCache.load_factor())};
455 if (shrunkLoadFactor < static_cast<double>(m_ItemCache.max_load_factor())) {
456 // The only way we can actually reclaim memory is to copy
457 // the contents of the cache into a new map.
458 TCompressedKeyCacheItemUMap copy{*shrunk - 1};
459 for (auto & [ compressedKey, value ] : m_ItemCache) {
460 copy.emplace(compressedKey, std::move(value));
461 }
462 m_ItemCache.swap(copy);
463 }
464 }

467 const VALUE* hit(const TCompressedKey& key) const {
468 auto item = m_ItemCache.find(key);
469 return item != m_ItemCache.end() ? &item->second.value() : nullptr;

472 void incrementCount(const TCompressedKey& key) {
473 auto item = m_ItemCache.find(key);
474 if (item != m_ItemCache.end()) {
475 auto node = m_ItemStats.extract(item->second.stats());
476 node.value().count(node.value().count() + 1);
477 item->second.stats(m_ItemStats.insert(std::move(node)));
478 }

481 void removeFromCache(TCacheItemStatsSetItr itemToEvict) {
482 m_ItemCache.erase(itemToEvict->key());
483 m_ItemsMemoryUsage -= itemToEvict->memoryUsage();
484 m_ItemStats.erase(itemToEvict);
485 }
486
487 const VALUE& insert(const TCompressedKey& key,
488 VALUE value,
489 std::size_t itemMemoryUsage,
490 std::size_t count) {
491 m_ItemsMemoryUsage += itemMemoryUsage;
492 auto item = m_ItemCache.emplace(key, std::move(value)).first;
493 auto stats = m_ItemStats.emplace_hint(m_ItemStats.begin(), key,
494 itemMemoryUsage, count);
495 item->second.stats(stats);
496 return item->second.value();
497 }

499 bool full(std::size_t itemMemoryUsage) const {
500 std::size_t memory{this->unguardedMemoryUsage() + itemMemoryUsage +
501 sizeof(typename TCompressedKeyCacheItemUMap::value_type) +
502 memory::storageNodeOverhead(m_ItemCache) +
503 sizeof(typename TCacheItemStatsSet::value_type) +
504 memory::storageNodeOverhead(m_ItemStats)};
505 if (m_ItemCache.bucket_count() > 0 && this->needToResizeItemCache()) {
506 memory += static_cast<std::size_t>(
507 (static_cast<double>(this->nextItemCacheBucketCount()) /
508 static_cast<double>(m_ItemCache.bucket_count()) -
509 1.0) *
510 static_cast<double>(memory::dynamicSize(m_ItemCache) -
511 m_ItemCache.size() *
512 sizeof(typename TCompressedKeyCacheItemUMap::value_type)));
513 }
514 return memory > m_MaximumMemory;
515 }
516
517 std::size_t unguardedMemoryUsage() const {
518 return m_ItemsMemoryUsage + // overheads
519 memory::dynamicSize(m_ItemCache) + memory::dynamicSize(m_ItemStats) +
520 memory::dynamicSize(m_BucketCountSequence);
521 }
522
523 bool needToResizeItemCache() const {
524 return static_cast<double>(m_ItemCache.load_factor()) +
525 1.0 / static_cast<double>(m_ItemCache.bucket_count()) >
526 static_cast<double>(m_ItemCache.max_load_factor());
527 }
528
529 std::size_t nextItemCacheBucketCount() const {
530 return *std::upper_bound(m_BucketCountSequence.begin(),
531 m_BucketCountSequence.end(), m_ItemCache.bucket_count());
532 }
533
534 void computeMemoryParameters() {
535 // We expect resizing to cycle through a fixed sequence of bucket counts.
536 // This deduces and caches that sequence.
537
538 typename TDictionary::template TWordTUMap<std::size_t> test;
539
540 m_BucketCountSequence.clear();
541 m_BucketCountSequence.push_back(test.bucket_count());
542
543 // Ignore heap memory allocated by VALUE since we only need a lower bound.
544 std::size_t memoryPerItem{2 * sizeof(TCompressedKey) +
545 sizeof(std::uint64_t) + sizeof(VALUE)};
546 for (std::size_t i = 0; memoryPerItem * i < m_MaximumMemory; ++i) {
547 auto translator = m_Dictionary.translator();
548 translator.add(i);
549 test.emplace(translator.word(), i);
550 if (test.bucket_count() != m_BucketCountSequence.back()) {
551 m_BucketCountSequence.push_back(test.bucket_count());
552 }
553 }
554
555 LOG_TRACE(<< "bucket sequence = " << m_BucketCountSequence);
556 }
557
558 void checkRestoredInvariants() const {
559 VIOLATES_INVARIANT(m_ItemCache.size(), !=, m_ItemStats.size());
560 for (const auto & [ key, item ] : m_ItemCache) {
561 VIOLATES_INVARIANT_NO_EVALUATION(item.stats(), ==, m_ItemStats.end());
562 VIOLATES_INVARIANT_NO_EVALUATION(item.stats(), ==, TCacheItemStatsSetItr{});
563 VIOLATES_INVARIANT_NO_EVALUATION(key, !=, item.stats()->key());
564 }
565 for (const auto& stats : m_ItemStats) {
566 VIOLATES_INVARIANT_NO_EVALUATION(m_ItemCache.find(stats.key()), ==,
567 m_ItemCache.end());
568 }
569 }

679private:
680 bool guardRead(bool timeOut, TReadFunc func) const override {
681 if (timeOut) {
682 std::shared_lock<std::shared_timed_mutex> lock{m_Mutex, m_MaximumWait};
683 return lock.owns_lock() && func();
684 }
685 std::shared_lock<std::shared_timed_mutex> lock{m_Mutex};
686 return func();
687 }
688
689 bool guardWrite(bool timeOut, TWriteFunc func) override {
690 if (timeOut) {
691 std::unique_lock<std::shared_timed_mutex> lock{m_Mutex, m_MaximumWait};
692 if (lock.owns_lock()) {
693 func();
694 return true;
695 }
696 return false;
697 }
698 std::unique_lock<std::shared_timed_mutex> lock{m_Mutex};
699 func();
700 return true;
701 }
702

705private:

707 mutable std::shared_timed_mutex m_Mutex;
