220 double logLikelihoodFromCluster(maths_t::EClusterWeightCalc calc,
221 const TPointPrecise& x) const {
222 double likelihood;
223 const TPointPrecise& mean = CBasicStatistics::mean(m_Covariances);
224 const TMatrixPrecise& covariances =
225 CBasicStatistics::maximumLikelihoodCovariances(m_Covariances);
226 maths_t::EFloatingPointErrorStatus status =
227 gaussianLogLikelihood(covariances, x - mean, likelihood, false);
228 if ((status & maths_t::E_FpFailed) != 0) {
229 LOG_ERROR(<< "Unable to compute likelihood for " << x
230 << " and cluster " << m_Index);
231 return core::constants::LOG_MIN_DOUBLE - 1.0;
232 }
233 if ((status & maths_t::E_FpOverflowed) != 0) {
234 return likelihood;
235 }
236 return likelihood + std::log(this->weight(calc));
237 }

251 TOptionalClusterClusterPr split(CPRNG::CXorOShiro128Plus& rng,
252 double minimumCount,
253 CClustererTypes::CIndexGenerator& indexGenerator) {
254 // We do our clustering top down to minimize space and avoid
255 // making splits before we are confident they exist. This is
256 // important for anomaly detection because we do *not* want
257 // to fit a cluster to an outlier and judge it to be not
258 // anomalous as a result.
259 //
260 // By analogy to x-means we choose a candidate split of the
261 // data by minimizing the total within class deviation of the
262 // two classes. In order to decide whether or not to split we
263 // 1) impose minimum count on the smaller cluster 2) use an
264 // information theoretic criterion. Specifically, we threshold
265 // the BIC gain of using the multi-mode distribution verses
266 // the single mode distribution.
267
268 LOG_TRACE(<< "split");
269
270 if (m_Structure.buffering()) {
271 return {};
272 }
273
274 std::size_t n = m_Structure.size();
275 if (n < 2) {
276 return {};
277 }
278
279 TSizeVecVec split;
280 if (this->splitSearch(rng, minimumCount, split) == false) {
281 return {};
282 }
283 LOG_TRACE(<< "split = " << split);
284
285 TCovariances covariances[]{TCovariances(N), TCovariances(N)};
286 TSphericalClusterVec clusters;
287 this->sphericalClusters(clusters);
288 for (std::size_t i = 0; i < 2; ++i) {
289 for (std::size_t j = 0; j < split[i].size(); ++j) {
290 covariances[i].add(clusters[split[i][j]]);
291 }
292 }
293 TKMeansOnlineVec structure;
294 m_Structure.split(split, structure);
295 LOG_TRACE(<< "Splitting cluster " << this->index() << " at "
296 << this->centre() << " left = " << structure[0].print()
297 << ", right = " << structure[1].print());
298
299 std::size_t index[]{indexGenerator.next(), indexGenerator.next()};
300 indexGenerator.recycle(m_Index);
301
302 return TOptionalClusterClusterPr{
303 std::in_place,
304 CCluster{index[0], m_DataType, m_DecayRate, covariances[0],
305 std::move(structure[0])},
306 CCluster{index[1], m_DataType, m_DecayRate, covariances[1],
307 std::move(structure[1])}};
308 }

383 bool splitSearch(CPRNG::CXorOShiro128Plus& rng, double minimumCount, TSizeVecVec& result) {
384 result.clear();
385
386 // The search visits a binary tree of candidate 2-splits of
387 // the data breadth first. If a suitable split is found on a
388 // level of the tree then the search terminates returning that
389 // split. Note that if a subset of the data can be split we
390 // also check that the corresponding full 2-split can be split
391 // subject to the same constraints (to avoid merging the two
392 // clusters straight away).
393
394 TSphericalClusterVec node;
395 this->sphericalClusters(node);
396 TSphericalClusterVec remainder;
397 remainder.reserve(node.size());
398 TSphericalClusterVecVec candidate;
399
400 for (;;) {
401 TKMeansOnline::kmeans(rng, node, 2, candidate);
402 LOG_TRACE(<< "candidate = " << candidate);
403
404 if (candidate.size() > 2) {
405 LOG_ERROR(<< "Expected 2-split: " << candidate);
406 break;
407 }
408 if (candidate[0].empty() || candidate[1].empty()) {
409 // This can happen if all the points are co-located,
410 // in which case we can't split this node anyway.
411 break;
412 }
413
414 // We use the Ledoit and Wolf optimal shrinkage estimate
415 // because the sample sizes here might be quite small in
416 // which case the variance of the covariance estimates can
417 // be large.
418
419 TCovariances covariances[]{TCovariances(N), TCovariances(N)};
420 CBasicStatistics::covariancesLedoitWolf(candidate[0], covariances[0]);
421 CBasicStatistics::covariancesLedoitWolf(candidate[1], covariances[1]);
422 double n[]{CBasicStatistics::count(covariances[0]),
423 CBasicStatistics::count(covariances[1])};
424 double nmin = std::min(n[0], n[1]);
425
426 // Check the count constraint.
427 bool satisfiesCount = (nmin >= minimumCount);
428 LOG_TRACE(<< "count = " << nmin << " (to split " << minimumCount << ")");
429
430 // Check the gain constraint.
431 double gain{BICGain(covariances[0], covariances[1])};
432 bool satisfiesGain{gain > MINIMUM_SPLIT_GAIN};
433 LOG_TRACE(<< "BIC(1) - BIC(2) = " << gain << " (to split "
434 << MINIMUM_SPLIT_GAIN << ")");
435
436 if (satisfiesCount == false) {
437 // Recurse to the (one) node with sufficient count.
438 if (n[0] > minimumCount && candidate[0].size() > 1) {
439 node.swap(candidate[0]);
440 remainder.insert(remainder.end(), candidate[1].begin(),
441 candidate[1].end());
442 continue;
443 }
444 if (n[1] > minimumCount && candidate[1].size() > 1) {
445 node.swap(candidate[1]);
446 remainder.insert(remainder.end(), candidate[0].begin(),
447 candidate[0].end());
448 continue;
449 }
450 } else if (satisfiesGain) {
451 LOG_TRACE(<< "Checking full split");
452
453 TSizeVec assignment(remainder.size());
454 for (std::size_t i = 0; i < remainder.size(); ++i) {
455 assignment[i] = nearest(remainder[i], covariances);
456 }
457 for (std::size_t i = 0; i < assignment.size(); ++i) {
458 std::size_t j = assignment[i];
459 TCovariances ci(N);
460 ci.add(remainder[i]);
461 candidate[j].push_back(remainder[i]);
462 covariances[j] += ci;
463 n[j] += CBasicStatistics::count(ci);
464 }
465
466 gain = BICGain(covariances[0], covariances[1]);
467 LOG_TRACE(<< "BIC(1) - BIC(2) = " << gain << " (to split "
468 << MINIMUM_SPLIT_GAIN << ")");
469
470 if (gain > MINIMUM_SPLIT_GAIN) {
471 LOG_TRACE(<< "splitting");
472
473 typename CSphericalCluster<TPoint>::SLess less;
474
475 result.resize(candidate.size());
476 TSphericalClusterVec clusters;
477 this->sphericalClusters(clusters);
478 TSizeVec indexes(clusters.size());
479 std::iota(indexes.begin(), indexes.end(), 0);
480 COrderings::simultaneousSortWith(
481 typename CSphericalCluster<TPoint>::SLess(), clusters, indexes);
482 for (std::size_t i = 0; i < candidate.size(); ++i) {
483 for (const auto& x : candidate[i]) {
484 std::size_t j = std::lower_bound(clusters.begin(),
485 clusters.end(), x, less) -
486 clusters.begin();
487 if (j >= clusters.size()) {
488 LOG_ERROR(<< "Missing " << x
489 << " from clusters = " << clusters);
490 return false;
491 }
492 result[i].push_back(indexes[j]);
493 }
494 }
495 }
496 }
497 break;
498 }
499
500 return result.size() > 0;
501 }

521 static std::size_t nearest(const TSphericalCluster& x,
522 const TCovariances (&c)[2]) {
523 TPrecise d[]{0, 0};
524 TPointPrecise x_(x);
525 inverseQuadraticForm(CBasicStatistics::maximumLikelihoodCovariances(c[0]),
526 x_ - CBasicStatistics::mean(c[0]), d[0]);
527 inverseQuadraticForm(CBasicStatistics::maximumLikelihoodCovariances(c[1]),
528 x_ - CBasicStatistics::mean(c[1]), d[1]);
529 return d[0] < d[1] ? 0 : 1;
530 }

535 static double BICGain(const TCovariances& lhs, const TCovariances& rhs) {
536 CGaussianInfoCriterion<TPoint, E_BIC> BIC1;
537 BIC1.add(lhs + rhs);
538 CGaussianInfoCriterion<TPoint, E_BIC> BIC2;
539 BIC2.add(lhs);
540 BIC2.add(rhs);
541 return BIC1.calculate() - BIC2.calculate();
542 }

710 bool remove(std::size_t index) override {
711
712 if (m_Clusters.size() <= 1) {
713 return false;
714 }
715
716 auto i = std::find_if(m_Clusters.begin(), m_Clusters.end(), [&](const auto& cluster) {
717 return cluster.index() == index;
718 });
719
720 if (i != m_Clusters.end()) {
721 CCluster& cluster = *i;
722 CCluster* nearest = this->nearest(cluster);
723 if (nearest != nullptr) {
724 LOG_TRACE(<< "Merging cluster " << cluster.index() << " at "
725 << cluster.centre() << " and cluster "
726 << nearest->index() << " at " << nearest->centre());
727 CCluster merge = nearest->merge(cluster, m_ClusterIndexGenerator);
728 (this->mergeFunc())(cluster.index(), nearest->index(), merge.index());
729 nearest->swap(merge);

785 void cluster(const TPointPrecise& point,
786 TSizeDoublePr2Vec& result,
787 double count = 1.0) const override {
788 result.clear();
789
790 if (m_Clusters.empty()) {
791 LOG_ERROR(<< "No clusters");
792 return;
793 }
794
795 // This does a soft assignment. Given we are finding a
796 // partitioning clustering (as a result of targeting
797 // the k-means objective) we only consider the case that
798 // the point comes from either the left or right cluster.
799 // A-priori the probability a randomly selected point
800 // comes from a cluster is proportional to its weight:
801 // P(i) = n(i) / Sum_j{ n(j) }
802 //
803 // Bayes theorem then immediately gives that the probability
804 // that a given point is from the i'th cluster
805 // P(i | x) = L(x | i) * P(i) / Z
806 //
807 // where Z is the normalization constant:
808 // Z = Sum_i{ P(i | x) }
809
810 result.reserve(m_Clusters.size());
811 double lmax = -std::numeric_limits<double>::max();
812 for (const auto& cluster : m_Clusters) {
813 double likelihood = cluster.logLikelihoodFromCluster(m_WeightCalc, point);
814 result.emplace_back(cluster.index(), likelihood);
815 lmax = std::max(lmax, likelihood);
816 }
817 double Z = 0.0;
818 for (auto& x : result) {
819 x.second = std::exp(x.second - lmax);
820 Z += x.second;
821 }
822 double pmax = 0.0;
823 for (auto& x : result) {
824 x.second /= Z;
825 pmax = std::max(pmax, x.second);
826 }
827 result.erase(std::remove_if(result.begin(), result.end(),
828 CProbabilityLessThan(HARD_ASSIGNMENT_THRESHOLD * pmax)),
829 result.end());
830 Z = 0.0;
831 for (const auto& x : result) {
832 Z += x.second;
833 }
834 for (auto& x : result) {
835 x.second *= count / Z;
836 }
837 }

841 void add(const TPointPrecise& x, TSizeDoublePr2Vec& clusters, double count = 1.0) override {
842 if (m_Clusters.size() == 1) {
843 LOG_TRACE(<< "Adding " << x << " to " << m_Clusters[0].centre());
844 m_Clusters[0].add(x, count);
845 clusters.emplace_back(m_Clusters[0].index(), count);
846 if (this->maybeSplit(m_Clusters.begin())) {
847 this->cluster(x, clusters, count);
848 }
849 } else {
850 using TDoubleSizePr = std::pair<double, std::size_t>;
851 using TMaxAccumulator = CBasicStatistics::SMax<TDoubleSizePr, 2>::TAccumulator;
852
853 TMaxAccumulator closest;
854 for (std::size_t i = 0; i < m_Clusters.size(); ++i) {
855 closest.add({m_Clusters[i].logLikelihoodFromCluster(m_WeightCalc, x), i});
856 }
857 closest.sort();
858 LOG_TRACE(<< "closest = " << closest);
859
860 double likelihood0 = closest[0].first;
861 double likelihood1 = closest[1].first;
862
863 // Normalize the likelihood values.
864 double p0 = 1.0;
865 double p1 = std::exp(likelihood1 - likelihood0);
866 double normalizer = p0 + p1;
867 p0 /= normalizer;
868 p1 /= normalizer;
869 LOG_TRACE(<< "probabilities = [" << p0 << "," << p1 << "]");
870
871 auto cluster0 = m_Clusters.begin() + closest[0].second;
872 auto cluster1 = m_Clusters.begin() + closest[1].second;
873
874 if (p1 < HARD_ASSIGNMENT_THRESHOLD * p0) {
875 LOG_TRACE(<< "Adding " << x << " to " << cluster0->centre());
876 cluster0->add(x, count);
877 clusters.emplace_back(cluster0->index(), count);
878 if (this->maybeSplit(cluster0) || this->maybeMerge(cluster0)) {
879 this->cluster(x, clusters, count);
880 }
881 } else {
882 // Get the weighted counts.
883 double count0 = count * p0;
884 double count1 = count * p1;
885 LOG_TRACE(<< "Soft adding " << x << " " << count0 << " to "
886 << cluster0->centre() << " and " << count1 << " to "
887 << cluster1->centre());
888
889 cluster0->add(x, count0);
890 cluster1->add(x, count1);
891 clusters.emplace_back(cluster0->index(), count0);
892 clusters.emplace_back(cluster1->index(), count1);
893 if (this->maybeSplit(cluster0) || this->maybeSplit(cluster1) ||
894 this->maybeMerge(cluster0) || this->maybeMerge(cluster1)) {
895 this->cluster(x, clusters, count);
896 }
897 }
898 }
899
900 if (this->prune()) {
901 this->cluster(x, clusters, count);
902 }

1109 }

1112 using TMinAccumulator =
1113 CBasicStatistics::COrderStatisticsStack<TDoubleSizePr, 1, COrderings::SFirstLess>;
1114
1115 bool result = false;
1116
1117 double minimumCount = this->minimumSplitCount() * CLUSTER_DELETE_FRACTION;
1118
1119 // Get the clusters to prune.
1120 for (;;) {
1121 TMinAccumulator prune;
1122 for (std::size_t i = 0; i < m_Clusters.size(); ++i) {
1123 if (m_Clusters[i].count() < minimumCount) {
1124 prune.add({m_Clusters[i].count(), i});
1125 }
1126 }
1127 if (prune.count() == 0) {
1128 break;
1129 }
1130 LOG_TRACE(<< "prune = " << prune);
1131
1132 result = true;
1133
1134 // Merge the clusters to prune in increasing count order.
1135 CCluster& cluster = m_Clusters[prune[0].second];
1136 CCluster* nearest = this->nearest(cluster);
1137 if (nearest != nullptr) {
1138 LOG_TRACE(<< "Merging cluster " << cluster.index() << " at "
1139 << cluster.centre() << " and cluster "
1140 << nearest->index() << " at " << nearest->centre());
1141 CCluster merge = nearest->merge(cluster, m_ClusterIndexGenerator);
1142 (this->mergeFunc())(cluster.index(), nearest->index(), merge.index());
1143 nearest->swap(merge);
1144 }
1145
1146 // Actually remove the pruned clusters.
1147 m_Clusters.erase(m_Clusters.begin() + prune[0].second);
1148 }
1149
1150 return result;
