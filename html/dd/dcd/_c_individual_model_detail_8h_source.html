24template<typename T>
25void CIndividualModel::currentBucketPersonIds(core_t::TTime time,
26 const T& featureData,
27 TSizeVec& result) const {
28 using TSizeUSet = boost::unordered_set<std::size_t>;
29
30 result.clear();
31
32 if (!this->bucketStatsAvailable(time)) {
33 LOG_ERROR(<< "No statistics at " << time
34 << ", current bucket = " << this->printCurrentBucket());
35 return;
36 }
37
38 TSizeUSet people;
39 for (const auto& feature : featureData) {
40 for (const auto& data : feature.second) {
41 people.insert(data.first);
42 }
43 }
44 result.reserve(people.size());
45 result.assign(people.begin(), people.end());
46}

48template<typename T>
49const T* CIndividualModel::featureData(
50 model_t::EFeature feature,
51 std::size_t pid,
52 core_t::TTime time,
53 const std::vector<std::pair<model_t::EFeature, std::vector<std::pair<std::size_t, T>>>>& featureData) const {
54 if (!this->bucketStatsAvailable(time)) {
55 LOG_ERROR(<< "No statistics at " << time
56 << ", current bucket = " << this->printCurrentBucket());
57 return nullptr;
58 }
59
60 auto i = std::lower_bound(featureData.begin(), featureData.end(), feature,
61 maths::common::COrderings::SFirstLess());
62 if (i == featureData.end() || i->first != feature) {
63 LOG_ERROR(<< "No data for feature " << model_t::print(feature));
64 return nullptr;
65 }
66
67 auto j = std::lower_bound(i->second.begin(), i->second.end(), pid,
68 maths::common::COrderings::SFirstLess());
69 return (j != i->second.end() && j->first == pid) ? &j->second : nullptr;
70}

72template<typename T, typename FILTER>
73void CIndividualModel::sampleBucketStatistics(core_t::TTime startTime,
74 core_t::TTime endTime,
75 const FILTER& filter,
76 std::vector<std::pair<model_t::EFeature, T>>& featureData,
77 CResourceMonitor& resourceMonitor) {
78 CDataGatherer& gatherer = this->dataGatherer();
79
80 if (!gatherer.dataAvailable(startTime)) {
81 return;
82 }
83
84 for (core_t::TTime time = startTime, bucketLength = gatherer.bucketLength();
85 time < endTime; time += bucketLength) {
86 this->CIndividualModel::sampleBucketStatistics(time, time + bucketLength, resourceMonitor);
87
88 gatherer.featureData(time, bucketLength, featureData);
89 for (auto& feature_ : featureData) {
90 T& data = feature_.second;
91 LOG_TRACE(<< model_t::print(feature_.first) << " data = " << data);
92 this->applyFilter(model_t::E_XF_By, false, filter, data);
93 }
94 }
95}

97template<typename PARAMS, typename INFLUENCES>
98bool CIndividualModel::addProbabilityAndInfluences(std::size_t pid,
99 PARAMS& params,
100 const INFLUENCES& influences,
101 CProbabilityAndInfluenceCalculator& pJoint,
102 CAnnotatedProbabilityBuilder& builder) const {
103 if (!pJoint.addAttributeProbability(EMPTY_STRING, model_t::INDIVIDUAL_ANALYSIS_ATTRIBUTE_ID,
104 params, builder)) {
105 LOG_ERROR(<< "Failed to compute P(" << params.describe()
106 << ", person = " << this->personName(pid) << ")");
107 return false;
108 } else {
109 LOG_TRACE(<< "P(" << params.describe() << ", person = " << this->personName(pid)
110 << ") = " << params.s_Probability);
111 }
112
113 if (!influences.empty()) {
114 const CDataGatherer& gatherer = this->dataGatherer();
115 for (std::size_t j = 0; j < influences.size(); ++j) {
116 if (const CInfluenceCalculator* influenceCalculator =
117 this->influenceCalculator(params.s_Feature, j)) {
118 pJoint.plugin(*influenceCalculator);
119 pJoint.addInfluences(*(gatherer.beginInfluencers() + j),
120 influences[j], params);
121 }
122 }
123 }
124 return true;
125}
126}
