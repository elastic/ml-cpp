196template<typename VECTOR>
197typename SCoordinate<VECTOR>::Type distance(const VECTOR& x, const VECTOR& y) {
198 using TCoordinate = typename SPromoted<typename SCoordinate<VECTOR>::Type>::Type;
199 TCoordinate result(0);
200 for (std::size_t i = 0; i < dimension(x); ++i) {
201 TCoordinate x_(y(i) - x(i));
202 result += x_ * x_;
203 }
204 return std::sqrt(result);
205}

382template<typename VECTOR>
383struct SEstimateVectorMemoryUsage {
384 static std::size_t value(std::size_t) { return 0; }
385};
386template<typename T>
387struct SEstimateVectorMemoryUsage<CVector<T>> {
388 static std::size_t value(std::size_t dimension) {
389 return dimension * sizeof(T);
390 }
391};
392template<typename SCALAR>
393struct SEstimateVectorMemoryUsage<CDenseVector<SCALAR>> {
394 static std::size_t value(std::size_t dimension) {
395 // Ignore pad for alignment.
396 return dimension * sizeof(SCALAR);
397 }
398};
399template<typename VECTOR, typename ANNOTATION>
400struct SEstimateVectorMemoryUsage<CAnnotatedVector<VECTOR, ANNOTATION>> {
401 static std::size_t value(std::size_t dimension) {
402 // Ignore any dynamic memory used by the annotation: we don't know how to
403 // compute this here. It will be up to the calling code to estimate this
404 // correctly.
405 return SEstimateVectorMemoryUsage<VECTOR>::value(dimension);
406 }
407};

409template<typename MATRIX>
410struct SEstimateMatrixMemoryUsage {
411 static std::size_t value(std::size_t, std::size_t) { return 0; }
412};
413template<typename T>
414struct SEstimateMatrixMemoryUsage<CSymmetricMatrix<T>> {
415 static std::size_t value(std::size_t rows, std::size_t) {
416 return sizeof(T) * rows * (rows + 1) / 2;
417 }
418};
419template<typename SCALAR>
420struct SEstimateMatrixMemoryUsage<CDenseMatrix<SCALAR>> {
421 static std::size_t value(std::size_t rows, std::size_t columns) {
422 // Ignore pad for alignment.
423 return sizeof(SCALAR) * rows * columns;
424 }
425};
