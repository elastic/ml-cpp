338 template<typename F, typename T>
339 static void remove(const TSizeVec& toRemove, const F& extractId, CBucketQueue<T>& queue) {
340 for (auto bucketItr = queue.begin(); bucketItr != queue.end(); ++bucketItr) {
341 T& bucket = *bucketItr;
342 for (auto i = bucket.begin(); i != bucket.end(); ) {
343 if (std::binary_search(toRemove.begin(), toRemove.end(), extractId(*i))) {
344 i = bucket.erase(i);
345 } else {
346 ++i;
347 }
348 }
349 }
350 }

357 template<typename F, typename T>
358 static void remove(const TSizeVec& toRemove,
359 const F& extractId,
360 CBucketQueue<std::vector<T>>& queue) {
361 for (auto bucketItr = queue.begin(); bucketItr != queue.end(); ++bucketItr) {
362 for (std::size_t i = 0; i < bucketItr->size(); ++i) {
363 T& bucket = (*bucketItr)[i];
364 for (auto j = bucket.begin(); j != bucket.end(); ) {
365 if (std::binary_search(toRemove.begin(), toRemove.end(),
366 extractId(j->first))) {
367 j = bucket.erase(j);
368 } else {
369 ++j;
370 }
371 }
372 }
373 }
