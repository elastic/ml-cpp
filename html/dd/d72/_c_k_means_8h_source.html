405 class CCentroidComputer {
406 public:
407 CCentroidComputer(const TPointVec& centres, TMeanAccumulatorVec& centroids)
408 : m_Centres(centres), m_Centroids(&centroids) {}
409
413 bool operator()(const TNode& node) {
414 m_Centres.prune(node.boundingBox());
415 const TSizeVec& filter = m_Centres.filter();
416 if (filter.size() == 1) {
417 if (node.centroid()) {
418 (*m_Centroids)[filter[0]] += *node.centroid();
419 }
420 return false;
421 } else {
422 const TPointVec& centres = m_Centres.centres();
423 const POINT& point = node.s_Point;
424 (*m_Centroids)[closest(centres, filter, point)].add(point);
425 }
426 return true;
427 }
428
429 private:
431 CCentreFilter m_Centres;
432
434 TMeanAccumulatorVec* m_Centroids;

444 class CClosestPointsCollector {
445 public:
446 CClosestPointsCollector(std::size_t numberPoints,
447 const TPointVec& centres,
448 TPointVecVec& closestPoints)
449 : m_Centres(&centres), m_ClosestPoints(&closestPoints) {
450 m_ClosestPoints->resize(centres.size());
451 for (std::size_t i = 0; i < m_ClosestPoints->size(); ++i) {
452 (*m_ClosestPoints)[i].clear();
453 (*m_ClosestPoints)[i].reserve(numberPoints / m_ClosestPoints->size() + 1);
454 }
455 }
456
459 void operator()(const TNode& node) {
460 std::size_t n = m_Centres->size();
461 const POINT& point = node.s_Point;
462 (*m_ClosestPoints)[closest(*m_Centres, boost::counting_iterator<std::size_t>(0),
463 boost::counting_iterator<std::size_t>(n), point)]
464 .push_back(point);
465 }
466
467 private:
468 const TPointVec* m_Centres;
469 TPointVecVec* m_ClosestPoints;
470 };
471
472private:
474 bool updateCentres(TMeanAccumulatorVec& newCentres) {
475 const TCoordinate precision{TCoordinate(5) *
476 std::numeric_limits<TCoordinate>::epsilon()};
477 newCentres.assign(m_Centres.size(), TMeanAccumulator(las::zero(m_Centres[0])));
478 CCentroidComputer computer(m_Centres, newCentres);
479 m_Points.preorderDepthFirst(computer);
480 bool changed = false;
481 POINT newCentre;
482 for (std::size_t i = 0; i < newCentres.size(); ++i) {
483 newCentre = CBasicStatistics::mean(newCentres[i]);
484 if (las::distance(m_Centres[i], newCentre) >
485 precision * las::norm(m_Centres[i])) {
486 using std::swap;
487 swap(m_Centres[i], newCentre);
488 changed = true;
489 }
490 }
491 return changed;
492 }

495 template<typename ITR>
496 static std::size_t
497 closest(const TPointVec& centres, ITR filter, ITR end, const POINT& point) {
498 std::size_t result = *filter;
499 double d = las::distance(point, centres[result]);
500 for (++filter; filter != end; ++filter) {
501 double di = las::distance(point, centres[*filter]);
502 if (di < d) {
503 result = *filter;
504 d = di;
505 }
506 }
507 return result;
508 }
509
511 static std::size_t
512 closest(const TPointVec& centres, const TSizeVec& filter, const POINT& point) {
513 return closest(centres, filter.begin(), filter.end(), point);
514 }

529template<typename POINT, typename RNG>
530class CKMeansPlusPlusInitialization : private core::CNonCopyable {
531public:
532 using TDoubleVec = std::vector<double>;
533 using TSizeVec = std::vector<std::size_t>;
534 using TPointVec = std::vector<POINT>;
535
536public:
537 explicit CKMeansPlusPlusInitialization(RNG& rng) : m_Rng(rng) {}
538
542 void run(const TPointVec& points, std::size_t k, TPointVec& result) const {
543 run(points.begin(), points.end(), k, result);
544 }
545
552 template<typename ITR>
553 void run(ITR beginPoints, ITR endPoints, std::size_t k, TPointVec& result) const {
554
555 result.clear();
556 if (beginPoints == endPoints || k == 0) {
557 return;
558 }
559
560 std::size_t n(std::distance(beginPoints, endPoints));
561 LOG_TRACE(<< "# points = " << n);
562
563 result.reserve(k);
564 result.push_back(beginPoints[CSampling::uniformSample(m_Rng, std::size_t(0), n)]);
565
566 m_Distances.assign(n, std::numeric_limits<double>::max());
567 for (std::size_t i = 1; i < k; ++i) {
568 this->updateDistances(result.back(), beginPoints, endPoints);
569 m_Probabilities.assign(m_Distances.begin(), m_Distances.end());
570 result.push_back(beginPoints[CSampling::categoricalSample(m_Rng, m_Probabilities)]);
571 }
572 LOG_TRACE(<< "selected = " << result);
573 }
574
575private:
576 template<typename ITR>
577 void updateDistances(const POINT& selected, ITR beginPoints, ITR endPoints) const {
578 std::size_t j{0};
579 for (ITR point = beginPoints; point != endPoints; ++j, ++point) {
580 m_Distances[j] = std::min(
581 m_Distances[j], CTools::pow2(las::distance(*point, selected)));
582 }
583 }
584
585private:
586 RNG& m_Rng;
587 mutable TDoubleVec m_Distances;
588 mutable TDoubleVec m_Probabilities;
589};
