104template<typename T, unsigned int ORDER>
105bool CBasicStatistics::SSampleCentralMoments<T, ORDER>::fromDelimited(const std::string& str) {
106 if (str.empty()) {
107 LOG_ERROR(<< "Empty accumulator representation");
108 return false;
109 }
110
111 // Reuse this same string to avoid as many memory allocations as
112 // possible. The reservation is 15 because:
113 // a) For string implementations using the short string
114 // optimisation we don't want to cause an unnecessary
115 // allocation
116 // b) The count comes first and it's likely to be shorter than
117 // the subsequent moments
118 std::string token;
119 token.reserve(15);
120 std::size_t delimPos{str.find(INTERNAL_DELIMITER, 0)};
121 if (delimPos == std::string::npos) {
122 token.assign(str, 0, str.length());
123 } else {
124 token.assign(str, 0, delimPos);
125 }
126
127 if (!basic_statistics_detail::stringToType(token, s_Count)) {
128 LOG_ERROR(<< "Invalid count : element " << token << " in " << str);
129 return false;
130 }
131
132 std::size_t lastDelimPos{delimPos};
133 std::size_t index{0};
134 while (lastDelimPos != std::string::npos && index < ORDER) {
135 delimPos = str.find(INTERNAL_DELIMITER, lastDelimPos + 1);
136 if (delimPos == std::string::npos) {
137 token.assign(str, lastDelimPos + 1, str.length() - lastDelimPos);
138 } else {
139 token.assign(str, lastDelimPos + 1, delimPos - lastDelimPos - 1);
140 }
141
142 if (!basic_statistics_detail::stringToType(token, s_Moments[index++])) {
143 LOG_ERROR(<< "Invalid moment " << index << " : element " << token
144 << " in " << str);
145 return false;
146 }
147
148 lastDelimPos = delimPos;
149 }
150
151 return true;
152}

177template<typename POINT>
178bool CBasicStatistics::SSampleCovariances<POINT>::fromDelimited(std::string str) {
179 std::size_t dimension{0u};
180 std::size_t pos{str.find_first_of(CLinearAlgebra::DELIMITER)};
181 if (!core::CStringUtils::stringToType(str.substr(0, pos), dimension)) {
182 LOG_ERROR(<< "Failed to extract dimension from " << str.substr(0, pos));
183 return false;
184 }
185 str = str.substr(pos + 1);
186
187 std::size_t count{0};
188 for (std::size_t i = 0; i < dimension; ++i) {
189 count = str.find_first_of(CLinearAlgebra::DELIMITER, count + 1);
190 }
191 if (!s_Count.fromDelimited(str.substr(0, count))) {
192 LOG_ERROR(<< "Failed to extract counts from " << str.substr(0, count));
193 return false;
194 }
195
196 str = str.substr(count + 1);
197 std::size_t means{0};
198 for (std::size_t i = 0; i < dimension; ++i) {
199 means = str.find_first_of(CLinearAlgebra::DELIMITER, means + 1);
200 }
201 if (!s_Mean.fromDelimited(str.substr(0, means))) {
202 LOG_ERROR(<< "Failed to extract means from " << str.substr(0, means));
203 return false;
204 }
205
206 str = str.substr(means + 1);
207 if (!s_Covariances.fromDelimited(str)) {
208 LOG_ERROR(<< "Failed to extract covariances from " << str);
209 return false;
210 }
211
212 return true;
213}

299template<typename T, typename CONTAINER, typename LESS>
300bool CBasicStatistics::COrderStatisticsImpl<T, CONTAINER, LESS>::fromDelimited(
301 const std::string& value,
302 const TFromString& fromString) {
303 this->clear();
304
305 if (value.empty()) {
306 return true;
307 }
308
309 T statistic;
310
311 std::size_t delimPos{value.find(INTERNAL_DELIMITER)};
312 if (delimPos == std::string::npos) {
313 if (fromString(value, statistic) == false) {
314 LOG_ERROR(<< "Invalid statistic in '" << value << "'");
315 return false;
316 }
317 m_Statistics[--m_UnusedCount] = statistic;
318 return true;
319 }
320
321 m_UnusedCount = m_Statistics.size();
322
323 std::string token;
324 token.reserve(15);
325 token.assign(value, 0, delimPos);
326 if (fromString(token, statistic) == false) {
327 LOG_ERROR(<< "Invalid statistic '" << token << "' in '" << value << "'");
328 return false;
329 }
330 m_Statistics[--m_UnusedCount] = statistic;
331
332 while (delimPos < value.size()) {
333 if (m_UnusedCount == 0) {
334 LOG_ERROR(<< "Too many statistics in '" << value
335 << "' - expected at most " << m_Statistics.size());
336 return false;
337 }
338 std::size_t nextDelimPos{
339 std::min(value.find(INTERNAL_DELIMITER, delimPos + 1), value.size())};
340 token.assign(value, delimPos + 1, nextDelimPos - delimPos - 1);
341 if (fromString(token, statistic) == false) {
342 LOG_ERROR(<< "Invalid statistic '" << token << "' in '" << value << "'");
343 return false;
344 }
345 m_Statistics[--m_UnusedCount] = statistic;
346 delimPos = nextDelimPos;
347 }
348
349 return true;
350}

359template<typename T, typename CONTAINER, typename LESS>
360std::string
361CBasicStatistics::COrderStatisticsImpl<T, CONTAINER, LESS>::toDelimited(const TToString& toString) const {
362 if (this->count() == 0) {
363 return std::string{};
364 }
365 std::size_t i{m_Statistics.size()};
366 std::string result{toString(m_Statistics[i - 1])};
367 for (--i; i > m_UnusedCount; --i) {
368 result += INTERNAL_DELIMITER;
369 result += toString(m_Statistics[i - 1]);
370 }
371 return result;

376CBasicStatistics::COrderStatisticsImpl<T, CONTAINER, LESS>::checksum(std::uint64_t seed) const {
377 if (this->count() == 0) {

385 raw << ' ';

387 }
388 core::CHashing::CSafeMurmurHash2String64 hasher(seed);
