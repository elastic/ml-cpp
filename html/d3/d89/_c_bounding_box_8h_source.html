33template<typename POINT>
34class CBoundingBox {
35public:
37 static constexpr bool dynamicSizeAlwaysZero() {
38 return core::memory_detail::SDynamicSizeAlwaysZero<POINT>::value();
39 }
40 using TCoordinate = typename SCoordinate<POINT>::Type;
41 using TPointPrecise = typename SFloatingPoint<POINT, double>::Type;
42
43public:
44 CBoundingBox() : m_Empty(true), m_A(), m_B() {}
45 explicit CBoundingBox(const POINT& x) : m_Empty(false), m_A(x), m_B(x) {}
46
48 void clear() {
49 m_Empty = true;
50 m_A = m_B = POINT();
51 }
52
55 void add(const POINT& p) {
56 if (m_Empty) {
57 m_A = m_B = p;
58 } else {
59 las::min(p, m_A);
60 las::max(p, m_B);
61 }
62 m_Empty = false;
63 }
64
67 void add(const CBoundingBox& other) {
68 if (m_Empty) {
69 *this = other;
70 } else if (!other.m_Empty) {
71 las::min(other.m_A, m_A);
72 las::max(other.m_B, m_B);
73 m_Empty = false;
74 }
75 }
76
78 const POINT& blc() const { return m_A; }
79
81 const POINT& trc() const { return m_B; }
82
84 POINT centre() const { return POINT(TPointPrecise(m_A + m_B) / 2.0); }
85
96 bool closerToX(const POINT& x, const POINT& y) const {
97 POINT xy = y - x;
98 POINT f(m_B);
99 for (std::size_t i = 0; i < las::dimension(x); ++i) {
100 if (xy(i) < TCoordinate{0}) {
101 f(i) = m_A(i);
102 }
103 }
104 return las::distance(f, x) <= las::distance(f, y);
105 }
106
108 std::string print() const {
109 std::ostringstream result;
110 result << "{" << m_A << ", " << m_B << "}";
111 return result.str();
112 }
113
114private:
116 bool m_Empty;
117
119 POINT m_A, m_B;
120};
