152template<typename FUNCTION, typename... ARGS>
153std::future<std::invoke_result_t<std::decay_t<FUNCTION>, std::decay_t<ARGS>...>>
154async(CExecutor& executor, FUNCTION&& f, ARGS&&... args) {
155 using R = std::invoke_result_t<std::decay_t<FUNCTION>, std::decay_t<ARGS>...>;
156
157 // Note g stores copies of the arguments in the pack, which are moved into place
158 // if possible, so this is safe to invoke later in the context of a packaged task.
159 auto g = std::bind<R>(std::forward<FUNCTION>(f), std::forward<ARGS>(args)...);
160
161 auto promise = std::make_shared<std::promise<R>>();
162 auto result = promise->get_future();
163
164 std::function<void()> task(
165 [ g_ = std::move(g), promise_ = std::move(promise) ]() mutable {
166 concurrency_detail::invokeAndWriteResultToPromise(
167 g_, promise_, std::is_same<R, void>{});
168 });
169
170 // Schedule the task to compute the result.
171 executor.schedule(std::move(task));
172
173 return result;
174}

228template<typename FUNCTION>
229void parallel_for_each(std::size_t start,
230 std::size_t end,
231 std::vector<FUNCTION>& functions,
232 const std::function<void(double)>& recordProgress) {
233
234 // Threads access the indices in the following pattern:
235 // [0, m, 2*m, ...]
236 // [1, m+1, 2*m+1, ...]
237 // [2, m+2, 2*m+2, ...]
238 // ...
239 // for m partitions. This is choosen because it is expected that the index
240 // will often be used to access elements in a contiguous block of memory.
241 // Provided the threads are doing roughly equal work per call this should
242 // ensure the best possible locality of reference for reads which occur
243 // at a similar time in the different threads.
244
245 std::vector<std::future<bool>> finished;
246 finished.reserve(functions.size());
247
248 for (std::size_t offset = 0, partitions = functions.size();
249 offset < partitions; ++offset, ++start) {
250
251 // Note there is one f for each thread so capture by reference is thread
252 // safe provided each f is thread safe. Also, we always wait until all
253 // calls are finished before returning so functions always live beyond
254 // these references.
255
256 CLoopProgress progress{end - offset, recordProgress,
257 1.0 / static_cast<double>(partitions)};
258
259 auto& f = functions[offset];
260 finished.emplace_back(
261 async(defaultAsyncExecutor(),
262 [&f, partitions, progress](std::size_t start_, std::size_t end_) mutable {
263 for (std::size_t i = start_; i < end_;
264 i += partitions, progress.increment(partitions)) {
265 f(i);
266 }
267 return true; // So we can check for exceptions via get.
268 },
269 start, end));
270 }
271
272 get_conjunction_of_all(finished);
273}

275template<typename ITR, typename FUNCTION>
276void parallel_for_each(ITR start,
277 std::size_t size,
278 std::vector<FUNCTION>& functions,
279 const std::function<void(double)>& recordProgress = concurrency_detail::noop) {
280
281 // See above for the rationale for this access pattern.
282
283 std::vector<std::future<bool>> finished;
284 finished.reserve(functions.size());
285
286 for (std::size_t offset = 0, partitions = functions.size();
287 offset < partitions; ++offset, ++start) {
288
289 // As above, capturing by reference here is safe.
290
291 CLoopProgress progress{size - offset, recordProgress,
292 1.0 / static_cast<double>(partitions)};
293
294 auto& f = functions[offset];
295 finished.emplace_back(async(
296 defaultAsyncExecutor(),
297 [&f, partitions, offset, size, progress](ITR start_) mutable {
298
299 std::size_t i{offset};
300
301 auto incrementByPartitions = [&i, partitions, size](ITR& j) {
302 if (i < size) {
303 std::advance(j, partitions);
304 }
305 };
306
307 for (ITR j = start_; i < size; i += partitions,
308 incrementByPartitions(j), progress.increment(partitions)) {
309 f(*j);
310 }
311 return true; // So we can check for exceptions via get.
312 },
313 start));
314 }
315
316 get_conjunction_of_all(finished);
317}

324template<typename FUNCTION>
325bool parallel_for_each(std::size_t start,
326 std::size_t end,
327 std::vector<FUNCTION>& functions,
328 const std::function<void(double)>& recordProgress = concurrency_detail::noop) {
329
330 if (end <= start) {
331 recordProgress(1.0);
332 return true;
333 }
334 if (functions.empty()) {
335 LOG_ERROR(<< "No task supplied");
336 recordProgress(1.0);
337 return false;
338 }
339
340 // This function is not re-entrant without the chance of deadlock if we always
341 // execute in parallel. In particular, we wait on the completion of tasks which
342 // are enqueued in the global thread pool. However, if this has already been
343 // called further up the stack we will enqueue functions which will wait on the
344 // results of these and will block the tasks they need to complete behind them
345 // in the queues. There are a couple strategies we could take, two obvious ones
346 // in order of increasing complexity and decreasing pessimism are:
347 // 1) Execute sequentially if there are any tasks in the thread pool, we then
348 // won't wait here and nothing will block in the queue,
349 // 2) Assign tasks a priority and ensure in the queue that tasks are executed
350 // in priority order. Then we can assign these tasks "highest priority in
351 // queue" + 1.
352 //
353 // This implements 1) because it is significantly simpler to and in practice we
354 // probably don't actually need to nest parallel_for_each. We're just interested
355 // in guarding against accidental deadlock in the case someone inadvertantly calls
356 // parallelised code in the function to execute.
357
358 concurrency_detail::CDefaultAsyncExecutorBusyForScope scope;
359
360 functions.resize(std::min(functions.size(), end - start));
361
362 if (functions.size() < 2 || scope.wasBusy()) {
363 functions.resize(1); // For the case scope was busy.
364 CLoopProgress progress{end - start, recordProgress};
365 for (std::size_t i = start; i < end; ++i, progress.increment()) {
366 functions[0](i);
367 }
368 return true;
369 }
370
371 concurrency_detail::parallel_for_each(start, end, functions, recordProgress);
372 return true;
373}
