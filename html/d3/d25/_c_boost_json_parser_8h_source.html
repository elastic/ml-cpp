38 static constexpr std::size_t JSON_PARSE_BUFFER_SIZE{8192};

40 static bool parse(const std::string& jsonString, json::value& doc) {
41 unsigned char buffer[JSON_PARSE_BUFFER_SIZE]; // Small stack buffer to avoid most allocations during parse
42 json::monotonic_resource mr(buffer); // This resource will use our local buffer first
43 json::error_code ec;
44 doc = json::parse(jsonString, ec, &mr);
45 if (ec) {
46 LOG_ERROR(<< "An error occurred while parsing JSON: \""
47 << jsonString << "\" :" << ec.message());
48 return false;
49 }
50 return true;
51 }

53 static json::error_code parse(std::istream& istream, json::value& doc) {
54 json::stream_parser p;
55
56 unsigned char buf[JSON_PARSE_BUFFER_SIZE]; // Now we need a buffer to hold the actual JSON values
57 json::monotonic_resource mr(buf); // The static resource is monotonic, using only a caller-provided buffer
58 p.reset(&mr); // Use the static resource for producing the value
59
60 json::error_code ec;
61 std::string line;
62 while (std::getline(istream, line)) {
63 LOG_TRACE(<< "write_some: " << line);
64 p.write_some(line);
65 }
66 p.finish(ec);
67
68 doc = p.release();
69
70 return ec;
71 }

73 static json::error_code parse(char* begin, std::size_t length, json::value& doc) {
74 json::stream_parser p;
75
76 unsigned char buf[JSON_PARSE_BUFFER_SIZE]; // Now we need a buffer to hold the actual JSON values
77 json::monotonic_resource mr(buf); // The static resource is monotonic, using only a caller-provided buffer
78 p.reset(&mr); // Use the static resource for producing the value
79
80 json::error_code ec;
81 std::size_t written{0};
82 p.reset();
83 while (written < length) {
84 written += p.write_some(begin, length, ec);
85 }
86
87 p.finish();
88
