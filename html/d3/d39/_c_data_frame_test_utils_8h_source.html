52 template<typename FACTORY, typename POINT>
53 static auto toDataFrame(FACTORY factory, const std::vector<POINT>& points) {
54
55 std::size_t dimension{maths::common::las::dimension(points[0])};
56
57 auto result = factory(dimension).first;
58
59 for (std::size_t i = 0; i < points.size(); ++i) {
60 result->writeRow([&](core::CDataFrame::TFloatVecItr column, std::int32_t& id) {
61 for (std::size_t d = 0; d < dimension; ++d, ++column) {
62 *column = points[i](d);
63 }
64 id = static_cast<std::int32_t>(i);
65 });
66 }
67 result->finishWritingRows();
68
69 return result;
70 }
71};
