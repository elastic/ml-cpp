24template<typename T, typename I, typename A>
25std::size_t
26storageNodeOverhead(const boost::multi_index::multi_index_container<T, I, A>&) {
27 // It's tricky to determine the container overhead of a multi-index
28 // container. It can have an arbitrary number of indices, each of which
29 // can be of a different type. To accurately determine the overhead
30 // would require some serious template metaprogramming to interpret the
31 // "typename I" template argument, and it's just not worth it given the
32 // infrequent and relatively simple usage (generally just two indices
33 // in our current codebase). Therefore there's an approximation here
34 // that the overhead is 2 pointers per entry per index.
35 using TMultiIndex = boost::multi_index::multi_index_container<T, I, A>;
36 constexpr std::size_t indexCount{
37 boost::mpl::size<typename TMultiIndex::index_type_list>::value};
38 return 2 * indexCount * sizeof(std::size_t);
39}

41template<typename T, typename I, typename A>
42std::size_t dynamicSize(const boost::multi_index::multi_index_container<T, I, A>& t) {
43 return memory::elementDynamicSize(t) +
44 t.size() * (sizeof(T) + memory::storageNodeOverhead(t));
45}
46}

49template<typename T, typename I, typename A>
50void dynamicSize(const char* name,
51 const boost::multi_index::multi_index_container<T, I, A>& t,
52 const CMemoryUsage::TMemoryUsagePtr& mem) {
53 std::string componentName(name);
54
55 std::size_t items = t.size();
56 CMemoryUsage::SMemoryUsage usage(componentName + "::" + typeid(T).name(),
57 items * (sizeof(T) + memory::storageNodeOverhead(t)));
58 CMemoryUsage::TMemoryUsagePtr ptr = mem->addChild();
59 ptr->setName(usage);
60
61 memory_debug::elementDynamicSize(std::move(componentName), t, mem);
62}
