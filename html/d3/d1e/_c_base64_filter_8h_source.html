105 template<typename SINK>
106 std::streamsize write(SINK& snk, const char_type* s, std::streamsize n) {
107 // copy into the buffer while there is data to read and space in the buffer
108 std::streamsize done = 0;
109 while (done < n) {
110 std::streamsize toCopy =
111 std::min(std::streamsize(n - done),
112 std::streamsize(m_Buffer.capacity() - m_Buffer.size()));
113 m_Buffer.insert(m_Buffer.end(), s + done, s + done + toCopy);
114 done += toCopy;
115 this->Encode(snk, false);
116 }
117 LOG_TRACE(<< "Base64 write " << n);
118 return n;
119 }
120
123 template<typename SINK>
124 void close(SINK& snk) {
125 this->Encode(snk, true);
126 }

128private:
131 template<typename SINK>
132 void Encode(SINK& snk, bool isFinal) {
133 using TUInt8BufCItrTransformItr =
134 boost::archive::iterators::transform_width<TUInt8BufCItr, 6, 8>;
135 using TBase64Text = boost::archive::iterators::base64_from_binary<TUInt8BufCItrTransformItr>;
136
137 TUInt8BufItr endItr = m_Buffer.end();
138 // Base64 turns 3 bytes into 4 characters - unless this is the final part
139 // of the string, we don't encode non-multiples of 3
140 if (isFinal == false) {
141 for (std::size_t i = (m_Buffer.size() % 3); i != 0; i--) {
142 --endItr;
143 }
144 }
145
146 // Do the conversion
147 std::string e(TBase64Text(m_Buffer.begin()), TBase64Text(endItr));
148
149 // Remove the encoded bytes from the buffer
150 m_Buffer.erase(m_Buffer.begin(), endItr);
151
152 // Pad the final string if necessary
153 if (isFinal && !e.empty()) {
154 std::size_t paddingCount = 4 - e.length() % 4;
155 for (std::size_t i = 0; i < paddingCount; i++) {
156 e += '=';
157 }
158 }
159 LOG_TRACE(<< "Encoded: " << e);
160 boost::iostreams::write(snk, e.c_str(), e.length());
161 }
162
163private:
165 TUInt8Buf m_Buffer;
166};

239 template<typename SOURCE>
240 std::streamsize read(SOURCE& src, char_type* s, std::streamsize n) {
241 // copy into the buffer while there is data to read and space in the buffer
242 std::streamsize done = 0;
243 char buf[4096];
244 while (done < n) {
245 std::streamsize toCopy = std::min(std::streamsize(m_BufferOut.size()),
246 std::streamsize(n - done));
247 LOG_TRACE(<< "Trying to copy " << toCopy << " bytes into stream, max "
248 << n << ", available " << m_BufferOut.size());
249 for (std::streamsize i = 0; i < toCopy; i++) {
250 s[done++] = m_BufferOut.front();
251 m_BufferOut.pop_front();
252 }
253 LOG_TRACE(<< "Eos: " << m_Eos << ", In: " << m_BufferIn.empty()
254 << ", Out: " << m_BufferOut.empty());
255 if (done == n) {
256 break;
257 }
258 if ((done > 0) && m_BufferIn.empty() && m_BufferOut.empty() && m_Eos) {
259 LOG_TRACE(<< "Base64 READ " << done << ", from n " << n
260 << ", left " << m_BufferOut.size());
261 return done;
262 }
263
264 // grab some data if we need it
265 if ((m_BufferIn.size() < 4) && (m_Eos == false)) {
266 std::streamsize readBytes = boost::iostreams::read(src, buf, 4096);
267 LOG_TRACE(<< "Read " << readBytes << " from input stream");
268 if (readBytes == -1) {
269 LOG_TRACE(<< "Got EOS from underlying store");
270 m_Eos = true;
271 } else {
272 for (std::streamsize i = 0; i < readBytes; i++) {
273 // Only copy Base64 characters - JSON punctuation is ignored
274 // The dechunker parses JSON and should give us only base64 strings,
275 // but we don't want to try and decode anything which might cause
276 // the decoder to choke
277 switch (buf[i]) {
278 case ']':
279 case '[':
280 case ',':
281 case '"':
282 case '{':
283 case '}':
284 case '\\':
285 case ' ':
286 case ':':
287 break;
288
289 default:
290 m_BufferIn.push_back(static_cast<uint8_t>(buf[i]));
291 break;
292 }
293 }
294 }
295 }
296 this->Decode(m_Eos);
297 if (m_Eos && m_BufferOut.empty() && m_BufferIn.empty() && (done == 0)) {
298 LOG_TRACE(<< "Returning -1 from read");
299 return -1;
300 }
301 }
302 LOG_TRACE(<< "Base64 READ " << done << ", from n " << n << ", left "
303 << m_BufferOut.size());
304 return done;
305 }

313 void Decode(bool isFinal) {
314 // Base64 turns 4 characters into 3 bytes
315 using TUInt8BufCItrBinaryBase64Itr =
316 boost::archive::iterators::binary_from_base64<TUInt8BufCItr>;
317 using TBase64Binary =
318 boost::archive::iterators::transform_width<TUInt8BufCItrBinaryBase64Itr, 8, 6, uint8_t>;
319
320 std::size_t inBytes = m_BufferIn.size();
321 if (inBytes == 0) {
322 return;
323 }
324
325 TUInt8BufItr endItr = m_BufferIn.end();
326 std::size_t paddingBytes = 0;
327 // Only try and decode multiples of 4 characters, unless this is the last
328 // data in the stream
329 if (isFinal == false) {
330 if (inBytes < 4) {
331 return;
332 }
333
334 for (std::size_t i = 0; i < inBytes % 4; i++) {
335 LOG_TRACE(<< "Ignoring end bytes of " << inBytes);
336 --endItr;
337 }
338 } else {
339 // We can only work with 4 or more bytes, so with fewer there is something
340 // wrong, and there can't be a sensible outcome
341 if (inBytes < 4) {
342 LOG_ERROR(<< "Invalid size of stream for decoding: " << inBytes);
343 m_BufferIn.clear();
344 return;
345 }
346 }
347
348 // Check for padding characters
349 {
350 TUInt8BufCRItr i = m_BufferIn.rbegin();
351 while ((i != m_BufferIn.rend()) && (*i == '=')) {
352 ++i;
353 paddingBytes++;
354 }
355 }
356 LOG_TRACE(<< "About to decode: " << std::string(m_BufferIn.begin(), endItr));
357
358 m_BufferOut.insert(m_BufferOut.end(), TBase64Binary(m_BufferIn.begin()),
359 TBase64Binary(endItr));
360
361 // Remove padding bytes off the back of the stream
362 m_BufferOut.erase_end(paddingBytes);
363
364 // Remove the encoded bytes from the buffer
365 m_BufferIn.erase(m_BufferIn.begin(), endItr);
366 }

368private:
370 TUInt8Buf m_BufferIn;
371
373 TUInt8Buf m_BufferOut;
374
376 bool m_Eos;
