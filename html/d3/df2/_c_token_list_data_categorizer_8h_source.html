95 void tokeniseString(const TStrStrUMap& fields,
96 const std::string& str,
97 TSizeSizePrVec& tokenIds,
98 TSizeSizeMap& tokenUniqueIds,
99 std::size_t& totalWeight,
100 std::size_t& minReweightedTotalWeight,
101 std::size_t& maxReweightedTotalWeight) override {
102 tokenIds.clear();
103 tokenUniqueIds.clear();
104 totalWeight = 0;
105
106 std::string temp;
107
108 // TODO - make more efficient
109 std::string::size_type nonHexPos(std::string::npos);
110 for (const char curChar : str) {
111
112 if (TRUNCATE_AT_NEWLINE && curChar == '\n') {
113 break;
114 }
115
116 // Basically tokenise into [a-zA-Z0-9/]+ strings, possibly
117 // allowing underscores, dots and dashes in the middle
118 if (std::isalnum(static_cast<unsigned char>(curChar)) ||
119 (!temp.empty() && ((ALLOW_UNDERSCORE && curChar == '_') ||
120 (ALLOW_DOT && curChar == '.') ||
121 (ALLOW_DASH && curChar == '-'))) ||
122 (ALLOW_FORWARD_SLASH && curChar == '/')) {
123 temp += curChar;
124 if (IGNORE_HEX) {
125 // Count dots and dashes as numeric
126 if (!std::isxdigit(static_cast<unsigned char>(curChar)) &&
127 curChar != '.' && curChar != '-') {
128 nonHexPos = temp.length() - 1;
129 }
130 }
131 } else {
132 if (!temp.empty()) {
133 this->considerToken(fields, nonHexPos, temp, tokenIds, tokenUniqueIds,
134 totalWeight, minReweightedTotalWeight,
135 maxReweightedTotalWeight);
136 temp.clear();
137 }
138
139 if (IGNORE_HEX) {
140 nonHexPos = std::string::npos;
141 }
142 }
143 }
144
145 if (!temp.empty()) {
146 this->considerToken(fields, nonHexPos, temp, tokenIds, tokenUniqueIds, totalWeight,
147 minReweightedTotalWeight, maxReweightedTotalWeight);
148 }
149
150 LOG_TRACE(<< str << " tokenised to " << tokenIds.size() << " tokens with total weight "
151 << totalWeight << ": " << SIdTranslater(*this, tokenIds, ' '));
152
153 m_DictionaryWeightFunc.reset();
154 }

158 void tokenToIdAndWeight(const std::string& token,
159 TSizeSizePrVec& tokenIds,
160 TSizeSizeMap& tokenUniqueIds,
161 std::size_t& totalWeight,
162 std::size_t& minReweightedTotalWeight,
163 std::size_t& maxReweightedTotalWeight) override {
164 TSizeSizePr idWithWeight(this->idForToken(token), 1);
165
166 if (token.length() >= MIN_DICTIONARY_LENGTH) {
167 // Give more weighting to tokens that are dictionary words.
168 idWithWeight.second += m_DictionaryWeightFunc(m_Dict.partOfSpeech(token));
169 }
170
171 tokenIds.push_back(idWithWeight);
172 tokenUniqueIds[idWithWeight.first] += idWithWeight.second;
173 totalWeight += idWithWeight.second;
174 minReweightedTotalWeight +=
175 m_DictionaryWeightFunc.minMatchingWeight(idWithWeight.second);
176 maxReweightedTotalWeight +=
177 m_DictionaryWeightFunc.maxMatchingWeight(idWithWeight.second);
178 }
179
180 void reset() override { m_DictionaryWeightFunc.reset(); }

183 double similarity(const TSizeSizePrVec& left,
184 std::size_t leftWeight,
185 const TSizeSizePrVec& right,
186 std::size_t rightWeight) const override {
187 double similarity(1.0);
188
189 std::size_t maxWeight(std::max(leftWeight, rightWeight));
190 if (maxWeight > 0) {
191 std::size_t diff(DO_WARPING ? m_SimilarityTester.weightedEditDistance(left, right)
192 : this->compareNoWarp(left, right));
193
194 similarity = 1.0 - double(diff) / double(maxWeight);
195 }
196
197 return similarity;
198 }

204 std::size_t compareNoWarp(const TSizeSizePrVec& left, const TSizeSizePrVec& right) const {
205 std::size_t minSize(std::min(left.size(), right.size()));
206 std::size_t maxSize(std::max(left.size(), right.size()));
207
208 std::size_t diff(0);
209
210 for (std::size_t index = 0; index < minSize; ++index) {
211 if (left[index].first != right[index].first) {
212 diff += std::max(left[index].second, right[index].second);
213 }
214 }
215
216 // Account for different length vector instances
217 if (left.size() < right.size()) {
218 for (std::size_t index = minSize; index < maxSize; ++index) {
219 diff += right[index].second;
220 }
221 } else if (left.size() > right.size()) {
222 for (std::size_t index = minSize; index < maxSize; ++index) {
223 diff += left[index].second;
224 }
225 }
226
227 return diff;
228 }

233 void considerToken(const TStrStrUMap& fields,
234 std::string::size_type nonHexPos,
235 std::string& token,
236 TSizeSizePrVec& tokenIds,
237 TSizeSizeMap& tokenUniqueIds,
238 std::size_t& totalWeight,
239 std::size_t& minReweightedTotalWeight,
240 std::size_t& maxReweightedTotalWeight) {
241 if (IGNORE_LEADING_DIGIT && std::isdigit(static_cast<unsigned char>(token[0]))) {
242 return;
243 }
244
245 // If configured, ignore pure hex numbers, with or without a 0x
246 // prefix.
247 if (IGNORE_HEX) {
248 if (nonHexPos == std::string::npos) {
249 // Implies hex without 0x prefix.
250 return;
251 }
252
253 // This second hex test is redundant if we're ignoring tokens
254 // with leading digits, and checking this first will cause the
255 // check to be completely compiled away as IGNORE_LEADING_DIGIT
256 // is a template argument
257 if (!IGNORE_LEADING_DIGIT && nonHexPos == 1 &&
258 token.compare(0, 2, "0x") == 0 && token.length() != 2) {
259 // Implies hex with 0x prefix.
260 return;
261 }
262 }
263
264 // If the last character is not alphanumeric, strip it.
265 while (!std::isalnum(static_cast<unsigned char>(token[token.length() - 1]))) {
266 token.erase(token.length() - 1);
267 }
268
269 if (IGNORE_DATE_WORDS && core::CTimeUtils::isDateWord(token)) {
270 return;
271 }
272
273 if (IGNORE_FIELD_NAMES && fields.find(token) != fields.end()) {
274 return;
275 }
276
277 this->tokenToIdAndWeight(token, tokenIds, tokenUniqueIds, totalWeight,
278 minReweightedTotalWeight, maxReweightedTotalWeight);
279 }
