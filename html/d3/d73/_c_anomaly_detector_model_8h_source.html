617 template<typename T, typename FILTER>
618 void applyFilter(model_t::EExcludeFrequent exclude,
619 bool updateStatistics,
620 const FILTER& filter,
621 T& data) const {
622 if (this->params().s_ExcludeFrequent & exclude) {
623 std::size_t initialSize = data.size();
624 data.erase(std::remove_if(data.begin(), data.end(), filter), data.end());
625 if (updateStatistics && data.size() != initialSize) {
626 ++core::CProgramCounters::counter(counter_t::E_TSADNumberExcludedFrequentInvocations);
627 }
628 }
