171 template<typename FACTORY>
172 void elements(const TNode& node, bool pivot, const FACTORY& factory, TTypePtrVec& result) {
173 result.clear();
174 if (this->isSimpleCount(node)) {
175 return;
176 }
177
178 if (pivot && this->isRoot(node)) {
179 TWord word = ms_Dictionary.word(node.s_Spec.s_PersonFieldName.value_or(""));
180 TWordTypePrVecItr i = element(m_InfluencerBucketSet, word);
181 if (i == m_InfluencerBucketSet.end() || i->first != word) {
182 i = m_InfluencerBucketSet.emplace(i, word, factory.make(node, pivot));
183 }
184 result.push_back(&i->second);
185 return;
186 }
187 if (pivot && !this->isRoot(node)) {
188 TWord word = ms_Dictionary.word(node.s_Spec.s_PersonFieldName.value_or(""));
189 TWordTypePrVecItr i = element(m_InfluencerSet, word);
190 if (i == m_InfluencerSet.end() || i->first != word) {
191 i = m_InfluencerSet.emplace(i, word, factory.make(node, pivot));
192 }
193 result.push_back(&i->second);
194 return;
195 }
196
197 if (this->isLeaf(node)) {
198 TWord word =
199 ms_Dictionary.word(node.s_Spec.s_PartitionFieldName.value_or(""),
200 node.s_Spec.s_PersonFieldName.value_or(""),
201 node.s_Spec.s_FunctionName.value_or(""),
202 node.s_Spec.s_ValueFieldName.value_or(""));
203 TWordTypePrVecItr i = element(m_LeafSet, word);
204 if (i == m_LeafSet.end() || i->first != word) {
205 i = m_LeafSet.emplace(i, word, factory.make(node, pivot));
206 }
207 result.push_back(&i->second);
208 }
209 if (this->isPerson(node)) {
210 TWord word =
211 ms_Dictionary.word(node.s_Spec.s_PartitionFieldName.value_or(""),
212 node.s_Spec.s_PersonFieldName.value_or(""));
213 TWordTypePrVecItr i = element(m_PersonSet, word);
214 if (i == m_PersonSet.end() || i->first != word) {
215 i = m_PersonSet.emplace(i, word, factory.make(node, pivot));
216 }
217 result.push_back(&i->second);
218 }
219 if (this->isPartition(node)) {
220 TWord word = ms_Dictionary.word(node.s_Spec.s_PartitionFieldName.value_or(""));
221
222 TWordTypePrVecItr i = element(m_PartitionSet, word);
223 if (i == m_PartitionSet.end() || i->first != word) {
224 i = m_PartitionSet.emplace(i, word, factory.make(node, pivot));
225 }
226 result.push_back(&i->second);
227 }
228 if (this->isRoot(node)) {
229 result.push_back(&m_BucketElement);
230 }
231 }

234 uint64_t checksum(uint64_t seed) const {
235 seed = maths::common::CChecksum::calculate(seed, m_BucketElement);
236 seed = maths::common::CChecksum::calculate(seed, m_InfluencerBucketSet);
237 seed = maths::common::CChecksum::calculate(seed, m_InfluencerSet);
238 seed = maths::common::CChecksum::calculate(seed, m_PartitionSet);
239 seed = maths::common::CChecksum::calculate(seed, m_PersonSet);
240 return maths::common::CChecksum::calculate(seed, m_LeafSet);
241 }

245 static const T* element(const TWordTypePrVec& set, const std::string& name) {
246 TWord word = ms_Dictionary.word(name);
247 TWordTypePrVecCItr i = element(set, word);
248 return (i != set.end() && i->first == word) ? &i->second : nullptr;
249 }

253 static TWordTypePrVecCItr element(const TWordTypePrVec& set, const TWord& word) {
254 return element(const_cast<TWordTypePrVec&>(set), word);
255 }
256
259 static TWordTypePrVecItr element(TWordTypePrVec& set, const TWord& word) {
260 return std::lower_bound(set.begin(), set.end(), word,
261 maths::common::COrderings::SFirstLess());
262 }
263
265 static void sort(TWordTypePrVec& set) {
266 std::sort(set.begin(), set.end(), maths::common::COrderings::SFirstLess());
267 }

270 template<typename F>
271 static void age(TWordTypePrVec& set, F doAge) {
272 for (std::size_t i = 0; i < set.size(); ++i) {
273 doAge(set[i].second);
274 }
