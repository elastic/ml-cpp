111 class CFiniteIterator {
112 public:
113 using iterator_category = std::forward_iterator_tag;
114 using value_type = typename std::iterator_traits<ITR>::value_type;
115 using difference_type = typename std::iterator_traits<ITR>::difference_type;
116 using pointer = typename std::iterator_traits<ITR>::pointer;
117 using reference = typename std::iterator_traits<ITR>::reference;
118
119 public:
120 CFiniteIterator() : m_Base(), m_End() {}
121 CFiniteIterator(const ITR& base, const ITR& end)
122 : m_Base(base), m_End(end) {
123 if (m_Base != m_End && !isFinite(*m_Base)) {
124 this->increment();
125 }
126 }
127
129 bool operator==(const CFiniteIterator& rhs) const {
130 return m_Base == rhs.m_Base;
131 }
133 bool operator!=(const CFiniteIterator& rhs) const {
134 return m_Base != rhs.m_Base;
135 }
136
138 reference operator*() const { return *m_Base; }
140 pointer operator->() const { return m_Base.operator->(); }
141
143 const CFiniteIterator& operator++() {
144 this->increment();
145 return *this;
146 }
148 CFiniteIterator operator++(int) {
149 CFiniteIterator result(*this);
150 this->increment();
151 return result;
152 }
153
154 private:
156 void increment() {
157 while (++m_Base != m_End) {
158 if (isFinite(*m_Base)) {
159 break;
160 }
161 }
162 }
163
164 private:
165 ITR m_Base;
166 ITR m_End;
167 };
