256 void add(core_t::TTime time,
257 const TDouble1Vec& value,
258 unsigned int count,
259 unsigned int /*sampleCount*/,
260 const TOptionalStrVec& influences) {
261 TSampleVec& sum = m_BucketSums.get(time);
262 if (sum.empty()) {
263 core_t::TTime bucketLength = m_BucketSums.bucketLength();
264 sum.push_back(CSample(maths::common::CIntegerTools::floor(time, bucketLength),
265 TDoubleVec(1, 0.0), 1.0, 0.0));
266 }
267 (sum[0].value())[0] += value[0];
268 sum[0].count() += static_cast<double>(count);
269 for (std::size_t i = 0; i < influences.size(); ++i) {
270 if (!influences[i]) {
271 continue;
272 }
273 TOptionalStrDoubleUMap& sums = m_InfluencerBucketSums[i].get(time);
274 sums[influences[i]] += value[0];
275 }
276 }
