49 static inline double inverseQuadraticInterpolate(const double a,
50 const double b,
51 const double c,
52 const double fa,
53 const double fb,
54 const double fc) {
55 // The bracketing is important here: if f(x) < (min double)^(1/2)
56 // the products will underflow.
57 return a * (fb / (fa - fb)) * (fc / (fa - fc)) +
58 b * (fa / (fb - fa)) * (fc / (fb - fc)) +
59 c * (fa / (fc - fa)) * (fb / (fc - fb));
60 }

66 return b - (b - a) * (fb / (fb - fa));

108 template<typename F>
109 static void minimize(double a,
110 double b,
111 double fa,
112 double fb,
113 const F& f,
114 double tolerance,
115 std::size_t& maxIterations,
116 double lb,
117 double& x,
118 double& fx) {
119 tolerance = std::max(tolerance, std::sqrt(std::numeric_limits<double>::epsilon()));
120 const double golden = 0.3819660;
121
122 if (fa < fb) {
123 x = a;
124 fx = fa;
125 } else {
126 x = b;
127 fx = fb;
128 }
129
130 double w = x;
131 double v = x;
132 double fw = fx;
133 double fv = fx;
134 double s = 0.0;
135 double sLast = 0.0;
136
137 std::size_t n = std::max(maxIterations, std::size_t{1});
138
139 do {
140 double xm = bisect(a, b);
141
142 double t1 = tolerance * (std::fabs(x) + 0.25);
143 double t2 = 2.0 * t1;
144 if (fx <= lb || std::fabs(x - xm) <= (t2 - (b - a) / 2.0)) {
145 break;
146 }
147
148 double sign = (x >= xm) ? -1.0 : 1.0;
149
150 if (sLast > t1) {
151 // Fit parabola to abscissa.
152 double r = (x - w) * (fx - fv);
153 double q = (x - v) * (fx - fw);
154 double p = (x - v) * q - (x - w) * r;
155 q = 2 * (q - r);
156 if (q > 0) {
157 p = -p;
158 }
159 q = std::fabs(q);
160
161 double td = sLast;
162 sLast = std::fabs(s);
163
164 if (std::fabs(p) >= q * td / 2.0 || p <= q * (a - x) || p >= q * (b - x)) {
165 // Minimum not in range or converging too slowly.
166 sLast = (x >= xm) ? x - a : b - x;
167 s = sign * std::max(golden * sLast, t1);
168 } else {
169 s = p / q;
170 double u = x + s;
171 if ((u - a) < t2 || (b - u) < t2) {
172 s = sign * t1;
173 }
174 }
175 } else {
176 // Don't have a suitable abscissa so just use golden
177 // section in to the larger of [a, x] and [x, b].
178 sLast = (x >= xm) ? x - a : b - x;
179 s = sign * std::max(golden * sLast, t1);
180 }
181
182 double u = x + s;
183 double fu = f(u);
184 LOG_TRACE(<< "s = " << s << ", u = " << u);
185 LOG_TRACE(<< "f(u) = " << fu << ", f(x) = " << fx);
186
187 if (fu <= fx) {
188 u >= x ? a = x : b = x;
189 shift(v, w, x, u);
190 shift(fv, fw, fx, fu);
191 } else {
192 u < x ? a = u : b = u;
193 if (fu <= fw || w == x) {
194 shift(v, w, u);
195 shift(fv, fw, fu);
196 } else if (fu <= fv || v == x || v == w) {
197 v = u;
198 fv = fu;
199 }
200 }
201 LOG_TRACE(<< "a = " << a << ", b = " << b);
202 LOG_TRACE(<< "x = " << x << ", v = " << v << ", w = " << w);
203 LOG_TRACE(<< "f(x) = " << fx << ", f(v) = " << fv << ", f(w) = " << fw);
204 } while (--n > 0);
205
206 maxIterations -= n;
207 }

211 template<typename F>
212 static bool bracket(double& a,
213 double& b,
214 double& fa,
215 double& fb,
216 const F& f,
217 const double direction,
218 std::size_t& maxIterations,
219 const double min,
220 const double max) {
221 if (a > b) {
222 std::swap(a, b);
223 std::swap(fa, fb);
224 }
225
226 // Simple doubling search which switches to the secant
227 // method if this strategy fails to produce a bracket
228 // quickly.
229
230 double step = b - a;
231 if (step == 0.0) {
232 step = 1.0;
233 }
234
235 std::size_t n = maxIterations;
236 for (; n > 0; --n) {
237 if (std::signbit(fa) != std::signbit(fb) || fa == 0.0 || fb == 0.0) {
238 break;
239 }
240
241 step *= 2.0;
242 if (n < (3 * maxIterations) / 4) {
243 double minStep = step;
244 double maxStep = step * step;
245 step = fa == fb
246 ? maxStep
247 : std::min(std::max(std::fabs(b - a) / std::fabs(fb - fa) *
248 std::fabs(fb),
249 minStep),
250 maxStep);
251 }
252 a = b;
253 fa = fb;
254 b += direction * step;
255 b = std::max(std::min(b, max), min);
256 if (a == b) {
257 // We've hit our domain constraints.
258 break;
259 }
260
261 fb = f(b);
262 }
263
264 if (a > b) {
265 std::swap(a, b);
266 std::swap(fa, fb);
267 }
268
269 maxIterations = maxIterations - n;
270 return std::signbit(fa) != std::signbit(fb) || fa == 0.0 || fb == 0.0;
271 }

410 template<typename F, typename EQUAL>
411 static void solve(double& a,
412 double& b,
413 double fa,
414 double fb,
415 const F& f,
416 std::size_t& maxIterations,
417 const EQUAL& equal,
418 double& bestGuess) {
419 if (equal(a, b)) {
420 // There is a bug in boost's solver for the case that
421 // a == b so trap and return early.
422 maxIterations = 0;
423 bestGuess = (a + b) / 2.0;
424 return;
425 }
426 try {
427 CTrapNaNArgument<F> fSafe(f);
428 // Need at least one step or the boost solver underflows
429 // size_t.
430 boost::uintmax_t n = std::max(maxIterations, std::size_t(1));
431 auto bracket = boost::math::tools::toms748_solve<const CTrapNaNArgument<F>&>(
432 fSafe, a, b, fa, fb, equal, n);
433 a = bracket.first;
434 b = bracket.second;
435 bestGuess = bisect(a, b);
436 maxIterations = static_cast<std::size_t>(n);
437 return;
438 } catch (const std::exception& e) {
439 LOG_TRACE(<< "Falling back to Brent's solver: " << e.what());
440 SUPPRESS_USAGE_WARNING(e);
441 }
442 if (!brent(a, b, fa, fb, f, maxIterations, equal, bestGuess)) {
443 throw std::invalid_argument("doesn't bracket root");
444 }
445 }

507 template<typename F, typename EQUAL>
508 static bool brent(double& a,
509 double& b,
510 double fa,
511 double fb,
512 const F& f,
513 std::size_t& maxIterations,
514 const EQUAL& equal,
515 double& bestGuess) {
516 std::size_t n = std::max(maxIterations, std::size_t{1});
517 if (fa == 0.0) {
518 // Root at left bracket.
519 bestGuess = b = a;
520 maxIterations -= n;
521 return true;
522 }
523 if (fb == 0.0) {
524 // Root at right bracket.
525 bestGuess = a = b;
526 maxIterations -= n;
527 return true;
528 }
529 if (std::signbit(fa) == std::signbit(fb)) {
530 // Not bracketed.
531 maxIterations -= n;
532 return false;
533 }
534
535 if (std::fabs(fa) < std::fabs(fb)) {
536 std::swap(a, b);
537 std::swap(fa, fb);
538 }
539
540 bool bisected = true;
541 double c = a;
542 double fc = fa;
543 double d = std::numeric_limits<double>::max();
544
545 do {
546 double s;
547 bool canUseInverseQuadratic = ((fa != fc) && (fb != fc));
548 bool canUseSecant = (fa != fb);
549 if (canUseInverseQuadratic || canUseSecant) {
550 s = canUseInverseQuadratic
551 ? inverseQuadraticInterpolate(a, b, c, fa, fb, fc)
552 : secantInterpolate(a, b, fa, fb);
553
554 double e = (3.0 * a + b) / 4.0;
555
556 if ((!(((s > e) && (s < b)) || ((s < e) && (s > b)))) ||
557 (bisected &&
558 ((std::fabs(s - b) >= std::fabs(b - c) / 2.0) || equal(b, c))) ||
559 (!bisected &&
560 ((std::fabs(s - b) >= std::fabs(c - d) / 2.0) || equal(c, d)))) {
561 // Use bisection.
562 s = bisect(a, b);
563 bisected = true;
564 } else {
565 bisected = false;
566 }
567 } else {
568 s = bisect(a, b);
569 bisected = true;
570 }
571
572 double fs = f(s);
573 shift(d, c, b);
574 fc = fb;
575
576 if (fs == 0.0) {
577 // Root at s.
578 bestGuess = a = b = s;
579 maxIterations -= (n - 1);
580 return true;
581 }
582
583 if (std::signbit(fa) == std::signbit(fs)) {
584 a = s;
585 fa = fs;
586 } else {
587 b = s;
588 fb = fs;
589 }
590
591 if (std::fabs(fa) < std::fabs(fb)) {
592 std::swap(a, b);
593 std::swap(fa, fb);
594 }
595 } while (--n > 0 && !equal(a, b));
596
597 if (b < a) {
598 std::swap(a, b);
599 std::swap(fa, fb);
600 }
601 bestGuess = (fa != fc) && (fb != fc)
602 ? inverseQuadraticInterpolate(a, b, c, fa, fb, fc)
603 : (fa != fb ? secantInterpolate(a, b, fa, fb) : bisect(a, b));
604 bestGuess = std::min(std::max(a, bestGuess), b);
605 maxIterations -= n;
606
607 return true;
608 }

676 template<typename F, typename EQUAL>
677 static bool bisection(double& a,
678 double& b,
679 double fa,
680 double fb,
681 const F& f,
682 std::size_t& maxIterations,
683 const EQUAL& equal,
684 double& bestGuess) {
685 std::size_t n = std::max(maxIterations, std::size_t{1});
686 if (fa == 0.0) {
687 // Root at left bracket.
688 bestGuess = b = a;
689 maxIterations -= n;
690 return true;
691 }
692 if (fb == 0.0) {
693 // Root at right bracket.
694 bestGuess = a = b;
695 maxIterations -= n;
696 return true;
697 }
698 if (std::signbit(fa) == std::signbit(fb)) {
699 // Not bracketed.
700 maxIterations -= n;
701 return false;
702 }
703
704 do {
705 const double c = bisect(a, b);
706 const double fc = f(c);
707 if (fc == 0.0) {
708 // Root at s.
709 bestGuess = a = b = c;
710 maxIterations -= (n - 1);
711 return true;
712 }
713
714 if (std::signbit(fa) == std::signbit(fc)) {
715 a = c;
716 fa = fc;
717 } else {
718 b = c;
719 fb = fc;
720 }
721 } while (--n > 0 && !equal(a, b));
722
723 bestGuess = fa != fb ? secantInterpolate(a, b, fa, fb) : bisect(a, b);
724 bestGuess = std::min(std::max(bestGuess, a), b);
725 maxIterations -= n;
726
727 return true;
728 }

822 template<typename T, typename F>
823 static bool globalMinimize(const T& p, const F& f, double& x, double& fx, double& fsd) {
824 using TMinAccumulator =
825 CBasicStatistics::SMin<std::pair<double, std::size_t>>::TAccumulator;
826 using TMeanVarAccumulator = CBasicStatistics::SSampleMeanVar<double>::TAccumulator;
827
828 std::size_t n = p.size();
829
830 if (n == 0) {
831 LOG_ERROR(<< "Must provide points at which to evaluate function");
832 return false;
833 }
834
835 TMinAccumulator min;
836 TMeanVarAccumulator moments;
837 T fp = p;
838 for (std::size_t i = 0; i < n; ++i) {
839 double fi = f(p[i]);
840 fp[i] = fi;
841 min.add({fi, i});
842 moments.add(fi);
843 }
844 fsd = std::sqrt(CBasicStatistics::maximumLikelihoodVariance(moments));
845 LOG_TRACE(<< "p = " << p);
846 LOG_TRACE(<< "f(p) = " << fp);
847
848 std::size_t i = min[0].second;
849 std::size_t maxIterations = 5;
850 if (i == 0) {
851 minimize(p[0], p[1], fp[0], fp[1], f, 0.0, maxIterations, x, fx);
852 } else if (i == n - 1) {
853 minimize(p[n - 2], p[n - 1], fp[n - 2], fp[n - 1], f, 0.0,
854 maxIterations, x, fx);
855 } else {
856 std::size_t ai = i - 1;
857 std::size_t bi = i + 1;
858 minimize(p[ai], p[bi], fp[ai], fp[bi], f, 0.0, maxIterations, x, fx);
859 if (fp[i] < fx) {
860 x = p[i];
861 fx = fp[i];
862 }
863 }
864 LOG_TRACE(<< "x = " << x << " fx = " << fx);
865 return true;
866 }

917 template<typename F>
918 static bool sublevelSet(double a,
919 double b,
920 double fa,
921 double fb,
922 const F& f,
923 const double fc,
924 std::size_t maxIterations,
925 TDoubleDoublePr& result) {
926 if (a > b) {
927 std::swap(a, b);
928 std::swap(fa, fb);
929 }
930
931 double x;
932 double fx;
933 {
934 std::size_t n = maxIterations;
935 minimize(a, b, fa, fb, f, 0.0, n, fc, x, fx);
936 }
937
938 result = {x, x};
939 if (fx > fc) {
940 return false;
941 }
942
943 // [a, x] and [b, r] bracket the sublevel set end points.
944
945 auto fMinusFc = [&f, fc](double x_) { return f(x_) - fc; };
946
947 LOG_TRACE(<< "a = " << a << ", x = " << x << ", b = " << b);
948 LOG_TRACE(<< "f_(a) = " << fa - fc << ", f_(x) = " << fx - fc
949 << ", f_(b) = " << fb - fc);
950
951 const double eps = std::sqrt(std::numeric_limits<double>::epsilon()) * b;
952 CEqualWithTolerance<double> equal(CToleranceTypes::E_AbsoluteTolerance, eps);
953 LOG_TRACE(<< "eps = " << eps);
954
955 try {
956 std::size_t n = maxIterations;
957 solve(a, x, fa - fc, fx - fc, fMinusFc, n, equal, result.first);
958 LOG_TRACE(<< "iterations = " << n);
959 } catch (const std::exception& e) {
960 LOG_ERROR(<< "Failed to find left end point: " << e.what());
961 return false;
962 }
963
964 try {
965 std::size_t n = maxIterations;
966 solve(x, b, fx - fc, fb - fc, fMinusFc, n, equal, result.second);
967 LOG_TRACE(<< "iterations = " << n);
968 } catch (std::exception& e) {
969 LOG_ERROR(<< "Failed to find right end point: " << e.what());
970 return false;
971 }
972
973 return true;
974 }
