147template<typename T>
148class CMixtureDistribution {
149public:
150 using TDoubleVec = std::vector<double>;
151 using TModeVec = std::vector<T>;
152
153public:
154 CMixtureDistribution() = default;
155
157 CMixtureDistribution(const TDoubleVec& weights, const TModeVec& modes)
158 : m_Weights(weights), m_Modes(modes) {
159 std::size_t w = m_Weights.size();
160 if (w != m_Modes.size()) {
161 LOG_ERROR(<< "# weights = " << w << ", # modes = " << m_Modes.size());
162 m_Weights.resize(m_Modes.size(), 0.0);
163 }
164
165 // Normalize the weights.
166 double weightSum = 0.0;
167 for (std::size_t i = 0; i < w; ++i) {
168 weightSum += m_Weights[i];
169 }
170 if (weightSum == 0.0) {
171 LOG_ERROR(<< "Expected non-zero weight sum");
172 }
173 for (std::size_t i = 0; i < w; ++i) {
174 m_Weights[i] = weightSum == 0.0 ? 1.0 / static_cast<double>(w)
175 : m_Weights[i] / weightSum;
176 }
177 }
178
179 void swap(CMixtureDistribution& other) {
180 m_Weights.swap(other.m_Weights);
181 m_Modes.swap(other.m_Modes);
182 }
183
184 inline const TDoubleVec& weights() const { return m_Weights; }
185 inline TDoubleVec& weights() { return m_Weights; }
186
187 inline const TModeVec& modes() const { return m_Modes; }
188 inline TModeVec& modes() { return m_Modes; }
189
190 std::string print() const {
191 std::string result;
192 for (std::size_t i = 0; i < m_Weights.size(); ++i) {
193 result +=
194 ' ' + core::CStringUtils::typeToStringPretty(m_Weights[i]) + '/' +
195 core::CStringUtils::typeToStringPretty(mean(m_Modes[i])) + '/' +
196 core::CStringUtils::typeToStringPretty(standard_deviation(m_Modes[i]));
197 }
198 result += (m_Weights.empty() ? "" : " ");
199 return result;
200 }
201
202private:
203 TDoubleVec m_Weights;
204 TModeVec m_Modes;
205};

224template<typename T>
225mixture_detail::TDoubleDoublePr support(const CMixtureDistribution<T>& distribution) {
226 using TModeVec = typename CMixtureDistribution<T>::TModeVec;
227
228 const TModeVec& modes = distribution.modes();
229
230 if (modes.empty()) {
231 return {std::numeric_limits<double>::lowest(), std::numeric_limits<double>::max()};
232 }
233
234 mixture_detail::TDoubleDoublePr result(std::numeric_limits<double>::max(),
235 std::numeric_limits<double>::lowest());
236
237 for (std::size_t i = 0; i < modes.size(); ++i) {
238 try {
239 mixture_detail::TDoubleDoublePr modeSupport = support(modes[i]);
240 result.first = std::min(result.first, modeSupport.first);
241 result.second = std::max(result.second, modeSupport.second);
242 } catch (const std::exception& e) {
243 LOG_ERROR(<< "Failed to compute support for mode: " << e.what());
244 }
245 }
246
247 return result;
248}

253template<typename T>
254double mode(const CMixtureDistribution<T>& distribution) {
255 using TDoubleVec = typename CMixtureDistribution<T>::TDoubleVec;
256 using TModeVec = typename CMixtureDistribution<T>::TModeVec;
257
258 static const std::size_t MAX_ITERATIONS = 20;
259
260 double result = 0.0;
261
262 const TDoubleVec& weights = distribution.weights();
263 const TModeVec& modes = distribution.modes();
264
265 if (weights.empty()) {
266 return result;
267 }
268 if (weights.size() == 1) {
269 return mode(modes[0]);
270 }
271
272 mixture_detail::CPdfAdpater<T> f(distribution);
273 double fMax = 0.0;
274 for (std::size_t i = 0; i < weights.size(); ++i) {
275 try {
276 double x25 = quantile(modes[i], 0.25);
277 double x75 = quantile(modes[i], 0.75);
278 std::size_t maxIterations = MAX_ITERATIONS;
279 double x;
280 double fx;
281 CSolvers::maximize(x25, x75, f(x25), f(x75), f, 0.0, maxIterations, x, fx);
282 if (fx > fMax) {
283 result = x;
284 fMax = fx;
285 }
286 } catch (const std::exception& e) {
287 LOG_ERROR(<< "Failed to compute f(x) at mode: " << e.what());
288 throw e;
289 }
290 }
291
292 return result;
293}

298template<typename T>
299double pdf(const CMixtureDistribution<T>& distribution, double x) {
300 using TDoubleVec = typename CMixtureDistribution<T>::TDoubleVec;
301 using TModeVec = typename CMixtureDistribution<T>::TModeVec;
302
303 if (CMathsFuncs::isNan(x)) {
304 LOG_ERROR(<< "Bad value x = " << x);
305 return 0.0;
306 }
307
308 double result = 0.0;
309
310 const TDoubleVec& weights = distribution.weights();
311 const TModeVec& modes = distribution.modes();
312
313 if (weights.empty()) {
314 return result;
315 }
316
317 for (std::size_t i = 0; i < weights.size(); ++i) {
318 mixture_detail::TDoubleDoublePr ms = support(modes[i]);
319 if (x >= ms.first && x <= ms.second) {
320 try {
321 double fx = pdf(modes[i], x);
322 LOG_TRACE(<< "x = " << x << ", w(" << i << ") = " << weights[i]
323 << ", f(x, " << i << ") " << fx);
324 result += weights[i] * fx;
325 } catch (const std::exception& e) {
326 LOG_ERROR(<< "Failed to compute f(x) for mode at " << x << ": "
327 << e.what());
328 throw e;
329 }
330 } else {
331 LOG_TRACE(<< "x = " << x << ", support = (" << ms.first << ","
332 << ms.second << ")");
333 }
334 }
335
336 return result;
337}

342template<typename T>
343double cdf(const CMixtureDistribution<T>& distribution, double x) {
344 using TDoubleVec = typename CMixtureDistribution<T>::TDoubleVec;
345 using TModeVec = typename CMixtureDistribution<T>::TModeVec;
346
347 if (CMathsFuncs::isNan(x)) {
348 LOG_ERROR(<< "Bad value x = " << x);
349 return 1.0;
350 }
351
352 const TDoubleVec& weights = distribution.weights();
353 const TModeVec& modes = distribution.modes();
354
355 if (weights.empty()) {
356 return 0.0;
357 }
358
359 double result = 0.0;
360 for (std::size_t i = 0; i < modes.size(); ++i) {
361 mixture_detail::TDoubleDoublePr ms = support(modes[i]);
362 if (x >= ms.second) {
363 result += weights[i];
364 } else if (x >= ms.first) {
365 try {
366 double fx = cdf(modes[i], x);
367 LOG_TRACE(<< "x = " << x << ", w(" << i << ") = " << weights[i]
368 << ", f(x, " << i << ") " << fx);
369 result += weights[i] * fx;
370 } catch (const std::exception& e) {
371 LOG_ERROR(<< "Failed to compute f(x) for mode at " << x << ": "
372 << e.what());
373 throw e;
374 }
375 } else {
376 LOG_TRACE(<< "x = " << x << ", support = (" << ms.first << ","
377 << ms.second << ")");
378 }
379 }
380
381 return result;

399

401 return 1.0;
402 }
403
404 double result = 0.0;
405 for (std::size_t i = 0; i < modes.size(); ++i) {
406 mixture_detail::TDoubleDoublePr ms = support(modes[i]);
407 if (x < ms.first) {
408 result += weights[i];

448template<typename T>
449double quantile(const CMixtureDistribution<T>& distribution, const double q) {
450 using TModeVec = typename CMixtureDistribution<T>::TModeVec;
451
452 mixture_detail::TDoubleDoublePr s = support(distribution);
453
454 if (q <= 0.0) {
455 if (q < 0.0) {
456 LOG_ERROR(<< "Bad quantile " << q);
457 }
458 return s.first;
459 } else if (q >= 1.0) {
460 if (q > 1.0) {
461 LOG_ERROR(<< "Bad quantile " << q);
462 }
463 return s.second;
464 }
465
466 const TModeVec& modes = distribution.modes();
467 if (modes.empty()) {
468 return q < 0.5 ? s.first : (q > 0.5 ? s.second : 0.0);
469 } else if (modes.size() == 1) {
470 return quantile(modes[0], q);
471 }
472
473 mixture_detail::CCdfAdapter<T> f(distribution);
474 CCompositeFunctions::CMinusConstant<mixture_detail::CCdfAdapter<T>> fq(f, q);
475
476 static const std::size_t MAX_ITERATIONS = 100;
477 static const double EPS = 1e-3;
478
479 double x0 = mode(distribution);
480 double result = x0;
481
482 try {
483 double f0 = fq(x0);
484 double a = x0, b = x0, fa = f0, fb = f0;
485 LOG_TRACE(<< "(a,b) = [" << a << "," << b << "], "
486 << ", (f(a),f(b)) = [" << fa << "," << fb << "]");
487
488 std::size_t maxIterations = MAX_ITERATIONS;
489 if ((f0 < 0 && !CSolvers::rightBracket(a, b, fa, fb, fq, maxIterations,
490 s.first, s.second)) ||
491 (f0 >= 0 && !CSolvers::leftBracket(a, b, fa, fb, fq, maxIterations,
492 s.first, s.second))) {
493 LOG_ERROR(<< "Unable to bracket quantile = " << q << ", (a,b) = ("
494 << a << "," << b << ")"
495 << ", (f(a),f(b)) = (" << fa << "," << fb << ")");
496 result = std::fabs(fa) < std::fabs(fb) ? a : b;
497 } else {
498 LOG_TRACE(<< "(a,b) = (" << a << "," << b << ")"
499 << ", (f(a),f(b)) = (" << fa << "," << fb << ")");
500 maxIterations = MAX_ITERATIONS - maxIterations;
501 CEqualWithTolerance<double> equal(
502 CToleranceTypes::E_AbsoluteTolerance,
503 std::min(std::numeric_limits<double>::epsilon() * b,
504 EPS * q / std::max(fa, fb)));
505 CSolvers::solve(a, b, fa, fb, fq, maxIterations, equal, result);
506 LOG_TRACE(<< "q = " << q << ", x = " << result << ", f(x) = " << fq(result)
507 << ", iterations = " << maxIterations);
508 }
509 } catch (const std::exception& e) {
510 LOG_ERROR(<< "Failed to compute quantile " << q);
511 throw e;
512 }
513
514 return result;
515}
516}
