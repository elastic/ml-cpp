222 template<typename T>
223 void makeModelT(const ml::model::SModelParams& params,
224 const ml::model_t::TFeatureVec& features,
225 ml::core_t::TTime startTime,
226 ml::model_t::EModelType modelType,
227 ml::model::CModelFactory::TDataGathererPtr& gatherer,
228 ml::model::CModelFactory::TModelPtr& model,
229 TOptionalUInt sampleCount = TOptionalUInt(),
230 const std::string& summaryCountField = EMPTY_STRING) {
231 if (m_InterimBucketCorrector == nullptr) {
232 m_InterimBucketCorrector =
233 std::make_shared<ml::model::CInterimBucketCorrector>(params.s_BucketLength);
234 }
235 if (m_Factory == nullptr) {
236 m_Factory.reset(new T(params, m_InterimBucketCorrector,
237 summaryCountField.empty() ? ml::model_t::E_None
238 : ml::model_t::E_Manual,
239 summaryCountField));
240 m_Factory->features(features);
241 }
242 ml::model::CModelFactory::SGathererInitializationData initData(startTime);
243 if (sampleCount) {
244 initData.s_SampleOverrideCount = *sampleCount;
245 }
246 gatherer.reset(m_Factory->makeDataGatherer(initData));
247 model.reset(m_Factory->makeModel({gatherer}));
248
249 BOOST_TEST_REQUIRE(model);
250 BOOST_REQUIRE_EQUAL(modelType, model->category());
251 BOOST_REQUIRE_EQUAL(params.s_BucketLength, model->bucketLength());
252 }
