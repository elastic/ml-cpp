155 using TVectorMeanAccumulatorVec = std::vector<TVectorMeanAccumulator>;

160private:
161 TSizeSizePr bucket(double prediction, double actual) const {
162 auto bucketWidth{this->bucketWidth()};
163 double bucketPrediction{(prediction - m_ExpPredictionMinMax.min()) /
164 bucketWidth.first};
165 std::size_t predictionBucketIndex{std::min(
166 static_cast<std::size_t>(bucketPrediction), m_Buckets.size() - 1)};
167
168 double bucketActual{(actual - m_LogActualMinMax.min()) / bucketWidth.second};
169 std::size_t actualBucketIndex{std::min(
170 static_cast<std::size_t>(bucketActual), m_Buckets[0].size() - 1)};
171
172 return std::make_pair(predictionBucketIndex, actualBucketIndex);
173 }
174
175 TDoubleDoublePr bucketWidth() const {
176 double predictionBucketWidth{m_ExpPredictionMinMax.range() /
177 static_cast<double>(m_Buckets.size())};
178 double actualBucketWidth{m_LogActualMinMax.range() /
179 static_cast<double>(m_Buckets[0].size())};
180 return std::make_pair(predictionBucketWidth, actualBucketWidth);
181 }
182

185 std::size_t m_CurrentPass{0};
186 TMinMaxAccumulator m_ExpPredictionMinMax;
187 TMinMaxAccumulator m_LogActualMinMax;
