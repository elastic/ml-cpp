73class MODEL_EXPORT CResultType {
74public:
75 enum EInterimOrFinal { E_Interim = 1, E_Final = 2 };
76
77 enum EConditionalOrUnconditional { E_Unconditional = 4, E_Conditional = 8 };
78
79public:
80 CResultType() : m_Type(0) {}
81 CResultType(EInterimOrFinal type) : m_Type(type) {}
82 CResultType(EConditionalOrUnconditional type) : m_Type(type) {}
83 explicit CResultType(unsigned int type) : m_Type(type) {}
84
86 void set(EInterimOrFinal type) {
87 m_Type = m_Type & (E_Unconditional | E_Conditional);
88 m_Type = m_Type | type;
89 }
90
92 void set(EConditionalOrUnconditional type) {
93 m_Type = m_Type & (E_Interim | E_Final);
94 m_Type = m_Type | type;
95 }
96
98 bool isInterim() const {
99 return (m_Type & static_cast<unsigned int>(E_Interim)) != 0;
100 }
101
103 EInterimOrFinal asInterimOrFinal() const {
104 return this->isInterim() ? E_Interim : E_Final;
105 }
106
108 bool isUnconditional() const {
109 return (m_Type & static_cast<unsigned int>(E_Unconditional)) != 0;
110 }
111
113 EConditionalOrUnconditional asConditionalOrUnconditional() const {
114 return this->isUnconditional() ? E_Unconditional : E_Conditional;
115 }
116
118 unsigned int asUint() const { return m_Type; }
119
120private:
122 unsigned int m_Type;
123};
