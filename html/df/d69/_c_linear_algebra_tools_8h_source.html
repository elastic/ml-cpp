45template<>
46struct SSqrt<VectorTag> {
47 template<typename VECTOR>
48 static void calculate(std::size_t d, VECTOR& result) {
49 for (std::size_t i = 0; i < d; ++i) {
50 result(i) = std::sqrt(result(i));
51 }

59 for (std::size_t i = 0; i < d; ++i) {
60 for (std::size_t j = 0; j <= i; ++j) {
61 result(i, j) = std::sqrt(result(i, j));
62 }
63 }
64 }
65};

646template<typename T, typename MATRIX>
647void scaleCovariances(std::size_t i, T scale, MATRIX& m) {
