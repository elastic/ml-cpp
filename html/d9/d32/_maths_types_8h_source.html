99template<typename VECTOR>
100struct SUnitWeightFactory {
101 static std::size_t dimension(const VECTOR& weight) { return weight.size(); }
102 static VECTOR weight(std::size_t dimension) {
103 return VECTOR(dimension, 1.0);
104 }
105};
107template<>
108struct SUnitWeightFactory<double> {
109 static std::size_t dimension(double) { return 1; }
110 static double weight(std::size_t) { return 1.0; }
111};

114template<typename VECTOR>
115struct SWeightArithmetic {
116 static void add(const VECTOR& lhs, VECTOR& rhs) {
117 for (std::size_t i = 0; i < lhs.size(); ++i) {
118 rhs[i] += lhs[i];
119 }
120 }
121 static void multiply(const VECTOR& lhs, VECTOR& rhs) {
122 for (std::size_t i = 0; i < lhs.size(); ++i) {
123 rhs[i] *= lhs[i];
124 }
125 }
126};
128template<>
129struct SWeightArithmetic<double> {
130 static void add(double lhs, double& rhs) { rhs += lhs; }
131 static void multiply(double lhs, double& rhs) { rhs *= lhs; }
132};
133}

143 template<typename VECTOR>
144 static TWeightsAry<VECTOR> unit(const VECTOR& weight) {
145 return unit<VECTOR>(maths_types_detail::SUnitWeightFactory<VECTOR>::dimension(weight));
146 }
148 template<typename VECTOR>
149 static TWeightsAry<VECTOR> unit(std::size_t dimension) {
150 TWeightsAry<VECTOR> result;
151 result.fill(maths_types_detail::SUnitWeightFactory<VECTOR>::weight(dimension));
152 return result;
153 }
155 template<typename VECTOR>
156 static core::CSmallVector<TWeightsAry<VECTOR>, 1> singleUnit(const VECTOR& weight) {
157 return {unit<VECTOR>(weight)};
158 }
160 template<typename VECTOR>
161 static core::CSmallVector<TWeightsAry<VECTOR>, 1> singleUnit(std::size_t dimension) {
162 return {unit<VECTOR>(dimension)};
163 }
164};
