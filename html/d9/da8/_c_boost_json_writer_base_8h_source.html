115 std::shared_ptr<CBoostJsonPoolAllocator> getAllocator() const {
116 TPoolAllocatorPtr allocator;
117 CBoostJsonPoolAllocator* rawAllocator = nullptr;
118 while (!m_JsonPoolAllocators.empty()) {
119 allocator = m_JsonPoolAllocators.top();
120
121 if (allocator && (rawAllocator = allocator.get())) {
122 break;
123 } else {
124 LOG_ERROR(<< "Invalid JSON memory allocator encountered. Removing.");
125 m_JsonPoolAllocators.pop();
126 }
127 }
128
129 // shouldn't ever happen as it indicates that the default allocator is invalid
130 if (rawAllocator == nullptr) {
131 LOG_ERROR(<< "No viable JSON memory allocator encountered. Recreating.");
132 allocator = std::make_shared<CBoostJsonPoolAllocator>();
133 m_JsonPoolAllocators.push(allocator);
134 }
135
136 return allocator;
137 }

465 virtual bool write(const TValue& doc) {
466 switch (doc.kind()) {
467 case json::kind::bool_:
468 return this->onBool(doc.as_bool());
469 case json::kind::null:
470 return this->onNull();
471 case json::kind::object:
472 if (this->onObjectBegin() == false) {
473 return false;
474 }
475 for (const auto& member : doc.as_object()) {
476 if (this->onKey(member.key()) == false) {
477 return false;
478 }
479 if (this->write(member.value()) == false) {
480 return false;
481 }
482 }
483 return this->onObjectEnd(doc.as_object().size());
484 case json::kind::array:
485 if (this->onArrayBegin() == false) {
486 return false;
487 }
488 for (const auto& member : doc.as_array()) {
489 if (this->write(member) == false) {
490 return false;
491 }
492 }
493 return this->onArrayEnd();
494 case json::kind::string: {
495 std::string str{json::serialize(doc)};
496 return this->onRawString(str);
497 }
498 case json::kind::double_:
499 return this->onDouble(doc.to_number<double>());
500 case json::kind::int64:
501 return this->onInt64(doc.to_number<std::int64_t>());
502 case json::kind::uint64:
503 return this->onUint64(doc.to_number<std::uint64_t>());
504 }
505 return true;
506 }
