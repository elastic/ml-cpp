53 template<typename VECTORS>
54 static bool orthonormalBasisImpl(VECTORS& vectors) {
55 if (vectors.empty()) {
56 return true;
57 }
58
59 int dimension{COrthogonaliser::dimension(vectors)};
60 if (dimension < 0) {
61 return false;
62 }
63
64 int rows{dimension};
65 int cols{static_cast<int>(vectors.size())};
66 CDenseMatrix<double> x(rows, cols);
67 for (std::size_t j = 0; j < vectors.size(); ++j) {
68 writeColumn(j, vectors[j], x);
69 }
70
71 auto qr = x.colPivHouseholderQr();
72 auto rank = qr.rank();
73 CDenseMatrix<double> thinQ{CDenseMatrix<double>::Identity(rows, rank)};
74 thinQ = qr.householderQ() * thinQ;
75
76 if (resize(vectors, rank) == false) {
77 return false;
78 }
79
80 for (long int j = 0; j < rank; ++j) {
81 readColumn(j, thinQ, vectors[j]);
82 }
83
84 return true;
85 }
