102 static std::string join(const CONTAINER& strings, const std::string& delimiter) {
103 if (strings.empty()) {
104 return std::string();
105 }
106 std::size_t requiredSpace = computeStringLength(strings.begin(), strings.end());
107 requiredSpace += (strings.size() - 1) * delimiter.length();
108 if (requiredSpace == 0) {
109 return std::string();
110 }
111 std::string output;
112 output.reserve(requiredSpace);
113 CStringUtils::join(strings.begin(), strings.end(), delimiter, output);
114 return output;
115 }
