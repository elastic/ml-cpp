93 template<typename T, std::size_t N>
94 bool operator()(const SDistributionRestoreParams& params,
95 const CClustererTypes::TSplitFunc& splitFunc,
96 const CClustererTypes::TMergeFunc& mergeFunc,
97 std::unique_ptr<CClusterer<CVectorNx1<T, N>>>& ptr,
98 core::CStateRestoreTraverser& traverser) const {
99 std::size_t numResults(0);
100
101 do {
102 const std::string& name = traverser.name();
103 if (name == CClustererTypes::X_MEANS_ONLINE_TAG) {
104 ptr.reset(CXMeansOnlineFactory::restore<T, N>(params, splitFunc,
105 mergeFunc, traverser));
106 ++numResults;
107 } else {
108 LOG_ERROR(<< "No clusterer corresponds to node name "
109 << traverser.name());
110 }
111 } while (traverser.next());
112
113 if (numResults != 1) {
114 LOG_ERROR(<< "Expected 1 (got " << numResults << ") clusterer tags");
115 ptr.reset();
116 return false;
117 }
118
119 return true;
120 }
