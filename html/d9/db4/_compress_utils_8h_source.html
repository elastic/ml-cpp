152 template<typename T>
153 bool processInput(bool finish, const T& input) {
154 if (input.empty() && m_State == E_Active && !finish) {
155 return true;
156 }
157
158 m_State = E_Active;
159
160 m_ZlibStrm.next_in = bytes(input);
161 m_ZlibStrm.avail_in = size(input);
162
163 int flush{finish ? Z_FINISH : Z_NO_FLUSH};
164 do {
165 if (this->processChunk(flush) == false) {
166 return false;
167 }
168 } while (m_ZlibStrm.avail_out == 0);
169
170 m_State = finish ? E_Finished : E_Active;
171
172 return true;
173 }
