40 CFuzzyTruthValue(double value = 0.5, double isTrueThreshold = 0.5)
41 : m_Value{value}, m_IsTrueThreshold{isTrueThreshold} {}
42
44 bool operator<(const CFuzzyTruthValue& rhs) const {
45 return rhs.m_IsTrueThreshold * m_Value < m_IsTrueThreshold * rhs.m_Value;
46 }
47
49 bool operator<=(const CFuzzyTruthValue& rhs) const {
50 return rhs.m_IsTrueThreshold * m_Value <= m_IsTrueThreshold * rhs.m_Value;
51 }
52
54 bool boolean() const { return m_Value > m_IsTrueThreshold; }
55
57 double value() const { return m_Value; }
58
60 friend inline CFuzzyTruthValue operator&&(const CFuzzyTruthValue& lhs,
61 const CFuzzyTruthValue& rhs) {
62 return {lhs.m_Value * rhs.m_Value, lhs.m_IsTrueThreshold * rhs.m_IsTrueThreshold};
63 }
64
66 friend inline CFuzzyTruthValue operator||(const CFuzzyTruthValue& lhs,
67 const CFuzzyTruthValue& rhs) {
68 // If possible rescale so both expressions have equal importance determining
69 // the expression truth value.
70 if (lhs.m_IsTrueThreshold < rhs.m_IsTrueThreshold && lhs.m_IsTrueThreshold > 0.0) {
71 double scale{lhs.m_IsTrueThreshold / rhs.m_IsTrueThreshold};
72 return {1.0 - (1.0 - lhs.m_Value) * (1.0 - scale * rhs.m_Value),
73 1.0 - (1.0 - lhs.m_IsTrueThreshold) * (1.0 - scale * rhs.m_IsTrueThreshold)};
74 }
75 if (rhs.m_IsTrueThreshold < lhs.m_IsTrueThreshold && rhs.m_IsTrueThreshold > 0.0) {
76 double scale{rhs.m_IsTrueThreshold / lhs.m_IsTrueThreshold};
77 return {1.0 - (1.0 - scale * lhs.m_Value) * (1.0 - rhs.m_Value),
78 1.0 - (1.0 - scale * lhs.m_IsTrueThreshold) * (1.0 - rhs.m_IsTrueThreshold)};
79 }
80 return {1.0 - (1.0 - lhs.m_Value) * (1.0 - rhs.m_Value),
81 1.0 - (1.0 - lhs.m_IsTrueThreshold) * (1.0 - rhs.m_IsTrueThreshold)};
82 }
83
85 std::string print() const {
86 return std::to_string(m_Value) + "/" + std::to_string(m_IsTrueThreshold);
87 }
88
89private:
90 double m_Value;
91 double m_IsTrueThreshold;
92};
