141 template<typename STRINGLIKE>
142 size_t levenshteinDistance(const STRINGLIKE& first, const STRINGLIKE& second) const {
143 // Levenshtein distance is the number of operations required to
144 // convert one string into another, where an operation means
145 // inserting 1 character, deleting 1 character or changing 1
146 // character.
147 //
148 // There are some examples with pretty pictures of the matrix on
149 // Wikipedia.
150 //
151 // This has been generalised to the case of vectors, where this
152 // method calculates the number of operations to convert one vector
153 // to another.
154
155 size_t firstLen(first.size());
156 size_t secondLen(second.size());
157
158 // The Berghel-Roach algorithm works in time O(n + d ^ 2), and the
159 // simple algorithm works in time O(m * n), where the shorter
160 // sequence length is m, the longer sequence length is m and the
161 // edit distance is d. Therefore the Berghel-Roach algorithm is
162 // much faster when the two sequences are similar, but in the case
163 // of them being very different, it's slower. Additionally, each
164 // operation it performs is much slower than the simple algorithm,
165 // because it has to do more complex calculations to get the matrix
166 // cell values. Therefore, we need a heuristic of when to use the
167 // Berghel-Roach algorithm and when to use the simple algorithm.
168 // The chosen heuristic is that if the longer sequence is double the
169 // length of the shorter sequence, we'll use the simple algorithm.
170
171 if (firstLen >= secondLen) {
172 // Rule out boundary case
173 if (secondLen == 0) {
174 return firstLen;
175 }
176
177 if (firstLen >= secondLen * 2) {
178 return this->levenshteinDistanceSimple(second, first);
179 }
180
181 return this->berghelRoachEditDistance(second, first);
182 }
183
184 if (secondLen >= firstLen * 2) {
185 // Rule out boundary case
186 if (firstLen == 0) {
187 return secondLen;
188 }
189
190 return this->levenshteinDistanceSimple(first, second);
191 }
192
193 return this->berghelRoachEditDistance(first, second);
194 }

212 template<typename PAIRCONTAINER>
213 size_t weightedEditDistance(const PAIRCONTAINER& first, const PAIRCONTAINER& second) const {
214 // This is similar to the levenshteinDistanceSimple() method below,
215 // but adding the concept of different costs for each element. If
216 // you are trying to understand this method, you should first make
217 // sure you fully understand the levenshteinDistance() method above
218 // (and the Wikipedia article referenced in it will help with that).
219
220 size_t firstLen(first.size());
221 size_t secondLen(second.size());
222
223 // Rule out boundary cases
224 if (firstLen == 0) {
225 size_t cost(0);
226 for (size_t index = 0; index < secondLen; ++index) {
227 cost += second[index].second;
228 }
229 return cost;
230 }
231
232 if (secondLen == 0) {
233 size_t cost(0);
234 for (size_t index = 0; index < firstLen; ++index) {
235 cost += first[index].second;
236 }
237 return cost;
238 }
239
240 // We need to store two columns of the matrix, but allocate both in
241 // one go for efficiency. Then the current and previous column
242 // pointers alternate between pointing and the first and second half
243 // of the memory block.
244 TScopedSizeArray data(new size_t[(secondLen + 1) * 2]);
245 size_t* currentCol(data.get());
246 size_t* prevCol(currentCol + (secondLen + 1));
247
248 // Populate the left column
249 currentCol[0] = 0;
250 for (size_t downMinusOne = 0; downMinusOne < secondLen; ++downMinusOne) {
251 currentCol[downMinusOne + 1] = currentCol[downMinusOne] +
252 second[downMinusOne].second;
253 }
254
255 // Calculate the other entries in the matrix
256 for (size_t acrossMinusOne = 0; acrossMinusOne < firstLen; ++acrossMinusOne) {
257 std::swap(currentCol, prevCol);
258 size_t firstCost(first[acrossMinusOne].second);
259 currentCol[0] = prevCol[0] + firstCost;
260
261 for (size_t downMinusOne = 0; downMinusOne < secondLen; ++downMinusOne) {
262 size_t secondCost(second[downMinusOne].second);
263
264 // There are 3 options, and due to the possible differences
265 // in the weightings, we must always evaluate all 3:
266
267 // 1) Deletion => cell to the left's value plus cost of
268 // deleting the element from the first sequence
269 size_t option1(prevCol[downMinusOne + 1] + firstCost);
270
271 // 2) Insertion => cell above's value plus cost of
272 // inserting the element from the second sequence
273 size_t option2(currentCol[downMinusOne] + secondCost);
274
275 // 3) Substitution => cell above left's value plus the
276 // higher of the two element weights
277 // OR
278 // No extra cost in the case where the corresponding
279 // elements are equal
280 size_t option3(
281 prevCol[downMinusOne] +
282 ((first[acrossMinusOne].first == second[downMinusOne].first)
283 ? 0
284 : std::max(firstCost, secondCost)));
285
286 // Take the cheapest option of the 3
287 currentCol[downMinusOne + 1] = std::min(std::min(option1, option2), option3);
288 }
289 }
290
291 // Result is the value in the bottom right hand corner of the matrix
292 return currentCol[secondLen];
293 }

306 template<typename STRINGLIKE>
307 size_t levenshteinDistanceSimple(const STRINGLIKE& first, const STRINGLIKE& second) const {
308 // This method implements the simple algorithm for calculating
309 // Levenshtein distance.
310 //
311 // There are some examples with pretty pictures of the matrix on
312 // Wikipedia here http://en.wikipedia.org/wiki/Levenshtein_distance
313
314 // It's best if secondLen >= firstLen. Although this uses more
315 // space for the array below, the total number of calculations will
316 // be the same, but the bigger array will make compiler
317 // optimisations such as loop unrolling and vectorisation more
318 // beneficial. Most internet pages will recommend the opposite,
319 // i.e. allocate the two arrays based on the size of the smaller
320 // sequence, but we're more interested in speed than space.
321 size_t firstLen(first.size());
322 size_t secondLen(second.size());
323
324 // We need to store two columns of the matrix, but allocate both in
325 // one go for efficiency. Then the current and previous column
326 // pointers alternate between pointing and the first and second half
327 // of the memory block.
328 TScopedSizeArray data(new size_t[(secondLen + 1) * 2]);
329 size_t* currentCol(data.get());
330 size_t* prevCol(currentCol + (secondLen + 1));
331
332 // Populate the left column
333 for (size_t down = 0; down <= secondLen; ++down) {
334 currentCol[down] = down;
335 }
336
337 // Calculate the other entries in the matrix
338 for (size_t acrossMinusOne = 0; acrossMinusOne < firstLen; ++acrossMinusOne) {
339 std::swap(currentCol, prevCol);
340 currentCol[0] = acrossMinusOne + 1;
341
342 for (size_t downMinusOne = 0; downMinusOne < secondLen; ++downMinusOne) {
343 // Do the strings differ at the point we've reached?
344 if (first[acrossMinusOne] == second[downMinusOne]) {
345 // No, they're the same => no extra cost
346
347 currentCol[downMinusOne + 1] = prevCol[downMinusOne];
348 } else {
349 // Yes, they differ, so there are 3 options:
350
351 // 1) Deletion => cell to the left's value plus 1
352 size_t option1(prevCol[downMinusOne + 1]);
353
354 // 2) Insertion => cell above's value plus 1
355 size_t option2(currentCol[downMinusOne]);
356
357 // 3) Substitution => cell above left's value plus 1
358 size_t option3(prevCol[downMinusOne]);
359
360 // Take the cheapest option of the 3
361 currentCol[downMinusOne + 1] =
362 std::min(std::min(option1, option2), option3) + 1;
363 }
364 }
365 }
366
367 // Result is the value in the bottom right hand corner of the matrix
368 return currentCol[secondLen];
369 }

375 template<typename STRINGLIKE>
376 size_t berghelRoachEditDistance(const STRINGLIKE& first, const STRINGLIKE& second) const {
377 // We need to do the calculation using signed variables
378 int shortLen(static_cast<int>(first.size()));
379 int maxDist(static_cast<int>(second.size()));
380
381 // Allocate the matrix memory, and setup pointers so that we can
382 // access it using negative arguments. This enables the
383 // implementation in this method to vaguely resemble the original
384 // paper.
385 TScopedIntArray dataArray;
386 TScopedIntPArray matrixArary;
387 int** matrix;
388 matrix = this->setupBerghelRoachMatrix(maxDist, dataArray, matrixArary);
389 if (matrix == nullptr) {
390 return 0;
391 }
392
393 // The remaining code corresponds to the pseudo-code in the
394 // sub-section titled "The Driver Algorithm" on
395 // http://berghel.net/publications/asm/asm.pdf
396
397 // k is the difference in lengths between the two sequences, i.e.
398 // the minimum distance
399 int k(maxDist - shortLen);
400 // p will end up storing the result
401 int p(k);
402 do {
403 int inc(p);
404 for (int tempP = 0; tempP < p; ++tempP, --inc) {
405 if (::abs(k - inc) <= tempP) {
406 this->calcDist(first, second, k - inc, tempP, matrix);
407 }
408 if (::abs(k + inc) <= tempP) {
409 this->calcDist(first, second, k + inc, tempP, matrix);
410 }
411 }
412 this->calcDist(first, second, k, p, matrix);
413
414 if (matrix[k][p] == shortLen) {
415 break;
416 }
417 } while (++p < maxDist);
418
419 return static_cast<size_t>(p);
420 }

424 template<typename STRINGLIKE>
425 void calcDist(const STRINGLIKE& first, const STRINGLIKE& second, int row, int column, int** matrix) const {
426 // 1) Substitution
427 int option1(matrix[row][column - 1] + 1);
428
429 // NB: Unlike the Berghel-Roach paper, we DO NOT consider
430 // transposition at this point
431
432 // 2) Insertion
433 int option2(matrix[row - 1][column - 1]);
434
435 // 3) Deletion
436 int option3(matrix[row + 1][column - 1] + 1);
437
438 int t(std::max(std::max(option1, option2), option3));
439 int limit(std::min(static_cast<int>(first.size()),
440 static_cast<int>(second.size()) - row));
441 while (t < limit && first[t] == second[t + row]) {
442 ++t;
443 }
444 matrix[row][column] = t;
445 }
