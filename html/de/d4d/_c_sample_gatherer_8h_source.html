113 void acceptPersistInserter(core::CStatePersistInserter& inserter) const {
114 inserter.insertValue(DIMENSION_TAG, m_Dimension);
115 inserter.insertLevel(CLASSIFIER_TAG,
116 std::bind(&CDataClassifier::acceptPersistInserter,
117 &m_Classifier, std::placeholders::_1));
118 if (m_SampleStats.empty() == false) {
119 inserter.insertLevel(SAMPLE_STATS_TAG,
120 std::bind(&TSampleQueue::acceptPersistInserter,
121 &m_SampleStats, std::placeholders::_1));
122 }
123 if (m_BucketStats.empty() == false) {
124 inserter.insertLevel(
125 BUCKET_STATS_TAG,
126 std::bind<void>(TStatBucketQueueSerializer(TMetricPartialStatistic(m_Dimension)),
127 std::cref(m_BucketStats), std::placeholders::_1));
128 }
129 for (const auto& stats : m_InfluencerBucketStats) {
130 inserter.insertLevel(
131 INFLUENCER_BUCKET_STATS_TAG,
132 std::bind<void>(TOptionalStrStatUMapBucketQueueSerializer(
133 TOptionalStrStatUMap(1),
134 COptionalStrStatUMapSerializer(m_Dimension)),
135 std::cref(stats), std::placeholders::_1));
136 }
137 }

140 bool acceptRestoreTraverser(core::CStateRestoreTraverser& traverser) {
141 std::size_t i = 0;
142 do {
143 const std::string& name = traverser.name();
144 TMetricPartialStatistic stat(m_Dimension);
145 RESTORE_BUILT_IN(DIMENSION_TAG, m_Dimension)
146 RESTORE(CLASSIFIER_TAG, traverser.traverseSubLevel(std::bind(
147 &CDataClassifier::acceptRestoreTraverser,
148 &m_Classifier, std::placeholders::_1)))
149 RESTORE(SAMPLE_STATS_TAG, traverser.traverseSubLevel(std::bind(
150 &TSampleQueue::acceptRestoreTraverser,
151 &m_SampleStats, std::placeholders::_1)))
152 RESTORE(BUCKET_STATS_TAG,
153 traverser.traverseSubLevel(std::bind<bool>(
154 TStatBucketQueueSerializer(TMetricPartialStatistic(m_Dimension)),
155 std::ref(m_BucketStats), std::placeholders::_1)))
156 RESTORE(INFLUENCER_BUCKET_STATS_TAG,
157 i < m_InfluencerBucketStats.size() &&
158 traverser.traverseSubLevel(std::bind<bool>(
159 TOptionalStrStatUMapBucketQueueSerializer(
160 TOptionalStrStatUMap(1), COptionalStrStatUMapSerializer(m_Dimension)),
161 std::ref(m_InfluencerBucketStats[i++]), std::placeholders::_1)))
162 } while (traverser.next());
163 return true;
164 }

176 SMetricFeatureData featureData(core_t::TTime time,
177 core_t::TTime /*bucketLength*/,
178 double effectiveSampleCount) const {
179 const TMetricPartialStatistic& bucketPartial = m_BucketStats.get(time);
180 double count = bucketPartial.count();
181 if (count > 0.0) {
182 core_t::TTime bucketTime = bucketPartial.time();
183 TDouble1Vec bucketValue = bucketPartial.value();
184 if (bucketValue.size() > 0) {
185 TStrCRefDouble1VecDoublePrPrVecVec influenceValues(
186 m_InfluencerBucketStats.size());
187 for (std::size_t i = 0; i < m_InfluencerBucketStats.size(); ++i) {
188 const TOptionalStrStatUMap& influencerStats =
189 m_InfluencerBucketStats[i].get(time);
190 influenceValues[i].reserve(influencerStats.size());
191 for (const auto& stat : influencerStats) {
192 influenceValues[i].emplace_back(
193 std::cref(*stat.first),
194 std::make_pair(
195 CMetricStatisticWrappers::influencerValue(stat.second),
196 CMetricStatisticWrappers::count(stat.second)));
197 }
198 }
199 return {bucketTime,
200 bucketValue,
201 model_t::varianceScale(FEATURE, effectiveSampleCount, count),
202 count,
203 influenceValues,
204 m_Classifier.isInteger(),
205 m_Classifier.isNonNegative(),
206 m_Samples};
207 }
208 }
209 return {m_Classifier.isInteger(), m_Classifier.isNonNegative(), m_Samples};
210 }

249 void add(core_t::TTime time,
250 const TDouble1Vec& statistic,
251 unsigned int count,
252 unsigned int sampleCount,
253 const TOptionalStrVec& influences) {
254 if (sampleCount > 0) {
255 m_SampleStats.add(time, statistic, count, sampleCount);
256 }
257 m_BucketStats.get(time).add(statistic, time, count);
258 m_Classifier.add(FEATURE, statistic, count);
259 std::size_t n = std::min(influences.size(), m_InfluencerBucketStats.size());
260 for (std::size_t i = 0; i < n; ++i) {
261 if (!influences[i]) {
262 continue;
263 }
264 TOptionalStrStatUMap& stats = m_InfluencerBucketStats[i].get(time);
265 auto j = stats
266 .emplace(influences[i],
267 CMetricStatisticWrappers::template make<STATISTIC>(m_Dimension))
268 .first;
269 CMetricStatisticWrappers::add(statistic, count, j->second);
270 }
271 }

292 bool isRedundant(core_t::TTime samplingCutoffTime) const {
293 if (m_SampleStats.latestEnd() >= samplingCutoffTime) {
294 return false;
295 }
296 for (const auto& bucket : m_BucketStats) {
297 if (bucket.count() > 0.0) {
298 return false;
299 }
300 }
301 return true;

305 uint64_t checksum() const {
306 uint64_t seed = static_cast<uint64_t>(m_Classifier.isInteger());
307 seed = maths::common::CChecksum::calculate(seed, m_Classifier.isNonNegative());
308 seed = maths::common::CChecksum::calculate(seed, m_SampleStats);
309 seed = maths::common::CChecksum::calculate(seed, m_BucketStats);
310 return maths::common::CChecksum::calculate(seed, m_InfluencerBucketStats);
311 }

361 class COptionalStrStatUMapSerializer {
362 public:
363 COptionalStrStatUMapSerializer(std::size_t dimension)
364 : m_Initial(CMetricStatisticWrappers::template make<STATISTIC>(dimension)) {}
365
366 void operator()(const TOptionalStrStatUMap& map,
367 core::CStatePersistInserter& inserter) const {
368 using TStatCRef = std::reference_wrapper<const STATISTIC>;
369 using TStrCRefStatCRefPr = std::pair<TStrCRef, TStatCRef>;
370 using TStrCRefStatCRefPrVec = std::vector<TStrCRefStatCRefPr>;
371 TStrCRefStatCRefPrVec ordered;
372 ordered.reserve(map.size());
373 for (const auto& stat : map) {
374 ordered.emplace_back(TStrCRef(*stat.first), TStatCRef(stat.second));
375 }
376 std::sort(ordered.begin(), ordered.end(),
377 maths::common::COrderings::SFirstLess());
378 for (const auto& stat : ordered) {
379 inserter.insertValue(MAP_KEY_TAG, stat.first);
380 CMetricStatisticWrappers::persist(stat.second.get(), MAP_VALUE_TAG, inserter);
381 }
382 }
383
384 bool operator()(TOptionalStrStatUMap& map,
385 core::CStateRestoreTraverser& traverser) const {
386 std::string key;
387 do {
388 const std::string& name = traverser.name();
389 RESTORE_NO_ERROR(MAP_KEY_TAG, key = traverser.value())
390 RESTORE(MAP_VALUE_TAG,
391 CMetricStatisticWrappers::restore(
392 traverser, map.insert({key, m_Initial}).first->second))
393 } while (traverser.next());
394 return true;
395 }
396
397 private:
398 STATISTIC m_Initial;
399 };
400 using TOptionalStrStatUMapBucketQueueSerializer =
401 typename TOptionalStrStatUMapBucketQueue::template CSerializer<COptionalStrStatUMapSerializer>;
