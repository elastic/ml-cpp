143template<typename T>
144class CTimeSeriesMultibucketMean final : public CTimeSeriesMultibucketFeature<T> {
145public:
146 using TT1Vec = typename CTimeSeriesMultibucketFeature<T>::TT1Vec;
147 using TWeightsAry1Vec = typename CTimeSeriesMultibucketFeature<T>::TWeightsAry1Vec;
148 using TT1VecTWeightAry1VecPr = typename CTimeSeriesMultibucketFeature<T>::TT1VecTWeightAry1VecPr;
149 using TPtr = typename CTimeSeriesMultibucketFeature<T>::TPtr;
150
151public:
152 explicit CTimeSeriesMultibucketMean(std::size_t length = 0)
153 : m_SlidingWindow(length) {}
154 ~CTimeSeriesMultibucketMean() override = default;
155
157 TPtr clone() const override {
158 return std::make_unique<CTimeSeriesMultibucketMean>(*this);
159 }
160
162 TT1VecTWeightAry1VecPr value() const override {
163 if (4 * m_SlidingWindow.size() >= 3 * m_SlidingWindow.capacity()) {
164 return {{this->mean()}, {maths_t::countWeight(this->count())}};
165 }
166 return {{}, {}};
167 }
168
170 double correlationWithBucketValue() const override {
171 // This follows from the weighting applied to values in the
172 // window and linearity of expectation.
173 double r{WINDOW_GEOMETRIC_WEIGHT * WINDOW_GEOMETRIC_WEIGHT};
174 double length{static_cast<double>(m_SlidingWindow.size())};
175 return length == 0.0 ? 0.0 : std::sqrt((1.0 - r) / (1.0 - std::pow(r, length)));
176 }
177
179 void clear() override { m_SlidingWindow.clear(); }
180
182 void add(core_t::TTime time,
183 core_t::TTime bucketLength,
184 const TT1Vec& values,
185 const TWeightsAry1Vec& weights) override {
186 // Remove any old samples.
187 core_t::TTime cutoff{time - this->windowInterval(bucketLength)};
188 while (m_SlidingWindow.size() > 0 && m_SlidingWindow.front().first < cutoff) {
189 m_SlidingWindow.pop_front();
190 }
191 if (values.size() > 0) {
192 using TStorage1Vec = core::CSmallVector<TStorage, 1>;
193
194 // Get the scales to apply to each value.
195 TStorage1Vec scales;
196 scales.reserve(weights.size());
197 for (const auto& weight : weights) {
198 using std::sqrt;
199 scales.push_back(sqrt(TStorage(maths_t::countVarianceScale(weight)) *
200 TStorage(maths_t::seasonalVarianceScale(weight))));
201 }
202
203 // Add the next sample.
204 TFloatMeanAccumulator next{this->conformable(TStorage(values[0]), 0.0F)};
205 for (std::size_t i = 0; i < values.size(); ++i) {
206 auto weight = maths_t::countForUpdate(weights[i]);
207 next.add(TStorage(values[i]) / scales[i], this->minweight(weight));
208 }
209 m_SlidingWindow.push_back({time, next});
210 }
211 }
212
214 std::uint64_t checksum(std::uint64_t seed = 0) const override {
215 seed = common::CChecksum::calculate(seed, m_SlidingWindow.capacity());
216 return common::CChecksum::calculate(seed, m_SlidingWindow);
217 }
218
220 void debugMemoryUsage(const core::CMemoryUsage::TMemoryUsagePtr& mem) const override {
221 mem->setName("CTimeSeriesMultibucketMean");
222 core::memory_debug::dynamicSize("m_SlidingWindow", m_SlidingWindow, mem);
223 }
224
226 std::size_t staticSize() const override { return sizeof(*this); }
227
229 std::size_t memoryUsage() const override {
230 return core::memory::dynamicSize(m_SlidingWindow);
231 }
232
234 bool acceptRestoreTraverser(core::CStateRestoreTraverser& traverser) override {
235 do {
236 const std::string& name{traverser.name()};
237 RESTORE_SETUP_TEARDOWN(CAPACITY_TAG, std::size_t capacity,
238 core::CStringUtils::stringToType(traverser.value(), capacity),
239 m_SlidingWindow.set_capacity(capacity))
240 RESTORE(SLIDING_WINDOW_TAG,
241 core::CPersistUtils::restore(SLIDING_WINDOW_TAG, m_SlidingWindow, traverser))
242 } while (traverser.next());
243 return true;
244 }
245
247 void acceptPersistInserter(core::CStatePersistInserter& inserter) const override {
248 inserter.insertValue(CAPACITY_TAG, m_SlidingWindow.capacity());
249 core::CPersistUtils::persist(SLIDING_WINDOW_TAG, m_SlidingWindow, inserter);
250 }
251
252private:
253 using TDoubleDoublePr = std::pair<double, double>;
254 using TStorage =
255 typename std::conditional<std::is_arithmetic<T>::value, common::CFloatStorage, common::CVector<common::CFloatStorage>>::type;
256 using TFloatMeanAccumulator =
257 typename common::CBasicStatistics::SSampleMean<TStorage>::TAccumulator;
258 using TDoubleMeanAccumulator = typename common::SPromoted<TFloatMeanAccumulator>::Type;
259 using TTimeFloatMeanAccumulatorPr = std::pair<core_t::TTime, TFloatMeanAccumulator>;
260 using TTimeFloatMeanAccumulatorPrCBuf = boost::circular_buffer<TTimeFloatMeanAccumulatorPr>;
261
262private:
264 constexpr static const double WINDOW_GEOMETRIC_WEIGHT = 0.9;
266 static const std::string CAPACITY_TAG;
267 static const std::string SLIDING_WINDOW_TAG;
268
269private:
271 core_t::TTime windowInterval(core_t::TTime bucketLength) const {
272 return static_cast<core_t::TTime>(m_SlidingWindow.capacity()) * bucketLength;
273 }
274
276 T mean() const {
277 return evaluateOnWindow(
278 [](const TFloatMeanAccumulator& value) {
279 return common::CBasicStatistics::mean(value);
280 },
281 [](const TFloatMeanAccumulator& value) {
282 return common::CBasicStatistics::count(value);
283 });
284 }
285
287 T count() const {
288 return evaluateOnWindow(
289 [&](const TFloatMeanAccumulator& value) {
290 return this->conformable(
291 common::CBasicStatistics::mean(value),
292 static_cast<double>(common::CBasicStatistics::count(value)));
293 },
294 [](const TFloatMeanAccumulator&) { return 1.0; });
295 }
296
298 template<typename VALUE, typename WEIGHT>
299 T evaluateOnWindow(VALUE value, WEIGHT weight) const {
300 auto[earliest, latest] = this->range();
301 double n{static_cast<double>(m_SlidingWindow.size())};
302 double scale{(n - 1.0) * (latest == earliest ? 1.0 : 1.0 / (latest - earliest))};
303 auto i = m_SlidingWindow.begin();
304 TDoubleMeanAccumulator mean{this->conformable(value(i->second), 0.0)};
305 for (double last{earliest}; i != m_SlidingWindow.end(); ++i) {
306 double dt{static_cast<double>(i->first) - last};
307 last = static_cast<double>(i->first);
308 mean.age(std::pow(WINDOW_GEOMETRIC_WEIGHT, scale * dt));
309 mean.add(value(i->second), weight(i->second));
310 }
311 return this->toVector(common::CBasicStatistics::mean(mean));
312 }
313
315 TDoubleDoublePr range() const {
316 auto range =
317 std::accumulate(m_SlidingWindow.begin(), m_SlidingWindow.end(),
318 common::CBasicStatistics::CMinMax<double>(),
319 [](common::CBasicStatistics::CMinMax<double> partial,
320 const TTimeFloatMeanAccumulatorPr& value) {
321 partial.add(static_cast<double>(value.first));
322 return partial;
323 });
324 return {range.min(), range.max()};
325 }
326
327private:
329 TTimeFloatMeanAccumulatorPrCBuf m_SlidingWindow;
330};
