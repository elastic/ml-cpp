188 CMultivariateMultimodalPrior(const CMultivariateMultimodalPrior& other)
189 : CMultivariatePrior(other.dataType(), other.decayRate()),
190 m_Clusterer(other.m_Clusterer->clone()),
191 m_SeedPrior(other.m_SeedPrior->clone()) {
192 // Register the split and merge callbacks.
193 m_Clusterer->splitFunc(CModeSplitCallback(*this));
194 m_Clusterer->mergeFunc(CModeMergeCallback(*this));

196 // Clone all the modes up front so we can implement strong exception safety.

201 }
202 m_Modes.swap(modes);
203
204 this->addSamples(other.numberSamples());
205 }

297 void addSamples(const TDouble10Vec1Vec& samples,
298 const TDouble10VecWeightsAry1Vec& weights) override {
299 if (samples.empty()) {
300 return;
301 }
302 if (!this->check(samples, weights)) {
303 return;
304 }
305
306 // See CMultimodalPrior::addSamples for discussion.
307
308 using TSizeDoublePr2Vec = core::CSmallVector<TSizeDoublePr, 2>;
309
310 // Declared outside the loop to minimize the number of times it
311 // is initialized.
312 TDouble10Vec1Vec sample(1);
313 TDouble10VecWeightsAry1Vec weight{TWeights::unit<TDouble10Vec>(N)};
314 TSizeDoublePr2Vec clusters;
315
316 try {
317 bool hasSeasonalScale = !this->isNonInformative() &&
318 maths_t::hasSeasonalVarianceScale(weights);
319
320 TPoint mean = hasSeasonalScale ? this->mean() : TPoint(0.0);
321
322 for (std::size_t i = 0; i < samples.size(); ++i) {
323 TPoint x(samples[i]);
324 if (!CMathsFuncs::isFinite(x)) {
325 LOG_ERROR(<< "Discarding " << x);
326 continue;
327 }
328
329 if (hasSeasonalScale) {
330 TPoint seasonalScale =
331 sqrt(TPoint(maths_t::seasonalVarianceScale(weights[i])));
332 x = mean + (x - mean) / seasonalScale;
333 }
334
335 sample[0] = x.template toVector<TDouble10Vec>();
336 weight[0] = weights[i];
337 maths_t::setSeasonalVarianceScale(1.0, N, weight[0]);
338
339 double smallestCountWeight = this->smallest(maths_t::count(weight[0]));
340 clusters.clear();
341 m_Clusterer->add(x, clusters, smallestCountWeight);
342
343 double Z = std::accumulate(m_Modes.begin(), m_Modes.end(), smallestCountWeight,
344 [](double sum, const TMode& mode) {
345 return sum + mode.weight();
346 });
347
348 double n = 0.0;
349 for (const auto& cluster : clusters) {
350 auto k = std::find_if(m_Modes.begin(), m_Modes.end(),
351 CSetTools::CIndexInSet(cluster.first));
352 if (k == m_Modes.end()) {
353 LOG_TRACE(<< "Creating mode with index " << cluster.first);
354 m_Modes.emplace_back(cluster.first, m_SeedPrior);
355 k = m_Modes.end() - 1;
356 }
357 maths_t::setCount(cluster.second, N, weight[0]);
358 if (maths_t::isWinsorised(weight)) {
359 TDouble10Vec ww = maths_t::outlierWeight(weight[0]);
360 double f = (k->weight() + cluster.second) / Z;
361 for (auto& w : ww) {
362 w = std::max(1.0 - (1.0 - w) / f, w * f);
363 }
364 maths_t::setOutlierWeight(ww, weight[0]);
365 }
366 k->s_Prior->addSamples(sample, weight);
367 n += this->smallest(maths_t::countForUpdate(weight[0]));
368 }
369 this->addSamples(n);
370 }
371 } catch (const std::exception& e) {
372 LOG_ERROR(<< "Failed to update likelihood: " << e.what());
373 }
374 }

377 void propagateForwardsByTime(double time) override {
378 if (CMathsFuncs::isFinite(time) == false || time < 0.0) {
379 LOG_ERROR(<< "Bad propagation time " << time);
380 return;
381 }
382 if (this->isNonInformative()) {
383 // Nothing to be done.
384 return;
385 }
386
387 // We want to hold the probabilities constant. Since the i'th
388 // probability:
389 // p(i) = w(i) / Sum_j{ w(j) }
390 //
391 // where w(i) is its weight we can achieve this by multiplying
392 // all weights by some factor f in the range [0, 1].
393
394 m_Clusterer->propagateForwardsByTime(time);
395 for (const auto& mode : m_Modes) {
396 mode.s_Prior->propagateForwardsByTime(time);
397 }
398
399 // Remove any mode which is non-informative.
400 while (m_Modes.size() > 1) {
401 // Calling remove with the mode's index triggers a callback
402 // which also removes it from s_Modes, see CModeMergeCallback.
403 auto i = std::find_if(m_Modes.begin(), m_Modes.end(), [](const auto& mode) {
404 return mode.s_Prior->isNonInformative();
405 });
406 if (i == m_Modes.end() || m_Clusterer->remove(i->s_Index) == false) {
407 break;
408 }
409 }
410
411 this->numberSamples(this->numberSamples() *
412 std::exp(-this->scaledDecayRate() * time));
413 LOG_TRACE(<< "numberSamples = " << this->numberSamples());
414 }

427 TUnivariatePriorPtrDoublePr univariate(const TSize10Vec& marginalize,
428 const TSizeDoublePr10Vec& condition) const override {
429 std::size_t n = m_Modes.size();
430
431 CMultimodalPrior::TPriorPtrVec modes;
432 TDouble5Vec weights;
433 CBasicStatistics::SMax<double>::TAccumulator maxWeight;
434 modes.reserve(n);
435 weights.reserve(n);
436
437 for (const auto& mode : m_Modes) {
438 TUnivariatePriorPtrDoublePr prior(mode.s_Prior->univariate(marginalize, condition));
439 if (prior.first == nullptr) {
440 return {};
441 }
442 if (prior.first->isNonInformative()) {
443 continue;
444 }
445 modes.push_back(std::move(prior.first));
446 weights.push_back(prior.second);
447 maxWeight.add(weights.back());
448 }
449
450 double Z = 0.0;
451 for (auto& weight : weights) {
452 weight = std::exp(weight - maxWeight[0]);
453 Z += weight;
454 }
455 for (std::size_t i = 0; i < weights.size(); ++i) {
456 modes[i]->numberSamples(weights[i] / Z * modes[i]->numberSamples());
457 }
458
459 return {std::make_unique<CMultimodalPrior>(this->dataType(), this->decayRate(), modes),
460 Z > 0.0 ? maxWeight[0] + std::log(Z) : 0.0};
461 }

475 TPriorPtrDoublePr bivariate(const TSize10Vec& marginalize,
476 const TSizeDoublePr10Vec& condition) const override {
477 if (N == 2) {
478 return {TPriorPtr(this->clone()), 0.0};
479 }

481 std::size_t n = m_Modes.size();
482
483 TPriorPtrVec modes;
484 TDouble5Vec weights;
485 modes.reserve(n);
486 weights.reserve(n);
487 CBasicStatistics::SMax<double>::TAccumulator maxWeight;
488
489 for (const auto& mode : m_Modes) {
490 TPriorPtrDoublePr prior(mode.s_Prior->bivariate(marginalize, condition));
491 if (prior.first == nullptr) {
492 return TPriorPtrDoublePr();

494 if (prior.first->isNonInformative()) {
495 continue;
496 }
497 modes.push_back(std::move(prior.first));
498 weights.push_back(prior.second);
499 maxWeight.add(weights.back());
500 }
501
502 double Z = 0.0;
503 for (auto& weight : weights) {
504 weight = std::exp(weight - maxWeight[0]);
505 Z += weight;
506 }
507 for (std::size_t i = 0; i < weights.size(); ++i) {
508 modes[i]->numberSamples(weights[i] / Z * modes[i]->numberSamples());

510
511 return {std::make_unique<CMultivariateMultimodalPrior<2>>(this->dataType(), modes),

516 TDouble10VecDouble10VecPr marginalLikelihoodSupport() const override {
517 if (m_Modes.empty()) {
518 return {TPoint::smallest().template toVector<TDouble10Vec>(),
519 TPoint::largest().template toVector<TDouble10Vec>()};
520 }
521 if (m_Modes.size() == 1) {
522 return m_Modes[0].s_Prior->marginalLikelihoodSupport();
523 }
524
525 TPoint lower = TPoint::largest();
526 TPoint upper = TPoint::smallest();
527
528 // We define this is as the union of the mode supports.
529 for (const auto& mode : m_Modes) {
530 TDouble10VecDouble10VecPr s = mode.s_Prior->marginalLikelihoodSupport();
531 lower = min(lower, TPoint(s.first));
532 upper = max(upper, TPoint(s.second));
533 }
534
535 return {lower.template toVector<TDouble10Vec>(),
536 upper.template toVector<TDouble10Vec>()};
537 }

577 TDouble10Vec marginalLikelihoodMode(const TDouble10VecWeightsAry& weight) const override {
578
579 if (m_Modes.empty()) {
580 return TDouble10Vec(N, 0.0);
581 }
582 if (m_Modes.size() == 1) {
583 return m_Modes[0].s_Prior->marginalLikelihoodMode(weight);
584 }
585
586 using TMaxAccumulator = CBasicStatistics::SMax<double>::TAccumulator;
587
588 // We'll approximate this as the mode with the maximum likelihood.
589 TPoint result(0.0);
590
591 TPoint seasonalScale = sqrt(TPoint(maths_t::seasonalVarianceScale(weight)));
592 TDouble10VecWeightsAry1Vec weight_{TWeights::unit<TDouble10Vec>(N)};
593 maths_t::setCountVarianceScale(maths_t::countVarianceScale(weight), weight_[0]);
594
595 // Declared outside the loop to minimize number of times it is created.
596 TDouble10Vec1Vec mode(1);
597
598 TMaxAccumulator modeLikelihood;
599 for (const auto& mode_ : m_Modes) {
600 double w = mode_.weight();
601 const TPriorPtr& prior = mode_.s_Prior;
602 mode[0] = prior->marginalLikelihoodMode(weight_[0]);
603 double likelihood;
604 if ((prior->jointLogMarginalLikelihood(mode, weight_, likelihood) &
605 maths_t::E_FpAllErrors) != 0) {
606 continue;
607 }
608 if (modeLikelihood.add(std::log(w) + likelihood)) {
609 result = TPoint(mode[0]);
610 }
611 }
612
613 TPoint mean = this->mean();
614 result = mean + seasonalScale * (result - mean);
615 return result.template toVector<TDouble10Vec>();
616 }

656 maths_t::EFloatingPointErrorStatus
657 jointLogMarginalLikelihood(const TDouble10Vec1Vec& samples,
658 const TDouble10VecWeightsAry1Vec& weights,
659 double& result) const override {
660 result = 0.0;
661
662 if (samples.empty()) {
663 LOG_ERROR(<< "Can't compute likelihood for empty sample set");
664 return maths_t::E_FpFailed;
665 }
666 if (!this->check(samples, weights)) {
667 return maths_t::E_FpFailed;
668 }
669 if (this->isNonInformative()) {
670 // The non-informative likelihood is improper and effectively
671 // zero everywhere. We use minus max double because
672 // log(0) = HUGE_VALUE, which causes problems for Windows.
673 // Calling code is notified when the calculation overflows
674 // and should avoid taking the exponential since this will
675 // underflow and pollute the floating point environment. This
676 // may cause issues for some library function implementations
677 // (see fe*exceptflag for more details).
678 result = std::numeric_limits<double>::lowest();
679 return maths_t::E_FpOverflowed;
680 }
681
682 if (m_Modes.size() == 1) {
683 // Apply a small penalty to kill off this model if the data are
684 // single mode.
685 maths_t::EFloatingPointErrorStatus status =
686 m_Modes[0].s_Prior->jointLogMarginalLikelihood(samples, weights, result);
687 result -= 10.0 * this->decayRate();
688 return status;
689 }
690
691 // See CMultimodalPrior::jointLogMarginalLikelihood for discussion.
692
693 namespace detail = multivariate_multimodal_prior_detail;
694
695 // Declared outside the loop to minimize number of times it is created.
696 TDouble10Vec1Vec sample(1);
697 detail::TSizeDoublePr3Vec modeLogLikelihoods;
698 modeLogLikelihoods.reserve(m_Modes.size());
699
700 TPoint mean = maths_t::hasSeasonalVarianceScale(weights) ? this->mean()
701 : TPoint(0.0);
702
703 TDouble10VecWeightsAry1Vec weight{TWeights::unit<TDouble10Vec>(N)};
704 try {
705 for (std::size_t i = 0; i < samples.size(); ++i) {
706 double n = this->smallest(maths_t::countForUpdate(weights[i]));
707 TPoint seasonalScale =
708 sqrt(TPoint(maths_t::seasonalVarianceScale(weights[i])));
709 double logSeasonalScale = 0.0;
710 for (std::size_t j = 0; j < seasonalScale.dimension(); ++j) {
711 logSeasonalScale += std::log(seasonalScale(j));
712 }
713
714 TPoint x(samples[i]);
715 x = mean + (x - mean) / seasonalScale;
716 sample[0] = x.template toVector<TDouble10Vec>();
717 maths_t::setCountVarianceScale(
718 maths_t::countVarianceScale(weights[i]), weight[0]);
719
720 double sampleLogLikelihood;
721 maths_t::EFloatingPointErrorStatus status = detail::jointLogMarginalLikelihood(
722 m_Modes, sample, weight, modeLogLikelihoods, sampleLogLikelihood);
723 if ((status & maths_t::E_FpOverflowed) != 0) {
724 result = std::numeric_limits<double>::lowest();
725 return status;
726 }
727 if ((status & maths_t::E_FpFailed) != 0) {
728 return status;
729 }
730 result += n * (sampleLogLikelihood - logSeasonalScale);
731 }
732 } catch (const std::exception& e) {
733 LOG_ERROR(<< "Failed to compute likelihood: " << e.what());
734 return maths_t::E_FpFailed;
735 }
736
737 LOG_TRACE(<< "Joint log likelihood = " << result);
738
739 maths_t::EFloatingPointErrorStatus status = CMathsFuncs::fpStatus(result);
740 if ((status & maths_t::E_FpFailed) != 0) {
741 LOG_ERROR(<< "Failed to compute likelihood (" << this->debugWeights() << ")");

888 void operator()(std::size_t sourceIndex,
889 std::size_t leftSplitIndex,
890 std::size_t rightSplitIndex) const {
891
892 LOG_TRACE(<< "Splitting mode with index " << sourceIndex);
893
894 TModeVec& modes = m_Prior->m_Modes;
895
896 // Remove the split mode.
897 auto mode = std::find_if(modes.begin(), modes.end(),
898 CSetTools::CIndexInSet(sourceIndex));
899 double numberSamples = mode != modes.end() ? mode->weight() : 0.0;
900 modes.erase(mode);
901
902 double pLeft = m_Prior->m_Clusterer->probability(leftSplitIndex);
903 double pRight = m_Prior->m_Clusterer->probability(rightSplitIndex);
904 double Z = pLeft + pRight;
905 if (Z > 0.0) {
906 pLeft /= Z;
907 pRight /= Z;
908 }
909 LOG_TRACE(<< "# samples = " << numberSamples
910 << ", pLeft = " << pLeft << ", pRight = " << pRight);
911
912 // Create the child modes.
913 LOG_TRACE(<< "Creating mode with index " << leftSplitIndex);
914 modes.emplace_back(leftSplitIndex, TPriorPtr(m_Prior->m_SeedPrior->clone()));
915 {
916 TPointVec samples;
917 if (!m_Prior->m_Clusterer->sample(
918 leftSplitIndex, MODE_SPLIT_NUMBER_SAMPLES, samples)) {
919 LOG_ERROR(<< "Couldn't find cluster for " << leftSplitIndex);
920 }
921 LOG_TRACE(<< "samples = " << samples);
922
923 double wl = pLeft * numberSamples;
924 double ws = std::min(wl, static_cast<double>(N + 2));
925 double n = static_cast<double>(samples.size());
926 LOG_TRACE(<< "# left = " << wl);
927
928 TDouble10Vec1Vec samples_;
929 samples_.reserve(samples.size());
930 for (const auto& sample : samples) {
931 samples_.push_back(sample.template toVector<TDouble10Vec>());
932 }
933 TDouble10VecWeightsAry1Vec weights(samples_.size(),
934 maths_t::countWeight(ws / n, N));
935 modes.back().s_Prior->addSamples(samples_, weights);
936 if (wl > ws) {
937 weights.assign(weights.size(), maths_t::countWeight((wl - ws) / n, N));
938 modes.back().s_Prior->addSamples(samples_, weights);
939 LOG_TRACE(<< modes.back().s_Prior->print());
940 }
941 }
942
943 LOG_TRACE(<< "Creating mode with index " << rightSplitIndex);
944 modes.emplace_back(rightSplitIndex, TPriorPtr(m_Prior->m_SeedPrior->clone()));
945 {
946 TPointVec samples;
947 if (!m_Prior->m_Clusterer->sample(
948 rightSplitIndex, MODE_SPLIT_NUMBER_SAMPLES, samples)) {
949 LOG_ERROR(<< "Couldn't find cluster for " << rightSplitIndex);
950 }
951 LOG_TRACE(<< "samples = " << samples);
952
953 double wr = pRight * numberSamples;
954 double ws = std::min(wr, static_cast<double>(N + 2));
955 double n = static_cast<double>(samples.size());
956 LOG_TRACE(<< "# right = " << wr);
957
958 TDouble10Vec1Vec samples_;
959 samples_.reserve(samples.size());
960 for (const auto& sample : samples) {
961 samples_.push_back(sample.template toVector<TDouble10Vec>());
962 }
963 TDouble10VecWeightsAry1Vec weights(samples_.size(),
964 maths_t::countWeight(ws / n, N));
965 modes.back().s_Prior->addSamples(samples_, weights);
966 if (wr > ws) {
967 weights.assign(weights.size(), maths_t::countWeight((wr - ws) / n, N));

969 LOG_TRACE(<< modes.back().s_Prior->print());
970 }
971 }

973 LOG_TRACE(<< m_Prior->print());
974 LOG_TRACE(<< "Split mode");

1085 TMatrix modeVariance(mode.s_Prior->marginalLikelihoodCovariance());
1086 result.add(modeMean.outer() - mean2 + modeVariance, weight);
1087 }
