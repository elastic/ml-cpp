97 bool fromDelimited(const std::string& delimited) {
98 // Expect N strings separated by commas.
99 std::size_t n{0};
100 std::size_t pos{0};
101 std::size_t comma{0};
102 while ((comma = delimited.find(DELIMITER, pos)) != std::string::npos && n < N) {
103 CStringUtils::stringToType(delimited.substr(pos, comma - pos),
104 m_Hash[n++]);
105 pos = comma + 1;
106 }
107 if (n < N) {
108 CStringUtils::stringToType(delimited.substr(pos, comma - pos),
109 m_Hash[n++]);
110 }
111 return n == N;
112 }
113
114 std::string toDelimited() const {
115 std::string result = CStringUtils::typeToString(m_Hash[0]);
116 for (std::size_t i = 1; i < N; ++i) {
117 result += DELIMITER;
118 result += CStringUtils::typeToString(m_Hash[i]);
119 }
120 return result;
121 }

149 template<typename T>
150 void add(const T& value) {
151 // We don't support POD types because the initialisation
152 // of padding bytes is undefined.
153 static_assert(std::is_arithmetic<T>::value || std::is_enum<T>::value,
154 "Unsupported type");
155 for (auto& hash : m_Hashes) {
156 hash = CHashing::safeMurmurHash64(&value, static_cast<int>(sizeof(T)), hash);
157 }

198public:
199 CCompressedDictionary() {
200 // 472882027 and 982451653 are prime numbers, so repeatedly
201 // multiplying them will give many distinct seeds modulo 2 ^ 64.
202 // This method of seed generation was chosen in preference
203 // to using uniform random numbers as the Boost random number
204 // generators can generate different numbers on different
205 // platforms, even with the same random number generator seed.
206 m_Seeds[0] = 472882027;
207 for (std::size_t i = 1; i < N; ++i) {
208 m_Seeds[i] = m_Seeds[i - 1] * 982451653ULL;
209 }
210 }
