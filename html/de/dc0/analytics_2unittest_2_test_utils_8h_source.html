205template<std::size_t N>
206class CUnitKernel {
207public:
208 CUnitKernel(const maths::common::CMultivariatePrior& prior)
209 : m_Prior(&prior), m_X(1) {}
210
211 bool operator()(const maths::common::CVectorNx1<double, N>& x, double& result) const {
212 m_X[0].assign(x.begin(), x.end());

234 CMeanKernel(const maths::common::CMultivariatePrior& prior)

261class CCovarianceKernel {
262public:
263 CCovarianceKernel(const maths::common::CMultivariatePrior& prior,
264 const maths::common::CVectorNx1<double, N>& mean)
265 : m_Prior(&prior), m_Mean(mean), m_X(1) {}
266
267 bool operator()(const maths::common::CVectorNx1<double, N>& x,
268 maths::common::CSymmetricMatrixNxN<double, N>& result) const {
269 m_X[0].assign(x.begin(), x.end());
270 double likelihood;
271 m_Prior->jointLogMarginalLikelihood(m_X, SINGLE_UNIT, likelihood);
272 likelihood = std::exp(likelihood);
273 result = (x - m_Mean).outer() * likelihood;
274 return true;
275 }
276
277private:
278 static ml::maths_t::TDouble10VecWeightsAry1Vec SINGLE_UNIT;
279
280private:
281 const maths::common::CMultivariatePrior* m_Prior;

283 mutable handy_typedefs::TDouble10Vec1Vec m_X;
