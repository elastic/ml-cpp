37template<typename PRIOR_PTR>
38struct SMultimodalPriorMode {
39 static const core::TPersistenceTag INDEX_TAG;
40 static const core::TPersistenceTag PRIOR_TAG;
41
42 SMultimodalPriorMode() : s_Index(0), s_Prior() {}
43 SMultimodalPriorMode(std::size_t index, const PRIOR_PTR& prior)
44 : s_Index(index), s_Prior(prior->clone()) {}
45 SMultimodalPriorMode(std::size_t index, PRIOR_PTR&& prior)
46 : s_Index(index), s_Prior(std::move(prior)) {}
47
49 double weight() const { return s_Prior->numberSamples(); }
50
52 std::uint64_t checksum(std::uint64_t seed) const {
53 seed = CChecksum::calculate(seed, s_Index);
54 return CChecksum::calculate(seed, s_Prior);
55 }
56
58 void debugMemoryUsage(const core::CMemoryUsage::TMemoryUsagePtr& mem) const {
59 mem->setName("CMultimodalPrior::SMode");
60 core::memory_debug::dynamicSize("s_Prior", s_Prior, mem);
61 }
62
64 std::size_t memoryUsage() const {
65 return core::memory::dynamicSize(s_Prior);
66 }
67
69 bool acceptRestoreTraverser(const SDistributionRestoreParams& params,
70 core::CStateRestoreTraverser& traverser) {
71 do {
72 const std::string& name = traverser.name();
73 RESTORE_BUILT_IN(INDEX_TAG, s_Index)
74 RESTORE(PRIOR_TAG, traverser.traverseSubLevel(
75 [&, serializer = CPriorStateSerialiser{} ](auto& traverser_) {
76 return serializer(params, s_Prior, traverser_);
77 }))
78 } while (traverser.next());
79
80 this->checkRestoredInvariants();
81
82 return true;
83 }
84
87 void checkRestoredInvariants() const {
88 VIOLATES_INVARIANT_NO_EVALUATION(s_Prior, ==, nullptr);
89 }
90
92 void acceptPersistInserter(core::CStatePersistInserter& inserter) const {
93 inserter.insertValue(INDEX_TAG, s_Index);
94 inserter.insertLevel(
95 PRIOR_TAG, [&, serializer = CPriorStateSerialiser{} ](auto& inserter_) {
96 serializer(*s_Prior, inserter_);
97 });
98 }
99
101 template<typename T>
102 static std::string debugWeights(const std::vector<SMultimodalPriorMode<T>>& modes) {
103 if (modes.empty()) {
104 return std::string();
105 }
106 std::ostringstream result;
107 result << std::scientific << std::setprecision(15) << modes[0].weight();
108 for (std::size_t i = 1; i < modes.size(); ++i) {
109 result << " " << modes[i].weight();
110 }
111 return result.str();
112 }
113
114 std::size_t s_Index;
115 PRIOR_PTR s_Prior;
116};
