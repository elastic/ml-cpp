35public:
40 explicit CConcurrentWrapper(T& resource,

42 std::size_t notifyCapacity = 50)
43 : m_Queue(queueCapacity, notifyCapacity), m_Resource(resource), m_Done(false) {
44 m_Worker = std::thread([this] {
45 while (!m_Done) {
46 m_Queue.pop()();
47 }
48 });
49 }
