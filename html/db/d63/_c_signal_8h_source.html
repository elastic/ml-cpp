67 struct SSeasonalComponentSummary {
68 SSeasonalComponentSummary() = default;
69 SSeasonalComponentSummary(std::size_t period,
70 std::size_t startOfWeek,
71 std::size_t windowRepeat,
72 const TSizeSizePr& window)
73 : s_Period{period}, s_StartOfWeek{startOfWeek},
74 s_WindowRepeat{windowRepeat}, s_Window{window} {}
75
76 bool operator==(const SSeasonalComponentSummary& rhs) const {
77 return s_Period == rhs.s_Period && s_StartOfWeek == rhs.s_StartOfWeek &&
78 s_WindowRepeat == rhs.s_WindowRepeat && s_Window == rhs.s_Window;
79 }
80 bool operator<(const SSeasonalComponentSummary& rhs) const {
81 return common::COrderings::lexicographicalCompare(
82 s_Period, s_StartOfWeek, s_WindowRepeat, s_Window, // lhs
83 rhs.s_Period, rhs.s_StartOfWeek, rhs.s_WindowRepeat, rhs.s_Window);
84 }
85
87 bool almostEqual(const SSeasonalComponentSummary& rhs, double eps) const {
88 return almostEqual(s_Period, rhs.s_Period, eps) &&
89 almostEqual(s_StartOfWeek, rhs.s_StartOfWeek, eps) &&
90 almostEqual(s_WindowRepeat, rhs.s_WindowRepeat, eps) &&
91 almostEqual(s_Window.first, rhs.s_Window.first, eps) &&
92 almostEqual(s_Window.second, rhs.s_Window.second, eps);
93 }
95 bool periodAlmostEqual(const SSeasonalComponentSummary& rhs, double eps) const {
96 return almostEqual(s_Period, rhs.s_Period, eps);
97 }
99 static bool almostEqual(std::size_t i, std::size_t j, double eps) {
100 std::size_t max{std::max(i, j)};
101 std::size_t min{std::min(i, j)};
102 return static_cast<double>(max - min) <= eps * static_cast<double>(max);
103 }
104
106 bool nested(const SSeasonalComponentSummary& other) const {
107 if (other.windowed()) {
108 return other.s_Window == s_Window && other.s_Period > s_Period &&
109 other.s_Period % s_Period == 0;
110 }
111 if (this->windowed()) {
112 return other.s_Period > s_WindowRepeat &&
113 other.s_Period % s_WindowRepeat == 0;
114 }
115 return other.s_Period > s_Period && other.s_Period % s_Period == 0;
116 }
117
119 bool contains(std::size_t index) const {
120 index = (s_WindowRepeat + index - s_StartOfWeek) % s_WindowRepeat;
121 return index >= s_Window.first && index < s_Window.second;
122 }
123
125 std::size_t fractionInWindow(std::size_t range) const {
126 if (s_WindowRepeat > 0) {
127 return ((s_Window.second - s_Window.first) * range) / s_WindowRepeat;
128 }
129 return range;
130 }
131
133 std::size_t offset(std::size_t index) const {
134 index = (s_WindowRepeat + index - s_StartOfWeek) % s_WindowRepeat;
135 return (index - s_Window.first) % this->period();
136 }
137
139 std::size_t nextRepeat(std::size_t index) const {
140 for (index += s_Period; this->contains(index) == false; index += s_Period) {
141 }
142 return index;
143 }
144
146 std::size_t period() const {
147 return std::min(s_Period, s_Window.second - s_Window.first);
148 }
149
151 bool windowed() const {
152 return s_Window.second - s_Window.first < s_WindowRepeat;
153 }
154
156 TSizeSizePr2Vec windows(std::size_t numberValues) const {
157 TSizeSizePr2Vec result;
158 if (this->windowed()) {
159 result.reserve((numberValues + s_WindowRepeat - 1) / s_WindowRepeat);
160 for (std::size_t i = 0; i < numberValues; i += s_WindowRepeat) {
161 result.emplace_back(s_StartOfWeek + s_Window.first + i,
162 s_StartOfWeek + s_Window.second + i);
163 }
164 } else {
165 result.emplace_back(0, numberValues);
166 }
167 return result;
168 }
169
171 template<typename MODEL>
172 double value(const MODEL& model, std::size_t index) const {
173 return this->contains(index) == false
174 ? 0.0
175 : common::CBasicStatistics::mean(model[this->offset(index)]);
176 }
177
179 std::string print() const {
180 std::ostringstream result;
181 result << s_Period << "/" << s_StartOfWeek << "/" << s_WindowRepeat
182 << "/" << core::CContainerPrinter::print(s_Window);
183 return result.str();
184 }
185
186 std::size_t s_Period = 0;
187 std::size_t s_StartOfWeek = 0;
188 std::size_t s_WindowRepeat = 0;
189 TSizeSizePr s_Window;
190 };

473 template<typename CONTAINER>
474 static void restrictTo(const TSizeSizePr2Vec& windows, CONTAINER& values) {
475 if (values.size() == 0) {
476 return;
477 }
478 if (windows.empty()) {
479 values.clear();
480 return;
481 }
482
483 std::size_t m{std::accumulate(windows.begin(), windows.end(), std::size_t{0},
484 [](std::size_t m_, const auto& window) {
485 return m_ + window.second - window.first;
486 })};
487 std::size_t n{values.size()};
488
489 std::size_t first{windows[0].first % n};
490 std::size_t i{windows[0].first % n};
491 for (const auto& window : windows) {
492 for (std::size_t j = window.first; j < window.second; ++j, ++i) {
493 values[i % n] = values[j % n];
494 }
495 }
496 std::rotate(values.begin(), values.begin() + first, values.end());
497 values.resize(m);
498 }
