CMakeDataFrameCategoryEncoder & ml::maths::analytics::CMakeDataFrameCategoryEncoder::redundancyWeight ( TOptionalDouble redundancyWeight)

double ml::maths::analytics::CMakeDataFrameCategoryEncoder::MINIMUM_FREQUENCY_TO_ONE_HOT_ENCODE {0.05}
