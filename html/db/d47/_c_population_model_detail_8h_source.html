22template<typename T>
23CPopulationModel::TSizeSizePr CPopulationModel::personRange(const T& data, std::size_t pid) {
24 const std::size_t minCid = 0;
25 const std::size_t maxCid = std::numeric_limits<std::size_t>::max();
26 auto begin = std::lower_bound(data.begin(), data.end(), std::make_pair(pid, minCid),
27 maths::common::COrderings::SFirstLess());
28 auto end = std::upper_bound(begin, data.end(), std::make_pair(pid, maxCid),
29 maths::common::COrderings::SFirstLess());
30 return {static_cast<std::size_t>(begin - data.begin()),
31 static_cast<std::size_t>(end - data.begin())};
32}

34template<typename T>

36CPopulationModel::find(const T& data, std::size_t pid, std::size_t cid) {
37 auto i = std::lower_bound(data.begin(), data.end(), std::make_pair(pid, cid),
38 maths::common::COrderings::SFirstLess());
39 if (i != data.end() && (CDataGatherer::extractPersonId(*i) != pid ||
40 CDataGatherer::extractAttributeId(*i) != cid)) {
41 i = data.end();

60template<typename T, typename PERSON_FILTER, typename ATTRIBUTE_FILTER>
61void CPopulationModel::applyFilters(bool updateStatistics,
62 const PERSON_FILTER& personFilter,
63 const ATTRIBUTE_FILTER& attributeFilter,
64 T& data) const {
65 std::size_t initialSize = data.size();
66 if (this->params().s_ExcludeFrequent & model_t::E_XF_Over) {
67 data.erase(std::remove_if(data.begin(), data.end(), personFilter), data.end());
68 }
69 if (this->params().s_ExcludeFrequent & model_t::E_XF_By) {
70 data.erase(std::remove_if(data.begin(), data.end(), attributeFilter), data.end());
71 }
72 if (updateStatistics && data.size() != initialSize) {
73 ++core::CProgramCounters::counter(counter_t::E_TSADNumberExcludedFrequentInvocations);
74 }
75}
76}
77}
