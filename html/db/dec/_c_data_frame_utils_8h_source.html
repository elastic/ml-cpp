43template<typename T>
44struct SRowTo {
45 static_assert(sizeof(T) < 0, "Vector type not supported");
46};
47
48template<typename T, Eigen::AlignmentType ALIGNMENT>
49struct SRowTo<common::CMemoryMappedDenseVector<T, ALIGNMENT>> {
50 static common::CMemoryMappedDenseVector<T, ALIGNMENT>
51 dispatch(const core::CDataFrame::TRowRef& row) {
52 return {row.data(), static_cast<long>(row.numberColumns())};
53 }
54};
55
56template<typename T>
57struct SRowTo<common::CDenseVector<T>> {
58 static common::CDenseVector<T> dispatch(const core::CDataFrame::TRowRef& row) {
59 std::size_t n{row.numberColumns()};
60 common::CDenseVector<T> result{
61 common::SConstant<common::CDenseVector<T>>::get(n, 0)};
62 for (std::size_t i = 0; i < n; ++i) {
63 result(i) = row[i];
64 }
65 return result;
66 }
67};
68}
