64 struct SSubSample {
65 static const std::string SAMPLE_START_TAG;
66 static const std::string SAMPLE_END_TAG;
67 static const std::string SAMPLE_TAG;
68
69 SSubSample(std::size_t dimension, core_t::TTime time)
70 : s_Statistic(dimension), s_Start(time), s_End(time) {}
71
72 void add(const TDouble1Vec& measurement, core_t::TTime time, unsigned int count) {
73 s_Statistic.add(measurement, time, count);
74 // Using explicit tests instead of std::min and std::max to work
75 // around g++ 4.1 optimiser bug
76 if (time < s_Start) {
77 s_Start = time;
78 }
79 if (time > s_End) {
80 s_End = time;
81 }
82 }
83
86 bool isClose(core_t::TTime time, core_t::TTime targetSpan) const {
87 if (time > s_End) {
88 return time < s_Start + targetSpan;
89 }
90 if (time >= s_Start) {
91 return true;
92 }
93 return time > s_End - targetSpan;
94 }
95
96 // This assumes that buckets are aligned to n * bucketLength
97 bool isInSameBucket(core_t::TTime time, core_t::TTime bucketLength) const {
98 core_t::TTime timeBucket = maths::common::CIntegerTools::floor(time, bucketLength);
99 core_t::TTime subSampleBucket =
100 maths::common::CIntegerTools::floor(s_Start, bucketLength);
101 return timeBucket == subSampleBucket;
102 }
103
105 const SSubSample& operator+=(const SSubSample& rhs) {
106 s_Statistic += rhs.s_Statistic;
107 s_Start = std::min(s_Start, rhs.s_Start);
108 s_End = std::max(s_End, rhs.s_End);
109 return *this;
110 }

113 void acceptPersistInserter(core::CStatePersistInserter& inserter) const {
114 inserter.insertLevel(SAMPLE_TAG,
115 std::bind(&TMetricPartialStatistic::persist,
116 &s_Statistic, std::placeholders::_1));
117 inserter.insertValue(SAMPLE_START_TAG, s_Start);
118 inserter.insertValue(SAMPLE_END_TAG, s_End);
119 }
120
122 bool acceptRestoreTraverser(core::CStateRestoreTraverser& traverser) {
123 do {
124 const std::string& name = traverser.name();
125 if (name == SAMPLE_TAG) {
126 if (traverser.traverseSubLevel(
127 std::bind(&TMetricPartialStatistic::restore, &s_Statistic,
128 std::placeholders::_1)) == false) {
129 LOG_ERROR(<< "Invalid sample value");
130 return false;
131 }
132 } else if (name == SAMPLE_START_TAG) {
133 if (core::CStringUtils::stringToType(traverser.value(), s_Start) == false) {
134 LOG_ERROR(<< "Invalid attribute identifier in "
135 << traverser.value());
136 return false;
137 }
138 } else if (name == SAMPLE_END_TAG) {
139 if (core::CStringUtils::stringToType(traverser.value(), s_End) == false) {
140 LOG_ERROR(<< "Invalid attribute identifier in "
141 << traverser.value());
142 return false;
143 }
144 }
145 } while (traverser.next());
146 return true;
147 }

200 CSampleQueue(std::size_t dimension,
201 std::size_t sampleCountFactor,
202 std::size_t latencyBuckets,
203 double growthFactor,
204 core_t::TTime bucketLength)
205 : m_Dimension(dimension),
206 m_Queue(std::max(sampleCountFactor * latencyBuckets, std::size_t(1))),
207 m_SampleCountFactor(sampleCountFactor), m_GrowthFactor(growthFactor),
208 m_BucketLength(bucketLength),
209 m_Latency(static_cast<core_t::TTime>(latencyBuckets) * bucketLength) {}

217 void add(core_t::TTime time, const TDouble1Vec& measurement, unsigned int count, unsigned int sampleCount) {
218 if (m_Queue.empty()) {
219 this->pushFrontNewSubSample(measurement, time, count);
220 } else if (time >= m_Queue[0].s_Start) {
221 this->addAfterLatestStartTime(measurement, time, count, sampleCount);
222 } else {
223 this->addHistorical(measurement, time, count, sampleCount);
224 }
225 }
226
228 bool canSample(core_t::TTime bucketStart) const {
229 core_t::TTime bucketEnd = bucketStart + m_BucketLength - 1;
230 return m_Queue.empty() ? false : m_Queue.back().s_End <= bucketEnd;
231 }
232
239 void sample(core_t::TTime bucketStart,
240 unsigned int sampleCount,
241 model_t::EFeature feature,
242 TSampleVec& samples) {
243 core_t::TTime latencyCutoff = bucketStart + m_BucketLength - 1;
244 TOptionalSubSample combinedSubSample;
245
246 while (m_Queue.empty() == false && m_Queue.back().s_End <= latencyCutoff) {
247 if (combinedSubSample) {
248 *combinedSubSample += m_Queue.back();
249 } else {
250 combinedSubSample = TOptionalSubSample(m_Queue.back());
251 }
252
253 m_Queue.pop_back();
254
255 double count = combinedSubSample->s_Statistic.count();
256 double countIncludingNext =
257 (m_Queue.empty()) ? count : count + m_Queue.back().s_Statistic.count();
258 double countRatio = sampleCount / count;
259 double countRatioIncludingNext = sampleCount / countIncludingNext;
260
261 if (countIncludingNext >= sampleCount &&
262 (std::abs(1.0 - countRatio) <= std::abs(1.0 - countRatioIncludingNext))) {
263 TDouble1Vec sample = combinedSubSample->s_Statistic.value();
264 core_t::TTime sampleTime = combinedSubSample->s_Statistic.time();
265 double vs = model_t::varianceScale(feature, sampleCount, count);
266 samples.emplace_back(sampleTime, sample, vs, count);
267 combinedSubSample = TOptionalSubSample();
268 }
269 }
270
271 if (combinedSubSample) {
272 m_Queue.push_back(*combinedSubSample);
273 }
274 }

276 void resetBucket(core_t::TTime bucketStart) {
277 // The queue is ordered in descending sub-sample start time.
278
279 iterator firstEarlierThanBucket =
280 std::upper_bound(m_Queue.begin(), m_Queue.end(), bucketStart, timeLater);
281
282 // This is equivalent to lower_bound(., ., bucketStart + m_BucketLength - 1, .);
283 iterator latestWithinBucket = std::upper_bound(
284 m_Queue.begin(), m_Queue.end(), bucketStart + m_BucketLength, timeLater);
285
286 m_Queue.erase(latestWithinBucket, firstEarlierThanBucket);
287 }

385 void resizeIfFull() {
386 if (m_Queue.full()) {
387 std::size_t currentSize = m_Queue.size();
388 std::size_t newSize = static_cast<std::size_t>(
389 static_cast<double>(currentSize) * (1.0 + m_GrowthFactor));
390 m_Queue.set_capacity(std::max(newSize, currentSize + 1));
391 }
392 }
393
394 void addAfterLatestStartTime(const TDouble1Vec& measurement,
395 core_t::TTime time,
396 unsigned int count,
397 unsigned int sampleCount) {
398 if (time >= m_Queue[0].s_End &&
399 this->shouldCreateNewSubSampleAfterLatest(time, sampleCount)) {
400 this->pushFrontNewSubSample(measurement, time, count);
401 } else {
402 m_Queue[0].add(measurement, time, count);
403 }
404 }
405
406 bool shouldCreateNewSubSampleAfterLatest(core_t::TTime time, unsigned int sampleCount) {
407 if (m_Queue[0].s_Statistic.count() >=
408 static_cast<double>(this->targetSubSampleCount(sampleCount))) {
409 return true;
410 }
411
412 // If latency is non-zero, we also want to check whether the new measurement
413 // is too far from the latest sub-sample or whether they belong in different buckets.
414 if (m_Latency > 0) {
415 if (!m_Queue[0].isClose(time, this->targetSubSampleSpan()) ||
416 !m_Queue[0].isInSameBucket(time, m_BucketLength)) {
417 return true;
418 }
419 }
420 return false;
421 }
422
423 core_t::TTime targetSubSampleSpan() const {
424 return (m_BucketLength + static_cast<core_t::TTime>(m_SampleCountFactor) - 1) /
425 static_cast<core_t::TTime>(m_SampleCountFactor);
426 }
427
428 std::size_t targetSubSampleCount(unsigned int sampleCount) const {
429 return static_cast<std::size_t>(sampleCount) / m_SampleCountFactor;
430 }
431
432 void addHistorical(const TDouble1Vec& measurement,
433 core_t::TTime time,
434 unsigned int count,
435 unsigned int sampleCount) {
436 // We have to resize before we do the search of the upper bound. Otherwise,
437 // a later resize will invalidate the upper bound iterator.
438 this->resizeIfFull();
439
440 reverse_iterator upperBound =
441 std::upper_bound(m_Queue.rbegin(), m_Queue.rend(), time, timeEarlier);
442 core_t::TTime targetSubSampleSpan = this->targetSubSampleSpan();
443
444 if (upperBound == m_Queue.rbegin()) {
445 if ((upperBound->s_Statistic.count() >=
446 static_cast<double>(this->targetSubSampleCount(sampleCount))) ||
447 !upperBound->isClose(time, targetSubSampleSpan) ||
448 !(*upperBound).isInSameBucket(time, m_BucketLength)) {
449 this->pushBackNewSubSample(measurement, time, count);
450 } else {
451 upperBound->add(measurement, time, count);
452 }
453 return;
454 }
455
456 SSubSample& left = *(upperBound - 1);
457 SSubSample& right = *upperBound;
458 if (time <= left.s_End) {
459 left.add(measurement, time, count);
460 return;
461 }
462 bool sameBucketWithLeft = left.isInSameBucket(time, m_BucketLength);
463 bool sameBucketWithRight = right.isInSameBucket(time, m_BucketLength);
464 std::size_t spaceLimit = this->targetSubSampleCount(sampleCount);
465 bool leftHasSpace = static_cast<std::size_t>(left.s_Statistic.count()) < spaceLimit;
466 bool rightHasSpace = static_cast<std::size_t>(right.s_Statistic.count()) < spaceLimit;
467 core_t::TTime leftDistance = time - left.s_End;
468 core_t::TTime rightDistance = right.s_Start - time;
469 SSubSample& candidate = maths::common::COrderings::lexicographicalCompare(
470 -static_cast<int>(sameBucketWithLeft),
471 -static_cast<int>(leftHasSpace), leftDistance,
472 -static_cast<int>(sameBucketWithRight),
473 -static_cast<int>(rightHasSpace), rightDistance)
474 ? left
475 : right;
476
477 if (candidate.isInSameBucket(time, m_BucketLength) &&
478 (candidate.isClose(time, targetSubSampleSpan) ||
479 right.s_Start <= left.s_End + targetSubSampleSpan)) {
480 candidate.add(measurement, time, count);
481 return;
482 }
483 this->insertNewSubSample(upperBound.base(), measurement, time, count);
484 }
485
486 static bool timeEarlier(core_t::TTime time, const SSubSample& subSample) {
487 return time < subSample.s_Start;
488 }
489
490 static bool timeLater(core_t::TTime time, const SSubSample& subSample) {
491 return time > subSample.s_Start;
492 }
