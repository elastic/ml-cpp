143 void add(double x, double y, double weight = 1.0) {
144 TVector d;
145 double xi{1.0};
146 for (std::size_t i = 0; i < N; ++i, xi *= x) {
147 d(i) = xi;
148 d(i + 2 * N - 1) = xi * y;
149 }
150 for (std::size_t i = N; i < 2 * N - 1; ++i, xi *= x) {
151 d(i) = xi;
152 }
153 if constexpr (R_2) {
154 d(3 * N - 1) = y * y;
155 }
156 m_S.add(d, weight);
157 }

299 bool r2(double& result, double maxCondition = MAX_CONDITION) const {
300
301 result = 0.0;
302
303 if constexpr (R_2 == false) {
304 return false;
305 }
306 if (CBasicStatistics::count(m_S) == T{0}) {
307 return true;
308 }
309
310 TMeanVarAccumulator residualMoments;
311 if (this->residualMoments(residualMoments, maxCondition) == false) {
312 return false;
313 }
314
315 const auto& s = CBasicStatistics::mean(m_S);
316 double variance{s(3 * N - 1) - CTools::pow2(s(2 * N - 1))};
317 double residualVariance{CBasicStatistics::maximumLikelihoodVariance(residualMoments)};
318 result = CTools::truncate(1.0 - residualVariance / variance, 0.0, 1.0);
319 return true;
320 }

347 TArray parameters(double x, double maxCondition = MAX_CONDITION) const {
348 TArray result;
349 TArray params;
350 if (this->parameters(params, maxCondition)) {
351 std::ptrdiff_t n{static_cast<std::ptrdiff_t>(params.size())};
352 for (std::ptrdiff_t i = n - 1; i >= 0; --i) {
353 result[i] = params[i];
354 for (std::ptrdiff_t j = i + 1; j < n; ++j) {
355 params[j] *= static_cast<double>(i + 1) /
356 static_cast<double>(j - i) * x;
357 result[i] += params[j];
358 }
359 }
360 }
361 return result;
362 }

390 double range() const {
391 // The magic 12 comes from assuming the independent
392 // variable X is uniform over the range (for our uses
393 // it typically is). We maintain mean X^2 and X. For
394 // a uniform variable on a range [a, b] we have that
395 // E[(X - E(X))^2] = E[X^2] - E[X]^2 = (b - a)^2 / 12.
396
397 double x1{CBasicStatistics::mean(m_S)(1)};
398 double x2{CBasicStatistics::mean(m_S)(2)};
399 return std::sqrt(12.0 * std::max(x2 - x1 * x1, 0.0));
400 }

403 void age(double factor, double meanRevertFactor = 0.0) {
404 if (meanRevertFactor > 0.0) {
405 TVector& s{CBasicStatistics::moment<0>(m_S)};
406 double alpha{1.0 - meanRevertFactor * (1.0 - factor)};
407 double beta{1.0 - alpha};
408 for (std::size_t i = 1; i < N; ++i) {
409 s(i + 2 * N - 1) = alpha * s(i + 2 * N - 1) + beta * s(i) * s(2 * N - 1);
410 }
411 }
412 m_S.age(factor);
413 }

422 double mean(double a, double b) const {
423 double result{0.0};
424
425 double interval{b - a};
426
427 TArray params;
428 this->parameters(params);
429
430 if (interval == 0.0) {
431 result = params[0];
432 double xi{a};
433 for (std::size_t i = 1; i < params.size(); ++i, xi *= a) {
434 result += params[i] * xi;
435 }
436 return result;
437 }
438
439 for (std::size_t i = 0; i < N; ++i) {
440 for (std::size_t j = 0; j <= i; ++j) {
441 result += CCategoricalTools::binomialCoefficient(i + 1, j + 1) *
442 params[i] / static_cast<double>(i + 1) *
443 std::pow(a, static_cast<double>(i - j)) *
444 std::pow(interval, static_cast<double>(j + 1));
445 }
446 }
447
448 return result / interval;
449 }
