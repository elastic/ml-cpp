85public:

88 enum ESearchType { E_LinearSearch, E_LogSearch };
89
90public:
91 explicit CBoostedTreeParameter(T value, ESearchType logSearch = E_LinearSearch)
92 : m_Value{value}, m_MinValue{value}, m_MaxValue{value}, m_LogSearch{logSearch == E_LogSearch} {
93 }
94
96 T value() const { return m_Scale * m_Value; }
97
101 CBoostedTreeParameter& set(T value) {
102 if (m_FixedToRange) {
103 value = common::CTools::truncate(value, m_MinValue, m_MaxValue);
104 }
105 m_Value = value;
106 return *this;
107 }
108
110 T scale() const { return m_Scale; }
111
113 CBoostedTreeParameter& scale(T scale) {
114 m_Scale = scale;
115 return *this;
116 }
117
123 CBoostedTreeParameter& captureScale() {
124 m_Value *= m_Scale;
125 m_MinValue *= m_Scale;
126 m_MaxValue *= m_Scale;
127 m_Scale = T{1};
128 return *this;
129 }
130
135 void setToRangeMidpointOr(T value) {
136 if (this->fixed() == false) {
137 m_Value = m_FixedToRange
138 ? this->fromSearchValue((this->toSearchValue(m_MinValue) +
139 this->toSearchValue(m_MaxValue)) /
140 T{2})
141 : value;
142 }
143 }
144
146 void fixTo(T value) {
147 m_Value = value;
148 m_MinValue = value;
149 m_MaxValue = value;
150 m_Scale = T{1};
151 m_FixedToRange = true;
152 }
153
157 void fixToRange(T minValue, T maxValue) {
158 m_Value = common::CTools::truncate(m_Value, minValue, maxValue);
159 m_MinValue = minValue;
160 m_MaxValue = maxValue;
161 m_Scale = T{1};
162 m_FixedToRange = true;
163 }
164
166 void fixTo(const std::vector<T>& value) {
167 if (value.empty()) {
168 return;
169 }
170 if (value.size() == 1) {
171 this->fixTo(value[0]);
172 return;
173 }
174 if (value[0] > value[1]) {
175 this->fixToRange(value[1], value[0]);
176 }
177 this->fixToRange(value[0], value[1]);
178 }
179
181 void fix() { this->fixTo(m_Value); }
182
184 bool fixed() const { return m_FixedToRange && m_MinValue == m_MaxValue; }
185
187 bool rangeFixed() const { return m_FixedToRange; }
188
190 bool valid(double value) const {
191 return m_LogSearch == false || value > 0.0;
192 }
193
195 double toSearchValue() const { return this->toSearchValue(m_Value); }
196
198 double toSearchValue(T value) const {
199 return m_LogSearch ? common::CTools::stableLog(static_cast<double>(value))
200 : static_cast<double>(value);
201 }
202
204 T fromSearchValue(double value) const {
205 return static_cast<T>(m_LogSearch ? common::CTools::stableExp(value) : value);
206 }
207
209 std::pair<T, T> searchRange() const {
210 return {this->toSearchValue(m_MinValue), this->toSearchValue(m_MaxValue)};
211 }
212
214 void save() {
215 m_SavedValue = m_Value;
216 m_SavedMinValue = m_MinValue;
217 m_SavedMaxValue = m_MaxValue;
218 m_SavedScale = m_Scale;
219 }

222 CBoostedTreeParameter& load() {
223 m_Value = m_SavedValue;
224 m_MinValue = m_SavedMinValue;
225 m_MaxValue = m_SavedMaxValue;
226 m_Scale = m_SavedScale;
227 return *this;
228 }

231 void acceptPersistInserter(core::CStatePersistInserter& inserter) const {

233 core::CPersistUtils::persist(LOG_SEARCH_TAG, m_LogSearch, inserter);

236 core::CPersistUtils::persist(SAVED_MAX_VALUE_TAG, m_SavedMaxValue, inserter);
237 core::CPersistUtils::persist(SAVED_MIN_VALUE_TAG, m_SavedMinValue, inserter);
238 core::CPersistUtils::persist(SAVED_SCALE_TAG, m_SavedScale, inserter);
239 core::CPersistUtils::persist(SAVED_VALUE_TAG, m_SavedValue, inserter);
240 core::CPersistUtils::persist(SCALE_TAG, m_Scale, inserter);
241 core::CPersistUtils::persist(VALUE_TAG, m_Value, inserter);
242 }

245 bool acceptRestoreTraverser(core::CStateRestoreTraverser& traverser) {
246 do {
247 const std::string& name{traverser.name()};
248 RESTORE(FIXED_TO_RANGE_TAG,
249 core::CPersistUtils::restore(FIXED_TO_RANGE_TAG, m_FixedToRange, traverser))
250 RESTORE(LOG_SEARCH_TAG,
251 core::CPersistUtils::restore(LOG_SEARCH_TAG, m_LogSearch, traverser))
252 RESTORE(MAX_VALUE_TAG,
253 core::CPersistUtils::restore(MAX_VALUE_TAG, m_MaxValue, traverser))
254 RESTORE(MIN_VALUE_TAG,
255 core::CPersistUtils::restore(MIN_VALUE_TAG, m_MinValue, traverser))
256 RESTORE(SAVED_MAX_VALUE_TAG,
257 core::CPersistUtils::restore(SAVED_MAX_VALUE_TAG, m_SavedMaxValue, traverser))
258 RESTORE(SAVED_MIN_VALUE_TAG,
259 core::CPersistUtils::restore(SAVED_MIN_VALUE_TAG, m_SavedMinValue, traverser))
260 RESTORE(SAVED_SCALE_TAG,
261 core::CPersistUtils::restore(SAVED_SCALE_TAG, m_SavedScale, traverser))
262 RESTORE(SAVED_VALUE_TAG,
263 core::CPersistUtils::restore(SAVED_VALUE_TAG, m_SavedValue, traverser))
264 RESTORE(SCALE_TAG, core::CPersistUtils::restore(SCALE_TAG, m_Scale, traverser))
265 RESTORE(VALUE_TAG, core::CPersistUtils::restore(VALUE_TAG, m_Value, traverser))
266 } while (traverser.next());
267 return true;
268 }

271 std::uint64_t checksum(std::uint64_t seed = 0) const {
272 seed = common::CChecksum::calculate(seed, m_FixedToRange);
273 seed = common::CChecksum::calculate(seed, m_LogSearch);
274 seed = common::CChecksum::calculate(seed, m_MaxValue);
275 seed = common::CChecksum::calculate(seed, m_MinValue);
276 seed = common::CChecksum::calculate(seed, m_SavedMaxValue);
277 seed = common::CChecksum::calculate(seed, m_SavedMinValue);
278 seed = common::CChecksum::calculate(seed, m_SavedScale);
279 seed = common::CChecksum::calculate(seed, m_SavedValue);
280 seed = common::CChecksum::calculate(seed, m_Scale);
281 return common::CChecksum::calculate(seed, m_Value);
282 }

288 (m_FixedToRange ? " fixed to [" + std::to_string(m_MinValue) +
289 "," + std::to_string(m_MaxValue) + "]"
290 : "");
291 }

344template<typename T>
345class CScopeBoostedTreeParameterOverrides {
346public:
347 CScopeBoostedTreeParameterOverrides() noexcept = default;
348 ~CScopeBoostedTreeParameterOverrides() {
349 // Undo changes in reverse order to which they were applied.
350 for (std::size_t i = m_Parameters.size(); i > 0; --i) {
351 m_Parameters[i - 1]->m_Value = m_ValuesToRestore[i - 1];
352 }
353 }
354
355 CScopeBoostedTreeParameterOverrides(const CScopeBoostedTreeParameterOverrides&) = delete;
356 CScopeBoostedTreeParameterOverrides&
357 operator=(const CScopeBoostedTreeParameterOverrides&) = delete;
358
359 void apply(CBoostedTreeParameter<T>& parameter, T value) {
360 m_ValuesToRestore.push_back(parameter.value());
361 m_Parameters.push_back(&parameter);
362 parameter.m_Value = value;
363 }
364
365private:
366 using TVec = std::vector<T>;
367 using TParameterPtrVec = std::vector<CBoostedTreeParameter<T>*>;
368
369private:
370 TParameterPtrVec m_Parameters;
371 TVec m_ValuesToRestore;
372};
