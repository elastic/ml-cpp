47template<std::size_t N, typename T, bool R_2>
48void CLeastSquaresOnlineRegression<N, T, R_2>::shiftAbscissa(double dx) {
49 if (CBasicStatistics::count(m_S) == 0.0) {
50 return;
51 }
52
53 // The update scheme is as follows:
54 //
55 // 1/n sum_i{ (t(i) + dx)^i }
56 // -> 1/n * ( sum_i{ t(i)^i }
57 // + sum_j{ (i j) * dx^(i - j) * sum_i{ t(i)^j } } )
58 //
59 // 1/n sum_i{ (t(i) + dx)^i * y(i) }
60 // -> 1/n * ( sum_i{ t(i)^i * y(i) }
61 // + sum_j{ (i j) * dx^(i - j) * sum_i{ t(i)^j y(i) } } )
62
63 double d[2 * N - 2]{dx};
64 for (std::size_t i = 1; i < 2 * N - 2; ++i) {
65 d[i] = d[i - 1] * dx;
66 }
67 LOG_TRACE(<< "d = " << d);
68
69 LOG_TRACE(<< "S(before) " << CBasicStatistics::mean(m_S));
70 for (std::size_t i = 2 * N - 2; i > 0; --i) {
71 LOG_TRACE(<< "i = " << i);
72 for (std::size_t j = 0; j < i; ++j) {
73 double bij{CCategoricalTools::binomialCoefficient(i, j) * d[i - j - 1]};
74 LOG_TRACE(<< "bij = " << bij);
75 CBasicStatistics::moment<0>(m_S)(i) += bij * CBasicStatistics::mean(m_S)(j);
76 if (i >= N) {
77 continue;
78 }
79 std::size_t yi{i + 2 * N - 1};
80 std::size_t yj{j + 2 * N - 1};
81 LOG_TRACE(<< "yi = " << yi << ", yj = " << yj);
82 CBasicStatistics::moment<0>(m_S)(yi) += bij * CBasicStatistics::mean(m_S)(yj);
83 }
84 }
85 LOG_TRACE(<< "S(after) = " << CBasicStatistics::mean(m_S));
86}

88template<std::size_t N, typename T, bool R_2>
89bool CLeastSquaresOnlineRegression<N, T, R_2>::residualMoments(TMeanVarAccumulator& result,
90 double maxCondition) const {
91
92 result = TMeanVarAccumulator{};
93
94 if constexpr (R_2 == false) {
95 return false;
96 }
97 if (CBasicStatistics::count(m_S) == T{0}) {
98 return true;
99 }
100
101 std::size_t n{N + 1};
102 while (--n > 0) {
103 switch (n) {
104 case 1: {
105 result = TMeanVarAccumulator{};
106 return true;
107 }
108 case N: {
109 constexpr int N_{static_cast<int>(N)};
110 Eigen::Matrix<double, N_, N_> x;
111 Eigen::Matrix<double, N_, 1> y;
112 Eigen::Matrix<double, N_, 1> z;
113 if (this->residualMoments(n, x, y, z, maxCondition, result)) {
114 return true;
115 }
116 break;
117 }
118 default: {
119 CDenseMatrix<double> x(n, n);
120 CDenseVector<double> y(n);
121 CDenseVector<double> z(n);
122 if (this->residualMoments(n, x, y, z, maxCondition, result)) {
123 return true;
124 }
125 break;
126 }
127 }
128 }
129 return false;
130}

138template<std::size_t N, typename T, bool R_2>
139bool CLeastSquaresOnlineRegression<N, T, R_2>::parameters(std::size_t n,
140 TArray& result,
141 double maxCondition) const {
142 result.fill(0.0);
143
144 // Define all parameters as zero if no data have been added.
145 if (CBasicStatistics::count(m_S) == 0.0) {
146 return true;
147 }
148
149 // Search for non-singular solution.
150 n += 1;
151 while (--n > 0) {
152 switch (n) {
153 case 1: {
154 result[0] = CBasicStatistics::mean(m_S)(2 * N - 1);
155 return true;
156 }
157 case N: {
158 constexpr int N_{static_cast<int>(N)};
159 Eigen::Matrix<double, N_, N_> x;
160 Eigen::Matrix<double, N_, 1> y;
161 if (this->parameters(N, x, y, maxCondition, result)) {
162 return true;
163 }
164 break;
165 }
166 default: {
167 CDenseMatrix<double> x(n, n);
168 CDenseVector<double> y(n);
169 if (this->parameters(n, x, y, maxCondition, result)) {
170 return true;
171 }
172 break;
173 }
174 }
175 }
176 return false;
177}

186template<std::size_t N, typename T, bool R_2>
187bool CLeastSquaresOnlineRegression<N, T, R_2>::covariances(std::size_t n,
188 double variance,
189 TMatrix& result,
190 double maxCondition) const {
191 result = TMatrix(0.0);
192
193 // Define covariance as zero matrix if no data have been added.
194 if (CBasicStatistics::count(m_S) == 0.0) {
195 return true;
196 }
197
198 // Search for the covariance matrix of a non-singular subproblem.
199 n += 1;
200 while (--n > 0) {
201 switch (n) {
202 case 1: {
203 result(0, 0) = variance / CBasicStatistics::count(m_S);
204 return true;
205 }
206 case N: {
207 constexpr int N_{static_cast<int>(N)};
208 Eigen::Matrix<double, N_, N_> x;
209 if (this->covariances(N, x, variance, maxCondition, result) == false) {
210 continue;
211 }
212 break;
213 }
214 default: {
215 CDenseMatrix<double> x(n, n);
216 if (this->covariances(n, x, variance, maxCondition, result) == false) {
217 continue;
218 }
219 break;
220 }
221 }
222 return true;
223 }
224 return false;
225}

242template<std::size_t N, typename T, bool R_2>
243template<typename MATRIX, typename VECTOR>
244bool CLeastSquaresOnlineRegression<N, T, R_2>::residualMoments(std::size_t n,
245 MATRIX& x,
246 VECTOR& y,
247 VECTOR& z,
248 double maxCondition,
249 TMeanVarAccumulator& result) const {
250 const auto& s = CBasicStatistics::mean(m_S);
251
252 if (n == 1) {
253 double count{CBasicStatistics::count(m_S)};
254 double mean{s(2 * N - 1)};
255 double variance{s(3 * N - 1) - CTools::pow2(mean)};
256 result = CBasicStatistics::momentsAccumulator(count, mean, variance);
257 return true;
258 }
259
260 this->gramian(n, x);
261 for (std::size_t i = 0; i < n; ++i) {
262 y(i) = s(i + 2 * N - 1);
263 z(i) = s(i);
264 }
265
266 typename SJacobiSvd<MATRIX>::Type svd{x.template selfadjointView<Eigen::Upper>(),
267 Eigen::ComputeFullU | Eigen::ComputeFullV};
268 if (svd.info() != Eigen::Success ||
269 svd.singularValues()(0) > maxCondition * svd.singularValues()(n - 1)) {
270 LOG_TRACE(<< "SVD compute failed with status " << svd.info()
271 << " singular values = " << svd.singularValues() << " for x =\n"
272 << x);
273 return false;
274 }
275
276 // The mean predictions are given by 1^t X p and the square residuals are given
277 // by
278 // ||y - X p - 1^t X p||^2 (1)
279 //
280 // and for least sqaures solution the parameters p = (X^t X)^-1 X^t y. We can
281 // calculate 1^t X from the statistics we maintain. To compute (1) we first use
282 // the fact that
283 //
284 // ||y - X p - 1^t X p||^2 = ||y - X p||^2 + ||1^t X p||^2
285 //
286 // Substituting for p in the first term we have
287 //
288 // ||y - X (X^t X)^-1 X^t y||^2
289 // = y^t y - 2 y^t X (X^t X)^-1 X^t y + y^t X (X^t X)^-1 X^t X (X^t X)^-1 X^t y
290 // = y^t y - 2 y^t X (X^t X)^-1 X^t y + y^t X (X^t X)^-1 X^t y
291 // = y^t y - y^t X (X^t X)^-1 X^t y
292
293 VECTOR r{svd.solve(y)};
294 double count{CBasicStatistics::count(m_S)};
295 double mean{s(2 * N - 1) - z.transpose() * r};
296 double variance{(s(3 * N - 1) - y.transpose() * r) - CTools::pow2(mean)};
297
298 result = CBasicStatistics::momentsAccumulator(count, mean, variance);
299 return true;
300}

302template<std::size_t N, typename T, bool R_2>
303template<typename MATRIX, typename VECTOR>
304bool CLeastSquaresOnlineRegression<N, T, R_2>::parameters(std::size_t n,
305 MATRIX& x,
306 VECTOR& y,
307 double maxCondition,
308 TArray& result) const {
309 const auto& s = CBasicStatistics::mean(m_S);
310
311 if (n == 1) {
312 result[0] = s(2 * N - 1);
313 return true;
314 }
315
316 this->gramian(n, x);
317 for (std::size_t i = 0; i < n; ++i) {
318 y(i) = s(i + 2 * N - 1);
319 }
320
321 typename SJacobiSvd<MATRIX>::Type svd{x.template selfadjointView<Eigen::Upper>(),
322 Eigen::ComputeFullU | Eigen::ComputeFullV};
323 if (svd.info() != Eigen::Success ||
324 svd.singularValues()(0) > maxCondition * svd.singularValues()(n - 1)) {
325 LOG_TRACE(<< "SVD compute failed with status " << svd.info()
326 << " singular values = " << svd.singularValues() << " for x =\n"
327 << x);
328 return false;
329 }
330
331 // Don't bother checking the solution since we check the matrix
332 // condition above.
333 VECTOR r{svd.solve(y)};
334 for (std::size_t i = 0; i < n; ++i) {
335 result[i] = r(i);
336 }
337
338 return true;
339}

341template<std::size_t N, typename T, bool R_2>
342template<typename MATRIX>
343bool CLeastSquaresOnlineRegression<N, T, R_2>::covariances(std::size_t n,
344 MATRIX& x,
345 double variance,
346 double maxCondition,
347 TMatrix& result) const {
348 if (n == 1) {
349 x(0) = variance / CBasicStatistics::count(m_S);
350 return true;
351 }
352
353 this->gramian(n, x);
354 typename SJacobiSvd<MATRIX>::Type svd{x.template selfadjointView<Eigen::Upper>(),
355 Eigen::ComputeFullU | Eigen::ComputeFullV};
356 if (svd.info() != Eigen::Success ||
357 svd.singularValues()(0) > maxCondition * svd.singularValues()(n - 1)) {
358 LOG_TRACE(<< "SVD compute failed with status " << svd.info()
359 << ", singular values = " << svd.singularValues() << " for x =\n"
360 << x);
361 return false;
362 }
363
364 // Don't bother checking for division by zero since we check
365 // the matrix condition above. Also, we zero initialize result
366 // in the calling code so any values we don't fill in the
367 // following loop are zero (as required).
368 x = (svd.matrixV() * svd.singularValues().cwiseInverse().asDiagonal() *
369 svd.matrixU().transpose()) *
370 variance / CBasicStatistics::count(m_S);
371 for (std::size_t i = 0; i < n; ++i) {
372 result(i, i) = x(i, i);
373 for (std::size_t j = 0; j < i; ++j) {
374 result(i, j) = x(i, j);
375 }
376 }
377
378 return true;
379}
