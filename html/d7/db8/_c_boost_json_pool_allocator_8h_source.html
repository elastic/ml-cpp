26class custom_resource : public boost::container::pmr::memory_resource {
27public:
28 custom_resource(std::size_t& allocatedBytes)
29 : m_AllocatedBytes{allocatedBytes} {}
30 std::size_t allocatedBytes() const { return m_AllocatedBytes; }
31
32private:
33 void* do_allocate(std::size_t bytes, std::size_t /*align*/) override {
34 m_AllocatedBytes += bytes;
35 return ::operator new(bytes);
36 }
37
38 void do_deallocate(void* ptr, std::size_t bytes, std::size_t /*align*/) override {
39 m_AllocatedBytes -= bytes;
40 return ::operator delete(ptr);
41 }
42
43 bool do_is_equal(memory_resource const& other) const noexcept override {
44 // since the global allocation and de-allocation functions are used,
45 // any instance of a custom_resource can deallocate memory allocated
46 // by another instance of a logging_resource
47 return dynamic_cast<custom_resource const*>(&other) != nullptr;
48 }
49
50private:
51 std::size_t& m_AllocatedBytes;
52};
