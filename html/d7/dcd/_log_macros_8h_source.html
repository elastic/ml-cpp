89#define LOG_WARN(message) \
90 do { \
91 std::size_t countOfWarnMessages; \
92 bool skipWarnMessage; \
93 std::tie(countOfWarnMessages, skipWarnMessage) = \
94 ml::core::CLogger::instance().throttler().skip(__FILE__, __LINE__); \
95 if (skipWarnMessage == false) { \
96 BOOST_LOG_STREAM_SEV(ml::core::CLogger::instance().logger(), \
97 ml::core::CLogger::E_Warn) \
98 LOG_LOCATION_INFO \
99 message << (countOfWarnMessages > 1 \
100 ? " | repeated [" + std::to_string(countOfWarnMessages) + "]" \
101 : ""); \
102 } \
103 } while (0)
