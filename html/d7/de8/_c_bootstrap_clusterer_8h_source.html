140 struct SCutState {
141 SCutState(std::size_t seed, const TGraph& graph)
142 : s_V(boost::num_vertices(graph)), s_ToVisit(1, seed),
143 s_Adjacency(s_V, 0), s_Cut(0.0), s_A(0) {
144 this->initializeQueue();
145 }
146
148 double cost() const {
149 return s_Cut / static_cast<double>(s_A * (s_V - s_A));
150 }
151
153 bool toVisit(std::size_t i) const {
154 return this->toVisit(s_ToVisit.size(), i);
155 }
156
158 bool toVisit(std::size_t n, std::size_t i) const {
159 return std::binary_search(s_ToVisit.begin(), s_ToVisit.begin() + n, i);
160 }
161
163 std::size_t next() const { return s_Queue.front().second; }
164
166 std::size_t nextToVisit(std::size_t i) const {
167 return static_cast<std::size_t>(
168 std::lower_bound(s_ToVisit.begin(), s_ToVisit.end(), i) -
169 s_ToVisit.begin());
170 }
171
173 void mergeAfter(std::size_t n) {
174 if (s_ToVisit.size() > n) {
175 std::sort(s_ToVisit.begin() + n, s_ToVisit.end());
176 std::inplace_merge(s_ToVisit.begin(), s_ToVisit.begin() + n,
177 s_ToVisit.end());
178 }
179 }
180
182 void initializeQueue() {
183 s_Queue.clear();
184 s_Queue.reserve(s_ToVisit.size());
185 for (std::size_t i = 0; i < s_ToVisit.size(); ++i) {
186 s_Queue.push_back(std::make_pair(s_Adjacency[s_ToVisit[i]], i));
187 }
188 std::make_heap(s_Queue.begin(), s_Queue.end(), std::less<TDoubleSizePr>());
189 }
190
192 void popQueue() {
193 std::pop_heap(s_Queue.begin(), s_Queue.end(), std::less<TDoubleSizePr>());
194 s_Queue.pop_back();
195 }

200 TSizeVec s_ToVisit;
202 TDoubleVec s_Adjacency;
204 TDoubleSizePrVec s_Queue;
206 double s_Cut;
208 std::size_t s_A;

225 template<typename CLUSTERER>
226 std::size_t bootstrapClusters(std::size_t b,
227 CLUSTERER& clusterer,
228 TPointVec& points,
229 TSizeVecVecVec& result) {
230 std::size_t n = points.size();
231 LOG_TRACE(<< "# points = " << n);
232
233 result.clear();
234 result.reserve(b);
235 result.push_back(TSizeVecVec());
236 clusterer.cluster(points, result.back());
237 LOG_TRACE(<< "Run 1: # clusters = " << result.back().size());
238
239 TSizeVec sampling;
240 TPointVec bootstrapPoints;
241 sampling.reserve(n);
242 bootstrapPoints.reserve(n);
243
244 for (std::size_t i = 1; i < b; ++i) {
245 sampling.clear();
246 CSampling::uniformSample(m_Rng, 0, n, n, sampling);
247 std::sort(sampling.begin(), sampling.end());
248 LOG_TRACE(<< "# samples = " << sampling.size());
249
250 bootstrapPoints.clear();
251 for (std::size_t j = 0; j < n; ++j) {
252 bootstrapPoints.push_back(points[sampling[j]]);
253 }
254
255 result.push_back(TSizeVecVec());
256 clusterer.cluster(bootstrapPoints, result.back());
257 for (std::size_t j = 0; j < result.back().size(); ++j) {
258 for (std::size_t k = 0; k < (result.back())[j].size(); ++k) {
259 (result.back())[j][k] = sampling[(result.back())[j][k]];
260 }
261 }
262 LOG_TRACE(<< "Run " << i + 1 << ": # clusters = " << result.back().size());
263 }
264
265 m_Offsets.clear();
266 m_Offsets.resize(result.size());
267 std::size_t k = 0;
268 for (std::size_t i = 0; i < result.size(); ++i) {
269 m_Offsets[i] = k;
270 k += result[i].size();
271 }
272 return k;
273 }

283 void buildClusterGraph(const TPointVec& points,
284 TSizeVecVecVec& bootstrapClusters,
285 TGraph& graph) const {
286 using TSizeSizePrUSet = boost::unordered_set<TSizeSizePr>;
287 using TSizeSizePrUSetCItr = TSizeSizePrUSet::const_iterator;
288
289 TSizeSizePrUSet edges;
290
291 // If there are no common points between a cluster and
292 // another bootstrap sampling we remember it and assign
293 // edges which are consistent with its overlap with other
294 // clusters.
295 TSizeSizePrUSet ambiguous;
296
297 for (std::size_t i = 0; i < bootstrapClusters.size(); ++i) {
298 for (std::size_t j = 0; j < bootstrapClusters[i].size(); ++j) {
299 std::sort(bootstrapClusters[i][j].begin(),
300 bootstrapClusters[i][j].end());
301 }
302 }
303 TSizeVec cik;
304 cik.reserve(points.size());
305 TDoubleVec overlaps;
306 for (std::size_t i = 0; i < bootstrapClusters.size(); ++i) {
307 for (std::size_t j = 0; j < bootstrapClusters.size(); ++j) {
308 if (i == j) {
309 continue;
310 }
311
312 for (std::size_t k = 0; k < bootstrapClusters[i].size(); ++k) {
313 cik = bootstrapClusters[i][k];
314 cik.erase(std::unique(cik.begin(), cik.end()), cik.end());
315 auto nik = static_cast<double>(cik.size());
316
317 overlaps.clear();
318 double sum = 0.0;
319
320 for (std::size_t l = 0;
321 !cik.empty() && l < bootstrapClusters[j].size(); ++l) {
322 const TSizeVec& cjl = bootstrapClusters[j][l];
323 auto o = static_cast<double>(cik.size());
324 cik.erase(std::set_difference(cik.begin(), cik.end(), cjl.begin(),
325 cjl.end(), cik.begin()),
326 cik.end());
327 o -= static_cast<double>(cik.size());
328 o /= nik;
329 overlaps.push_back(o);
330 sum += o;
331 }
332
333 if (sum == 0.0) {
334 ambiguous.insert(std::make_pair(this->toVertex(i, k), j));
335 } else {
336 for (std::size_t l = 0; l < overlaps.size(); ++l) {
337 if (overlaps[l] > m_OverlapThreshold * sum) {
338 std::size_t u = this->toVertex(i, k);
339 std::size_t v = this->toVertex(j, l);
340 if (u > v) {
341 std::swap(u, v);
342 }
343 if (edges.insert(std::make_pair(u, v)).second) {
344 boost::put(boost::edge_weight, graph,
345 boost::add_edge(u, v, graph).first,
346 std::min(m_ChainingFactor *
347 (overlaps[l] - m_OverlapThreshold * sum),
348 1.0));
349 }
350 }
351 }
352 }
353 }
354 }
355 }
356 LOG_TRACE(<< "ambiguous = " << ambiguous);
357
358 TDoubleSizePrVec consistent;
359 for (TSizeSizePrUSetCItr i = ambiguous.begin(); i != ambiguous.end(); ++i) {
360 std::size_t u = i->first;
361
362 consistent.clear();
363 TOutEdgeItr j, endj;
364 for (boost::tie(j, endj) = boost::out_edges(u, graph); j != endj; ++j) {
365 std::size_t v = boost::target(*j, graph);
366 double weight = boost::get(boost::edge_weight, graph, *j);
367
368 TOutEdgeItr k, endk;
369 for (boost::tie(k, endk) = boost::out_edges(v, graph); k != endk; ++k) {
370 std::size_t w = boost::target(*k, graph);
371 if (this->fromVertex(w).first == i->second) {
372 consistent.push_back(std::make_pair(
373 weight * boost::get(boost::edge_weight, graph, *k), w));
374 }
375 }
376 }
377 std::sort(consistent.begin(), consistent.end(), COrderings::SSecondLess());
378 consistent.erase(std::unique(consistent.begin(), consistent.end(), SSecondEqual()),
379 consistent.end());
380 LOG_TRACE(<< "consistent = " << consistent);
381
382 for (std::size_t k = 0; k < consistent.size(); ++k) {
383 boost::put(boost::edge_weight, graph,
384 boost::add_edge(u, consistent[k].second, graph).first,
385 consistent[k].first);
386 }
387 }
388 }

400 void buildClusters(const TPointVec& points,
401 const TSizeVecVecVec& bootstrapClusters,
402 const TGraph& graph,
403 TPointVecVec& result) const {
404 using TSizeSizeUMap = boost::unordered_map<std::size_t, std::size_t>;
405 using TSizeSizeUMapCItr = TSizeSizeUMap::const_iterator;
406 using TSizeSizeUMapVec = std::vector<TSizeSizeUMap>;
407
408 // Find the maximum connected components.
409 TSizeVec components(boost::num_vertices(graph));
410 std::size_t n = boost::connected_components(graph, &components[0]);
411 LOG_TRACE(<< "# vertices = " << components.size());
412 LOG_TRACE(<< "Connected components = " << n);
413
414 // Find components which aren't easily separable. These will
415 // be the voting population.
416 n = this->thickets(n, graph, components);
417 LOG_TRACE(<< "thickets = " << n);
418
419 // Build a map from voters to point indices.
420 TSizeSizeUMapVec voters(n);
421 for (std::size_t i = 0; i < components.size(); ++i) {
422 TSizeSizeUMap& cluster = voters[components[i]];
423 const TSizeVec& vertex = this->fromVertex(bootstrapClusters, i);
424 for (std::size_t j = 0; j < vertex.size(); ++j) {
425 ++cluster[vertex[j]];
426 }
427 }
428
429 // Extract clusters via majority vote.
430 result.clear();
431 result.resize(voters.size());
432 for (std::size_t i = 0; i < points.size(); ++i) {
433 std::size_t jmax = 0;
434 std::size_t cmax = 0;
435 std::size_t nmax = 0;
436 for (std::size_t j = 0; j < n; ++j) {
437 TSizeSizeUMapCItr k = voters[j].find(i);
438 if (k == voters[j].end()) {
439 continue;
440 }
441 std::size_t c = k->second;
442 std::size_t n_ = voters[j].size();
443 if (COrderings::lexicographicalCompareWith(std::greater<>(), c,
444 n_, cmax, nmax)) {
445 jmax = j;
446 cmax = c;
447 nmax = n_;
448 }
449 }
450 if (cmax == 0) {
451 LOG_ERROR(<< "Failed to find cluster for " << points[i]);
452 continue;
453 }
454
455 result[jmax].push_back(points[i]);
456 }
457
458 // It is possible that after voting clusters contain
459 // no points. Remove these.
460 result.erase(std::remove_if(result.begin(), result.end(), SIsEmpty()),
461 result.end());
462 }

473 std::size_t thickets(std::size_t n, const TGraph& graph, TSizeVec& components) const {
474 std::size_t V = boost::num_vertices(graph);
475
476 TSizeVec mapping(V);
477 TSizeVec inverse;
478 TBoolVec parities(V);
479 TGraph component(1);
480
481 for (std::size_t i = 0; i < n; ++i) {
482 LOG_TRACE(<< "component = " << i);
483
484 // Extract the component vertices.
485 inverse.clear();
486 for (std::size_t j = 0; j < V; ++j) {
487 if (components[j] == i) {
488 inverse.push_back(j);
489 mapping[j] = inverse.size() - 1;
490 }
491 }
492
493 std::size_t Vi = inverse.size();
494
495 if (Vi < 3) {
496 continue;
497 }
498
499 // Build the component graph.
500 this->copy(graph, mapping, inverse, component);
501
502 // Find the partitions of the component which are difficult
503 // to separate (by removing edges).
504 if (this->separate(component, parities)) {
505 LOG_TRACE(<< "Separated component");
506 LOG_TRACE(<< "parities = "
507 << core::CContainerPrinter::print(parities.begin(),
508 parities.begin() + Vi));
509 for (std::size_t j = 0; j < Vi; ++j) {
510 if (parities[j]) {
511 components[inverse[j]] = n;
512 }
513 }
514 LOG_TRACE(<< "components = " << components);
515 ++n;
516 }
517 }
518
519 return n;
520 }

536 bool separate(const TGraph& graph, TBoolVec& result) const {
537 std::size_t V = boost::num_vertices(graph);
538 std::size_t E = boost::num_edges(graph);
539
540 result.assign(V, true);
541
542 std::size_t D = V;
543 for (std::size_t i = 0; i < V; ++i) {
544 D = std::min(D, boost::out_degree(i, graph));
545 }
546 TDoubleVec weights;
547 weights.reserve(E);
548 double totalWeight = 0.0;
549 {
550 TEdgeItr i, end;
551 for (boost::tie(i, end) = boost::edges(graph); i != end; ++i) {
552 double weight = boost::get(boost::edge_weight, graph, *i);
553 weights.push_back(weight);
554 totalWeight += weight;
555 }
556 }
557
558 double p = totalWeight / static_cast<double>(V * (V - 1) / 2);
559 double threshold = SEPARATION_THRESHOLD * p;
560
561 // We can bound the ratio of the cut size to the typical
562 // by noting that in most separable configuration we'd
563 // remove all edges we're short of a complete graph from
564 // the cut. This is a poor bound when the split is uneven,
565 // i.e. unless the graph is close to complete
566 // V (V - 1) / 2 - E > V - 1 so the graph could disconnected.
567 // In this case, we can bound the ratio based on the minimum
568 // vertex degree D for cuts in which one component contains
569 // fewer than D + 1 vertices.
570 double bound = std::numeric_limits<double>::max();
571 std::sort(weights.begin(), weights.end());
572 for (std::size_t i = 1; i < weights.size(); ++i) {
573 weights[i] += weights[i - 1];
574 }
575 for (std::size_t i = 1; i <= V / 2 + 1; ++i) {
576 std::size_t C = std::max(
577 i * (D - std::min(D, i - 1)),
578 (i * (V - i)) - std::min(i * (V - i), (V * (V - 1)) / 2 - E));
579 bound = std::min(bound, weights[C] / static_cast<double>(i * (V - i)));
580 }
581 LOG_TRACE(<< "bound = " << bound << " threshold = " << threshold);
582
583 if (bound >= threshold) {
584 LOG_TRACE(<< "Short circuit: D = " << D << ", V = " << V << ", bound = " << bound
585 << ", threshold = " << SEPARATION_THRESHOLD * p);
586 return false;
587 }
588
589 TDoubleVec seeds;
590 CSampling::uniformSample(m_Rng, 0.0, 1.0, 6, seeds);
591
592 TSizeSizePrVec cut;
593 TSizeSizePrVec newCut;
594 for (std::size_t i = 0; i < seeds.size(); ++i) {
595 if (cut.empty()) {
596 TEdgeItr seed = boost::edges(graph).first;
597 for (std::size_t j = 0;
598 j < static_cast<std::size_t>(seeds[i] * static_cast<double>(E));
599 ++j, ++seed) {
600 }
601 cut.push_back(std::make_pair(boost::source(*seed, graph),
602 boost::target(*seed, graph)));
603 }
604
605 double cost;
606 if (this->cutSearch(cut.back().first, cut.back().second, graph,
607 threshold, cost, result)) {
608 return true;
609 }
610
611 cut.pop_back();
612 std::size_t n = cut.size();
613
614 TEdgeItr j, end;
615 for (boost::tie(j, end) = boost::edges(graph); j != end; ++j) {
616 std::size_t u = boost::source(*j, graph);
617 std::size_t v = boost::target(*j, graph);
618 if (result[u] != result[v]) {
619 cut.push_back(std::make_pair(u, v));
620 }
621 }
622 if (n > 0) {
623 std::sort(cut.begin() + n, cut.end());
624 newCut.clear();
625 std::set_intersection(cut.begin(), cut.begin() + n, cut.begin() + n,
626 cut.end(), std::back_inserter(newCut));
627 cut.swap(newCut);
628 }
629 }
630
631 return false;
632 }

661 bool cutSearch(std::size_t u,
662 std::size_t v,
663 const TGraph& graph,
664 double threshold,
665 double& cost,
666 TBoolVec& parities) const {
667 LOG_TRACE(<< "Seed edge = (" << u << "," << v << ")");
668
669 std::size_t V = boost::num_vertices(graph);
670
671 parities.assign(V, true);
672
673 SCutState state(u, graph);
674 std::size_t next = state.next();
675 this->visit(next, graph, parities, state);
676 next = state.nextToVisit(v);
677 this->visit(next, graph, parities, state);
678
679 double lowestCost = state.cost();
680 double bestCut = state.s_Cut;
681 std::size_t bestA = state.s_A;
682 TBoolVec best = parities;
683
684 while (state.s_A + 1 < V) {
685 if (!this->findNext(parities, graph, state)) {
686 LOG_TRACE(<< "The positive subgraph is already disconnected");
687
688 TSizeVec components;
689 std::size_t c = this->positiveSubgraphConnected(graph, parities, components);
690 LOG_TRACE(<< "components = " << components);
691
692 // Find the smallest component.
693 TSizeVec sizes(c, 0);
694 for (std::size_t i = 0; i < components.size(); ++i) {
695 if (parities[i]) {
696 ++sizes[components[i]];
697 }
698 }
699 std::size_t smallest = static_cast<std::size_t>(
700 std::min_element(sizes.begin(), sizes.end()) - sizes.begin());
701 LOG_TRACE(<< "sizes = " << sizes);
702 LOG_TRACE(<< "smallest = " << smallest);
703
704 // Add all its vertices to the "to visit" set.
705 std::size_t n = state.s_ToVisit.size();
706 for (std::size_t i = 0; i < components.size(); ++i) {
707 if (parities[i] && components[i] == smallest && !state.toVisit(i)) {
708 state.s_ToVisit.push_back(i);
709 }
710 }
711
712 state.mergeAfter(n);
713
714 for (std::size_t i = 0; i < components.size(); ++i) {
715 if (parities[i] && components[i] == smallest) {
716 next = state.nextToVisit(i);
717 this->visit(next, graph, parities, state);
718 }
719 }
720 } else {
721 next = state.next();
722 this->visit(next, graph, parities, state);
723 }
724
725 double cutCost = state.cost();
726 if (cutCost < lowestCost) {
727 lowestCost = cutCost;
728 bestCut = state.s_Cut;
729 bestA = state.s_A;
730 best = parities;
731 }
732 }
733
734 cost = lowestCost;
735 parities.swap(best);
736
737 LOG_TRACE(<< "Best cut = " << bestCut << ", |A| = " << bestA << ", |B| = " << V - bestA
738 << ", cost = " << cost << ", threshold = " << threshold);
739
740 return cost < threshold;
741 }

781 void copy(const TGraph& graph, const TSizeVec& mapping, const TSizeVec& inverse, TGraph& result) const {
782 result = TGraph(inverse.size());
783 for (std::size_t i = 0; i < inverse.size(); ++i) {
784 TOutEdgeItr j, end;
785 for (boost::tie(j, end) = boost::out_edges(inverse[i], graph); j != end; ++j) {
786 std::size_t u = boost::source(*j, graph);
787 std::size_t v = boost::target(*j, graph);
788 if (u < v && std::binary_search(inverse.begin(), inverse.end(), v)) {
789 boost::put(boost::edge_weight, result,
790 boost::add_edge(mapping[u], mapping[v], result).first,
791 boost::get(boost::edge_weight, graph, *j));
792 }
793 }
794 }

801 bool findNext(const TBoolVec& parities, const TGraph& graph, SCutState& state) const {
802 state.initializeQueue();
803 TSizeVec components;
804 for (std::size_t i = 0; i < state.s_ToVisit.size(); ++i) {
805 std::size_t candidate = state.next();
806 std::size_t v = state.s_ToVisit[candidate];
807 const_cast<TBoolVec&>(parities)[v] = false;
808 bool connected =
809 (this->positiveSubgraphConnected(graph, parities, components) == 1);
810 const_cast<TBoolVec&>(parities)[v] = true;
811 if (connected) {
812 return true;
813 }
814 state.popQueue();
815 }
816 return false;
817 }

824 void visit(std::size_t next, const TGraph& graph, TBoolVec& parities, SCutState& state) const {
825 std::size_t u = state.s_ToVisit[next];
826 LOG_TRACE(<< "Visiting " << u);
827
828 parities[u] = false;
829 state.s_ToVisit.erase(state.s_ToVisit.begin() + next);
830
831 std::size_t n = state.s_ToVisit.size();
832 TOutEdgeItr i, end;
833 for (boost::tie(i, end) = boost::out_edges(u, graph); i != end; ++i) {
834 double weight = boost::get(boost::edge_weight, graph, *i);
835 std::size_t v = boost::target(*i, graph);
836 if (parities[v]) {
837 state.s_Adjacency[v] += weight;
838 state.s_Cut += weight;
839 if (!state.toVisit(n, v)) {
840 state.s_ToVisit.push_back(v);
841 }
842 } else {
843 state.s_Cut -= weight;
844 }
845 }
846
847 state.mergeAfter(n);
848 state.s_Adjacency[u] = 0;
849 ++state.s_A;
850 }

856 using TParityGraph = boost::filtered_graph<TGraph, CParityFilter, CParityFilter>;
857 CParityFilter parityFilter(graph, parities, true);
858 TParityGraph parityGraph(graph, parityFilter, parityFilter);
859 components.resize(boost::num_vertices(graph));
860 return boost::connected_components(parityGraph, &components[0]);
861 }

911class CBootstrapClustererFacadeExtractClusters {
912public:
913 using TSizeVec = std::vector<std::size_t>;
914 using TSizeVecVec = std::vector<TSizeVec>;
915 using TPointVec = std::vector<POINT>;
916 using TPointVecCItr = typename TPointVec::const_iterator;
917
918public:
925 template<typename CLUSTERS>
926 void extract(const TPointVec& points, const CLUSTERS& clusters, TSizeVecVec& result) {
927
928 result.resize(clusters.size());
929
930 for (std::size_t i = 0; i < clusters.size(); ++i) {
931 const TPointVec& clusterPoints = clusters[i];
932
933 result[i].clear();
934 result[i].reserve(clusterPoints.size());
935
936 for (std::size_t j = 0; j < clusterPoints.size(); ++j) {
937 std::size_t k = points.size();
938 for (TPointVecCItr l = this->begin(points, clusterPoints[j]),
939 end = this->end(points, clusterPoints[j]);
940 l != end; ++l) {
941 if (*l == clusterPoints[j]) {
942 k = static_cast<std::size_t>(l - points.begin());
943 break;
944 }
945 }
946
947 if (k == points.size()) {
948 LOG_ERROR(<< "Didn't find point " << clusterPoints[j]);
949 continue;
950 }
951
952 result[i].push_back(k);
953 }
954 }
955 }
956
957private:
959 TPointVecCItr begin(const TPointVec& points, const POINT& x) const {
960 return std::lower_bound(points.begin(), points.end(), x);
961 }
962
964 TPointVecCItr end(const TPointVec& points, const POINT& x) const {
965 return std::upper_bound(points.begin(), points.end(), x);
966 }
