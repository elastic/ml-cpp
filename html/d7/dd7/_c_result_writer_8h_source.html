89 template<std::size_t N>
90 void writePrediction(const ::torch::Tensor& prediction, TStringBufWriter& jsonWriter) {
91
92 // Creating the accessor will throw if the tensor does not have exactly
93 // N dimensions. Do this before writing any output so the error message
94 // isn't mingled with a partial result.
95
96 if (prediction.dtype() == ::torch::kFloat32) {
97 auto accessor = prediction.accessor<float, N>();
98 this->writeInferenceResults(accessor, jsonWriter);
99
100 } else if (prediction.dtype() == ::torch::kFloat64) {
101 auto accessor = prediction.accessor<double, N>();
102 this->writeInferenceResults(accessor, jsonWriter);
103
104 } else if (prediction.dtype() == ::c10::kBFloat16) {
105 // cast the tensor to float32 and then write it
106 auto float32Tensor = prediction.toType(::torch::kFloat32);
107 auto accessor = float32Tensor.accessor<float, N>();
108 this->writeInferenceResults(accessor, jsonWriter);
109
110 } else {
111 std::ostringstream ss;
112 ss << "Cannot process result tensor of type [" << prediction.dtype() << ']';
113 writeInnerError(ss.str(), jsonWriter);
114 }
115 }

145 template<typename T>
146 void writeTensor(const ::torch::TensorAccessor<T, 1UL>& accessor,
147 TStringBufWriter& jsonWriter) {
148 jsonWriter.onArrayBegin();
149 for (int i = 0; i < accessor.size(0); ++i) {
150 jsonWriter.onDouble(static_cast<double>(accessor[i]));
151 }
152 jsonWriter.onArrayEnd();
153 }

156 template<typename T, std::size_t N_DIMS>
157 void writeTensor(const ::torch::TensorAccessor<T, N_DIMS>& accessor,
158 TStringBufWriter& jsonWriter) {
159 jsonWriter.onArrayBegin();
160 for (int i = 0; i < accessor.size(0); ++i) {
161 this->writeTensor(accessor[i], jsonWriter);
162 }
163 jsonWriter.onArrayEnd();
164 }

167 template<typename T>
168 void writeInferenceResults(const ::torch::TensorAccessor<T, 3UL>& accessor,
169 TStringBufWriter& jsonWriter) {
170
171 jsonWriter.onKey(RESULT);
172 jsonWriter.onObjectBegin();
173 jsonWriter.onKey(INFERENCE);
174 this->writeTensor(accessor, jsonWriter);
175 jsonWriter.onObjectEnd();
176 }
177
179 template<typename T>
180 void writeInferenceResults(const ::torch::TensorAccessor<T, 2UL>& accessor,
181 TStringBufWriter& jsonWriter) {
182
183 jsonWriter.onKey(RESULT);
184 jsonWriter.onObjectBegin();
185 jsonWriter.onKey(INFERENCE);
186 // The Java side requires a 3D array, so wrap the 2D result in an
187 // extra outer array.
188 jsonWriter.onArrayBegin();
189 this->writeTensor(accessor, jsonWriter);
190 jsonWriter.onArrayEnd();
191 jsonWriter.onObjectEnd();
192 }
