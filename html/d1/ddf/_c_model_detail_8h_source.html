34template<typename TREND, typename VECTOR>
35std::optional<VECTOR> CModel::predictionError(const TREND& trend, const VECTOR& sample) {
36 std::optional<VECTOR> result;
37 std::size_t dimension = sample.size();
38 for (std::size_t i = 0; i < dimension; ++i) {
39 if (trend[i]->initialized()) {
40 result.emplace(dimension, 0.0);
41 for (; i < dimension; ++i) {
42 if (trend[i]->initialized()) {
43 (*result)[i] = sample[i];
44 }
45 }
46 }
47 }
48 return result;
49}

51template<typename PRIOR, typename VECTOR>
52std::optional<VECTOR> CModel::predictionError(double propagationInterval,
53 const PRIOR& prior,
54 const VECTOR& sample) {
55 std::optional<VECTOR> result;
56 if (prior->numberSamples() > 20.0 / propagationInterval) {
57 std::size_t dimension{sample.size()};
58 result.emplace(sample);
59 VECTOR mean(marginalLikelihoodMean<VECTOR>(*prior));
60 for (std::size_t d = 0; d < dimension; ++d) {
61 (*result)[d] -= mean[d];
62 }
63 }
64 return result;
65}
