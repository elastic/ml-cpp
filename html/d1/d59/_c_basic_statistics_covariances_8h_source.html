47template<typename POINT>
48struct SCovariancesLedoitWolf {
49 template<typename OTHER_POINT>
50 static void estimate(const std::vector<POINT>& points,
51 CBasicStatistics::SSampleCovariances<OTHER_POINT>& covariances) {
52 if (points.empty()) {
53 return;
54 }
55
56 using TCoordinate = typename SCoordinate<OTHER_POINT>::Type;
57 using TMatrix = typename SConformableMatrix<OTHER_POINT>::Type;
58
59 std::size_t dimension{las::dimension(points[0])};
60 TCoordinate d{static_cast<TCoordinate>(dimension)};
61
62 TCoordinate n{CBasicStatistics::count(covariances)};
63 if (n == TCoordinate{0}) {
64 return;
65 }
66
67 const OTHER_POINT& m{CBasicStatistics::mean(covariances)};
68 const TMatrix& s{CBasicStatistics::maximumLikelihoodCovariances(covariances)};
69
70 TCoordinate mn{s.trace() / d};
71 TCoordinate norm(0);
72 for (std::size_t i = 0; i < dimension; ++i) {
73 norm += pow2(s(i, i) - mn);
74 for (std::size_t j = 0; j < i; ++j) {
75 norm += TCoordinate(2) * pow2(s(i, j));
76 }
77 }
78 TCoordinate dn{norm / d};
79 TCoordinate bn(0);
80 TCoordinate z{n * n};
81 for (const auto& point : points) {
82 norm = TCoordinate(0);
83 for (std::size_t i = 0; i < dimension; ++i) {
84 norm += pow2(pow2(TCoordinate(point(i)) - m(i)) - s(i, i));
85 for (std::size_t j = 0; j < i; ++j) {
86 norm += TCoordinate(2) * pow2((TCoordinate(point(i)) - m(i)) *
87 (TCoordinate(point(j)) - m(j)) -
88 s(i, j));
89 }
90 }
91 bn += norm / d / z;
92 }
93 bn = std::min(bn, dn);
94 LOG_TRACE(<< "m = " << mn << ", d = " << dn << ", b = " << bn);
95
96 covariances.s_Covariances *=
97 std::max((TCoordinate(1) - (bn == dn ? 1.0 : bn / dn)), 0.0);
98 for (std::size_t i = 0; i < dimension; ++i) {
99 covariances.s_Covariances(i, i) += (bn == dn ? 1.0 : bn / dn) * mn;
100 }
101 }

138template<typename POINT>
139void CBasicStatistics::SSampleCovariances<POINT>::add(const TVector& x, const TVector& n, int) {
140 if (!las::isZero(n)) {
141 s_Count += n;
142
143 // Note we don't trap the case alpha is less than epsilon,
144 // because then we'd have to compute epsilon and it is very
145 // unlikely the count will get big enough.
146 TVector alpha{las::componentwise(n) / las::componentwise(s_Count)};
147 TVector beta{las::ones(alpha) - alpha};
148
149 TVector mean{s_Mean};
150 s_Mean = las::componentwise(beta) * las::componentwise(mean) +
151 las::componentwise(alpha) * las::componentwise(x);
152
153 TVector r{x - s_Mean};
154 TMatrix r2{las::outer(r)};
155 TVector dMean{mean - s_Mean};
156 TMatrix dMean2{las::outer(dMean)};
157
158 s_Covariances += dMean2;
159 scaleCovariances(beta, s_Covariances);
160 scaleCovariances(alpha, r2);
161 s_Covariances += r2;
162 }
163}

167template<typename POINT>
168template<typename OTHER_POINT>
169const CBasicStatistics::SSampleCovariances<POINT>&
170CBasicStatistics::SSampleCovariances<POINT>::
171operator+=(const SSampleCovariances<OTHER_POINT>& rhs) {
172 s_Count = s_Count + rhs.s_Count;
173 if (!las::isZero(s_Count)) {
174 // Note we don't trap the case alpha is less than epsilon,
175 // because then we'd have to compute epsilon and it is very
176 // unlikely the count will get big enough.
177 TVector alpha{las::componentwise(rhs.s_Count) / las::componentwise(s_Count)};
178 TVector beta{las::ones(alpha) - alpha};
179
180 TVector meanLhs{s_Mean};
181
182 s_Mean = las::componentwise(beta) * las::componentwise(meanLhs) +
183 las::componentwise(alpha) * las::componentwise(rhs.s_Mean);
184
185 TVector dMeanLhs{meanLhs - s_Mean};
186 TMatrix dMean2Lhs{las::outer(dMeanLhs)};
187 TVector dMeanRhs{rhs.s_Mean - s_Mean};
188 TMatrix dMean2Rhs{las::outer(dMeanRhs)};
189
190 s_Covariances += dMean2Lhs;
191 scaleCovariances(beta, s_Covariances);
192 dMean2Rhs += rhs.s_Covariances;
193 scaleCovariances(alpha, dMean2Rhs);
194 s_Covariances += dMean2Rhs;
195 }
196
197 return *this;
198}

200template<typename POINT>
201template<typename OTHER_POINT>
202const CBasicStatistics::SSampleCovariances<POINT>&
203CBasicStatistics::SSampleCovariances<POINT>::
204operator-=(const SSampleCovariances<OTHER_POINT>& rhs) {
205 using TCoordinate = typename SCoordinate<POINT>::Type;
206
207 s_Count = las::max(s_Count - rhs.s_Count, las::zero(s_Count));
208 if (!las::isZero(s_Count)) {
209 // Note we don't trap the case alpha is less than epsilon,
210 // because then we'd have to compute epsilon and it is very
211 // unlikely the count will get big enough.
212 TVector alpha{las::componentwise(rhs.s_Count) / las::componentwise(s_Count)};
213 TVector beta{las::ones(alpha) + alpha};
214
215 TVector meanLhs(s_Mean);
216
217 s_Mean = las::componentwise(beta) * las::componentwise(meanLhs) -
218 las::componentwise(alpha) * las::componentwise(rhs.s_Mean);
219
220 TVector dMeanLhs{s_Mean - meanLhs};
221 TMatrix dMean2Lhs{las::outer(dMeanLhs)};
222 TVector dMeanRhs{rhs.s_Mean - meanLhs};
223 TMatrix dMean2Rhs{las::outer(dMeanRhs)};
224
225 s_Covariances = s_Covariances - dMean2Lhs;
226 scaleCovariances(beta, s_Covariances);
227 dMean2Rhs += rhs.s_Covariances - dMean2Lhs;
228 scaleCovariances(alpha, dMean2Rhs);
229 s_Covariances -= dMean2Rhs;
230
231 // If any of the diagonal elements are negative round them
232 // up to zero and zero the corresponding row and column.
233 for (std::size_t i = 0, dimension = las::dimension(s_Mean); i < dimension; ++i) {
234 if (s_Covariances(i, i) < TCoordinate{0}) {
235 for (std::size_t j = 0; j < dimension; ++j) {
236 s_Covariances(i, j) = s_Covariances(j, i) = TCoordinate{0};
237 }
238 }
239 }
240 } else {
241 s_Mean = las::zero(s_Mean);
242 s_Covariances = las::zero(s_Covariances);
243 }
244
245 return *this;
246}

256template<typename POINT>
257typename SConformableMatrix<POINT>::Type
258CBasicStatistics::covariances(const SSampleCovariances<POINT>& accumulator) {
259 using TCoordinate = typename SCoordinate<POINT>::Type;
260 using TMatrix = typename SConformableMatrix<POINT>::Type;
261
262 POINT bias(accumulator.s_Count);
263 for (std::size_t i = 0; i < las::dimension(bias); ++i) {
264 if (bias(i) > TCoordinate{1}) {
265 bias(i) /= bias(i) - TCoordinate(1);
266 } else {
267 bias(i) = TCoordinate{0};
268 }
269 }
270
271 TMatrix result{accumulator.s_Covariances};
272 scaleCovariances(bias, result);
273
274 return result;
275}
