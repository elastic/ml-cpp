335 template<typename T>
336 bool featureData(core_t::TTime time,
337 core_t::TTime bucketLength,
338 std::vector<std::pair<model_t::EFeature, T>>& result) const {
339 TFeatureAnyPrVec rawFeatureData;
340 m_BucketGatherer->featureData(time, bucketLength, rawFeatureData);
341
342 bool succeeded = true;
343
344 result.clear();
345 result.reserve(rawFeatureData.size());
346 for (std::size_t i = 0; i < rawFeatureData.size(); ++i) {
347 TFeatureAnyPr& feature = rawFeatureData[i];
348
349 // Check the typeid before attempting the cast so we
350 // don't use throw to handle failure, which is slow.
351 if (feature.second.type() != typeid(T)) {
352 LOG_ERROR(<< "Bad type for feature = " << model_t::print(feature.first)
353 << ", expected " << typeid(T).name() << " got "
354 << feature.second.type().name());
355 succeeded = false;
356 continue;
357 }
358
359 // We emulate move semantics here to avoid the expensive
360 // copy if T is large (as we expect it might be sometimes).
361 // We have to adopt the using std::swap idiom (contravening
362 // coding guidelines) because T can be a built in type.
363 // Unfortunately, this implementation requires T to be
364 // default constructible.
365 using std::swap;
366 result.push_back(std::pair<model_t::EFeature, T>(feature.first, T()));
367 T& tmp = std::any_cast<T&>(feature.second);
368 swap(result.back().second, tmp);
369 }
370
371 return succeeded;
372 }
