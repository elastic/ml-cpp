25// The logic in this function is rather subtle because we want to sort the
26// collections in place. In particular, we create a sorted collection of
27// indices where each index tells us where to get the element from at that
28// location and we want to re-order all the collections by that ordering in
29// place. If an index matches its position then we can move to the next
30// position. Otherwise, we need to swap the items at the index in to its
31// position. To work in place we need to do something with the items which
32// are displaced. If these are the items required at the swapped in position
33// then we are done. Otherwise, we just repeat until we find this position.
34// It is easy to verify that this process finds a closed cycle with at most
35// N steps. Each time a swap is made at least one more item is in its correct
36// place, and we update the ordering accordingly. So the containers are sorted
37// in at most O(N) additional steps to the N * log(N) taken to sort the indices.
38template<typename PRED, typename K, typename... V>
39bool COrderings::simultaneousSortWith(const PRED& pred, K&& keys, V&&... values) {
40 if (((keys.size() != values.size()) && ...)) {
41 return false;
42 }
43 if (std::is_sorted(keys.begin(), keys.end(), pred)) {
44 return true;
45 }
46
47 std::vector<std::size_t> ordering(keys.size());
48 std::iota(ordering.begin(), ordering.end(), 0);
49 std::stable_sort(ordering.begin(), ordering.end(), [&](auto lhs, auto rhs) {
50 return pred(keys[lhs], keys[rhs]);
51 });
52
53 using std::swap;
54 auto swapValue = [](auto& value, auto j_, auto j) {
55 swap(value[j_], value[j]);
56 };
57 for (std::size_t i = 0; i < ordering.size(); ++i) {
58 std::size_t j_ = i;
59 std::size_t j = ordering[j_];
60 while (i != j) {
61 swap(keys[j_], keys[j]);
62 (swapValue(values, j_, j), ...);
63 ordering[j_] = j_;
64 j_ = j;
65 j = ordering[j_];
66 }
67 ordering[j_] = j_;
68 }
69 return true;
70}
71}
