54 template<typename LOGF>
55 class CSmoothedKernel : private core::CNonCopyable {
56 public:
57 CSmoothedKernel(LOGF logf, double logF0, double k)
58 : m_LogF(logf), m_LogF0(logF0), m_K(k),
59 m_Scale(std::exp(m_LogF0) * (1.0 + std::exp(-k))) {}
60
61 void k(double k) {
62 double f0 = m_Scale / (1.0 + std::exp(-m_K));
63 m_K = k;
64 m_Scale = f0 * (1.0 + std::exp(-k));
65 }
66
67 bool operator()(double x, double& result) const {
68 // We use the fact that if:
69 // 1 + exp(-k(f(x)/f0 - 1)) < (1 + eps) * exp(-k(f(x)/f0 - 1))
70 //
71 // then the kernel = scale to working precision. Canceling
72 // O(1) terms in the exponential, taking logs and using the
73 // fact that the log is monotonic increasing function this
74 // reduces to
75 // 0 < -k(f(x)/f0 - 1) + log(eps)
76 //
77 // which implies that we can simplify if
78 // f(x)/f0 < 1 + log(eps)/k
79
80 result = 0.0;
81 double logFx;
82 if (!m_LogF(x, logFx)) {
83 LOG_ERROR(<< "Failed to calculate likelihood at " << x);
84 return false;
85 }
86 logFx -= m_LogF0;
87 if (m_K * (logFx - 1.0) >= core::constants::LOG_MAX_DOUBLE) {
88 return true;
89 }
90 double fx = std::exp(logFx);
91 if (fx < 1.0 + core::constants::LOG_DOUBLE_EPSILON / m_K) {
92 result = m_Scale * fx;
93 return true;
94 }
95 result = m_Scale / (1.0 + std::exp(m_K * (fx - 1.0))) * fx;
96 return true;
97 }
98
99 private:
100 LOGF m_LogF;
101 double m_LogF0;
102 double m_K;
103 double m_Scale;
104 };

145 template<typename LOGF, typename EQUAL>
146 bool leftTail(const LOGF& logf, std::size_t iterations, const EQUAL& equal, double& result) const {
147 if (m_X <= m_A) {
148 result = m_X;
149 return true;
150 }
151
152 CCompositeFunctions::CMinusConstant<const LOGF&> f(logf, m_LogFx);
153
154 try {
155 double xr = m_A;
156 double fr = f(xr);
157 if (fr < 0.0) {
158 result = m_A;
159 return true;
160 }
161 double xl = xr;
162 double fl = fr;
163 if (m_MaxDeviation.count() > 0) {
164 xl = xr - m_MaxDeviation[0];
165 fl = f(xl);
166 }
167
168 iterations = std::max(iterations, std::size_t(4));
169 std::size_t n = iterations - 2;
170 if (!CSolvers::leftBracket(xl, xr, fl, fr, f, n)) {
171 result = xl;
172 return false;
173 }
174 n = iterations - n;
175 CSolvers::solve(xl, xr, fl, fr, f, n, equal, result);
176 } catch (const std::exception& e) {
177 LOG_ERROR(<< "Failed to find left root: " << e.what() << ", a = " << m_A
178 << ", logf(x) = " << m_LogFx << ", logf(a) = " << logf(m_A) << ", max deviation = "
179 << (m_MaxDeviation.count() > 0 ? m_MaxDeviation[0] : 0.0));
180 return false;
181 }
182 return true;
183 }

200 template<typename LOGF, typename EQUAL>
201 bool rightTail(const LOGF& logf, std::size_t iterations, const EQUAL& equal, double& result) const {
202 if (m_X >= m_B) {
203 result = m_X;
204 return true;
205 }
206
207 CCompositeFunctions::CMinusConstant<const LOGF&> f(logf, m_LogFx);
208
209 try {
210 double xl = m_B;
211 double fl = f(xl);
212 if (fl < 0.0) {
213 result = m_B;
214 return true;
215 }
216 double xr = xl;
217 double fr = fl;
218 if (m_MaxDeviation.count() > 0) {
219 xr = xl + m_MaxDeviation[0];
220 fr = f(xr);
221 }
222
223 iterations = std::max(iterations, std::size_t(4));
224 std::size_t n = iterations - 2;
225 if (!CSolvers::rightBracket(xl, xr, fl, fr, f, n)) {
226 result = xr;
227 return false;
228 }
229 n = iterations - n;
230 CSolvers::solve(xl, xr, fl, fr, f, n, equal, result);
231 } catch (const std::exception& e) {
232 LOG_ERROR(<< "Failed to find right root: " << e.what() << ",b = " << m_B
233 << ", logf(x) = " << m_LogFx << ", logf(b) = " << logf(m_B));
234 return false;
235 }
236 return true;

252 template<typename LOGF>
253 double calculate(const LOGF& logf, double pTails) {
254 TDoubleDoublePrVec intervals;
255 this->intervals(intervals);
256
257 double p = 0.0;
258 TDoubleVec pIntervals(intervals.size(), 0.0);
259 CSmoothedKernel<const LOGF&> kernel(logf, m_LogFx, 3.0);
260 for (std::size_t i = 0; i < intervals.size(); ++i) {
261 if (!CIntegration::gaussLegendre<CIntegration::OrderFour>(
262 kernel, intervals[i].first, intervals[i].second, pIntervals[i])) {
263 LOG_ERROR(<< "Couldn't integrate kernel over " << intervals[i]);
264 }
265 }
266
267 p += pTails;
268 kernel.k(15.0);
269 CIntegration::adaptiveGaussLegendre<CIntegration::OrderTwo>(kernel, intervals, pIntervals,
270 2, // refinements
271 3, // splits
272 1e-2, // tolerance
273 p);
274 return truncate(p - pTails, 0.0, 1.0);
275 }

302template<typename T>
303double CTools::differentialEntropy(const CMixtureDistribution<T>& mixture) {
304 using TModeVec = typename CMixtureDistribution<T>::TModeVec;
305
306 static const double EPS = 1e-5;
307 static const std::size_t INTERVALS = 8;
308
309 const TDoubleVec& weights = mixture.weights();
310 const TModeVec& modes = mixture.modes();
311
312 if (weights.empty()) {
313 return 0.0;
314 }
315
316 TDoubleDoublePrVec range;
317 for (std::size_t i = 0; i < modes.size(); ++i) {
318 range.push_back(TDoubleDoublePr(quantile(modes[i], EPS),
319 quantile(modes[i], 1.0 - EPS)));
320 }
321 std::sort(range.begin(), range.end(), COrderings::SFirstLess());
322 LOG_TRACE(<< "range = " << range);
323 std::size_t left = 0;
324 for (std::size_t i = 1; i < range.size(); ++i) {
325 if (range[left].second < range[i].first) {
326 ++left;
327 std::swap(range[left], range[i]);
328 } else {
329 range[left].second = std::max(range[left].second, range[i].second);
330 }
331 }
332 range.erase(range.begin() + left + 1, range.end());
333 LOG_TRACE(<< "range = " << range);
334
335 double result = 0.0;
336
337 CDifferentialEntropyKernel<T> kernel(mixture);
338 for (auto[a, b] : range) {
339 double d = (b - a) / static_cast<double>(INTERVALS);
340 for (std::size_t j = 0; j < INTERVALS; ++j, a += d) {
341 double integral;
342 if (CIntegration::gaussLegendre<CIntegration::OrderFive>(kernel, a, a + d, integral)) {
343 result += integral;
344 }
345 }
346 }
347
348 LOG_TRACE(<< "result = " << result);
349 return result;
350}

352template<typename COLLECTION>
353void CTools::inplaceSoftmax(COLLECTION& z) {
354 double Z{0.0};
355 double zmax{*std::max_element(z.begin(), z.end())};
356 for (auto& zi : z) {
357 zi = stableExp(zi - zmax);
358 Z += zi;
359 }
360 for (auto& zi : z) {
361 zi /= Z;
362 }
363}

365template<typename COLLECTION>
366void CTools::inplaceLogSoftmax(COLLECTION& z) {
367 double zmax{*std::max_element(z.begin(), z.end())};
368 for (auto& zi : z) {
369 zi -= zmax;
370 }
371 double logZ{std::log(std::accumulate(z.begin(), z.end(), 0.0, [](double sum, const auto& zi) {
372 return sum + std::exp(zi);
373 }))};
374 for (auto& zi : z) {
375 zi -= logZ;
376 }
377}

379template<typename T>
380void CTools::inplaceSoftmax(CDenseVector<T>& z) {
381 double zmax{z.maxCoeff()};
382 z.array() -= zmax;
383 z.array() = z.array().exp();
384 z /= z.sum();
385}

387template<typename SCALAR>
388void CTools::inplaceLogSoftmax(CDenseVector<SCALAR>& z) {
389 // Handle under/overflow when taking exponentials by subtracting zmax.
390 double zmax{z.maxCoeff()};
391 z.array() -= zmax;
392 double Z{z.array().exp().sum()};
393 z.array() -= stableLog(Z);
394}
